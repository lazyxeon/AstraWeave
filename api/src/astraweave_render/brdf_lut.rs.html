1//! BRDF Integration LUT for split-sum IBL approximation.
2//!
3//! Precomputes a 2D lookup table parameterized by (NdotV, roughness) that stores
4//! the integral of the GGX BRDF split into (scale, bias) terms:
5//!   `specular = prefiltered_color * (F0 * scale + bias)`
6//!
7//! This LUT is resolution-independent and only needs computation once (or when
8//! sample count changes).
9
10use bytemuck::{Pod, Zeroable};
11use wgpu::util::DeviceExt;
12
13// ---------------------------------------------------------------------------
14// GPU types
15// ---------------------------------------------------------------------------
16
17/// Uniform parameters for BRDF LUT generation.
18#[repr(C)]
19#[derive(Debug, Clone, Copy, Pod, Zeroable)]
20pub struct BrdfLutParams {
21    pub lut_size: u32,
22    pub num_samples: u32,
23    pub _pad0: u32,
24    pub _pad1: u32,
25}
26
27// ---------------------------------------------------------------------------
28// Configuration
29// ---------------------------------------------------------------------------
30
31/// Configuration for the BRDF LUT.
32#[derive(Debug, Clone, Copy)]
33pub struct BrdfLutConfig {
34    /// LUT resolution (square). Default: 256.
35    pub size: u32,
36    /// Number of importance samples per pixel. Default: 1024.
37    pub num_samples: u32,
38}
39
40impl Default for BrdfLutConfig {
41    fn default() -> Self {
42        Self {
43            size: 256,
44            num_samples: 1024,
45        }
46    }
47}
48
49// ---------------------------------------------------------------------------
50// BRDF LUT Pass
51// ---------------------------------------------------------------------------
52
53/// Manages GPU resources for BRDF LUT generation.
54pub struct BrdfLutPass {
55    config: BrdfLutConfig,
56    pipeline: wgpu::ComputePipeline,
57    bgl: wgpu::BindGroupLayout,
58    params_buf: wgpu::Buffer,
59    lut_texture: wgpu::Texture,
60    lut_view: wgpu::TextureView,
61    generated: bool,
62}
63
64impl BrdfLutPass {
65    pub fn new(device: &wgpu::Device, config: BrdfLutConfig) -> Self {
66        let fmt = wgpu::TextureFormat::Rgba16Float;
67
68        let lut_texture = device.create_texture(&wgpu::TextureDescriptor {
69            label: Some("brdf_lut"),
70            size: wgpu::Extent3d {
71                width: config.size,
72                height: config.size,
73                depth_or_array_layers: 1,
74            },
75            mip_level_count: 1,
76            sample_count: 1,
77            dimension: wgpu::TextureDimension::D2,
78            format: fmt,
79            usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
80            view_formats: &[],
81        });
82        let lut_view = lut_texture.create_view(&wgpu::TextureViewDescriptor::default());
83
84        let params = BrdfLutParams {
85            lut_size: config.size,
86            num_samples: config.num_samples,
87            _pad0: 0,
88            _pad1: 0,
89        };
90        let params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
91            label: Some("brdf_lut_params"),
92            contents: bytemuck::bytes_of(&params),
93            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
94        });
95
96        let bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
97            label: Some("brdf_lut_bgl"),
98            entries: &[
99                wgpu::BindGroupLayoutEntry {
100                    binding: 0,
101                    visibility: wgpu::ShaderStages::COMPUTE,
102                    ty: wgpu::BindingType::Buffer {
103                        ty: wgpu::BufferBindingType::Uniform,
104                        has_dynamic_offset: false,
105                        min_binding_size: None,
106                    },
107                    count: None,
108                },
109                wgpu::BindGroupLayoutEntry {
110                    binding: 1,
111                    visibility: wgpu::ShaderStages::COMPUTE,
112                    ty: wgpu::BindingType::StorageTexture {
113                        access: wgpu::StorageTextureAccess::WriteOnly,
114                        format: fmt,
115                        view_dimension: wgpu::TextureViewDimension::D2,
116                    },
117                    count: None,
118                },
119            ],
120        });
121
122        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
123            label: Some("brdf_lut_shader"),
124            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/pbr/brdf_lut.wgsl").into()),
125        });
126        let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
127            label: Some("brdf_lut_pl"),
128            bind_group_layouts: &[&bgl],
129            push_constant_ranges: &[],
130        });
131        let pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
132            label: Some("brdf_lut_pipeline"),
133            layout: Some(&pl),
134            module: &shader,
135            entry_point: Some("brdf_lut_main"),
136            compilation_options: Default::default(),
137            cache: None,
138        });
139
140        Self {
141            config,
142            pipeline,
143            bgl,
144            params_buf,
145            lut_texture,
146            lut_view,
147            generated: false,
148        }
149    }
150
151    /// Get the BRDF LUT texture view for binding in material shaders.
152    pub fn lut_view(&self) -> &wgpu::TextureView {
153        &self.lut_view
154    }
155
156    /// Whether the LUT has been generated.
157    pub fn is_generated(&self) -> bool {
158        self.generated
159    }
160
161    pub fn config(&self) -> &BrdfLutConfig {
162        &self.config
163    }
164
165    /// Generate the BRDF LUT. Only needs to be called once.
166    pub fn generate(&mut self, device: &wgpu::Device, encoder: &mut wgpu::CommandEncoder) {
167        if self.generated {
168            return;
169        }
170
171        let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
172            label: Some("brdf_lut_bg"),
173            layout: &self.bgl,
174            entries: &[
175                wgpu::BindGroupEntry {
176                    binding: 0,
177                    resource: self.params_buf.as_entire_binding(),
178                },
179                wgpu::BindGroupEntry {
180                    binding: 1,
181                    resource: wgpu::BindingResource::TextureView(&self.lut_view),
182                },
183            ],
184        });
185
186        let wg = (self.config.size + 7) / 8;
187        let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
188            label: Some("brdf_lut_gen"),
189            timestamp_writes: None,
190        });
191        pass.set_pipeline(&self.pipeline);
192        pass.set_bind_group(0, &bg, &[]);
193        pass.dispatch_workgroups(wg, wg, 1);
194
195        self.generated = true;
196    }
197}

237    #[test]
238    fn hammersley_cpu_validation() {
239        // Validate the radical inverse Van der Corput sequence
240        fn radical_inverse_vdc(mut bits: u32) -> f32 {
241            bits = (bits << 16) | (bits >> 16);
242            bits = ((bits & 0x55555555) << 1) | ((bits & 0xAAAAAAAA) >> 1);
243            bits = ((bits & 0x33333333) << 2) | ((bits & 0xCCCCCCCC) >> 2);
244            bits = ((bits & 0x0F0F0F0F) << 4) | ((bits & 0xF0F0F0F0) >> 4);
245            bits = ((bits & 0x00FF00FF) << 8) | ((bits & 0xFF00FF00) >> 8);
246            bits as f32 * 2.328_306_4e-10
247        }
248
249        // First few values should be well-distributed in [0, 1)
250        let v0 = radical_inverse_vdc(0);
251        let v1 = radical_inverse_vdc(1);
252        let v2 = radical_inverse_vdc(2);
253        assert!(v0 >= 0.0 && v0 < 1.0);
254        assert!(v1 >= 0.0 && v1 < 1.0);
255        assert!(v2 >= 0.0 && v2 < 1.0);
256        // They should be different
257        assert!((v0 - v1).abs() > 0.01);
258        assert!((v1 - v2).abs() > 0.01);
259    }

261    #[test]
262    fn importance_sample_ggx_cpu() {
263        // Validate GGX importance sampling produces valid directions
264        fn importance_sample_ggx(xi: [f32; 2], roughness: f32) -> [f32; 3] {
265            let a = roughness * roughness;
266            let phi = 2.0 * std::f32::consts::PI * xi[0];
267            let cos_theta = ((1.0 - xi[1]) / (1.0 + (a * a - 1.0) * xi[1])).sqrt();
268            let sin_theta = (1.0 - cos_theta * cos_theta).sqrt();
269            [phi.cos() * sin_theta, phi.sin() * sin_theta, cos_theta]
270        }
271
272        let h = importance_sample_ggx([0.5, 0.5], 0.5);
273        let len = (h[0] * h[0] + h[1] * h[1] + h[2] * h[2]).sqrt();
274        assert!((len - 1.0).abs() < 1e-5, "Should be unit vector: {len}");
275        assert!(h[2] >= 0.0, "z should be positive (hemisphere)");
276    }
