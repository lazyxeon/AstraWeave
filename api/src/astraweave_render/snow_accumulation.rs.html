1//! Snow accumulation system — per-chunk compute + PBR snow blending configuration.
2//!
3//! # Architecture
4//!
5//! 1. **Compute pass** (`snow_accumulation.wgsl`): Each frame, updates a
6//!    per-chunk R32Float accumulation heightmap based on snowfall state and
7//!    terrain normals. Upward-facing surfaces gain depth; steep/melting lose it.
8//!
9//! 2. **PBR integration**: The global `snow_amount` in [`SceneEnvironmentUBO`]
10//!    (`fog_range_pad.w`) drives a material blend in the fragment shader:
11//!    albedo → snow white, roughness → 0.8, metallic → 0.0.
12//!
13//! 3. **Configuration**: [`SnowMaterial`] holds snow albedo, roughness, and
14//!    blend thresholds. [`SnowAccumulationConfig`] controls compute-side rates.
15
16use bytemuck::{Pod, Zeroable};
17
18// ─── GPU Uniform ─────────────────────────────────────────────────────────
19
20/// GPU-side parameters for the snow accumulation compute shader.
21///
22/// Must match `SnowParams` in `snow_accumulation.wgsl`. 32 bytes, 4-byte aligned.
23#[repr(C)]
24#[derive(Debug, Clone, Copy, Pod, Zeroable)]
25pub struct SnowParamsGpu {
26    /// Snow accumulation rate per second (world depth units).
27    pub accumulate_rate: f32,
28    /// Melt rate per second.
29    pub melt_rate: f32,
30    /// Frame delta time.
31    pub dt: f32,
32    /// 1.0 if snow is actively falling, 0.0 otherwise.
33    pub snow_active: f32,
34    /// Chunk width in texels.
35    pub width: u32,
36    /// Chunk height in texels.
37    pub height: u32,
38    /// Minimum slope cosine for accumulation (surfaces steeper than
39    /// `acos(min_slope_cos)` shed snow). Typically ~0.5 (60°).
40    pub min_slope_cos: f32,
41    /// Maximum accumulation depth (caps the heightmap value).
42    pub max_depth: f32,
43}
44
45impl Default for SnowParamsGpu {
46    fn default() -> Self {
47        Self {
48            accumulate_rate: 0.15,
49            melt_rate: 0.03,
50            dt: 1.0 / 60.0,
51            snow_active: 0.0,
52            width: 256,
53            height: 256,
54            min_slope_cos: 0.5, // ~60° slope limit
55            max_depth: 1.0,
56        }
57    }
58}

60// ─── Snow Material Constants ─────────────────────────────────────────────
61
62/// Snow material properties for PBR blending in the fragment shader.
63///
64/// These constants define how snow appears when blended over terrain/mesh materials.
65/// The PBR shader uses `snow_amount` from SceneEnvironmentUBO to interpolate
66/// between the base material and these snow properties.
67#[derive(Debug, Clone, Copy)]
68pub struct SnowMaterial {
69    /// Snow albedo (linear RGB). Fresh snow is ~0.9 reflectance.
70    pub albedo: [f32; 3],
71    /// Snow roughness. Fresh snow is quite rough (~0.8), compacted ~0.6.
72    pub roughness: f32,
73    /// Threshold of `snow_amount * up_facing` below which no snow appears.
74    /// Creates a sharp transition edge. Typically 0.1–0.2.
75    pub threshold: f32,
76    /// Sharpness of the blend transition. Higher = sharper edge.
77    /// `snow_weight = saturate((raw - threshold) * sharpness)`.
78    pub sharpness: f32,
79}
80
81impl Default for SnowMaterial {
82    fn default() -> Self {
83        Self {
84            albedo: [0.95, 0.96, 0.98], // Slightly blue-white
85            roughness: 0.8,
86            threshold: 0.1,
87            sharpness: 5.0,
88        }
89    }
90}
91
92impl SnowMaterial {
93    /// Packed snow — slightly darker, smoother.
94    pub fn packed() -> Self {
95        Self {
96            albedo: [0.85, 0.87, 0.90],
97            roughness: 0.6,
98            threshold: 0.15,
99            sharpness: 4.0,
100        }
101    }
102
103    /// Light dusting — very thin coverage with soft transition.
104    pub fn dusting() -> Self {
105        Self {
106            albedo: [0.92, 0.93, 0.95],
107            roughness: 0.75,
108            threshold: 0.3,
109            sharpness: 3.0,
110        }
111    }
112}

114// ─── CPU-side Configuration ──────────────────────────────────────────────
115
116/// Controls the compute-side snow accumulation behavior.
117#[derive(Debug, Clone, Copy)]
118pub struct SnowAccumulationConfig {
119    /// Rate at which snow accumulates on upward surfaces (depth/second).
120    pub accumulate_rate: f32,
121    /// Rate at which snow melts when not snowing (depth/second).
122    pub melt_rate: f32,
123    /// Maximum snow depth before capping.
124    pub max_depth: f32,
125    /// Minimum slope cosine (dot(N, UP)) to allow accumulation.
126    /// 0.0 = all surfaces, 1.0 = perfectly flat only.
127    pub min_slope_cos: f32,
128    /// Resolution of accumulation heightmap per chunk (square).
129    pub map_resolution: u32,
130}
131
132impl Default for SnowAccumulationConfig {
133    fn default() -> Self {
134        Self {
135            accumulate_rate: 0.15,
136            melt_rate: 0.03,
137            max_depth: 1.0,
138            min_slope_cos: 0.5,
139            map_resolution: 256,
140        }
141    }
142}
143
144impl SnowAccumulationConfig {
145    /// Build GPU params from this config + runtime state.
146    pub fn to_gpu_params(&self, dt: f32, snow_active: bool) -> SnowParamsGpu {
147        SnowParamsGpu {
148            accumulate_rate: self.accumulate_rate,
149            melt_rate: self.melt_rate,
150            dt,
151            snow_active: if snow_active { 1.0 } else { 0.0 },
152            width: self.map_resolution,
153            height: self.map_resolution,
154            min_slope_cos: self.min_slope_cos,
155            max_depth: self.max_depth,
156        }
157    }
158
159    /// Arctic environment — fast accumulation, slow melt.
160    pub fn arctic() -> Self {
161        Self {
162            accumulate_rate: 0.25,
163            melt_rate: 0.01,
164            max_depth: 2.0,
165            min_slope_cos: 0.3,
166            map_resolution: 256,
167        }
168    }
169
170    /// Temperate environment — moderate accumulation, faster melt.
171    pub fn temperate() -> Self {
172        Self {
173            accumulate_rate: 0.10,
174            melt_rate: 0.08,
175            max_depth: 0.5,
176            min_slope_cos: 0.6,
177            map_resolution: 256,
178        }
179    }
180
181    /// Compute dispatch workgroup count for the accumulation shader.
182    /// Workgroup size is (8, 8), so divide resolution by 8, rounding up.
183    pub fn dispatch_size(&self) -> (u32, u32) {
184        let wg = 8u32;
185        (
186            self.map_resolution.div_ceil(wg),
187            self.map_resolution.div_ceil(wg),
188        )
189    }
190}
