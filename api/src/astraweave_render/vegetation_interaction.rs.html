1//! Vegetation interaction stamp system.
2//!
3//! Generates a persistent R8 interaction texture that encodes entity proximity,
4//! consumed by the grass blade shader to bend blades away from characters.
5//!
6//! Each frame:
7//! 1. **Decay pass**: fades previous stamps toward zero (trail dissipation).
8//! 2. **Stamp pass**: stamps circular footprints at entity world positions.
9//!
10//! The texture covers a square world region centered on the camera, with
11//! extent = 2 × `interaction_radius` in each axis.
12
13use bytemuck::{Pod, Zeroable};
14
15// ── Constants ───────────────────────────────────────────────────────────────
16
17/// WGSL shader source.
18const INTERACTION_SHADER: &str = include_str!("../shaders/vegetation_interaction.wgsl");
19
20/// Default interaction texture resolution (128×128 is sufficient for 16m radius).
21const DEFAULT_TEX_SIZE: u32 = 128;
22
23/// Maximum entity stamps per frame.
24const MAX_ENTITIES: usize = 64;
25
26// ── GPU structs (match WGSL layout) ─────────────────────────────────────────
27
28/// Stamp parameters uniform (32 bytes, matches WGSL `StampParams`).
29#[repr(C)]
30#[derive(Copy, Clone, Debug, Pod, Zeroable)]
31pub struct StampParams {
32    pub camera_x: f32,
33    pub camera_z: f32,
34    pub interaction_radius: f32,
35    pub tex_size: f32,
36    pub decay_rate: f32,
37    pub entity_count: u32,
38    pub stamp_radius: f32,
39    pub stamp_intensity: f32,
40}
41
42impl Default for StampParams {
43    fn default() -> Self {
44        Self {
45            camera_x: 0.0,
46            camera_z: 0.0,
47            interaction_radius: 8.0,
48            tex_size: DEFAULT_TEX_SIZE as f32,
49            decay_rate: 0.92,
50            entity_count: 0,
51            stamp_radius: 0.5,
52            stamp_intensity: 0.8,
53        }
54    }
55}

57/// Entity position for stamping (8 bytes, matches WGSL `EntityPos`).
58#[repr(C)]
59#[derive(Copy, Clone, Debug, Pod, Zeroable)]
60pub struct EntityPos {
61    pub x: f32,
62    pub z: f32,
63}

65// ── Pipeline ────────────────────────────────────────────────────────────────
66
67/// Manages the vegetation interaction stamp compute pipeline and texture.
68pub struct VegetationInteraction {
69    decay_pipeline: wgpu::ComputePipeline,
70    stamp_pipeline: wgpu::ComputePipeline,
71    bind_group_layout: wgpu::BindGroupLayout,
72
73    // R8 interaction texture (persistent across frames).
74    interaction_texture: wgpu::Texture,
75    interaction_view: wgpu::TextureView,
76    tex_size: u32,
77
78    // Uniform + entity storage buffers.
79    params_buffer: wgpu::Buffer,
80    entity_buffer: wgpu::Buffer,
81}
82
83impl VegetationInteraction {
84    /// Create the interaction stamp system.
85    ///
86    /// `tex_size`: interaction texture resolution (default 128).
87    pub fn new(device: &wgpu::Device, tex_size: Option<u32>) -> Self {
88        let tex_size = tex_size.unwrap_or(DEFAULT_TEX_SIZE);
89
90        // ── Shader ──────────────────────────────────────────────────────
91        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
92            label: Some("vegetation_interaction.wgsl"),
93            source: wgpu::ShaderSource::Wgsl(INTERACTION_SHADER.into()),
94        });
95
96        // ── Bind group layout ───────────────────────────────────────────
97        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
98            label: Some("VegInteraction BGL"),
99            entries: &[
100                // 0: StampParams uniform
101                wgpu::BindGroupLayoutEntry {
102                    binding: 0,
103                    visibility: wgpu::ShaderStages::COMPUTE,
104                    ty: wgpu::BindingType::Buffer {
105                        ty: wgpu::BufferBindingType::Uniform,
106                        has_dynamic_offset: false,
107                        min_binding_size: None,
108                    },
109                    count: None,
110                },
111                // 1: EntityPos storage (read)
112                wgpu::BindGroupLayoutEntry {
113                    binding: 1,
114                    visibility: wgpu::ShaderStages::COMPUTE,
115                    ty: wgpu::BindingType::Buffer {
116                        ty: wgpu::BufferBindingType::Storage { read_only: true },
117                        has_dynamic_offset: false,
118                        min_binding_size: None,
119                    },
120                    count: None,
121                },
122                // 2: Interaction texture (storage, read_write)
123                wgpu::BindGroupLayoutEntry {
124                    binding: 2,
125                    visibility: wgpu::ShaderStages::COMPUTE,
126                    ty: wgpu::BindingType::StorageTexture {
127                        access: wgpu::StorageTextureAccess::ReadWrite,
128                        format: wgpu::TextureFormat::R8Unorm,
129                        view_dimension: wgpu::TextureViewDimension::D2,
130                    },
131                    count: None,
132                },
133            ],
134        });
135
136        // ── Pipeline layout ─────────────────────────────────────────────
137        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
138            label: Some("VegInteraction PipelineLayout"),
139            bind_group_layouts: &[&bind_group_layout],
140            push_constant_ranges: &[],
141        });
142
143        // ── Compute pipelines ───────────────────────────────────────────
144        let decay_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
145            label: Some("VegInteraction Decay"),
146            layout: Some(&pipeline_layout),
147            module: &shader,
148            entry_point: Some("decay"),
149            compilation_options: Default::default(),
150            cache: None,
151        });
152
153        let stamp_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
154            label: Some("VegInteraction Stamp"),
155            layout: Some(&pipeline_layout),
156            module: &shader,
157            entry_point: Some("stamp"),
158            compilation_options: Default::default(),
159            cache: None,
160        });
161
162        // ── Interaction texture ─────────────────────────────────────────
163        let interaction_texture = device.create_texture(&wgpu::TextureDescriptor {
164            label: Some("VegInteraction Texture"),
165            size: wgpu::Extent3d {
166                width: tex_size,
167                height: tex_size,
168                depth_or_array_layers: 1,
169            },
170            mip_level_count: 1,
171            sample_count: 1,
172            dimension: wgpu::TextureDimension::D2,
173            format: wgpu::TextureFormat::R8Unorm,
174            usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
175            view_formats: &[],
176        });
177
178        let interaction_view =
179            interaction_texture.create_view(&wgpu::TextureViewDescriptor::default());
180
181        // ── Buffers ─────────────────────────────────────────────────────
182        let params_buffer = device.create_buffer(&wgpu::BufferDescriptor {
183            label: Some("VegInteraction Params"),
184            size: std::mem::size_of::<StampParams>() as u64,
185            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
186            mapped_at_creation: false,
187        });
188
189        let entity_buffer = device.create_buffer(&wgpu::BufferDescriptor {
190            label: Some("VegInteraction Entities"),
191            size: (MAX_ENTITIES * std::mem::size_of::<EntityPos>()) as u64,
192            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
193            mapped_at_creation: false,
194        });
195
196        Self {
197            decay_pipeline,
198            stamp_pipeline,
199            bind_group_layout,
200            interaction_texture,
201            interaction_view,
202            tex_size,
203            params_buffer,
204            entity_buffer,
205        }
206    }
207
208    /// Dispatch decay + stamp passes for this frame.
209    ///
210    /// `camera_xz`: camera world position (x, z).
211    /// `entity_positions`: world XZ positions of entities that bend grass.
212    /// `interaction_radius`: world-space radius covered by the texture.
213    /// `decay_rate`: per-frame decay multiplier (0.9–0.98 typical).
214    #[allow(clippy::too_many_arguments)]
215    pub fn dispatch(
216        &self,
217        device: &wgpu::Device,
218        queue: &wgpu::Queue,
219        encoder: &mut wgpu::CommandEncoder,
220        camera_xz: [f32; 2],
221        entity_positions: &[EntityPos],
222        interaction_radius: f32,
223        decay_rate: f32,
224    ) {
225        let entity_count = entity_positions.len().min(MAX_ENTITIES);
226
227        let params = StampParams {
228            camera_x: camera_xz[0],
229            camera_z: camera_xz[1],
230            interaction_radius,
231            tex_size: self.tex_size as f32,
232            decay_rate,
233            entity_count: entity_count as u32,
234            stamp_radius: 0.5,
235            stamp_intensity: 0.8,
236        };
237
238        queue.write_buffer(&self.params_buffer, 0, bytemuck::bytes_of(&params));
239
240        if entity_count > 0 {
241            queue.write_buffer(
242                &self.entity_buffer,
243                0,
244                bytemuck::cast_slice(&entity_positions[..entity_count]),
245            );
246        }
247
248        // Build bind group (shared by both passes).
249        let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
250            label: Some("VegInteraction BG"),
251            layout: &self.bind_group_layout,
252            entries: &[
253                wgpu::BindGroupEntry {
254                    binding: 0,
255                    resource: self.params_buffer.as_entire_binding(),
256                },
257                wgpu::BindGroupEntry {
258                    binding: 1,
259                    resource: self.entity_buffer.as_entire_binding(),
260                },
261                wgpu::BindGroupEntry {
262                    binding: 2,
263                    resource: wgpu::BindingResource::TextureView(&self.interaction_view),
264                },
265            ],
266        });
267
268        let wg_x = self.tex_size.div_ceil(8);
269        let wg_y = self.tex_size.div_ceil(8);
270
271        // Pass 1: Decay
272        {
273            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
274                label: Some("VegInteraction Decay"),
275                timestamp_writes: None,
276            });
277            pass.set_pipeline(&self.decay_pipeline);
278            pass.set_bind_group(0, &bg, &[]);
279            pass.dispatch_workgroups(wg_x, wg_y, 1);
280        }
281
282        // Pass 2: Stamp entities
283        if entity_count > 0 {
284            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
285                label: Some("VegInteraction Stamp"),
286                timestamp_writes: None,
287            });
288            pass.set_pipeline(&self.stamp_pipeline);
289            pass.set_bind_group(0, &bg, &[]);
290            pass.dispatch_workgroups(wg_x, wg_y, 1);
291        }
292    }
293
294    /// Get the interaction texture view for binding to the grass blade shader.
295    pub fn interaction_view(&self) -> &wgpu::TextureView {
296        &self.interaction_view
297    }
298
299    /// Get the raw texture (for external sampling or debugging).
300    pub fn interaction_texture(&self) -> &wgpu::Texture {
301        &self.interaction_texture
302    }
303
304    /// Texture resolution.
305    pub fn tex_size(&self) -> u32 {
306        self.tex_size
307    }
308}

316    #[test]
317    fn test_stamp_params_size() {
318        assert_eq!(std::mem::size_of::<StampParams>(), 32);

322    fn test_entity_pos_size() {
323        assert_eq!(std::mem::size_of::<EntityPos>(), 8);
324    }
325
326    #[test]
327    fn test_stamp_params_default() {
328        let p = StampParams::default();
329        assert!(p.decay_rate > 0.0 && p.decay_rate < 1.0);
330        assert!(p.interaction_radius > 0.0);
331        assert!(p.stamp_radius > 0.0);
332        assert!(p.stamp_intensity > 0.0 && p.stamp_intensity <= 1.0);
333        assert_eq!(p.entity_count, 0);
334    }
335
336    #[test]
337    fn test_interaction_shader_present() {
338        assert!(
339            INTERACTION_SHADER.contains("decay"),
340            "shader must contain decay entry point"
341        );
342        assert!(
343            INTERACTION_SHADER.contains("stamp"),
344            "shader must contain stamp entry point"
345        );
346        assert!(
347            INTERACTION_SHADER.contains("StampParams"),
348            "shader must contain StampParams struct"
349        );
350        assert!(
351            INTERACTION_SHADER.contains("interaction_tex"),
352            "shader must reference interaction texture"
353        );
354    }
355
356    #[test]
357    fn test_max_entities_reasonable() {
358        // 64 entities × 8 bytes = 512 bytes storage buffer
