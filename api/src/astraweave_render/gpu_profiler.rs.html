1//! GPU timestamp profiler for per-pass performance analysis.
2//!
3//! Wraps wgpu's timestamp query mechanism to automatically capture timing for
4//! each render/compute pass. Results are read back asynchronously and exposed
5//! as per-pass millisecond durations.
6//!
7//! Requires `Features::TIMESTAMP_QUERY` on the device.
8
9use std::collections::BTreeMap;
10use std::sync::{
11    atomic::{AtomicBool, Ordering},
12    Arc,
13};
14
15/// Maximum number of timestamp pairs (begin + end per pass).
16const MAX_PASSES: u32 = 16;
17const MAX_QUERIES: u32 = MAX_PASSES * 2;
18
19/// Per-pass GPU timing result.
20#[derive(Debug, Clone)]
21pub struct PassTiming {
22    pub name: String,
23    pub duration_ms: f32,
24}
25
26/// Manages GPU timestamp queries across render/compute passes.
27pub struct GpuProfiler {
28    query_set: wgpu::QuerySet,
29    /// Buffer for resolved query results (GPU-only, COPY_SRC).
30    resolve_buf: wgpu::Buffer,
31    /// Staging buffer for CPU readback (MAP_READ | COPY_DST).
32    readback_buf: wgpu::Buffer,
33    /// Nanoseconds per GPU timestamp tick.
34    timestamp_period: f32,
35    /// Next query index to allocate this frame.
36    next_query: u32,
37    /// Map from pass name → (begin_query, end_query) indices.
38    pass_queries: Vec<(String, u32, u32)>,
39    /// Most recent readback results (updated asynchronously).
40    latest_results: Vec<PassTiming>,
41    /// Whether a readback is currently pending (buffer mapped).
42    readback_pending: bool,
43    /// Signalled by the map callback when the readback buffer is ready.
44    map_ready: Arc<AtomicBool>,
45}
46
47impl GpuProfiler {
48    /// Create a new GPU profiler. Only call when `TIMESTAMP_QUERY` is available.
49    pub fn new(device: &wgpu::Device, queue: &wgpu::Queue) -> Self {
50        let query_set = device.create_query_set(&wgpu::QuerySetDescriptor {
51            label: Some("gpu_profiler_queries"),
52            ty: wgpu::QueryType::Timestamp,
53            count: MAX_QUERIES,
54        });
55
56        let buf_size = (MAX_QUERIES as u64) * std::mem::size_of::<u64>() as u64;
57
58        let resolve_buf = device.create_buffer(&wgpu::BufferDescriptor {
59            label: Some("gpu_profiler_resolve"),
60            size: buf_size,
61            usage: wgpu::BufferUsages::QUERY_RESOLVE | wgpu::BufferUsages::COPY_SRC,
62            mapped_at_creation: false,
63        });
64
65        let readback_buf = device.create_buffer(&wgpu::BufferDescriptor {
66            label: Some("gpu_profiler_readback"),
67            size: buf_size,
68            usage: wgpu::BufferUsages::COPY_DST | wgpu::BufferUsages::MAP_READ,
69            mapped_at_creation: false,
70        });
71
72        let timestamp_period = queue.get_timestamp_period();
73
74        Self {
75            query_set,
76            resolve_buf,
77            readback_buf,
78            timestamp_period,
79            next_query: 0,
80            pass_queries: Vec::with_capacity(MAX_PASSES as usize),
81            latest_results: Vec::new(),
82            readback_pending: false,
83            map_ready: Arc::new(AtomicBool::new(false)),
84        }
85    }
86
87    /// Reset per-frame state. Call at the start of each frame before recording passes.
88    pub fn begin_frame(&mut self) {
89        self.next_query = 0;
90        self.pass_queries.clear();
91    }
92
93    /// Expose the underlying query set for constructing timestamp writes
94    /// separately from the allocation step. This enables borrow splitting
95    /// in the Renderer: allocate indices via `&mut self`, then build
96    /// `TimestampWrites` from `&self.query_set` without conflicting borrows.
97    pub fn query_set(&self) -> &wgpu::QuerySet {
98        &self.query_set
99    }
100
101    /// Allocate a begin/end query index pair for a named pass.
102    /// Returns `(begin_index, end_index)` or `None` if slots are exhausted.
103    /// The returned indices can be used with `query_set()` to construct
104    /// `RenderPassTimestampWrites` or `ComputePassTimestampWrites` inline.
105    pub fn allocate_pass(&mut self, name: &str) -> Option<(u32, u32)> {
106        if self.next_query + 2 > MAX_QUERIES {
107            return None;
108        }
109        let begin = self.next_query;
110        let end = self.next_query + 1;
111        self.next_query += 2;
112        self.pass_queries.push((name.to_owned(), begin, end));
113        Some((begin, end))
114    }
115
116    /// Allocate a timestamp write pair for a render pass.
117    /// Returns `RenderPassTimestampWrites` to embed in the pass descriptor,
118    /// or `None` if we've run out of query slots.
119    pub fn render_pass_timestamps(
120        &mut self,
121        name: &str,
122    ) -> Option<wgpu::RenderPassTimestampWrites<'_>> {
123        if self.next_query + 2 > MAX_QUERIES {
124            return None;
125        }
126        let begin = self.next_query;
127        let end = self.next_query + 1;
128        self.next_query += 2;
129        self.pass_queries.push((name.to_owned(), begin, end));
130
131        Some(wgpu::RenderPassTimestampWrites {
132            query_set: &self.query_set,
133            beginning_of_pass_write_index: Some(begin),
134            end_of_pass_write_index: Some(end),
135        })
136    }
137
138    /// Allocate a timestamp write pair for a compute pass.
139    /// Returns `ComputePassTimestampWrites` to embed in the pass descriptor,
140    /// or `None` if we've run out of query slots.
141    pub fn compute_pass_timestamps(
142        &mut self,
143        name: &str,
144    ) -> Option<wgpu::ComputePassTimestampWrites<'_>> {
145        if self.next_query + 2 > MAX_QUERIES {
146            return None;
147        }
148        let begin = self.next_query;
149        let end = self.next_query + 1;
150        self.next_query += 2;
151        self.pass_queries.push((name.to_owned(), begin, end));
152
153        Some(wgpu::ComputePassTimestampWrites {
154            query_set: &self.query_set,
155            beginning_of_pass_write_index: Some(begin),
156            end_of_pass_write_index: Some(end),
157        })
158    }
159
160    /// After all passes are recorded, resolve and copy timestamps.
161    /// Call before `queue.submit()`.
162    pub fn end_frame(&self, encoder: &mut wgpu::CommandEncoder) {
163        if self.next_query == 0 {
164            return;
165        }
166        // Resolve timestamps into the GPU buffer.
167        encoder.resolve_query_set(&self.query_set, 0..self.next_query, &self.resolve_buf, 0);
168        // Copy to the CPU-readable staging buffer.
169        let byte_count = (self.next_query as u64) * std::mem::size_of::<u64>() as u64;
170        encoder.copy_buffer_to_buffer(&self.resolve_buf, 0, &self.readback_buf, 0, byte_count);
171    }
172
173    /// Initiate an async readback of the timestamp buffer.
174    /// Call after `queue.submit()`.
175    pub fn request_readback(&mut self) {
176        if self.next_query == 0 || self.readback_pending {
177            return;
178        }
179        let slice = self.readback_buf.slice(..);
180        self.readback_pending = true;
181        self.map_ready.store(false, Ordering::Release);
182        let flag = self.map_ready.clone();
183        slice.map_async(wgpu::MapMode::Read, move |_result| {
184            flag.store(true, Ordering::Release);
185        });
186    }
187
188    /// Poll for completed readback and update `latest_results`.
189    /// Returns `true` if new data is available.
190    pub fn poll_readback(&mut self, device: &wgpu::Device) -> bool {
191        if !self.readback_pending {
192            return false;
193        }
194        // Non-blocking poll — drives the map callback.
195        let _ = device.poll(wgpu::PollType::Poll);
196
197        if !self.map_ready.load(Ordering::Acquire) {
198            // Mapping not complete yet.
199            return false;
200        }
201
202        let slice = self.readback_buf.slice(..);
203        let data = slice.get_mapped_range();
204        let timestamps: &[u64] =
205            bytemuck::cast_slice(&data[..self.next_query as usize * std::mem::size_of::<u64>()]);
206
207        let ns_per_tick = self.timestamp_period as f64;
208        let mut results = Vec::with_capacity(self.pass_queries.len());
209
210        for (name, begin_idx, end_idx) in &self.pass_queries {
211            let t0 = timestamps[*begin_idx as usize];
212            let t1 = timestamps[*end_idx as usize];
213            let delta_ns = (t1.wrapping_sub(t0)) as f64 * ns_per_tick;
214            results.push(PassTiming {
215                name: name.clone(),
216                duration_ms: (delta_ns / 1_000_000.0) as f32,
217            });
218        }
219
220        drop(data);
221        self.readback_buf.unmap();
222        self.readback_pending = false;
223        self.map_ready.store(false, Ordering::Release);
224        self.latest_results = results;
225        true
226    }
227
228    /// Get the most recent per-pass timing results.
229    pub fn results(&self) -> &[PassTiming] {
230        &self.latest_results
231    }
232
233    /// Get total GPU time across all instrumented passes (ms).
234    pub fn total_gpu_ms(&self) -> f32 {
235        self.latest_results.iter().map(|r| r.duration_ms).sum()
236    }
237
238    /// Get per-pass timings as a sorted map (name → ms).
239    pub fn results_map(&self) -> BTreeMap<String, f32> {
240        self.latest_results
241            .iter()
242            .map(|r| (r.name.clone(), r.duration_ms))
243            .collect()
244    }
245}
