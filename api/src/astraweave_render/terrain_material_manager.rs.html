1//! Terrain splat-array material pipeline manager (Phase 2.2 — Issue #9 fix).
2//!
3//! Real-Fix.D 2026-05-08: bumped from 8 to 32 canonical layers + 2 to 8 splat
4//! textures per Andrew-gate decision (h) Option D-2 (canonical material
5//! library). Capacity now derives from `crate::material_library::MAX_TERRAIN_LAYERS`.
6//!
7//! Binds the 32-layer splat shader in `shaders/pbr_terrain.wgsl` (plus the
8//! companion vertex stage in `shaders/pbr_terrain_vs.wgsl`) into a complete
9//! render pipeline and owns the GPU resources required to draw multi-material
10//! terrain chunks:
11//!
12//! * Four `texture_2d_array<f32>` shared across all chunks — `layer_albedo`,
13//!   `layer_normal`, `layer_orm`, `layer_height`, each with up to 32 layers
14//!   (configurable resolution via [`TerrainMaterialConfig`]).
15//! * Eight per-chunk `texture_2d<f32>` splat maps (`splat_0` = RGBA weights
16//!   for layers 0..3, `splat_1` = layers 4..7, ..., `splat_7` = layers 28..31).
17//! * A single `TerrainMaterialGpu` uniform buffer describing the active layer
18//!   count, triplanar/height-blend settings, and per-layer material factors.
19//!
20//! This module is gated behind the `terrain-splat-arrays` Cargo feature and is
21//! a no-op additive path: the existing single-texture terrain renderer used by
22//! the game-path renderer is unaffected when the feature is off.
23//!
24//! # High-level flow (editor/game integration)
25//!
26//! 1. Call [`TerrainMaterialManager::new`] once per device to allocate the
27//!    shared texture arrays, sampler, uniform buffer, and bind group layouts.
28//! 2. Before the first draw, call [`TerrainMaterialManager::ensure_pipeline`]
29//!    with the target color + depth formats to lazily build the render
30//!    pipeline.
31//! 3. Call [`TerrainMaterialManager::set_material`] to upload the 8 layer
32//!    textures (albedo + normal + orm + height) for the current biome pack and
33//!    write the `TerrainMaterialGpu` uniform.
34//! 4. For each terrain chunk, call [`TerrainMaterialManager::set_chunk_splat`]
35//!    to upload the two RGBA8 splat maps (typically rasterised from the
36//!    per-vertex `biome_weights_0/1` in the editor).
37//! 5. During the render pass, call [`TerrainMaterialManager::draw_chunk`] with
38//!    the chunk's vertex/index buffers and the camera bind group.

60/// Configuration controlling the shared layer array resolutions.
61///
62/// Defaults chosen for editor use: 1024² albedo, 512² for the other channels,
63/// giving roughly 56 MB of VRAM for the 8-layer arrays. Games that need higher
64/// fidelity can construct a custom config with 2048² albedo.
65#[derive(Debug, Clone, Copy)]
66pub struct TerrainMaterialConfig {
67    /// Albedo array resolution (per-layer square dim). Power-of-two.
68    pub albedo_resolution: u32,
69    /// Normal / ORM / height array resolution (per-layer square dim).
70    pub aux_resolution: u32,
71    /// Number of layers to allocate (1..=8). Must match shader expectations.
72    pub layer_count: u32,
73}

77        Self {
78            albedo_resolution: 1024,
79            aux_resolution: 512,
80            layer_count: MAX_TERRAIN_LAYERS,
81        }
82    }
83}

85impl TerrainMaterialConfig {
86    /// Validate the configuration; returns an error for out-of-range values.
87    pub fn validate(&self) -> Result<()> {
88        if !(1..=MAX_TERRAIN_LAYERS).contains(&self.layer_count) {
89            anyhow::bail!(
90                "TerrainMaterialConfig.layer_count must be in 1..={}, got {}",
91                MAX_TERRAIN_LAYERS,
92                self.layer_count
93            );
94        }
95        if self.albedo_resolution == 0 || !self.albedo_resolution.is_power_of_two() {
96            anyhow::bail!(
97                "TerrainMaterialConfig.albedo_resolution must be power-of-two, got {}",
98                self.albedo_resolution
99            );
100        }
101        if self.aux_resolution == 0 || !self.aux_resolution.is_power_of_two() {
102            anyhow::bail!(
103                "TerrainMaterialConfig.aux_resolution must be power-of-two, got {}",
104                self.aux_resolution
105            );
106        }
107        Ok(())
108    }
109
110    /// Approximate GPU memory footprint of the shared texture arrays in bytes.
111    pub fn approx_memory_bytes(&self) -> u64 {
112        let albedo = (self.albedo_resolution as u64).pow(2) * BYTES_PER_TEXEL as u64;
113        let aux = (self.aux_resolution as u64).pow(2) * BYTES_PER_TEXEL as u64;
114        (albedo + aux * 3) * self.layer_count as u64
115    }

388    pub fn new(device: &wgpu::Device, config: TerrainMaterialConfig) -> Result<Self> {
389        config.validate().context("invalid TerrainMaterialConfig")?;
390
391        let shared = create_shared_resources(device, &config);
392
393        let camera_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
394            label: Some("terrain-splat-camera-bgl"),
395            entries: &[wgpu::BindGroupLayoutEntry {
396                binding: 0,
397                visibility: wgpu::ShaderStages::VERTEX_FRAGMENT,
398                ty: wgpu::BindingType::Buffer {
399                    ty: wgpu::BufferBindingType::Uniform,
400                    has_dynamic_offset: false,
401                    min_binding_size: None,
402                },
403                count: None,
404            }],
405        });
406
407        let terrain_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
408            label: Some("terrain-splat-uniform-bgl"),
409            entries: &[wgpu::BindGroupLayoutEntry {
410                binding: 0,
411                visibility: wgpu::ShaderStages::VERTEX_FRAGMENT,
412                ty: wgpu::BindingType::Buffer {
413                    ty: wgpu::BufferBindingType::Uniform,
414                    has_dynamic_offset: false,
415                    min_binding_size: None,
416                },
417                count: None,
418            }],
419        });
420
421        // Real-Fix.D 2026-05-08: bumped from 2 to NUM_SPLAT_MAPS=8 splat
422        // textures + shifted layer-array bindings from 3..6 to 9..12 so the
423        // bindings align with `pbr_terrain.wgsl`.
424        let mut splat_entries = vec![
425            // 0: sampler (shared)
426            wgpu::BindGroupLayoutEntry {
427                binding: 0,
428                visibility: wgpu::ShaderStages::FRAGMENT,
429                ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
430                count: None,
431            },
432        ];
433        // 1..=NUM_SPLAT_MAPS: per-chunk splat textures

439        splat_entries.push(array_entry(array_base));
440        splat_entries.push(array_entry(array_base + 1));

442        splat_entries.push(array_entry(array_base + 3));

450            layout: &terrain_bgl,
451            entries: &[wgpu::BindGroupEntry {
452                binding: 0,
453                resource: shared.material_uniform.as_entire_binding(),
454            }],
455        });
456
457        let camera_bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
458            label: Some("terrain-splat-camera-bg"),
459            layout: &camera_bgl,
460            entries: &[wgpu::BindGroupEntry {
461                binding: 0,
462                resource: shared.camera_uniform.as_entire_binding(),
463            }],
464        });

469            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {

473                    visibility: wgpu::ShaderStages::VERTEX_FRAGMENT,

489                entries: &[
490                    // 0: TerrainMaterialGpu UBO

497                            min_binding_size: None,
498                        },
499                        count: None,
500                    },

502                    wgpu::BindGroupLayoutEntry {
503                        binding: 1,
504                        visibility: wgpu::ShaderStages::FRAGMENT,

510                        count: None,
511                    },

516                        visibility: wgpu::ShaderStages::FRAGMENT,

671    /// Upload the eight layer textures and write the `TerrainMaterialGpu`
672    /// uniform buffer.
673    ///
674    /// `layers` must have length `config.layer_count`; surplus entries are
675    /// ignored, missing entries are padded with a solid-grey fallback.
676    pub fn set_material(
677        &mut self,
678        queue: &wgpu::Queue,
679        gpu_material: &TerrainMaterialGpu,
680        layers: &[LayerTextures<'_>],
681    ) -> Result<()> {
682        // Validate each layer payload matches the expected byte size.
683        let albedo_bytes =
684            (self.config.albedo_resolution as usize).pow(2) * BYTES_PER_TEXEL as usize;
685        let aux_bytes = (self.config.aux_resolution as usize).pow(2) * BYTES_PER_TEXEL as usize;
686
687        let layer_count = self.config.layer_count as usize;
688        for (i, layer) in layers.iter().take(layer_count).enumerate() {
689            if let Some(data) = layer.albedo {
690                if data.len() != albedo_bytes {
691                    anyhow::bail!(
692                        "layer {} albedo size mismatch: expected {} bytes, got {}",
693                        i,
694                        albedo_bytes,
695                        data.len()
696                    );
697                }
698            }
699            for (name, data) in [
700                ("normal", layer.normal),
701                ("orm", layer.orm),
702                ("height", layer.height),
703            ] {
704                if let Some(data) = data {
705                    if data.len() != aux_bytes {
706                        anyhow::bail!(
707                            "layer {} {} size mismatch: expected {} bytes, got {}",
708                            i,
709                            name,
710                            aux_bytes,
711                            data.len()
712                        );
713                    }
714                }
715            }
716        }
717
718        // Default grey / flat-normal / neutral-ORM payloads for missing channels.
719        let grey_albedo = vec![128u8; albedo_bytes];
720        let flat_normal = build_flat_normal(self.config.aux_resolution);
721        let neutral_orm = build_neutral_orm(self.config.aux_resolution);
722        let flat_height = vec![128u8; aux_bytes];
723
724        for i in 0..layer_count {
725            let payload = layers.get(i).cloned().unwrap_or_default();
726            upload_layer_slice(
727                queue,
728                &self.shared._albedo_tex,
729                i as u32,
730                self.config.albedo_resolution,
731                payload.albedo.unwrap_or(&grey_albedo),
732            );
733            upload_layer_slice(
734                queue,
735                &self.shared._normal_tex,
736                i as u32,
737                self.config.aux_resolution,
738                payload.normal.unwrap_or(&flat_normal),
739            );
740            upload_layer_slice(
741                queue,
742                &self.shared._orm_tex,
743                i as u32,
744                self.config.aux_resolution,
745                payload.orm.unwrap_or(&neutral_orm),
746            );
747            upload_layer_slice(
748                queue,
749                &self.shared._height_tex,
750                i as u32,
751                self.config.aux_resolution,
752                payload.height.unwrap_or(&flat_height),
753            );
754        }
755
756        // Write the uniform buffer (576 B).
757        self.material_cache = *gpu_material;
758        self.material_cache.active_layer_count = self.config.layer_count;
759        queue.write_buffer(
760            &self.shared.material_uniform,
761            0,
762            bytemuck::bytes_of(&self.material_cache),
763        );
764
765        Ok(())
766    }

768    /// Register or replace the splat maps for a chunk.
769    ///
770    /// Real-Fix.D 2026-05-08: takes [`NUM_SPLAT_MAPS`] = 8 splat slices (was 2).
771    ///
772    /// Each slice in `splats` must be `dims.0 * dims.1 * 4` bytes (RGBA8).
773    /// Channel mapping matches `pbr_terrain.wgsl`:
774    /// * `splats[0]`: R=layer0,  G=layer1,  B=layer2,  A=layer3
775    /// * `splats[1]`: R=layer4,  G=layer5,  B=layer6,  A=layer7
776    /// * `splats[2]`: R=layer8,  G=layer9,  B=layer10, A=layer11
777    /// * `splats[3]`: R=layer12, G=layer13, B=layer14, A=layer15
778    /// * `splats[4]`: R=layer16, G=layer17, B=layer18, A=layer19
779    /// * `splats[5]`: R=layer20, G=layer21, B=layer22, A=layer23
780    /// * `splats[6]`: R=layer24, G=layer25, B=layer26, A=layer27
781    /// * `splats[7]`: R=layer28, G=layer29, B=layer30, A=layer31
782    pub fn set_chunk_splat(
783        &mut self,
784        device: &wgpu::Device,
785        queue: &wgpu::Queue,
786        chunk: ChunkKey,
787        splats: &[&[u8]],
788        dims: (u32, u32),
789    ) -> Result<()> {
790        let (w, h) = dims;
791        if w == 0 || h == 0 {
792            anyhow::bail!("chunk splat dims must be non-zero, got {w}x{h}");
793        }
794        if splats.len() != NUM_SPLAT_MAPS {
795            anyhow::bail!(
796                "chunk splat slice count mismatch: expected {}, got {}",
797                NUM_SPLAT_MAPS,
798                splats.len()
799            );
800        }
801        let expected = (w as usize) * (h as usize) * BYTES_PER_TEXEL as usize;
802        for (i, slice) in splats.iter().enumerate() {
803            if slice.len() != expected {
804                anyhow::bail!(
805                    "chunk splat[{}] payload mismatch: expected {} bytes, got {}",
806                    i,
807                    expected,
808                    slice.len()
809                );
810            }
811        }
812
813        let tex_desc = wgpu::TextureDescriptor {
814            label: Some("terrain-chunk-splat"),
815            size: wgpu::Extent3d {
816                width: w,
817                height: h,
818                depth_or_array_layers: 1,
819            },
820            mip_level_count: 1,
821            sample_count: 1,
822            dimension: wgpu::TextureDimension::D2,
823            format: wgpu::TextureFormat::Rgba8Unorm,
824            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
825            view_formats: &[],
826        };
827
828        let textures: [wgpu::Texture; NUM_SPLAT_MAPS] =
829            std::array::from_fn(|_| device.create_texture(&tex_desc));
830        for (i, slice) in splats.iter().enumerate() {
831            upload_full_2d(queue, &textures[i], w, h, slice);
832        }
833
834        let views: [wgpu::TextureView; NUM_SPLAT_MAPS] = std::array::from_fn(|i| {
835            textures[i].create_view(&wgpu::TextureViewDescriptor::default())
836        });
837
838        // Real-Fix.D 2026-05-08: build bind group entries dynamically — sampler
839        // at 0, NUM_SPLAT_MAPS=8 splat textures at 1..=8, then 4 layer arrays
840        // at 9..=12.
841        let mut entries: Vec<wgpu::BindGroupEntry> = Vec::with_capacity(NUM_SPLAT_MAPS + 5);
842        entries.push(wgpu::BindGroupEntry {
843            binding: 0,
844            resource: wgpu::BindingResource::Sampler(&self.shared.sampler),
845        });
846        for (i, view) in views.iter().enumerate() {
847            entries.push(wgpu::BindGroupEntry {
848                binding: (i + 1) as u32,
849                resource: wgpu::BindingResource::TextureView(view),
850            });
851        }
852        let array_base = NUM_SPLAT_MAPS as u32 + 1;
853        entries.push(wgpu::BindGroupEntry {
854            binding: array_base,
855            resource: wgpu::BindingResource::TextureView(&self.shared.layer_albedo),
856        });
857        entries.push(wgpu::BindGroupEntry {
858            binding: array_base + 1,
859            resource: wgpu::BindingResource::TextureView(&self.shared.layer_normal),
860        });
861        entries.push(wgpu::BindGroupEntry {
862            binding: array_base + 2,
863            resource: wgpu::BindingResource::TextureView(&self.shared.layer_orm),
864        });
865        entries.push(wgpu::BindGroupEntry {
866            binding: array_base + 3,
867            resource: wgpu::BindingResource::TextureView(&self.shared.layer_height),
868        });
869        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
870            label: Some("terrain-splat-chunk-bg"),
871            layout: &self.splat_bgl,
872            entries: &entries,
873        });
874
875        self.chunk_splats.insert(
876            chunk,
877            ChunkSplat {
878                _splats: textures,
879                bind_group,
880                dims,
881            },
882        );
883        Ok(())
884    }

1159            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
1160            view_formats: &[],
1161        };

1163        let textures: [wgpu::Texture; NUM_SPLAT_MAPS] =
1164            std::array::from_fn(|_| device.create_texture(&tex_desc));
1165        for (i, slice) in splats.iter().enumerate() {
1166            upload_full_2d(queue, &textures[i], w, h, slice);
1167        }

1169        let views: [wgpu::TextureView; NUM_SPLAT_MAPS] = std::array::from_fn(|i| {
1170            textures[i].create_view(&wgpu::TextureViewDescriptor::default())
1171        });
1172
1173        // Real-Fix.D 2026-05-08: 8 splat textures at bindings 0..7 + sampler
1174        // at NUM_SPLAT_MAPS=8.
1175        let mut entries: Vec<wgpu::BindGroupEntry> = views
1176            .iter()
1177            .enumerate()
1178            .map(|(i, view)| wgpu::BindGroupEntry {
1179                binding: i as u32,
1180                resource: wgpu::BindingResource::TextureView(view),
1181            })
1182            .collect();
1183        // Phase 1 re-cleanup Issue 1 fix: ClampToEdge splat sampler
1184        // (shared across chunks, lives in SharedResources).
1185        entries.push(wgpu::BindGroupEntry {
1186            binding: NUM_SPLAT_MAPS as u32,
1187            resource: wgpu::BindingResource::Sampler(&self.shared.splat_sampler),
1188        });
1189        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
1190            label: Some("terrain-forward-chunk-splat-bg"),
1191            layout: &self.forward_splat_bgl,
1192            entries: &entries,
1193        });
1194
1195        self.forward_chunk_splats.insert(
1196            chunk,
1197            ChunkSplatForward {
1198                _splats: textures,
1199                bind_group,
1200                dims,
1201            },
1202        );

1216    /// Issue a draw call for a chunk's forward-lit splat render.
1217    ///
1218    /// Prerequisites (caller must have satisfied):
1219    /// 1. [`Self::ensure_forward_pipeline`] was called with a compatible format.
1220    /// 2. [`Self::update_forward_camera`] and [`Self::update_forward_scene`]
1221    ///    were called earlier this frame (or sensible defaults are in the UBOs).
1222    /// 3. [`Self::set_chunk_splat_forward`] was called for this chunk key.
1223    ///
1224    /// Returns `false` when any prerequisite is missing; the caller can fall
1225    /// through to the legacy path without further state changes. Returns
1226    /// `true` when the draw was issued. The render pass itself must be
1227    /// configured for the `color_format` + `depth_format` passed to
1228    /// `ensure_forward_pipeline` — a mismatch is a wgpu validation error.
1229    pub fn draw_chunk_forward<'a>(
1230        &'a self,
1231        rpass: &mut wgpu::RenderPass<'a>,
1232        chunk: ChunkKey,
1233        vertex_buffer: &'a wgpu::Buffer,
1234        index_buffer: &'a wgpu::Buffer,
1235        index_count: u32,
1236    ) -> bool {
1237        let Some(pipeline) = self.forward_pipeline.as_ref() else {
1238            return false;
1239        };
1240        let Some(splat) = self.forward_chunk_splats.get(&chunk) else {
1241            return false;
1242        };
1243        rpass.set_pipeline(pipeline);
1244        rpass.set_bind_group(0, &self.forward_camera_bg, &[]);
1245        rpass.set_bind_group(1, &self.forward_terrain_bg, &[]);
1246        rpass.set_bind_group(2, &splat.bind_group, &[]);
1247        rpass.set_vertex_buffer(0, vertex_buffer.slice(..));
1248        rpass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint32);
1249        rpass.draw_indexed(0..index_count, 0, 0..1);
1250        true
1251    }
1252
1253    /// Issue a draw call for a chunk that has had its splats uploaded.
1254    ///
1255    /// Uses the manager's internal camera bind group (see
1256    /// [`Self::update_camera`]). Returns `false` when the chunk has no
1257    /// registered splat pair (caller should fall back to the legacy path).
1258    pub fn draw_chunk<'a>(
1259        &'a self,
1260        rpass: &mut wgpu::RenderPass<'a>,
1261        chunk: ChunkKey,
1262        vertex_buffer: &'a wgpu::Buffer,
1263        index_buffer: &'a wgpu::Buffer,
1264        index_count: u32,
1265    ) -> bool {
1266        let Some(pipeline) = self.pipeline.as_ref() else {
1267            return false;
1268        };
1269        let Some(splat) = self.chunk_splats.get(&chunk) else {
1270            return false;
1271        };
1272        rpass.set_pipeline(pipeline);
1273        rpass.set_bind_group(0, &self.camera_bg, &[]);
1274        rpass.set_bind_group(1, &self.terrain_bg, &[]);
1275        rpass.set_bind_group(2, &splat.bind_group, &[]);
1276        rpass.set_vertex_buffer(0, vertex_buffer.slice(..));
1277        rpass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint32);
1278        rpass.draw_indexed(0..index_count, 0, 0..1);
1279        true
1280    }
