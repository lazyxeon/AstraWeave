1//! Temporal Anti-Aliasing (TAA) with anti-ghosting.
2//!
3//! Provides production TAA: Halton(2,3) jitter sequence, history reprojection
4//! via motion vectors, YCoCg neighborhood clamping, Catmull-Rom history sampling,
5//! velocity-dependent blending, and optional RCAS sharpening.
6
7/// GPU-side uniform for the TAA resolve pass.
8#[repr(C)]
9#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
10pub struct TaaUniforms {
11    pub resolution: [f32; 2],
12    pub inv_resolution: [f32; 2],
13    /// x: history blend factor, y: clamp margin, z: sharpen strength, w: frame index
14    pub config: [f32; 4],
15}
16
17/// TAA configuration.
18#[derive(Debug, Clone)]
19pub struct TaaConfig {
20    pub enabled: bool,
21    /// History blend factor (0.9 = responsive, 0.98 = very stable).
22    pub blend_factor: f32,
23    /// AABB clamp margin (larger = less ghosting but more aliasing).
24    pub clamp_margin: f32,
25    /// Jitter scale multiplier.
26    pub jitter_scale: f32,
27    /// RCAS sharpening strength (0 = off, 0.5 = moderate, 1.0 = strong).
28    pub sharpen_strength: f32,
29    /// Number of jitter samples in the Halton sequence (typically 8 or 16).
30    pub jitter_samples: u32,
31}
32
33impl Default for TaaConfig {
34    fn default() -> Self {
35        Self {
36            enabled: true,
37            blend_factor: 0.95,
38            clamp_margin: 0.02,
39            jitter_scale: 1.0,
40            sharpen_strength: 0.5,
41            jitter_samples: 8,
42        }
43    }
44}

46/// Halton sequence for sub-pixel jitter.
47/// Low-discrepancy sequence that provides good coverage over N frames.
48pub fn halton(index: u32, base: u32) -> f32 {
49    let mut result = 0.0f32;
50    let mut f = 1.0f32;
51    let mut i = index;
52    let b = base as f32;
53
54    while i > 0 {
55        f /= b;
56        result += f * (i % base) as f32;
57        i /= base;
58    }
59
60    result
61}

63/// Get the jitter offset for a given frame index.
64/// Returns (x, y) in pixel units, centered around 0.
65pub fn get_jitter(frame_index: u32, jitter_samples: u32, jitter_scale: f32) -> (f32, f32) {
66    let idx = (frame_index % jitter_samples) + 1; // Halton(0) = 0, start from 1
67    let jx = (halton(idx, 2) - 0.5) * jitter_scale;
68    let jy = (halton(idx, 3) - 0.5) * jitter_scale;
69    (jx, jy)
70}

72/// Apply TAA jitter to a projection matrix.
73/// Offsets the projection by sub-pixel amounts to enable temporal supersampling.
74pub fn apply_jitter_to_projection(
75    proj: glam::Mat4,
76    jitter_x: f32,
77    jitter_y: f32,
78    width: f32,
79    height: f32,
80) -> glam::Mat4 {
81    let mut jittered = proj;
82    // Convert pixel offset to NDC offset
83    let ndc_x = 2.0 * jitter_x / width;
84    let ndc_y = 2.0 * jitter_y / height;
85    // Add to the projection matrix's translation (column 2, rows 0 and 1)
86    let cols = jittered.to_cols_array_2d();
87    let mut modified = cols;
88    modified[2][0] += ndc_x;
89    modified[2][1] += ndc_y;
90    jittered = glam::Mat4::from_cols_array_2d(&modified);
91    jittered
92}

94/// Manages TAA GPU resources: history buffers, compute pipelines, jitter state.
95pub struct TaaPass {
96    config: TaaConfig,
97    /// TAA resolve compute pipeline.
98    resolve_pipeline: wgpu::ComputePipeline,
99    /// RCAS sharpen compute pipeline.
100    sharpen_pipeline: wgpu::ComputePipeline,
101    /// Uniform buffer.
102    params_buf: wgpu::Buffer,
103    /// History texture (previous resolved frame).
104    history_texture: wgpu::Texture,
105    history_view: wgpu::TextureView,
106    /// Resolved output texture.
107    resolved_texture: wgpu::Texture,
108    resolved_view: wgpu::TextureView,
109    /// Sharpened output texture (post-RCAS).
110    /// Stored to keep the GPU allocation alive for sharpened_view.
111    #[allow(dead_code)]
112    sharpened_texture: wgpu::Texture,
113    sharpened_view: wgpu::TextureView,
114    /// Bind group layout.
115    bgl: wgpu::BindGroupLayout,
116    /// Frame counter for jitter sequence.
117    frame_index: u32,
118    width: u32,
119    height: u32,
120    /// Cached sampler (static — never changes).
121    sampler: wgpu::Sampler,
122    /// Cached resolve bind group (rebuilt on generation change).
123    cached_bg: crate::bind_group_cache::CachedBindGroup,
124    /// Cached sharpen bind group (rebuilt on generation change).
125    cached_sharpen_bg: crate::bind_group_cache::CachedBindGroup,
126}

128impl TaaPass {
129    pub fn new(device: &wgpu::Device, width: u32, height: u32) -> Self {
130        let config = TaaConfig::default();
131        let fmt = wgpu::TextureFormat::Rgba16Float;
132        let size = wgpu::Extent3d {
133            width,
134            height,
135            depth_or_array_layers: 1,
136        };
137
138        let make_tex = |label: &str| {
139            device.create_texture(&wgpu::TextureDescriptor {
140                label: Some(label),
141                size,
142                mip_level_count: 1,
143                sample_count: 1,
144                dimension: wgpu::TextureDimension::D2,
145                format: fmt,
146                usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
147                view_formats: &[],
148            })
149        };
150
151        let history_texture = make_tex("taa_history");
152        let history_view = history_texture.create_view(&wgpu::TextureViewDescriptor::default());
153        let resolved_texture = make_tex("taa_resolved");
154        let resolved_view = resolved_texture.create_view(&wgpu::TextureViewDescriptor::default());
155        let sharpened_texture = make_tex("taa_sharpened");
156        let sharpened_view = sharpened_texture.create_view(&wgpu::TextureViewDescriptor::default());
157
158        use wgpu::util::DeviceExt;
159        let uniforms = TaaUniforms {
160            resolution: [width as f32, height as f32],
161            inv_resolution: [1.0 / width as f32, 1.0 / height as f32],
162            config: [
163                config.blend_factor,
164                config.clamp_margin,
165                config.sharpen_strength,
166                0.0,
167            ],
168        };
169        let params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
170            label: Some("taa_params"),
171            contents: bytemuck::bytes_of(&uniforms),
172            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
173        });
174
175        let bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
176            label: Some("taa_bgl"),
177            entries: &[
178                tex_entry(0),          // current / sharpen input
179                tex_entry(1),          // history
180                tex_entry(2),          // velocity
181                tex_entry(3),          // depth
182                sampler_entry(4),      // sampler
183                uniform_entry(5),      // params
184                storage_entry(6, fmt), // output
185            ],
186        });
187
188        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
189            label: Some("taa_shader"),
190            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/taa.wgsl").into()),
191        });
192
193        let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
194            label: Some("taa_pl"),
195            bind_group_layouts: &[&bgl],
196            push_constant_ranges: &[],
197        });
198
199        let resolve_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
200            label: Some("taa_resolve"),
201            layout: Some(&pl),
202            module: &shader,
203            entry_point: Some("taa_resolve"),
204            compilation_options: Default::default(),
205            cache: None,
206        });
207
208        let sharpen_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
209            label: Some("taa_rcas"),
210            layout: Some(&pl),
211            module: &shader,
212            entry_point: Some("rcas_sharpen"),
213            compilation_options: Default::default(),
214            cache: None,
215        });
216
217        Self {
218            config,
219            resolve_pipeline,
220            sharpen_pipeline,
221            params_buf,
222            history_texture,
223            history_view,
224            resolved_texture,
225            resolved_view,
226            sharpened_texture,
227            sharpened_view,
228            bgl,
229            frame_index: 0,
230            width,
231            height,
232            sampler: device.create_sampler(&wgpu::SamplerDescriptor {
233                label: Some("taa_sampler"),
234                mag_filter: wgpu::FilterMode::Linear,
235                min_filter: wgpu::FilterMode::Linear,
236                ..Default::default()
237            }),
238            cached_bg: crate::bind_group_cache::CachedBindGroup::new(),
239            cached_sharpen_bg: crate::bind_group_cache::CachedBindGroup::new(),
240        }
241    }

278    /// Update uniforms for this frame.
279    pub fn update_params(&self, queue: &wgpu::Queue) {
280        let uniforms = TaaUniforms {
281            resolution: [self.width as f32, self.height as f32],
282            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
283            config: [
284                self.config.blend_factor,
285                self.config.clamp_margin,
286                self.config.sharpen_strength,
287                self.frame_index as f32,
288            ],
289        };
290        queue.write_buffer(&self.params_buf, 0, bytemuck::bytes_of(&uniforms));
291    }

293    /// Execute the TAA resolve pass: blend current frame with history using velocity.
294    ///
295    /// After execution, `resolved_view()` contains the anti-aliased result.
296    /// Call `copy_to_history()` and `next_frame()` after this.
297    /// `resource_gen` is the renderer's generation counter; the bind group is
298    /// rebuilt only when it changes (e.g., after a resize).
299    pub fn execute(
300        &mut self,
301        device: &wgpu::Device,
302        encoder: &mut wgpu::CommandEncoder,
303        current_view: &wgpu::TextureView,
304        depth_view: &wgpu::TextureView,
305        velocity_view: &wgpu::TextureView,
306        resource_gen: crate::bind_group_cache::Generation,
307    ) {
308        if !self.config.enabled {
309            return;
310        }
311
312        if !self.cached_bg.is_valid(resource_gen) {
313            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
314                label: Some("taa_resolve_bg"),
315                layout: &self.bgl,
316                entries: &[
317                    wgpu::BindGroupEntry {
318                        binding: 0,
319                        resource: wgpu::BindingResource::TextureView(current_view),
320                    },
321                    wgpu::BindGroupEntry {
322                        binding: 1,
323                        resource: wgpu::BindingResource::TextureView(&self.history_view),
324                    },
325                    wgpu::BindGroupEntry {
326                        binding: 2,
327                        resource: wgpu::BindingResource::TextureView(velocity_view),
328                    },
329                    wgpu::BindGroupEntry {
330                        binding: 3,
331                        resource: wgpu::BindingResource::TextureView(depth_view),
332                    },
333                    wgpu::BindGroupEntry {
334                        binding: 4,
335                        resource: wgpu::BindingResource::Sampler(&self.sampler),
336                    },
337                    wgpu::BindGroupEntry {
338                        binding: 5,
339                        resource: self.params_buf.as_entire_binding(),
340                    },
341                    wgpu::BindGroupEntry {
342                        binding: 6,
343                        resource: wgpu::BindingResource::TextureView(&self.resolved_view),
344                    },
345                ],
346            });
347            self.cached_bg =
348                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
349        }
350
351        let wg_x = self.width.div_ceil(8);
352        let wg_y = self.height.div_ceil(8);
353
354        let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
355            label: Some("taa_resolve"),
356            timestamp_writes: None,
357        });
358        pass.set_pipeline(&self.resolve_pipeline);
359        let bg = self
360            .cached_bg
361            .get_or_rebuild(resource_gen, || unreachable!());
362        pass.set_bind_group(0, bg, &[]);
363        pass.dispatch_workgroups(wg_x, wg_y, 1);
364        drop(pass);
365
366        // --- RCAS sharpen pass: reads resolved, writes sharpened ---
367        if self.config.sharpen_strength > 0.0 {
368            if !self.cached_sharpen_bg.is_valid(resource_gen) {
369                let sharpen_bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
370                    label: Some("taa_sharpen_bg"),
371                    layout: &self.bgl,
372                    entries: &[
373                        wgpu::BindGroupEntry {
374                            binding: 0,
375                            resource: wgpu::BindingResource::TextureView(&self.resolved_view),
376                        },
377                        wgpu::BindGroupEntry {
378                            binding: 1,
379                            resource: wgpu::BindingResource::TextureView(&self.history_view),
380                        },
381                        wgpu::BindGroupEntry {
382                            binding: 2,
383                            resource: wgpu::BindingResource::TextureView(velocity_view),
384                        },
385                        wgpu::BindGroupEntry {
386                            binding: 3,
387                            resource: wgpu::BindingResource::TextureView(depth_view),
388                        },
389                        wgpu::BindGroupEntry {
390                            binding: 4,
391                            resource: wgpu::BindingResource::Sampler(&self.sampler),
392                        },
393                        wgpu::BindGroupEntry {
394                            binding: 5,
395                            resource: self.params_buf.as_entire_binding(),
396                        },
397                        wgpu::BindGroupEntry {
398                            binding: 6,
399                            resource: wgpu::BindingResource::TextureView(&self.sharpened_view),
400                        },
401                    ],
402                });
403                self.cached_sharpen_bg = crate::bind_group_cache::CachedBindGroup::with_bind_group(
404                    sharpen_bg,
405                    resource_gen,
406                );
407            }
408
409            let mut sharpen_pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
410                label: Some("taa_rcas_sharpen"),
411                timestamp_writes: None,
412            });
413            sharpen_pass.set_pipeline(&self.sharpen_pipeline);
414            let sbg = self
415                .cached_sharpen_bg
416                .get_or_rebuild(resource_gen, || unreachable!());
417            sharpen_pass.set_bind_group(0, sbg, &[]);
418            sharpen_pass.dispatch_workgroups(wg_x, wg_y, 1);
419        }
420    }

422    /// Copy resolved output to history for next frame's reprojection.
423    pub fn copy_to_history(&self, encoder: &mut wgpu::CommandEncoder) {
424        encoder.copy_texture_to_texture(
425            wgpu::TexelCopyTextureInfo {
426                texture: &self.resolved_texture,
427                mip_level: 0,
428                origin: wgpu::Origin3d::ZERO,
429                aspect: wgpu::TextureAspect::All,
430            },
431            wgpu::TexelCopyTextureInfo {
432                texture: &self.history_texture,
433                mip_level: 0,
434                origin: wgpu::Origin3d::ZERO,
435                aspect: wgpu::TextureAspect::All,
436            },
437            wgpu::Extent3d {
438                width: self.width,
439                height: self.height,
440                depth_or_array_layers: 1,
441            },
442        );
443    }

445    /// Get the output view (sharpened if RCAS active, otherwise resolved).
446    pub fn output_view(&self) -> &wgpu::TextureView {
447        if self.config.sharpen_strength > 0.0 {
448            &self.sharpened_view
449        } else {
450            &self.resolved_view
451        }
452    }

454    pub fn resize(&mut self, device: &wgpu::Device, width: u32, height: u32) {
455        if self.width == width && self.height == height {
456            return;
457        }
458        *self = Self::new(device, width, height);
459    }

466        ty: wgpu::BindingType::Texture {

541    fn jitter_centered() {
542        // Jitter values should be in [-0.5, 0.5] range (at scale 1.0)
543        for i in 0..16 {
544            let (jx, jy) = get_jitter(i, 8, 1.0);
545            assert!(jx >= -0.5 && jx <= 0.5, "jx={} out of range", jx);
546            assert!(jy >= -0.5 && jy <= 0.5, "jy={} out of range", jy);
