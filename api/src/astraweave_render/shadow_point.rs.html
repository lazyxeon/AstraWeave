1//! Point and spot light shadow map management.
2//!
3//! Provides shadow cubemaps for point lights and 2D shadow maps for spot lights.
4//! Uses a priority system to select which lights get shadow maps based on
5//! distance and screen-area contribution.
6
7use glam::{Mat4, Vec3};
8
9/// Maximum number of shadow-casting point lights per frame.
10pub const MAX_POINT_SHADOW_LIGHTS: usize = 8;
11
12/// Maximum number of shadow-casting spot lights per frame.
13pub const MAX_SPOT_SHADOW_LIGHTS: usize = 8;
14
15/// Resolution of each point light shadow cubemap face.
16pub const POINT_SHADOW_RESOLUTION: u32 = 512;
17
18/// Resolution of each spot light shadow map.
19pub const SPOT_SHADOW_RESOLUTION: u32 = 1024;
20
21/// A light candidate for shadow map allocation.
22#[derive(Debug, Clone)]
23pub struct ShadowLightCandidate {
24    /// World-space position.
25    pub position: Vec3,
26    /// Light radius (attenuation range).
27    pub radius: f32,
28    /// Priority score (higher = more important for shadows).
29    pub priority: f32,
30    /// Index into the scene light array.
31    pub light_index: usize,
32}
33
34/// Configuration for the point/spot shadow system.
35#[derive(Debug, Clone)]
36pub struct PointShadowConfig {
37    /// Maximum point lights with shadow maps.
38    pub max_point_shadows: usize,
39    /// Maximum spot lights with shadow maps.
40    pub max_spot_shadows: usize,
41    /// Point shadow cubemap face resolution.
42    pub point_resolution: u32,
43    /// Spot shadow map resolution.
44    pub spot_resolution: u32,
45    /// Depth bias for point lights.
46    pub point_bias: f32,
47    /// Depth bias for spot lights.
48    pub spot_bias: f32,
49    /// PCF radius for spot lights.
50    pub spot_pcf_radius: f32,
51}
52
53impl Default for PointShadowConfig {
54    fn default() -> Self {
55        Self {
56            max_point_shadows: MAX_POINT_SHADOW_LIGHTS,
57            max_spot_shadows: MAX_SPOT_SHADOW_LIGHTS,
58            point_resolution: POINT_SHADOW_RESOLUTION,
59            spot_resolution: SPOT_SHADOW_RESOLUTION,
60            point_bias: 0.01,
61            spot_bias: 0.005,
62            spot_pcf_radius: 1.5,
63        }
64    }
65}

67/// Select which lights should receive shadow maps based on priority.
68///
69/// Priority is computed from distance to camera and light radius (screen coverage).
70pub fn select_shadow_casters(
71    candidates: &[ShadowLightCandidate],
72    camera_pos: Vec3,
73    max_count: usize,
74) -> Vec<usize> {
75    let mut scored: Vec<(usize, f32)> = candidates
76        .iter()
77        .enumerate()
78        .map(|(i, c)| {
79            let dist = (c.position - camera_pos).length();
80            // Priority: larger radius and closer distance = higher priority
81            let screen_area = c.radius / (dist + 0.1);
82            let score = screen_area * c.priority;
83            (i, score)
84        })
85        .collect();
86
87    // Sort by score descending
88    scored.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
89
90    scored.iter().take(max_count).map(|&(i, _)| i).collect()
91}

93/// Compute the 6 view matrices for a point light cubemap.
94///
95/// Returns [+X, -X, +Y, -Y, +Z, -Z] view matrices.
96pub fn point_light_view_matrices(light_pos: Vec3) -> [Mat4; 6] {
97    [
98        Mat4::look_at_rh(light_pos, light_pos + Vec3::X, -Vec3::Y), // +X
99        Mat4::look_at_rh(light_pos, light_pos - Vec3::X, -Vec3::Y), // -X
100        Mat4::look_at_rh(light_pos, light_pos + Vec3::Y, Vec3::Z),  // +Y
101        Mat4::look_at_rh(light_pos, light_pos - Vec3::Y, -Vec3::Z), // -Y
102        Mat4::look_at_rh(light_pos, light_pos + Vec3::Z, -Vec3::Y), // +Z
103        Mat4::look_at_rh(light_pos, light_pos - Vec3::Z, -Vec3::Y), // -Z
104    ]
105}

107/// Compute the perspective projection for a point light cubemap face.
108pub fn point_light_projection(near: f32, far: f32) -> Mat4 {
109    Mat4::perspective_rh(std::f32::consts::FRAC_PI_2, 1.0, near, far)
110}
111
112/// Compute the view-projection matrix for a spot light.
113pub fn spot_light_vp(light_pos: Vec3, light_dir: Vec3, fov: f32, near: f32, far: f32) -> Mat4 {
114    let up = if light_dir.y.abs() > 0.99 {
115        Vec3::X
116    } else {
117        Vec3::Y
118    };
119    let view = Mat4::look_at_rh(light_pos, light_pos + light_dir, up);
120    let proj = Mat4::perspective_rh(fov, 1.0, near, far);
121    proj * view
122}

124/// Manages point and spot shadow map GPU resources.
125pub struct PointShadowPass {
126    config: PointShadowConfig,
127    /// Point light shadow cubemap array (depth).
128    point_shadow_texture: wgpu::Texture,
129    point_shadow_view: wgpu::TextureView,
130    /// Spot light shadow 2D array (depth).
131    spot_shadow_texture: wgpu::Texture,
132    spot_shadow_view: wgpu::TextureView,
133    /// Comparison sampler.
134    shadow_sampler: wgpu::Sampler,
135}
136
137impl PointShadowPass {
138    pub fn new(device: &wgpu::Device, config: PointShadowConfig) -> Self {
139        // Point light shadow cubemap array
140        let point_shadow_texture = device.create_texture(&wgpu::TextureDescriptor {
141            label: Some("point_shadow_cubemap_array"),
142            size: wgpu::Extent3d {
143                width: config.point_resolution,
144                height: config.point_resolution,
145                // 6 faces per cubemap * max lights
146                depth_or_array_layers: 6 * config.max_point_shadows as u32,
147            },
148            mip_level_count: 1,
149            sample_count: 1,
150            dimension: wgpu::TextureDimension::D2,
151            format: wgpu::TextureFormat::Depth32Float,
152            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
153            view_formats: &[],
154        });
155        let point_shadow_view = point_shadow_texture.create_view(&wgpu::TextureViewDescriptor {
156            dimension: Some(wgpu::TextureViewDimension::CubeArray),
157            ..Default::default()
158        });
159
160        // Spot light shadow 2D array
161        let spot_shadow_texture = device.create_texture(&wgpu::TextureDescriptor {
162            label: Some("spot_shadow_array"),
163            size: wgpu::Extent3d {
164                width: config.spot_resolution,
165                height: config.spot_resolution,
166                depth_or_array_layers: config.max_spot_shadows as u32,
167            },
168            mip_level_count: 1,
169            sample_count: 1,
170            dimension: wgpu::TextureDimension::D2,
171            format: wgpu::TextureFormat::Depth32Float,
172            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
173            view_formats: &[],
174        });
175        let spot_shadow_view =
176            spot_shadow_texture.create_view(&wgpu::TextureViewDescriptor::default());
177
178        let shadow_sampler = device.create_sampler(&wgpu::SamplerDescriptor {
179            label: Some("point_spot_shadow_sampler"),
180            compare: Some(wgpu::CompareFunction::LessEqual),
181            mag_filter: wgpu::FilterMode::Linear,
182            min_filter: wgpu::FilterMode::Linear,
183            ..Default::default()
184        });
185
186        Self {
187            config,
188            point_shadow_texture,
189            point_shadow_view,
190            spot_shadow_texture,
191            spot_shadow_view,
192            shadow_sampler,
193        }
194    }
195
196    pub fn point_shadow_view(&self) -> &wgpu::TextureView {
197        &self.point_shadow_view
198    }
199
200    pub fn spot_shadow_view(&self) -> &wgpu::TextureView {
201        &self.spot_shadow_view
202    }
203
204    pub fn shadow_sampler(&self) -> &wgpu::Sampler {
205        &self.shadow_sampler
206    }
207
208    pub fn config(&self) -> &PointShadowConfig {
209        &self.config
210    }
211
212    /// Get a view for a specific point light cubemap face.
213    pub fn point_face_view(&self, light_index: usize, face: u32) -> wgpu::TextureView {
214        let layer = light_index as u32 * 6 + face;
215        self.point_shadow_texture
216            .create_view(&wgpu::TextureViewDescriptor {
217                label: Some(&format!("point_shadow_face_{light_index}_{face}")),
218                dimension: Some(wgpu::TextureViewDimension::D2),
219                base_array_layer: layer,
220                array_layer_count: Some(1),
221                ..Default::default()
222            })
223    }
224
225    /// Get a view for a specific spot light shadow map.
226    pub fn spot_layer_view(&self, light_index: usize) -> wgpu::TextureView {
227        self.spot_shadow_texture
228            .create_view(&wgpu::TextureViewDescriptor {
229                label: Some(&format!("spot_shadow_{light_index}")),
230                dimension: Some(wgpu::TextureViewDimension::D2),
231                base_array_layer: light_index as u32,
232                array_layer_count: Some(1),
233                ..Default::default()
234            })
235    }

251    #[test]
252    fn select_shadow_casters_prioritizes_close_large_lights() {
253        let camera = Vec3::ZERO;
254        let candidates = vec![
255            ShadowLightCandidate {
256                position: Vec3::new(100.0, 0.0, 0.0),
257                radius: 5.0,
258                priority: 1.0,
259                light_index: 0,
260            },
261            ShadowLightCandidate {
262                position: Vec3::new(2.0, 0.0, 0.0),
263                radius: 10.0,
264                priority: 1.0,
265                light_index: 1,
266            },
267            ShadowLightCandidate {
268                position: Vec3::new(5.0, 0.0, 0.0),
269                radius: 20.0,
270                priority: 1.0,
271                light_index: 2,
272            },
273        ];
274
275        let selected = select_shadow_casters(&candidates, camera, 2);
276        assert_eq!(selected.len(), 2);
277        // Light 1 (close, medium) and light 2 (medium dist, large) should win
278        assert!(selected.contains(&1) || selected.contains(&2));
279    }
