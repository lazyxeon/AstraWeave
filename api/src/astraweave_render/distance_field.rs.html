1//! Distance Field Ambient Occlusion (DFAO) for Lumen GI.
2//!
3//! Generates a 3D signed distance field (SDF) from scene geometry and provides
4//! a compute-based cone-traced AO pass that extends far beyond screen-space methods.
5//! The SDF is stored in a 3D texture and updated incrementally.
6//!
7//! DFAO produces soft, physically-plausible occlusion with infinite range — perfect
8//! for large-scale architectural occlusion that GTAO/SSAO cannot capture.
9
10use bytemuck::{Pod, Zeroable};
11use glam::Vec3;
12use wgpu::util::DeviceExt;

23    pub view_pos: [f32; 3],

30    pub num_steps: u32,

35}

41/// A bounding box for SDF generation.
42#[derive(Debug, Clone, Copy)]
43pub struct SdfBox {
44    pub center: Vec3,
45    pub half_extents: Vec3,
46}
47
48/// Configuration for the SDF volume.
49#[derive(Debug, Clone)]
50pub struct SdfConfig {
51    /// Resolution of the 3D SDF texture (per axis).
52    pub dims: [u32; 3],
53    /// World-space origin (min corner).
54    pub origin: Vec3,
55    /// World-space size of the entire volume.
56    pub world_size: f32,
57}
58
59impl Default for SdfConfig {
60    fn default() -> Self {
61        Self {
62            dims: [64, 32, 64],
63            origin: Vec3::new(-64.0, -16.0, -64.0),
64            world_size: 128.0,
65        }
66    }
67}
68
69impl SdfConfig {
70    /// Voxel size in world units.
71    pub fn voxel_size(&self) -> f32 {
72        self.world_size / self.dims[0].max(self.dims[1]).max(self.dims[2]) as f32
73    }
74}
75
76/// CPU-side SDF volume for baking distance fields from geometry.
77pub struct SdfVolume {
78    config: SdfConfig,
79    /// Distance values: positive = outside, negative = inside.
80    data: Vec<f32>,
81}
82
83impl SdfVolume {
84    /// Create a new SDF volume initialized to max distance.
85    pub fn new(config: SdfConfig) -> Self {
86        let total = config.dims[0] as usize * config.dims[1] as usize * config.dims[2] as usize;
87        Self {
88            config,
89            data: vec![f32::MAX; total],
90        }
91    }
92
93    /// Total voxel count.
94    pub fn total_voxels(&self) -> usize {
95        self.data.len()
96    }
97
98    /// Get the SDF value at a voxel index.
99    pub fn get(&self, x: u32, y: u32, z: u32) -> f32 {
100        let idx = self.linear_index(x, y, z);
101        self.data[idx]
102    }
103
104    /// World position of a voxel center.
105    pub fn voxel_center(&self, x: u32, y: u32, z: u32) -> Vec3 {
106        let voxel_size = Vec3::new(
107            self.config.world_size / self.config.dims[0] as f32,
108            self.config.world_size / self.config.dims[1] as f32,
109            self.config.world_size / self.config.dims[2] as f32,
110        );
111        self.config.origin + Vec3::new(x as f32 + 0.5, y as f32 + 0.5, z as f32 + 0.5) * voxel_size
112    }
113
114    fn linear_index(&self, x: u32, y: u32, z: u32) -> usize {
115        (z as usize * self.config.dims[1] as usize + y as usize) * self.config.dims[0] as usize
116            + x as usize
117    }
118
119    /// Bake SDF from a list of axis-aligned bounding boxes.
120    /// Uses brute-force signed distance computation: for each voxel, find the
121    /// minimum distance to any box surface.
122    pub fn bake_from_boxes(&mut self, boxes: &[SdfBox]) {
123        let [dx, dy, dz] = self.config.dims;
124        for z in 0..dz {
125            for y in 0..dy {
126                for x in 0..dx {
127                    let pos = self.voxel_center(x, y, z);
128                    let mut min_dist = f32::MAX;
129
130                    for b in boxes {
131                        let d = sdf_box(pos, b.center, b.half_extents);
132                        min_dist = min_dist.min(d);
133                    }
134
135                    let idx = self.linear_index(x, y, z);
136                    self.data[idx] = min_dist;
137                }
138            }
139        }
140    }
141
142    /// Raw float data for GPU upload.
143    pub fn data(&self) -> &[f32] {
144        &self.data
145    }
146
147    pub fn config(&self) -> &SdfConfig {
148        &self.config
149    }
150}

152/// Signed distance from point `p` to an axis-aligned box at `center` with `half_extents`.
153fn sdf_box(p: Vec3, center: Vec3, half_extents: Vec3) -> f32 {
154    let q = (p - center).abs() - half_extents;
155    let outside = Vec3::new(q.x.max(0.0), q.y.max(0.0), q.z.max(0.0)).length();
156    let inside = q.x.max(q.y.max(q.z)).min(0.0);
157    outside + inside
158}

194/// Manages GPU resources for the DFAO compute pass.
195pub struct DfaoPass {
196    config: DfaoConfig,
197    sdf_config: SdfConfig,
198    pipeline: wgpu::ComputePipeline,
199    bgl: wgpu::BindGroupLayout,
200    params_buf: wgpu::Buffer,
201    /// 3D SDF texture.
202    sdf_texture: wgpu::Texture,
203    sdf_view: wgpu::TextureView,
204    /// Output AO texture (R16Float).
205    ao_texture: wgpu::Texture,
206    ao_view: wgpu::TextureView,
207    width: u32,
208    height: u32,
209}
210
211impl DfaoPass {
212    pub fn new(device: &wgpu::Device, width: u32, height: u32, sdf_config: SdfConfig) -> Self {
213        let config = DfaoConfig::default();
214        let [sx, sy, sz] = sdf_config.dims;
215
216        // Create 3D SDF texture
217        let sdf_texture = device.create_texture(&wgpu::TextureDescriptor {
218            label: Some("sdf_volume"),
219            size: wgpu::Extent3d {
220                width: sx,
221                height: sy,
222                depth_or_array_layers: sz,
223            },
224            mip_level_count: 1,
225            sample_count: 1,
226            dimension: wgpu::TextureDimension::D3,
227            format: wgpu::TextureFormat::R32Float,
228            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
229            view_formats: &[],
230        });
231        let sdf_view = sdf_texture.create_view(&wgpu::TextureViewDescriptor::default());
232
233        // AO output
234        let ao_texture = device.create_texture(&wgpu::TextureDescriptor {
235            label: Some("dfao_output"),
236            size: wgpu::Extent3d {
237                width,
238                height,
239                depth_or_array_layers: 1,
240            },
241            mip_level_count: 1,
242            sample_count: 1,
243            dimension: wgpu::TextureDimension::D2,
244            format: wgpu::TextureFormat::Rgba16Float,
245            usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
246            view_formats: &[],
247        });
248        let ao_view = ao_texture.create_view(&wgpu::TextureViewDescriptor::default());
249
250        let params = DfaoParams {
251            inv_view_proj: glam::Mat4::IDENTITY.to_cols_array_2d(),
252            view_pos: [0.0; 3],
253            max_distance: config.max_distance,
254            resolution: [width as f32, height as f32],
255            inv_resolution: [1.0 / width as f32, 1.0 / height as f32],
256            sdf_origin: sdf_config.origin.to_array(),
257            sdf_inv_size: 1.0 / sdf_config.world_size,
258            sdf_dims: sdf_config.dims,
259            num_steps: config.num_steps,
260            ao_power: config.ao_power,
261            ao_intensity: config.ao_intensity,
262            near_plane: 0.1,
263            far_plane: 1000.0,
264        };
265        let params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
266            label: Some("dfao_params"),
267            contents: bytemuck::bytes_of(&params),
268            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
269        });
270
271        let bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
272            label: Some("dfao_bgl"),
273            entries: &[
274                // 0: params
275                wgpu::BindGroupLayoutEntry {
276                    binding: 0,
277                    visibility: wgpu::ShaderStages::COMPUTE,
278                    ty: wgpu::BindingType::Buffer {
279                        ty: wgpu::BufferBindingType::Uniform,
280                        has_dynamic_offset: false,
281                        min_binding_size: None,
282                    },
283                    count: None,
284                },
285                // 1: depth
286                wgpu::BindGroupLayoutEntry {
287                    binding: 1,
288                    visibility: wgpu::ShaderStages::COMPUTE,
289                    ty: wgpu::BindingType::Texture {
290                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
291                        view_dimension: wgpu::TextureViewDimension::D2,
292                        multisampled: false,
293                    },
294                    count: None,
295                },
296                // 2: normals
297                wgpu::BindGroupLayoutEntry {
298                    binding: 2,
299                    visibility: wgpu::ShaderStages::COMPUTE,
300                    ty: wgpu::BindingType::Texture {
301                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
302                        view_dimension: wgpu::TextureViewDimension::D2,
303                        multisampled: false,
304                    },
305                    count: None,
306                },
307                // 3: SDF 3D texture
308                wgpu::BindGroupLayoutEntry {
309                    binding: 3,
310                    visibility: wgpu::ShaderStages::COMPUTE,
311                    ty: wgpu::BindingType::Texture {
312                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
313                        view_dimension: wgpu::TextureViewDimension::D3,
314                        multisampled: false,
315                    },
316                    count: None,
317                },
318                // 4: sampler
319                wgpu::BindGroupLayoutEntry {
320                    binding: 4,
321                    visibility: wgpu::ShaderStages::COMPUTE,
322                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
323                    count: None,
324                },
325                // 5: AO output (storage)
326                wgpu::BindGroupLayoutEntry {
327                    binding: 5,
328                    visibility: wgpu::ShaderStages::COMPUTE,
329                    ty: wgpu::BindingType::StorageTexture {
330                        access: wgpu::StorageTextureAccess::WriteOnly,
331                        format: wgpu::TextureFormat::Rgba16Float,
332                        view_dimension: wgpu::TextureViewDimension::D2,
333                    },
334                    count: None,
335                },
336            ],
337        });
338
339        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
340            label: Some("dfao_shader"),
341            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/lumen/dfao.wgsl").into()),
342        });
343        let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
344            label: Some("dfao_pl"),
345            bind_group_layouts: &[&bgl],
346            push_constant_ranges: &[],
347        });
348        let pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
349            label: Some("dfao_pipeline"),
350            layout: Some(&pl),
351            module: &shader,
352            entry_point: Some("dfao_main"),
353            compilation_options: Default::default(),
354            cache: None,
355        });
356
357        Self {
358            config,
359            sdf_config,
360            pipeline,
361            bgl,
362            params_buf,
363            sdf_texture,
364            sdf_view,
365            ao_texture,
366            ao_view,
367            width,
368            height,
369        }
370    }

389    /// Upload SDF volume data to the GPU.
390    pub fn upload_sdf(&self, queue: &wgpu::Queue, volume: &SdfVolume) {
391        let [sx, sy, sz] = volume.config().dims;
392        let bytes_per_row = sx * 4; // R32Float = 4 bytes per texel
393
394        queue.write_texture(
395            wgpu::TexelCopyTextureInfo {
396                texture: &self.sdf_texture,
397                mip_level: 0,
398                origin: wgpu::Origin3d::ZERO,
399                aspect: wgpu::TextureAspect::All,
400            },
401            bytemuck::cast_slice(volume.data()),
402            wgpu::TexelCopyBufferLayout {
403                offset: 0,
404                bytes_per_row: Some(bytes_per_row),
405                rows_per_image: Some(sy),
406            },
407            wgpu::Extent3d {
408                width: sx,
409                height: sy,
410                depth_or_array_layers: sz,
411            },
412        );
413    }
414
415    /// Update params for this frame.
416    pub fn update_params(
417        &self,
418        queue: &wgpu::Queue,
419        inv_view_proj: glam::Mat4,
420        view_pos: Vec3,
421        near: f32,
422        far: f32,
423    ) {
424        let params = DfaoParams {
425            inv_view_proj: inv_view_proj.to_cols_array_2d(),
426            view_pos: view_pos.to_array(),
427            max_distance: self.config.max_distance,
428            resolution: [self.width as f32, self.height as f32],
429            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
430            sdf_origin: self.sdf_config.origin.to_array(),
431            sdf_inv_size: 1.0 / self.sdf_config.world_size,
432            sdf_dims: self.sdf_config.dims,
433            num_steps: self.config.num_steps,
434            ao_power: self.config.ao_power,
435            ao_intensity: self.config.ao_intensity,
436            near_plane: near,
437            far_plane: far,
438        };
439        queue.write_buffer(&self.params_buf, 0, bytemuck::bytes_of(&params));
440    }

442    /// Dispatch the DFAO compute pass.
443    pub fn execute(
444        &self,
445        device: &wgpu::Device,
446        encoder: &mut wgpu::CommandEncoder,
447        depth_view: &wgpu::TextureView,
448        normal_view: &wgpu::TextureView,
449    ) {
450        if !self.config.enabled {
451            return;
452        }
453
454        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
455            label: Some("dfao_sampler"),
456            mag_filter: wgpu::FilterMode::Linear,
457            min_filter: wgpu::FilterMode::Linear,
458            mipmap_filter: wgpu::FilterMode::Linear,
459            address_mode_u: wgpu::AddressMode::ClampToEdge,
460            address_mode_v: wgpu::AddressMode::ClampToEdge,
461            address_mode_w: wgpu::AddressMode::ClampToEdge,
462            ..Default::default()
463        });
464
465        let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
466            label: Some("dfao_bg"),
467            layout: &self.bgl,
468            entries: &[
469                wgpu::BindGroupEntry {
470                    binding: 0,
471                    resource: self.params_buf.as_entire_binding(),
472                },
473                wgpu::BindGroupEntry {
474                    binding: 1,
475                    resource: wgpu::BindingResource::TextureView(depth_view),
476                },
477                wgpu::BindGroupEntry {
478                    binding: 2,
479                    resource: wgpu::BindingResource::TextureView(normal_view),
480                },
481                wgpu::BindGroupEntry {
482                    binding: 3,
483                    resource: wgpu::BindingResource::TextureView(&self.sdf_view),
484                },
485                wgpu::BindGroupEntry {
486                    binding: 4,
487                    resource: wgpu::BindingResource::Sampler(&sampler),
488                },
489                wgpu::BindGroupEntry {
490                    binding: 5,
491                    resource: wgpu::BindingResource::TextureView(&self.ao_view),
492                },
493            ],
494        });
495
496        let wg_x = (self.width + 7) / 8;
497        let wg_y = (self.height + 7) / 8;
498
499        let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
500            label: Some("dfao"),
501            timestamp_writes: None,
502        });
503        pass.set_pipeline(&self.pipeline);
504        pass.set_bind_group(0, &bg, &[]);
505        pass.dispatch_workgroups(wg_x, wg_y, 1);
506    }

508    /// Resize AO output texture.
509    pub fn resize(&mut self, device: &wgpu::Device, width: u32, height: u32) {
510        if self.width == width && self.height == height {
511            return;
512        }
513        let sdf_config = self.sdf_config.clone();
514        *self = Self::new(device, width, height, sdf_config);
515    }

529    #[test]
530    fn sdf_box_distance() {
531        // Point at origin, box centered at origin with half-extents 1
532        let d = sdf_box(Vec3::ZERO, Vec3::ZERO, Vec3::ONE);
533        assert!(d < 0.0, "Point inside box should be negative: {d}");
