1//! Canonical terrain material library — single source of truth for material
2//! identity at the UI/renderer boundary.
3//!
4//! Real-Fix.D 2026-05-08: established to resolve Defect Class 4 (Material
5//! Library Coverage Gap; FOURTH §7.7 instance per Andrew-gate decision (h)
6//! Option D-2). Both UI panel material rendering AND renderer texture-array
7//! allocation AND splat-builder cap derive from this module — eliminating the
8//! prior three-tier capacity boundary mismatch (UI=22; splat-build=8;
9//! renderer=8).
10//!
11//! # Library shape
12//!
13//! 22 named materials + 10 reserved slots = 32 total layers. The canonical
14//! [`MAX_TERRAIN_LAYERS`] = 32 capacity provides headroom: future content can
15//! occupy IDs 22..32 without reopening the §7.7 wrapped-component resource
16//! identity trap.
17//!
18//! # Layer ID stability
19//!
20//! Layer IDs 0..7 preserve the historical biome→layer mapping from the
21//! pre-Real-Fix.D 8-layer splat pipeline (grass=0, sand=1, forest_floor=2,
22//! mountain_rock=3, snow=4, mud=5, wood_planks=6, stone=7). IDs 8..21 add the
23//! materials that were previously dropped by the splat builder cap (Round-8
24//! evidence). IDs 22..31 are reserved for future expansion.
25
26/// Maximum number of terrain material layers supported by the splat pipeline.
27///
28/// 32 = 22 named materials + 10 reserved slots. The WGSL shaders
29/// (`pbr_terrain.wgsl`, `pbr_terrain_forward.wgsl`) declare matching
30/// `array<TerrainLayer, 32>` and 8 splat textures (32 channels @ RGBA each).
31/// The Rust GPU mirrors (`TerrainMaterialGpu`) match byte-for-byte.
32pub const MAX_TERRAIN_LAYERS: u32 = 32;
33
34/// Number of RGBA8 splat textures used to encode per-vertex layer weights.
35///
36/// Each texture carries 4 channels of layer weights:
37/// `splats[i]` channel R..A → layers `i*4+0 .. i*4+3`.
38/// `MAX_TERRAIN_LAYERS / 4 = 32 / 4 = 8`.
39pub const NUM_SPLAT_MAPS: usize = (MAX_TERRAIN_LAYERS as usize) / 4;
40
41/// A single material entry in the canonical terrain material library.
42#[derive(Debug, Clone, Copy, PartialEq, Eq)]
43pub struct Material {
44    /// Canonical layer ID (0..MAX_TERRAIN_LAYERS).
45    pub id: u32,
46    /// Asset path stem; texture loader expects `assets/materials/{name}.png`,
47    /// `{name}_n.png`, `{name}_mra.png`.
48    pub name: &'static str,
49    /// Human-readable label for UI display.
50    pub display_name: &'static str,
51}
52

177    /// Total slot count (named + reserved). Equals [`MAX_TERRAIN_LAYERS`].
178    pub const fn len() -> usize {
179        MAX_TERRAIN_LAYERS as usize
180    }
181
182    /// Number of named (non-reserved) materials.
183    pub const fn named_count() -> usize {
184        Self::MATERIALS.len()
185    }
186
187    /// All canonical named materials in slot order.
188    pub const fn named() -> &'static [Material] {
189        &Self::MATERIALS
190    }
191
192    /// Look up a material by canonical ID. Returns `None` for reserved slots
193    /// (id ≥ [`MaterialLibrary::named_count()`]) and out-of-range IDs.
194    pub fn get(id: u32) -> Option<&'static Material> {
195        let idx = id as usize;
196        Self::MATERIALS.get(idx).filter(|m| m.id == id)
197    }
198
199    /// Look up a material name (asset path stem) by canonical ID.
200    pub fn name(id: u32) -> Option<&'static str> {
201        Self::get(id).map(|m| m.name)
202    }
203
204    /// Look up the display label for a canonical ID.
205    pub fn display_name(id: u32) -> Option<&'static str> {
206        Self::get(id).map(|m| m.display_name)
207    }
208
209    /// True if `id` is a named (non-reserved) layer.
210    pub fn is_named(id: u32) -> bool {
211        Self::get(id).is_some()
212    }
213
214    /// True if `id` is within the canonical capacity (named OR reserved).
215    pub fn is_in_range(id: u32) -> bool {
216        id < MAX_TERRAIN_LAYERS
217    }
218}

277#[cfg(test)]
278mod tests {
279    use super::*;
280
281    #[test]
282    fn library_capacity_is_32() {
283        assert_eq!(MAX_TERRAIN_LAYERS, 32);
284        assert_eq!(MaterialLibrary::len(), 32);
285        assert_eq!(NUM_SPLAT_MAPS, 8);
286    }
287
288    #[test]
289    fn library_has_22_named_materials() {
290        assert_eq!(MaterialLibrary::named_count(), 22);
291        assert_eq!(MaterialLibrary::named().len(), 22);
292        assert_eq!(MATERIAL_NAMES.len(), 22);
293        assert_eq!(MATERIAL_DISPLAY_NAMES.len(), 22);
294    }
295
296    #[test]
297    fn library_ids_match_slot_order() {
298        for (slot, material) in MaterialLibrary::named().iter().enumerate() {
299            assert_eq!(material.id as usize, slot);
300        }
301    }
302
303    #[test]
304    fn flat_arrays_match_canonical_library() {
305        for (i, material) in MaterialLibrary::named().iter().enumerate() {
306            assert_eq!(material.name, MATERIAL_NAMES[i]);
307            assert_eq!(material.display_name, MATERIAL_DISPLAY_NAMES[i]);
308        }
309    }
310
311    #[test]
312    fn get_returns_named_for_valid_ids() {
313        for id in 0..22 {
314            assert!(MaterialLibrary::get(id).is_some());
315            assert!(MaterialLibrary::is_named(id));
316        }
317    }
318
319    #[test]
320    fn get_returns_none_for_reserved_ids() {
321        for id in 22..32 {
322            assert!(MaterialLibrary::get(id).is_none());
323            assert!(!MaterialLibrary::is_named(id));
324            assert!(MaterialLibrary::is_in_range(id));
325        }
326    }
327
328    #[test]
329    fn get_returns_none_for_out_of_range_ids() {
330        assert!(MaterialLibrary::get(32).is_none());
331        assert!(MaterialLibrary::get(100).is_none());
332        assert!(MaterialLibrary::get(u32::MAX).is_none());
333        assert!(!MaterialLibrary::is_in_range(32));
334    }
335
336    #[test]
337    fn name_lookup_examples() {
338        assert_eq!(MaterialLibrary::name(0), Some("grass"));
339        assert_eq!(MaterialLibrary::name(8), Some("rock_slate"));
340        assert_eq!(MaterialLibrary::name(21), Some("tree_leaves"));
341        assert_eq!(MaterialLibrary::name(22), None);
342    }
343
344    #[test]
345    fn display_name_lookup_examples() {
346        assert_eq!(MaterialLibrary::display_name(0), Some("Grass"));
347        assert_eq!(MaterialLibrary::display_name(8), Some("Rock Slate"));
348        assert_eq!(MaterialLibrary::display_name(21), Some("Tree Leaves"));
349        assert_eq!(MaterialLibrary::display_name(22), None);
350    }
