1//! Screen-Space Global Illumination (SSGI) with temporal reprojection.
2//!
3//! Provides one-bounce indirect diffuse lighting by tracing rays in screen space
4//! against the depth buffer. Includes spatial bilateral filtering and temporal
5//! accumulation via motion vectors for stable, noise-free results.
6
7/// GPU-side uniform parameters for the SSGI compute pass.
8#[repr(C)]
9#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
10pub struct SsgiParams {
11    pub inv_proj: [[f32; 4]; 4],
12    pub proj: [[f32; 4]; 4],
13    pub resolution: [f32; 2],
14    pub inv_resolution: [f32; 2],
15    pub max_ray_distance: f32,
16    pub ray_step_size: f32,
17    pub num_rays: u32,
18    pub max_steps: u32,
19    pub thickness: f32,
20    pub intensity: f32,
21    pub frame_index: u32,
22    pub _pad: u32,
23}
24
25impl Default for SsgiParams {
26    fn default() -> Self {
27        Self {
28            inv_proj: glam::Mat4::IDENTITY.to_cols_array_2d(),
29            proj: glam::Mat4::IDENTITY.to_cols_array_2d(),
30            resolution: [1920.0, 1080.0],
31            inv_resolution: [1.0 / 1920.0, 1.0 / 1080.0],
32            max_ray_distance: 50.0,
33            ray_step_size: 0.5,
34            num_rays: 4,
35            max_steps: 32,
36            thickness: 0.5,
37            intensity: 1.0,
38            frame_index: 0,
39            _pad: 0,
40        }
41    }
42}

95/// Manages SSGI GPU resources.
96#[allow(dead_code)]
97pub struct SsgiPass {
98    config: SsgiConfig,
99    /// SSGI compute pipeline.
100    gi_pipeline: wgpu::ComputePipeline,
101    /// Denoise compute pipeline.
102    denoise_pipeline: wgpu::ComputePipeline,
103    /// SSGI params buffer.
104    params_buf: wgpu::Buffer,
105    /// Denoise params buffer.
106    denoise_buf: wgpu::Buffer,
107    /// Current frame GI output (rgba16float).
108    #[allow(dead_code)] // texture must be kept alive for view to remain valid
109    gi_texture: wgpu::Texture,
110    gi_view: wgpu::TextureView,
111    /// History GI texture for temporal accumulation.
112    #[allow(dead_code)] // texture must be kept alive for view to remain valid
113    history_texture: wgpu::Texture,
114    history_view: wgpu::TextureView,
115    /// Denoised output.
116    #[allow(dead_code)] // texture must be kept alive for view to remain valid
117    denoised_texture: wgpu::Texture,
118    denoised_view: wgpu::TextureView,
119    /// Bind group layouts.
120    gi_bgl: wgpu::BindGroupLayout,
121    #[allow(dead_code)]
122    denoise_bgl: wgpu::BindGroupLayout,
123    frame_index: u32,
124    width: u32,
125    height: u32,
126}
127
128impl SsgiPass {
129    pub fn new(device: &wgpu::Device, width: u32, height: u32) -> Self {
130        let config = SsgiConfig::default();
131        let size = wgpu::Extent3d {
132            width,
133            height,
134            depth_or_array_layers: 1,
135        };
136        let gi_fmt = wgpu::TextureFormat::Rgba16Float;
137
138        let make_tex = |label: &str| {
139            device.create_texture(&wgpu::TextureDescriptor {
140                label: Some(label),
141                size,
142                mip_level_count: 1,
143                sample_count: 1,
144                dimension: wgpu::TextureDimension::D2,
145                format: gi_fmt,
146                usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
147                view_formats: &[],
148            })
149        };
150
151        let gi_texture = make_tex("ssgi_output");
152        let gi_view = gi_texture.create_view(&wgpu::TextureViewDescriptor::default());
153        let history_texture = make_tex("ssgi_history");
154        let history_view = history_texture.create_view(&wgpu::TextureViewDescriptor::default());
155        let denoised_texture = make_tex("ssgi_denoised");
156        let denoised_view = denoised_texture.create_view(&wgpu::TextureViewDescriptor::default());
157
158        use wgpu::util::DeviceExt;
159        let params = SsgiParams::default();
160        let params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
161            label: Some("ssgi_params"),
162            contents: bytemuck::bytes_of(&params),
163            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
164        });
165
166        let denoise_params = SsgiDenoiseParams::default();
167        let denoise_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
168            label: Some("ssgi_denoise_params"),
169            contents: bytemuck::bytes_of(&denoise_params),
170            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
171        });
172
173        // GI bind group layout
174        let gi_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
175            label: Some("ssgi_gi_bgl"),
176            entries: &[
177                bgl_texture(0),            // depth
178                bgl_texture(1),            // normals
179                bgl_texture(2),            // color
180                bgl_sampler(3),            // sampler
181                bgl_uniform(4),            // params
182                bgl_storage_rw(5, gi_fmt), // output
183            ],
184        });
185
186        // Denoise bind group layout
187        let denoise_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
188            label: Some("ssgi_denoise_bgl"),
189            entries: &[
190                bgl_texture(0),            // gi current
191                bgl_texture(1),            // gi history
192                bgl_texture(2),            // depth
193                bgl_texture(3),            // velocity
194                bgl_sampler(4),            // sampler
195                bgl_uniform(5),            // denoise params
196                bgl_storage_rw(6, gi_fmt), // output
197            ],
198        });
199
200        // Pipelines
201        let gi_shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
202            label: Some("ssgi_shader"),
203            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/ssgi.wgsl").into()),
204        });
205        let gi_pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
206            label: Some("ssgi_gi_pl"),
207            bind_group_layouts: &[&gi_bgl],
208            push_constant_ranges: &[],
209        });
210        let gi_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
211            label: Some("ssgi_gi_pipeline"),
212            layout: Some(&gi_pl),
213            module: &gi_shader,
214            entry_point: Some("ssgi_main"),
215            compilation_options: Default::default(),
216            cache: None,
217        });
218
219        let denoise_shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
220            label: Some("ssgi_denoise_shader"),
221            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/ssgi_denoise.wgsl").into()),
222        });
223        let denoise_pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
224            label: Some("ssgi_denoise_pl"),
225            bind_group_layouts: &[&denoise_bgl],
226            push_constant_ranges: &[],
227        });
228        let denoise_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
229            label: Some("ssgi_denoise_pipeline"),
230            layout: Some(&denoise_pl),
231            module: &denoise_shader,
232            entry_point: Some("denoise_main"),
233            compilation_options: Default::default(),
234            cache: None,
235        });
236
237        Self {
238            config,
239            gi_pipeline,
240            denoise_pipeline,
241            params_buf,
242            denoise_buf,
243            gi_texture,
244            gi_view,
245            history_texture,
246            history_view,
247            denoised_texture,
248            denoised_view,
249            gi_bgl,
250            denoise_bgl,
251            frame_index: 0,
252            width,
253            height,
254        }
255    }
256
257    /// Get the final denoised GI texture view.
258    #[allow(clippy::misnamed_getters)]
259    pub fn gi_view(&self) -> &wgpu::TextureView {
260        &self.denoised_view
261    }
262
263    pub fn config(&self) -> &SsgiConfig {
264        &self.config
265    }
266
267    pub fn set_config(&mut self, config: SsgiConfig) {
268        self.config = config;
269    }
270
271    pub fn dimensions(&self) -> (u32, u32) {
272        (self.width, self.height)
273    }
274
275    /// Update parameters for this frame.
276    pub fn update_params(&mut self, queue: &wgpu::Queue, proj: glam::Mat4) {
277        let params = SsgiParams {
278            inv_proj: proj.inverse().to_cols_array_2d(),
279            proj: proj.to_cols_array_2d(),
280            resolution: [self.width as f32, self.height as f32],
281            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
282            max_ray_distance: 50.0,
283            ray_step_size: self.config.ray_step_size,
284            num_rays: self.config.num_rays,
285            max_steps: self.config.max_steps,
286            thickness: self.config.thickness,
287            intensity: self.config.intensity,
288            frame_index: self.frame_index,
289            _pad: 0,
290        };
291        queue.write_buffer(&self.params_buf, 0, bytemuck::bytes_of(&params));
292
293        let denoise = SsgiDenoiseParams {
294            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
295            temporal_blend: self.config.temporal_blend,
296            ..SsgiDenoiseParams::default()
297        };
298        queue.write_buffer(&self.denoise_buf, 0, bytemuck::bytes_of(&denoise));
299        self.frame_index = self.frame_index.wrapping_add(1);
300    }
301
302    /// Resize all SSGI textures.
303    pub fn resize(&mut self, device: &wgpu::Device, width: u32, height: u32) {
304        if self.width == width && self.height == height {
305            return;
306        }
307        *self = Self::new(device, width, height);
308    }
309}

311// Helper functions for bind group layout entries
312fn bgl_texture(binding: u32) -> wgpu::BindGroupLayoutEntry {
313    wgpu::BindGroupLayoutEntry {
314        binding,
315        visibility: wgpu::ShaderStages::COMPUTE,
316        ty: wgpu::BindingType::Texture {
317            sample_type: wgpu::TextureSampleType::Float { filterable: true },
318            view_dimension: wgpu::TextureViewDimension::D2,
319            multisampled: false,
320        },
321        count: None,
322    }
323}
324
325fn bgl_sampler(binding: u32) -> wgpu::BindGroupLayoutEntry {
326    wgpu::BindGroupLayoutEntry {
327        binding,
328        visibility: wgpu::ShaderStages::COMPUTE,
329        ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
330        count: None,
331    }
332}
333
334fn bgl_uniform(binding: u32) -> wgpu::BindGroupLayoutEntry {
335    wgpu::BindGroupLayoutEntry {
336        binding,
337        visibility: wgpu::ShaderStages::COMPUTE,
338        ty: wgpu::BindingType::Buffer {
339            ty: wgpu::BufferBindingType::Uniform,
340            has_dynamic_offset: false,
341            min_binding_size: None,
342        },
343        count: None,
344    }
345}
346
347fn bgl_storage_rw(binding: u32, format: wgpu::TextureFormat) -> wgpu::BindGroupLayoutEntry {
348    wgpu::BindGroupLayoutEntry {
349        binding,
350        visibility: wgpu::ShaderStages::COMPUTE,
351        ty: wgpu::BindingType::StorageTexture {
352            access: wgpu::StorageTextureAccess::WriteOnly,
353            format,
354            view_dimension: wgpu::TextureViewDimension::D2,
355        },
356        count: None,
357    }
358}
