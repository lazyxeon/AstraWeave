1//! Subgroup operation support for GPU compute shaders.
2//!
3//! Provides capability detection and shader variant selection for subgroup-optimized
4//! compute passes (auto-exposure, prefix sum, bitonic sort).
5//!
6//! Subgroup operations (`subgroupAdd`, `subgroupExclusiveAdd`, `subgroupShuffleXor`)
7//! collapse `log2(subgroup_size)` tree-reduction steps into single instructions,
8//! eliminating `workgroupBarrier()` calls and global memory round-trips.
9//!
10//! # Feature Gating
11//!
12//! Subgroup operations require `wgpu::Features::SUBGROUP` (Vulkan/Metal only — not WebGPU).
13//! All subgroup shaders have non-subgroup fallbacks; the renderer selects the variant at
14//! pipeline creation time based on adapter capabilities.
15//!
16//! # Optimized Shaders
17//!
18//! | Shader | Subgroup Optimization | Estimated Speedup |
19//! |--------|----------------------|-------------------|
20//! | Auto-exposure (average pass) | `subgroupAdd` reduction | 2–3× fewer barriers |
21//! | MegaLights prefix sum | `subgroupExclusiveAdd` scan | ~30% fewer barriers |
22//! | Bitonic sort (inner stages) | `subgroupShuffleXor` swap | ~30–40% fewer dispatches |
23
24/// WGSL source for the subgroup-optimized auto-exposure shader.
25pub const AUTO_EXPOSURE_SUBGROUP_WGSL: &str =
26    include_str!("../shaders/subgroup/auto_exposure_subgroup.wgsl");
27
28/// WGSL source for the subgroup-optimized prefix sum shader.
29pub const PREFIX_SUM_SUBGROUP_WGSL: &str =
30    include_str!("../shaders/subgroup/prefix_sum_subgroup.wgsl");
31
32/// WGSL source for the subgroup-optimized bitonic sort shader.
33pub const BITONIC_SORT_SUBGROUP_WGSL: &str =
34    include_str!("../shaders/subgroup/bitonic_sort_subgroup.wgsl");

36/// Subgroup capability information for the current device.
37#[derive(Debug, Clone, Default)]
38pub struct SubgroupCapabilities {
39    /// Whether the device supports subgroup operations at all.
40    pub supported: bool,
41    /// Minimum subgroup size (typically 4 on Intel, 32 on NVIDIA, 64 on AMD).
42    pub min_subgroup_size: u32,
43    /// Maximum subgroup size.
44    pub max_subgroup_size: u32,
45}
46
47impl SubgroupCapabilities {
48    /// Query subgroup capabilities from wgpu adapter features.
49    ///
50    /// Returns capabilities with `supported = true` only if the adapter
51    /// advertises `wgpu::Features::SUBGROUP`.
52    pub fn from_features(features: wgpu::Features) -> Self {
53        let supported = features.contains(wgpu::Features::SUBGROUP);
54        if supported {
55            // wgpu doesn't expose subgroup size directly in Features;
56            // actual size is determined at shader execution time via
57            // @builtin(subgroup_size). We report common GPU ranges.
58            Self {
59                supported: true,
60                min_subgroup_size: 4,  // Intel iGPU min
61                max_subgroup_size: 64, // AMD RDNA max
62            }
63        } else {
64            Self {
65                supported: false,
66                min_subgroup_size: 0,
67                max_subgroup_size: 0,
68            }
69        }
70    }
71
72    /// Check if subgroup operations are available.
73    pub fn is_available(&self) -> bool {
74        self.supported
75    }
76}
77
78/// Select the appropriate shader source based on subgroup capabilities.
79///
80/// Returns `(shader_source, is_subgroup_variant)`.
81pub fn select_auto_exposure_shader(caps: &SubgroupCapabilities) -> (&'static str, bool) {
82    if caps.is_available() {
83        (AUTO_EXPOSURE_SUBGROUP_WGSL, true)
84    } else {
85        (include_str!("../shaders/auto_exposure.wgsl"), false)
86    }
87}
88
89/// Select the appropriate prefix sum shader source.
90pub fn select_prefix_sum_shader(caps: &SubgroupCapabilities) -> (&'static str, bool) {
91    if caps.is_available() {
92        (PREFIX_SUM_SUBGROUP_WGSL, true)
93    } else {
94        (include_str!("../shaders/megalights/prefix_sum.wgsl"), false)
95    }
96}
97
98/// Select the appropriate bitonic sort shader source.
99pub fn select_bitonic_sort_shader(caps: &SubgroupCapabilities) -> (&'static str, bool) {
100    if caps.is_available() {
101        (BITONIC_SORT_SUBGROUP_WGSL, true)
102    } else {
103        (
104            include_str!("../shaders/particles/bitonic_sort.wgsl"),
105            false,
106        )
107    }
108}

110#[cfg(test)]
111mod tests {
112    use super::*;
113
114    #[test]
115    fn capabilities_default_no_subgroup() {
116        let caps = SubgroupCapabilities::default();
117        assert!(!caps.supported);
118        assert!(!caps.is_available());
119        assert_eq!(caps.min_subgroup_size, 0);
120    }
121
122    #[test]
123    fn capabilities_from_empty_features() {
124        let caps = SubgroupCapabilities::from_features(wgpu::Features::empty());
125        assert!(!caps.supported);
126    }
127
128    #[test]
129    fn capabilities_from_subgroup_feature() {
130        let caps = SubgroupCapabilities::from_features(wgpu::Features::SUBGROUP);
131        assert!(caps.supported);
132        assert!(caps.is_available());
133        assert!(caps.min_subgroup_size > 0);
134        assert!(caps.max_subgroup_size >= caps.min_subgroup_size);
135    }
136
137    #[test]
138    fn select_auto_exposure_without_subgroup() {
139        let caps = SubgroupCapabilities::default();
140        let (source, is_subgroup) = select_auto_exposure_shader(&caps);
141        assert!(!is_subgroup);
142        assert!(source.contains("histogram_pass"));
143        assert!(!source.contains("enable subgroups"));
144    }
145
146    #[test]
147    fn select_auto_exposure_with_subgroup() {
148        let caps = SubgroupCapabilities::from_features(wgpu::Features::SUBGROUP);
149        let (source, is_subgroup) = select_auto_exposure_shader(&caps);
150        assert!(is_subgroup);
151        assert!(source.contains("enable subgroups"));
152        assert!(source.contains("subgroupAdd"));
153    }
