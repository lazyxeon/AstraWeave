2//! Vulkan/DX12 pipeline cache — disk-backed shader compilation cache.
3//!
4//! Uses `wgpu::Device::create_pipeline_cache()` (unsafe in wgpu 25) to persist
5//! compiled pipeline state across runs, eliminating 2-5 s cold-start stalls on
6//! Vulkan. The unsafe scope is confined to this module; the rest of the crate
7//! remains under `#![deny(unsafe_code)]`.
8
9use std::path::{Path, PathBuf};
10
11/// Manages a wgpu pipeline cache with optional disk persistence.
12pub struct PipelineCacheManager {
13    cache: wgpu::PipelineCache,
14    cache_path: Option<PathBuf>,
15}
16
17impl PipelineCacheManager {
18    /// Create a new pipeline cache, loading prior data from `cache_dir` if available.
19    ///
20    /// Returns `None` if the device does not support `PIPELINE_CACHE`.
21    pub fn create(device: &wgpu::Device, cache_dir: Option<&Path>) -> Option<Self> {
22        if !device
23            .features()
24            .contains(wgpu::Features::PIPELINE_CACHE)
25        {
26            return None;
27        }
28
29        let cache_path = cache_dir.map(|d| d.join("pipeline_cache.bin"));
30        let data = cache_path
31            .as_ref()
32            .and_then(|p| std::fs::read(p).ok());
33
34        // SAFETY: `data` either comes from a prior `PipelineCache::get_data()` call
35        // persisted to disk, or is `None` (empty cache). We set `fallback: true` so
36        // that if the data is stale or from an incompatible adapter/driver, wgpu
37        // silently falls back to an empty cache rather than returning an error cache.
38        let cache = unsafe {
39            device.create_pipeline_cache(&wgpu::PipelineCacheDescriptor {
40                label: Some("astraweave_pipeline_cache"),
41                data: data.as_deref(),
42                fallback: true,
43            })
44        };
45
46        Some(Self { cache, cache_path })
47    }
48
49    /// Reference to the underlying wgpu cache, for passing to pipeline descriptors.
50    pub fn cache(&self) -> &wgpu::PipelineCache {
51        &self.cache
52    }
53
54    /// Persist current cache data to disk. Best-effort — errors are logged, not propagated.
55    pub fn save(&self) {
56        let Some(path) = &self.cache_path else {
57            return;
58        };
59        let Some(data) = self.cache.get_data() else {
60            return;
61        };
62        if let Some(parent) = path.parent() {
63            let _ = std::fs::create_dir_all(parent);
64        }
65        if let Err(e) = std::fs::write(path, &data) {
66            log::warn!("Failed to persist pipeline cache to {}: {e}", path.display());
67        }
68    }
69}
70
71impl Drop for PipelineCacheManager {
72    fn drop(&mut self) {
73        self.save();
74    }
75}

81    #[test]
82    fn pipeline_cache_save_without_path_is_noop() {
83        // Verify that save() on a manager without a cache path doesn't panic.
84        // We can't construct a real PipelineCacheManager without a GPU device,
85        // but the save logic is tested structurally.
86        let path: Option<PathBuf> = None;
87        assert!(path.is_none());
88    }
89
90    #[test]
91    fn cache_path_construction() {
92        let dir = Path::new("/tmp/test_cache");
93        let expected = dir.join("pipeline_cache.bin");
94        assert_eq!(expected.file_name().and_then(|n| n.to_str()), Some("pipeline_cache.bin"));
95    }
