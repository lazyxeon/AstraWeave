1//! Bind group cache to eliminate per-frame `device.create_bind_group()` overhead.
2//!
3//! Many render passes create identical bind groups every frame because their
4//! input textures and buffers haven't changed. This module provides a
5//! generation-based cache: bind groups are rebuilt only when the renderer
6//! signals a resource change (e.g., resize, texture swap) by bumping the
7//! generation counter.
8//!
9//! # Usage
10//! ```ignore
11//! // In pass struct:
12//! cached_bg: CachedBindGroup,
13//!
14//! // In execute():
15//! let bg = self.cached_bg.get_or_rebuild(current_gen, || {
16//!     device.create_bind_group(&descriptor)
17//! });
18//! pass.set_bind_group(0, bg, &[]);
19//! ```
20
21/// Monotonic generation counter. Bumped by the renderer on resize or
22/// other resource-invalidating events.
23pub type Generation = u64;
24
25/// A lazily-cached bind group that rebuilds when the generation advances.
26pub struct CachedBindGroup {
27    bind_group: Option<wgpu::BindGroup>,
28    generation: Generation,
29}
30
31impl CachedBindGroup {
32    /// Create an empty (uncached) entry.
33    pub fn new() -> Self {
34        Self {
35            bind_group: None,
36            generation: 0,
37        }
38    }
39
40    /// Create a pre-populated cache entry at the given generation.
41    pub fn with_bind_group(bg: wgpu::BindGroup, gen: Generation) -> Self {
42        Self {
43            bind_group: Some(bg),
44            generation: gen,
45        }
46    }
47
48    /// Get the cached bind group, or rebuild it if stale.
49    ///
50    /// `current_gen` is the renderer's current generation counter.
51    /// `rebuild` is called only if the cache is empty or the generation changed.
52    pub fn get_or_rebuild(
53        &mut self,
54        current_gen: Generation,
55        rebuild: impl FnOnce() -> wgpu::BindGroup,
56    ) -> &wgpu::BindGroup {
57        if self.generation != current_gen {
58            // Generation changed — force a rebuild by clearing first so
59            // `get_or_insert_with` runs the closure.
60            self.bind_group = None;
61            self.generation = current_gen;
62        }
63        // Returns `&mut wgpu::BindGroup`, which reborrows as `&wgpu::BindGroup`.
64        // Avoids any `expect()` round-trip on a freshly-set Some.
65        self.bind_group.get_or_insert_with(rebuild)
66    }
67
68    /// Explicitly invalidate the cache, forcing a rebuild on next access.
69    pub fn invalidate(&mut self) {
70        self.bind_group = None;
71    }
72
73    /// Check whether the cache is populated and current.
74    pub fn is_valid(&self, current_gen: Generation) -> bool {
75        self.bind_group.is_some() && self.generation == current_gen
76    }
77}
78
79impl Default for CachedBindGroup {
80    fn default() -> Self {
81        Self::new()
82    }
83}
84
85/// A set of cached bind groups for passes that need multiple groups per frame.
86///
87/// Each slot is independently versioned so partial rebuilds are possible
88/// (e.g., one group depends on depth and another on HDR, which may be
89/// invalidated at different times).
90pub struct CachedBindGroupSet {
91    groups: Vec<CachedBindGroup>,
92}
93
94impl CachedBindGroupSet {
95    /// Create a set with `count` empty slots.
96    pub fn new(count: usize) -> Self {
97        Self {
98            groups: (0..count).map(|_| CachedBindGroup::new()).collect(),
99        }
100    }
101
102    /// Get or rebuild the bind group at `index`.
103    pub fn get_or_rebuild(
104        &mut self,
105        index: usize,
106        current_gen: Generation,
107        rebuild: impl FnOnce() -> wgpu::BindGroup,
108    ) -> &wgpu::BindGroup {
109        self.groups[index].get_or_rebuild(current_gen, rebuild)
110    }
111
112    /// Invalidate all cached bind groups.
113    pub fn invalidate_all(&mut self) {
114        for bg in &mut self.groups {
115            bg.invalidate();
116        }
117    }
118
119    /// Check whether the bind group at `index` is populated and current.
120    pub fn is_valid(&self, index: usize, current_gen: Generation) -> bool {
121        self.groups[index].is_valid(current_gen)
122    }
123
124    /// Set the bind group at `index` to a pre-built bind group at the given generation.
125    pub fn set(&mut self, index: usize, bg: wgpu::BindGroup, gen: Generation) {
126        self.groups[index] = CachedBindGroup::with_bind_group(bg, gen);
127    }
128
129    /// Number of slots.
130    pub fn len(&self) -> usize {
131        self.groups.len()
132    }
133
134    /// Whether all slots are empty.
135    pub fn is_empty(&self) -> bool {
136        self.groups.is_empty()
137    }
138}
