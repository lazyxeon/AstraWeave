1//! Screen-space God Rays (Crepuscular Rays / Light Shafts).
2//!
3//! Projects the sun position to screen space and performs radial blur
4//! from the sun, sampling the depth buffer to determine occlusion.
5//! Produces bright shafts where light passes between occluders.
6//!
7//! The effect is additive — it should be composited after the main
8//! lighting pass (and optionally after volumetric fog).
9
10use bytemuck::{Pod, Zeroable};
11use glam::{Mat4, Vec3};
12use wgpu::util::DeviceExt;

19#[repr(C)]
20#[derive(Debug, Clone, Copy, Pod, Zeroable)]
21pub struct GodRayParams {
22    pub resolution: [f32; 2],
23    pub inv_resolution: [f32; 2],
24    pub sun_screen_pos: [f32; 2],
25    pub sun_visible: f32,
26    pub num_samples: u32,
27    pub density: f32,
28    pub weight: f32,
29    pub decay: f32,
30    pub exposure: f32,
31    pub sun_color: [f32; 3],
32    pub _pad: f32,
33}

39/// Configuration for god rays.
40#[derive(Debug, Clone)]
41pub struct GodRayConfig {
42    pub enabled: bool,
43    /// Number of radial samples per pixel (32–128).
44    pub num_samples: u32,
45    /// Ray density multiplier (controls ray length).
46    pub density: f32,
47    /// Per-sample contribution weight.
48    pub weight: f32,
49    /// Exponential decay per sample (0.9–0.99).
50    pub decay: f32,
51    /// Final brightness multiplier.
52    pub exposure: f32,
53    /// Render at half resolution for performance (bilinear upscale on composite).
54    pub half_res: bool,
55}
56
57impl Default for GodRayConfig {
58    fn default() -> Self {
59        Self {
60            enabled: true,
61            num_samples: 48,
62            density: 0.96,
63            weight: 0.25,
64            decay: 0.97,
65            exposure: 0.3,
66            half_res: false,
67        }
68    }
69}

75/// Manages GPU resources for the god rays compute pass.
76pub struct GodRayPass {
77    config: GodRayConfig,
78    pipeline: wgpu::ComputePipeline,
79    bgl: wgpu::BindGroupLayout,
80    params_buf: wgpu::Buffer,
81    #[allow(dead_code)] // texture must be kept alive for view to remain valid
82    output_texture: wgpu::Texture,
83    output_view: wgpu::TextureView,
84    width: u32,
85    height: u32,
86    /// Cached sampler (static — never changes).
87    sampler: wgpu::Sampler,
88    /// Cached bind group (rebuilt on generation change).
89    cached_bg: crate::bind_group_cache::CachedBindGroup,
90}

92impl GodRayPass {
93    pub fn new(device: &wgpu::Device, width: u32, height: u32) -> Self {
94        Self::with_config(device, width, height, GodRayConfig::default())
95    }

97    pub fn with_config(
98        device: &wgpu::Device,
99        width: u32,
100        height: u32,
101        config: GodRayConfig,
102    ) -> Self {
103        // Compute effective resolution (half-res if enabled)
104        let tex_w = if config.half_res {
105            width.div_ceil(2)
106        } else {
107            width
108        };
109        let tex_h = if config.half_res {
110            height.div_ceil(2)
111        } else {
112            height
113        };
114        let fmt = wgpu::TextureFormat::Rgba16Float;
115
116        let output_texture = device.create_texture(&wgpu::TextureDescriptor {
117            label: Some("god_rays_output"),
118            size: wgpu::Extent3d {
119                width: tex_w,
120                height: tex_h,
121                depth_or_array_layers: 1,
122            },
123            mip_level_count: 1,
124            sample_count: 1,
125            dimension: wgpu::TextureDimension::D2,
126            format: fmt,
127            usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
128            view_formats: &[],
129        });
130        let output_view = output_texture.create_view(&wgpu::TextureViewDescriptor::default());
131
132        let params = GodRayParams {
133            resolution: [tex_w as f32, tex_h as f32],
134            inv_resolution: [1.0 / tex_w as f32, 1.0 / tex_h as f32],
135            sun_screen_pos: [0.5, 0.5],
136            sun_visible: 0.0,
137            num_samples: config.num_samples,
138            density: config.density,
139            weight: config.weight,
140            decay: config.decay,
141            exposure: config.exposure,
142            sun_color: [1.0, 0.95, 0.85],
143            _pad: 0.0,
144        };
145        let params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
146            label: Some("god_ray_params"),
147            contents: bytemuck::bytes_of(&params),
148            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
149        });
150
151        let bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
152            label: Some("god_ray_bgl"),
153            entries: &[
154                // 0: params
155                wgpu::BindGroupLayoutEntry {
156                    binding: 0,
157                    visibility: wgpu::ShaderStages::COMPUTE,
158                    ty: wgpu::BindingType::Buffer {
159                        ty: wgpu::BufferBindingType::Uniform,
160                        has_dynamic_offset: false,
161                        min_binding_size: None,
162                    },
163                    count: None,
164                },
165                // 1: depth (Depth32Float)
166                wgpu::BindGroupLayoutEntry {
167                    binding: 1,
168                    visibility: wgpu::ShaderStages::COMPUTE,
169                    ty: wgpu::BindingType::Texture {
170                        sample_type: wgpu::TextureSampleType::Depth,
171                        view_dimension: wgpu::TextureViewDimension::D2,
172                        multisampled: false,
173                    },
174                    count: None,
175                },
176                // 2: scene color
177                wgpu::BindGroupLayoutEntry {
178                    binding: 2,
179                    visibility: wgpu::ShaderStages::COMPUTE,
180                    ty: wgpu::BindingType::Texture {
181                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
182                        view_dimension: wgpu::TextureViewDimension::D2,
183                        multisampled: false,
184                    },
185                    count: None,
186                },
187                // 3: sampler
188                wgpu::BindGroupLayoutEntry {
189                    binding: 3,
190                    visibility: wgpu::ShaderStages::COMPUTE,
191                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
192                    count: None,
193                },
194                // 4: output
195                wgpu::BindGroupLayoutEntry {
196                    binding: 4,
197                    visibility: wgpu::ShaderStages::COMPUTE,
198                    ty: wgpu::BindingType::StorageTexture {
199                        access: wgpu::StorageTextureAccess::WriteOnly,
200                        format: fmt,
201                        view_dimension: wgpu::TextureViewDimension::D2,
202                    },
203                    count: None,
204                },
205            ],
206        });
207
208        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
209            label: Some("god_rays_shader"),
210            source: wgpu::ShaderSource::Wgsl(
211                include_str!("../shaders/volumetrics/god_rays.wgsl").into(),
212            ),
213        });
214        let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
215            label: Some("god_rays_pl"),
216            bind_group_layouts: &[&bgl],
217            push_constant_ranges: &[],
218        });
219        let pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
220            label: Some("god_rays_pipeline"),
221            layout: Some(&pl),
222            module: &shader,
223            entry_point: Some("god_rays_main"),
224            compilation_options: Default::default(),
225            cache: None,
226        });
227
228        Self {
229            config,
230            pipeline,
231            bgl,
232            params_buf,
233            output_texture,
234            output_view,
235            width: tex_w,
236            height: tex_h,
237            sampler: device.create_sampler(&wgpu::SamplerDescriptor {
238                label: Some("god_rays_sampler"),
239                mag_filter: wgpu::FilterMode::Linear,
240                min_filter: wgpu::FilterMode::Linear,
241                ..Default::default()
242            }),
243            cached_bg: crate::bind_group_cache::CachedBindGroup::new(),
244        }
245    }

264    /// Project the sun direction to screen space and update parameters.
265    pub fn update_params(
266        &self,
267        queue: &wgpu::Queue,
268        view_proj: Mat4,
269        sun_dir: Vec3,
270        sun_color: Vec3,
271    ) {
272        // Project a distant point in the sun direction to screen space
273        let sun_world = sun_dir * 10000.0; // far enough to be "at infinity"
274        let clip = view_proj * sun_world.extend(1.0);
275
276        let (sun_screen, visible) = if clip.w > 0.0 {
277            let ndc = glam::Vec2::new(clip.x / clip.w, clip.y / clip.w);
278            let uv = ndc * 0.5 + 0.5;
279            // Check if on screen (with some margin)
280            let on_screen = uv.x >= -0.2 && uv.x <= 1.2 && uv.y >= -0.2 && uv.y <= 1.2;
281            (uv.to_array(), if on_screen { 1.0_f32 } else { 0.0 })
282        } else {
283            ([0.5, 0.5], 0.0)
284        };
285
286        let params = GodRayParams {
287            resolution: [self.width as f32, self.height as f32],
288            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
289            sun_screen_pos: sun_screen,
290            sun_visible: visible,
291            num_samples: self.config.num_samples,
292            density: self.config.density,
293            weight: self.config.weight,
294            decay: self.config.decay,
295            exposure: self.config.exposure,
296            sun_color: sun_color.to_array(),
297            _pad: 0.0,
298        };
299        queue.write_buffer(&self.params_buf, 0, bytemuck::bytes_of(&params));
300    }

302    /// Dispatch the god rays compute pass.
303    ///
304    /// `resource_gen` is the renderer's generation counter; the bind group is
305    /// rebuilt only when it changes (e.g., after a resize).
306    pub fn execute(
307        &mut self,
308        device: &wgpu::Device,
309        encoder: &mut wgpu::CommandEncoder,
310        depth_view: &wgpu::TextureView,
311        scene_color_view: &wgpu::TextureView,
312        resource_gen: crate::bind_group_cache::Generation,
313    ) {
314        if !self.config.enabled {
315            return;
316        }
317
318        if !self.cached_bg.is_valid(resource_gen) {
319            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
320                label: Some("god_rays_bg"),
321                layout: &self.bgl,
322                entries: &[
323                    wgpu::BindGroupEntry {
324                        binding: 0,
325                        resource: self.params_buf.as_entire_binding(),
326                    },
327                    wgpu::BindGroupEntry {
328                        binding: 1,
329                        resource: wgpu::BindingResource::TextureView(depth_view),
330                    },
331                    wgpu::BindGroupEntry {
332                        binding: 2,
333                        resource: wgpu::BindingResource::TextureView(scene_color_view),
334                    },
335                    wgpu::BindGroupEntry {
336                        binding: 3,
337                        resource: wgpu::BindingResource::Sampler(&self.sampler),
338                    },
339                    wgpu::BindGroupEntry {
340                        binding: 4,
341                        resource: wgpu::BindingResource::TextureView(&self.output_view),
342                    },
343                ],
344            });
345            self.cached_bg =
346                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
347        }
348
349        let wg_x = self.width.div_ceil(8);
350        let wg_y = self.height.div_ceil(8);
351
352        let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
353            label: Some("god_rays"),
354            timestamp_writes: None,
355        });
356        pass.set_pipeline(&self.pipeline);
357        let bg = self
358            .cached_bg
359            .get_or_rebuild(resource_gen, || unreachable!());
360        pass.set_bind_group(0, bg, &[]);
361        pass.dispatch_workgroups(wg_x, wg_y, 1);
362    }

364    /// Resize output texture (preserves current config including half_res).
365    pub fn resize(&mut self, device: &wgpu::Device, width: u32, height: u32) {
366        let tex_w = if self.config.half_res {
367            width.div_ceil(2)
368        } else {
369            width
370        };
371        let tex_h = if self.config.half_res {
372            height.div_ceil(2)
373        } else {
374            height
375        };
376        if self.width == tex_w && self.height == tex_h {
377            return;
378        }
379        let config = self.config.clone();
380        *self = Self::with_config(device, width, height, config);
381    }

384/// Project a sun direction to screen-space UV (CPU utility).
385pub fn sun_to_screen(view_proj: Mat4, sun_dir: Vec3) -> Option<[f32; 2]> {
386    let clip = view_proj * (sun_dir * 10000.0).extend(1.0);
387    if clip.w <= 0.0 {
388        return None;
389    }
390    let ndc = glam::Vec2::new(clip.x / clip.w, clip.y / clip.w);
391    let uv = ndc * 0.5 + 0.5;
392    if uv.x >= -0.2 && uv.x <= 1.2 && uv.y >= -0.2 && uv.y <= 1.2 {
393        Some(uv.to_array())
394    } else {
395        None
396    }

417    #[test]
418    fn sun_to_screen_behind_camera() {
419        // Sun behind camera (negative Z in view space = behind)
420        let vp = Mat4::perspective_rh(1.0, 1.0, 0.1, 1000.0)
421            * Mat4::look_at_rh(Vec3::ZERO, Vec3::NEG_Z, Vec3::Y);
422        // Sun in +Z direction (behind camera looking at -Z)
423        let result = sun_to_screen(vp, Vec3::Z);
424        assert!(result.is_none(), "Sun behind camera should be None");
425    }

427    #[test]
428    fn sun_to_screen_in_front() {
429        let vp = Mat4::perspective_rh(1.0, 1.0, 0.1, 1000.0)
430            * Mat4::look_at_rh(Vec3::ZERO, Vec3::NEG_Z, Vec3::Y);
431        // Sun in -Z direction (in front of camera)
432        let result = sun_to_screen(vp, Vec3::NEG_Z);
433        assert!(result.is_some(), "Sun in front should be visible");
434        let [u, v] = result.unwrap();
435        // Should be near center of screen
436        assert!((u - 0.5).abs() < 0.1, "u={u}");
437        assert!((v - 0.5).abs() < 0.1, "v={v}");
438    }
