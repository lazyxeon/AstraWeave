1//! Water rendering system with animated Gerstner waves
2//!
3//! Provides a chunked, camera-distance LOD ocean surface with:
4//! - 4 summed Gerstner wave components
5//! - Fresnel-based reflections
6//! - Depth-based color blending
7//! - Animated foam on wave crests
8//! - Discrete chunk-grid LOD (finer near the camera, coarser toward the horizon)
9//! - Per-chunk skirts that hide LOD-boundary cracks
10//! - Real water-level control via a shader uniform (no baked mesh Y)
11//!
12//! ## LOD / chunking model (W-series W.2a)
13//!
14//! The surface is a discrete grid of square chunks (`CHUNK_SIZE` units each)
15//! centered on the camera. Each frame, every chunk in the `(2*GRID_RADIUS+1)²`
16//! block around the camera picks an LOD by its distance, and chunks are drawn
17//! **instanced per LOD** — one indexed-instanced draw call per LOD level,
18//! independent of how many chunks chose that level. Wave displacement is a pure
19//! function of world XZ, so chunk meshes are world-stable (no swimming) and
20//! shared LOD-boundary vertices agree exactly; the only mismatch is curve-vs-chord
21//! along a coarse neighbour's edge (≤ wave amplitude), which the per-chunk skirt
22//! covers. This replaces the former single hardcoded `generate_water_plane(500,128)`
23//! plane.
24
25use glam::{Mat4, Vec2, Vec3};
26use wgpu::util::DeviceExt;

28// ── Chunked LOD configuration ────────────────────────────────────────────────
29
30/// World-space edge length of one water chunk (units).
31const CHUNK_SIZE: f32 = 64.0;
32/// Chunk radius around the camera: a `(2*GRID_RADIUS+1)²` block is active.
33const GRID_RADIUS: i32 = 8;
34/// Maximum simultaneously-active chunks (instance-buffer capacity per LOD).
35const MAX_CHUNKS: usize = ((2 * GRID_RADIUS + 1) * (2 * GRID_RADIUS + 1)) as usize; // 289
36/// Per-LOD grid subdivision count (finest → coarsest).
37const LOD_SUBDIVS: [u32; 4] = [32, 16, 8, 4];
38/// Upper distance bound (world units, camera→chunk-center) for each LOD band.
39/// The last entry is the catch-all for everything farther out.
40const LOD_DISTANCES: [f32; 4] = [110.0, 220.0, 360.0, f32::INFINITY];
41/// How far skirt vertices drop below the displaced surface edge (units).
42/// Must exceed the maximum LOD-boundary height mismatch (≤ total wave
43/// amplitude ≈ 1.65 units) by a wide margin so no crack outruns the skirt.
44const SKIRT_DEPTH: f32 = 8.0;
45/// Default water level (world Y). Matches the former baked plane Y so existing
46/// consumers that never call [`WaterRenderer::set_water_level`] are unchanged.
47const DEFAULT_WATER_LEVEL: f32 = 2.0;
48
49// ── Weave-response deformation (W.2c) ────────────────────────────────────────
50
51/// Maximum simultaneously-active weave-deformation instances (W.2c #4 ratification).
52/// Built to 8 — trivially raisable later, but the shader loops to exactly this bound
53/// so unused slots cost nothing beyond a `kind == None` early-continue.
54pub const MAX_WEAVE_INSTANCES: usize = 8;
55
56/// Peak world-space displacement a single weave can apply, in units. Equals
57/// [`SKIRT_DEPTH`] so a `Part`/`Raise` at full intensity can never outrun the LOD
58/// skirt and re-expose a seam (W.2c.2 skirt constraint). `intensity ∈ [0,1]` scales
59/// against this; the shader bounds each instance's height contribution to ±this.
60pub const WEAVE_MAX_DEFORM: f32 = SKIRT_DEPTH;
61
62/// Which weave-response deformation a [`WeaveInstance`] applies. Encoded as `u32`
63/// in the GPU instance; `None` (0) marks an inactive slot the shader skips.
64#[derive(Debug, Clone, Copy, PartialEq, Eq)]
65#[repr(u32)]
66pub enum WeaveKind {
67    /// Inactive slot — identity (no deformation).
68    None = 0,
69    /// Displace the surface *down*, exposing a corridor/well (additive negative height).
70    Part = 1,
71    /// Lift the surface *up* — a dome/ridge (additive positive height).
72    Raise = 2,
73    /// Lock/flatten the surface (suppress waves) + frozen material look. A mask, not
74    /// a displacement.
75    Freeze = 3,
76}
77
78/// A runtime weave-deformation instance: a normalized analytical profile placed at a
79/// world location. **Location lives only here.** The profile (selected by
80/// [`kind`](Self::kind)) is a position-agnostic shape in *local* space; the shader
81/// maps `world → local = rotate(world_xz − position, −orientation) / radius`,
82/// evaluates the profile, and scales by [`intensity`](Self::intensity). A single
83/// authored `Part` profile therefore serves a part-*anywhere* verb — only the
84/// instance carries where / how-big / which-way / how-strong.
85///
86/// Representation-agnostic (W.2c #2): a future deformation-texture profile would add a
87/// `profile_id` atlas index here without changing position/radius/orientation/
88/// intensity or the shader-feeding interface.
89#[derive(Debug, Clone, Copy)]
90pub struct WeaveInstance {
91    /// Which deformation this is (also selects the analytical profile in W.2c.2).
92    pub kind: WeaveKind,
93    /// World-space XZ center where the effect is placed (runtime input).
94    pub position: Vec2,
95    /// World-space footprint radius in units (runtime input): local `r = 1` maps here.
96    pub radius: f32,
97    /// Yaw orientation in radians (runtime input) — orients directional profiles.
98    pub orientation: f32,
99    /// Normalized magnitude in `[0, 1]` (runtime input). For `Part`/`Raise` it scales
100    /// the height against [`WEAVE_MAX_DEFORM`]; for `Freeze` it is the mask strength.
101    pub intensity: f32,
102    /// Animation phase / age in seconds (runtime input) — drives shader-side ramps.
103    pub phase: f32,
104}
105
106impl WeaveInstance {
107    fn to_raw(self) -> WeaveInstanceRaw {
108        WeaveInstanceRaw {
109            position: self.position.into(),
110            radius: self.radius.max(0.0),
111            orientation: self.orientation,
112            // Bound magnitude to [0,1]; with WEAVE_MAX_DEFORM == SKIRT_DEPTH in the
113            // shader this keeps any single deformation within skirt tolerance.
114            intensity: self.intensity.clamp(0.0, 1.0),
115            phase: self.phase,
116            kind: self.kind as u32,
117            _pad: 0,
118        }
119    }
120}

122/// GPU-side weave instance (std140-compatible: 32 B = 2×vec4, 16-aligned so it packs

134    kind: u32,          // 24..28
135    _pad: u32,          // 28..32
136}

305impl WaterRenderer {
306    /// Create a new water renderer
307    pub fn new(
308        device: &wgpu::Device,
309        surface_format: wgpu::TextureFormat,
310        depth_format: wgpu::TextureFormat,
311    ) -> Self {
312        // Load shader
313        let shader_source = include_str!("shaders/water.wgsl");
314        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
315            label: Some("water_shader"),
316            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
317        });
318
319        // Uniform buffer
320        let uniforms = WaterUniforms::default();
321        let uniform_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
322            label: Some("water_uniforms"),
323            contents: bytemuck::bytes_of(&uniforms),
324            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
325        });
326
327        // Bind group layout — W.2b: 4 entries (uniform + scene-color + scene-depth
328        // + sampler) so the fragment shader can refract the scene behind the water
329        // and compute the depth-delta shoreline foam.
330        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
331            label: Some("water_bind_group_layout"),
332            entries: &[
333                wgpu::BindGroupLayoutEntry {
334                    binding: 0,
335                    visibility: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
336                    ty: wgpu::BindingType::Buffer {
337                        ty: wgpu::BufferBindingType::Uniform,
338                        has_dynamic_offset: false,
339                        min_binding_size: None,
340                    },
341                    count: None,
342                },
343                // @1 scene color (opaque HDR snapshot) — sampled for refraction.
344                wgpu::BindGroupLayoutEntry {
345                    binding: 1,
346                    visibility: wgpu::ShaderStages::FRAGMENT,
347                    ty: wgpu::BindingType::Texture {
348                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
349                        view_dimension: wgpu::TextureViewDimension::D2,
350                        multisampled: false,
351                    },
352                    count: None,
353                },
354                // @2 scene depth — read via textureLoad for the foam delta.
355                wgpu::BindGroupLayoutEntry {
356                    binding: 2,
357                    visibility: wgpu::ShaderStages::FRAGMENT,
358                    ty: wgpu::BindingType::Texture {
359                        sample_type: wgpu::TextureSampleType::Depth,
360                        view_dimension: wgpu::TextureViewDimension::D2,
361                        multisampled: false,
362                    },
363                    count: None,
364                },
365                // @3 sampler for the scene-color tap.
366                wgpu::BindGroupLayoutEntry {
367                    binding: 3,
368                    visibility: wgpu::ShaderStages::FRAGMENT,
369                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
370                    count: None,
371                },
372            ],
373        });
374
375        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
376            label: Some("water_scene_sampler"),
377            address_mode_u: wgpu::AddressMode::ClampToEdge,
378            address_mode_v: wgpu::AddressMode::ClampToEdge,
379            address_mode_w: wgpu::AddressMode::ClampToEdge,
380            mag_filter: wgpu::FilterMode::Linear,
381            min_filter: wgpu::FilterMode::Linear,
382            ..Default::default()
383        });
384
385        // 1×1 dummy scene textures so the initial bind group is valid before the
386        // renderer wires real scene-color/depth views via `prepare_scene`.
387        let dummy_scene_color = device.create_texture(&wgpu::TextureDescriptor {
388            label: Some("water_dummy_scene_color"),
389            size: wgpu::Extent3d { width: 1, height: 1, depth_or_array_layers: 1 },
390            mip_level_count: 1,
391            sample_count: 1,
392            dimension: wgpu::TextureDimension::D2,
393            format: wgpu::TextureFormat::Rgba16Float,
394            usage: wgpu::TextureUsages::TEXTURE_BINDING,
395            view_formats: &[],
396        });
397        let dummy_scene_color_view =
398            dummy_scene_color.create_view(&wgpu::TextureViewDescriptor::default());
399        let dummy_depth = device.create_texture(&wgpu::TextureDescriptor {
400            label: Some("water_dummy_depth"),
401            size: wgpu::Extent3d { width: 1, height: 1, depth_or_array_layers: 1 },
402            mip_level_count: 1,
403            sample_count: 1,
404            dimension: wgpu::TextureDimension::D2,
405            format: wgpu::TextureFormat::Depth32Float,
406            usage: wgpu::TextureUsages::TEXTURE_BINDING,
407            view_formats: &[],
408        });
409        let dummy_depth_view = dummy_depth.create_view(&wgpu::TextureViewDescriptor::default());
410
411        // Initial bind group references the dummy scene textures.
412        let bind_group = Self::build_bind_group(
413            device,
414            &bind_group_layout,
415            &uniform_buffer,
416            &dummy_scene_color_view,
417            &dummy_depth_view,
418            &sampler,
419        );
420
421        // Pipeline layout
422        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
423            label: Some("water_pipeline_layout"),
424            bind_group_layouts: &[&bind_group_layout],
425            push_constant_ranges: &[],
426        });
427
428        // Render pipeline with alpha blending.
429        // Two vertex buffers: per-vertex tile geometry + per-instance chunk offset.
430        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
431            label: Some("water_pipeline"),
432            layout: Some(&pipeline_layout),
433            vertex: wgpu::VertexState {
434                module: &shader,
435                entry_point: Some("vs_main"),
436                buffers: &[WaterVertex::desc(), ChunkInstance::desc()],
437                compilation_options: Default::default(),
438            },
439            fragment: Some(wgpu::FragmentState {
440                module: &shader,
441                entry_point: Some("fs_main"),
442                targets: &[Some(wgpu::ColorTargetState {
443                    format: surface_format,
444                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
445                    write_mask: wgpu::ColorWrites::ALL,
446                })],
447                compilation_options: Default::default(),
448            }),
449            primitive: wgpu::PrimitiveState {
450                topology: wgpu::PrimitiveTopology::TriangleList,
451                strip_index_format: None,
452                front_face: wgpu::FrontFace::Ccw,
453                // Top surface (normal +Y) is CCW-front when viewed from above and
454                // skirt walls are wound outward-front, so back-face culling keeps
455                // the correct faces. The submerged/underside two-sided case is a
456                // deferred effects-phase concern (W-series Gemini triage §E).
457                cull_mode: Some(wgpu::Face::Back),
458                unclipped_depth: false,
459                polygon_mode: wgpu::PolygonMode::Fill,
460                conservative: false,
461            },
462            depth_stencil: Some(wgpu::DepthStencilState {
463                format: depth_format,
464                depth_write_enabled: false, // Transparent, don't write depth
465                depth_compare: wgpu::CompareFunction::LessEqual, // Normal depth testing
466                stencil: wgpu::StencilState::default(),
467                bias: wgpu::DepthBiasState::default(),
468            }),
469            multisample: wgpu::MultisampleState::default(),
470            multiview: None,
471            cache: None,
472        });
473
474        // Pre-bake one tile mesh per LOD level, plus an instance buffer per LOD.
475        let mut lod_meshes = Vec::with_capacity(LOD_SUBDIVS.len());
476        let mut instance_buffers = Vec::with_capacity(LOD_SUBDIVS.len());
477        let mut instance_counts = Vec::with_capacity(LOD_SUBDIVS.len());
478        for (lod, &subdiv) in LOD_SUBDIVS.iter().enumerate() {
479            let (vertices, indices) = Self::generate_tile(CHUNK_SIZE, subdiv);
480            let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
481                label: Some(&format!("water_tile_vb_lod{lod}")),
482                contents: bytemuck::cast_slice(&vertices),
483                usage: wgpu::BufferUsages::VERTEX,
484            });
485            let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
486                label: Some(&format!("water_tile_ib_lod{lod}")),
487                contents: bytemuck::cast_slice(&indices),
488                usage: wgpu::BufferUsages::INDEX,
489            });
490            lod_meshes.push(LodMesh {
491                vertex_buffer,
492                index_buffer,
493                index_count: indices.len() as u32,
494            });
495            instance_buffers.push(device.create_buffer(&wgpu::BufferDescriptor {
496                label: Some(&format!("water_instances_lod{lod}")),
497                size: (MAX_CHUNKS * std::mem::size_of::<ChunkInstance>()) as wgpu::BufferAddress,
498                usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
499                mapped_at_creation: false,
500            }));
501            instance_counts.push(0u32);
502        }
503
504        Self {
505            pipeline,
506            bind_group_layout,
507            bind_group,
508            uniform_buffer,
509            sampler,
510            _dummy_scene_color: dummy_scene_color,
511            _dummy_depth: dummy_depth,
512            scene_gen: u64::MAX,
513            lod_meshes,
514            instance_buffers,
515            instance_counts,
516            uniforms,
517        }
518    }

520    /// Build the 4-entry water bind group (uniform + scene-color + scene-depth + sampler).
521    fn build_bind_group(
522        device: &wgpu::Device,
523        layout: &wgpu::BindGroupLayout,
524        uniform_buffer: &wgpu::Buffer,
525        scene_color_view: &wgpu::TextureView,

527        sampler: &wgpu::Sampler,
528    ) -> wgpu::BindGroup {
529        device.create_bind_group(&wgpu::BindGroupDescriptor {
530            label: Some("water_bind_group"),
531            layout,
532            entries: &[
533                wgpu::BindGroupEntry {
534                    binding: 0,
535                    resource: uniform_buffer.as_entire_binding(),
536                },
537                wgpu::BindGroupEntry {
538                    binding: 1,
539                    resource: wgpu::BindingResource::TextureView(scene_color_view),
540                },
541                wgpu::BindGroupEntry {

543                    resource: wgpu::BindingResource::TextureView(scene_depth_view),

546                    binding: 3,
547                    resource: wgpu::BindingResource::Sampler(sampler),
548                },
549            ],
550        })
551    }

553    /// Generate a subdivided water plane (surface grid only, local Y = 0,
554    /// centered at the origin). Reusable mesh primitive; [`Self::generate_tile`]
555    /// wraps it with a skirt.
556    fn generate_water_plane(size: f32, subdivisions: u32) -> (Vec<WaterVertex>, Vec<u32>) {
557        let mut vertices = Vec::new();
558        let mut indices = Vec::new();
559
560        let half_size = size / 2.0;
561        let step = size / subdivisions as f32;
562
563        // Generate vertices
564        for z in 0..=subdivisions {
565            for x in 0..=subdivisions {
566                let pos_x = -half_size + x as f32 * step;
567                let pos_z = -half_size + z as f32 * step;
568                let u = x as f32 / subdivisions as f32;
569                let v = z as f32 / subdivisions as f32;
570
571                vertices.push(WaterVertex {
572                    position: [pos_x, 0.0, pos_z], // local Y = 0; world Y from uniform
573                    uv: [u, v],
574                });
575            }
576        }
577
578        // Generate indices
579        for z in 0..subdivisions {
580            for x in 0..subdivisions {
581                let top_left = z * (subdivisions + 1) + x;
582                let top_right = top_left + 1;
583                let bottom_left = (z + 1) * (subdivisions + 1) + x;
584                let bottom_right = bottom_left + 1;
585
586                // First triangle
587                indices.push(top_left);
588                indices.push(bottom_left);
589                indices.push(top_right);
590
591                // Second triangle
592                indices.push(top_right);
593                indices.push(bottom_left);
594                indices.push(bottom_right);
595            }
596        }
597
598        (vertices, indices)
599    }
600
601    /// Append one outward-facing skirt wall segment between two adjacent top-edge
602    /// vertices. Bottom twins reuse the top XZ with sentinel local Y = -1.0. The
603    /// fixed winding `(a_top, b_top, a_bot)` + `(b_top, b_bot, a_bot)` yields an
604    /// outward horizontal normal when callers traverse each edge in the direction
605    /// documented in [`Self::generate_tile`].
606    fn push_wall(
607        vertices: &mut Vec<WaterVertex>,
608        indices: &mut Vec<u32>,
609        a_top: u32,
610        b_top: u32,
611    ) {
612        let a = vertices[a_top as usize];
613        let b = vertices[b_top as usize];
614        let a_bot = vertices.len() as u32;
615        vertices.push(WaterVertex {
616            position: [a.position[0], -1.0, a.position[2]],
617            uv: a.uv,
618        });
619        let b_bot = vertices.len() as u32;
620        vertices.push(WaterVertex {
621            position: [b.position[0], -1.0, b.position[2]],
622            uv: b.uv,
623        });
624        indices.extend_from_slice(&[a_top, b_top, a_bot, b_top, b_bot, a_bot]);
625    }
626
627    /// Generate a chunk tile = surface grid + a perimeter skirt that hides
628    /// LOD-boundary cracks. Edges are traversed so the fixed skirt winding faces
629    /// outward (away from the chunk center): TOP +x, BOTTOM -x, LEFT -z, RIGHT +z.
630    fn generate_tile(size: f32, subdivisions: u32) -> (Vec<WaterVertex>, Vec<u32>) {
631        let (mut vertices, mut indices) = Self::generate_water_plane(size, subdivisions);
632        let n = subdivisions;
633        let stride = n + 1;
634
635        // TOP edge (z = 0 row), traverse x: 0 → n
636        for x in 0..n {
637            Self::push_wall(&mut vertices, &mut indices, x, x + 1);
638        }
639        // BOTTOM edge (z = n row), traverse x: n → 0
640        let zbase = n * stride;
641        for x in (1..=n).rev() {
642            Self::push_wall(&mut vertices, &mut indices, zbase + x, zbase + x - 1);
643        }
644        // LEFT edge (x = 0 col), traverse z: n → 0
645        for z in (1..=n).rev() {
646            Self::push_wall(&mut vertices, &mut indices, z * stride, (z - 1) * stride);
647        }
648        // RIGHT edge (x = n col), traverse z: 0 → n
649        for z in 0..n {
650            Self::push_wall(&mut vertices, &mut indices, z * stride + n, (z + 1) * stride + n);
651        }
652
653        (vertices, indices)
654    }

656    /// Whether any LOD band has chunks to draw this frame. False before the first
657    /// [`Self::update`] (e.g. the editor dormant-water case where `update_water`
658    /// is never called), letting the renderer skip the scene-color snapshot + pass.
659    pub fn has_visible_chunks(&self) -> bool {
660        self.instance_counts.iter().any(|&c| c > 0)
661    }

663    /// Select an LOD band (index into `LOD_SUBDIVS`) for a camera→chunk distance.
664    fn lod_for_distance(dist: f32) -> usize {
665        LOD_DISTANCES
666            .iter()
667            .position(|&d| dist < d)
668            .unwrap_or(LOD_SUBDIVS.len() - 1)
669    }

671    /// Update water state for animation and recompute the per-LOD chunk set from
672    /// the camera position. Call once per frame before [`Self::render`].
673    pub fn update(&mut self, queue: &wgpu::Queue, view_proj: Mat4, camera_pos: Vec3, time: f32) {
674        self.uniforms.view_proj = view_proj.to_cols_array_2d();
675        // Inverse view-proj reconstructs the world position of the opaque scene
676        // behind the water from its sampled depth (W.2b depth-delta foam).
677        self.uniforms.inv_view_proj = view_proj.inverse().to_cols_array_2d();
678        self.uniforms.camera_pos = camera_pos.into();
679        self.uniforms.time = time;
680        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::bytes_of(&self.uniforms));
681
682        // Assign the active chunk block around the camera to LOD bands.
683        let cam_cx = (camera_pos.x / CHUNK_SIZE).floor() as i32;
684        let cam_cz = (camera_pos.z / CHUNK_SIZE).floor() as i32;
685        let mut per_lod: Vec<Vec<ChunkInstance>> =
686            (0..LOD_SUBDIVS.len()).map(|_| Vec::new()).collect();
687
688        for dz in -GRID_RADIUS..=GRID_RADIUS {
689            for dx in -GRID_RADIUS..=GRID_RADIUS {
690                let cx = cam_cx + dx;
691                let cz = cam_cz + dz;
692                let center_x = (cx as f32 + 0.5) * CHUNK_SIZE;
693                let center_z = (cz as f32 + 0.5) * CHUNK_SIZE;
694                let dx_w = center_x - camera_pos.x;
695                let dz_w = center_z - camera_pos.z;
696                let dist = (dx_w * dx_w + dz_w * dz_w).sqrt();
697                let lod = Self::lod_for_distance(dist);
698                per_lod[lod].push(ChunkInstance {
699                    offset: [center_x, center_z],
700                });
701            }
702        }
703
704        for (lod, chunks) in per_lod.iter().enumerate() {
705            self.instance_counts[lod] = chunks.len() as u32;
706            if !chunks.is_empty() {
707                queue.write_buffer(
708                    &self.instance_buffers[lod],
709                    0,
710                    bytemuck::cast_slice(chunks),
711                );
712            }
713        }
714    }

716    /// Wire the scene-color snapshot + scene-depth views into the water bind group
717    /// and upload the screen size, immediately before the water pass (W.2b). The
718    /// bind group is only rebuilt when `resource_gen` changes (resize); the uniform
719    /// upload is cheap and unconditional. Must be called each frame before
720    /// [`Self::render`] in the split water pass.
721    pub fn prepare_scene(
722        &mut self,
723        device: &wgpu::Device,
724        queue: &wgpu::Queue,
725        scene_color_view: &wgpu::TextureView,
726        scene_depth_view: &wgpu::TextureView,
727        screen_size: [f32; 2],
728        resource_gen: u64,
729    ) {
730        if self.scene_gen != resource_gen {
731            self.bind_group = Self::build_bind_group(
732                device,
733                &self.bind_group_layout,
734                &self.uniform_buffer,
735                scene_color_view,
736                scene_depth_view,
737                &self.sampler,
738            );
739            self.scene_gen = resource_gen;
740        }
741        // Patch only screen_size — the rest of the block was uploaded by `update()`
742        // this frame, so re-uploading all 240 B here would be redundant.
743        self.uniforms.screen_size = screen_size;
744        let off = std::mem::offset_of!(WaterUniforms, screen_size) as u64;
745        queue.write_buffer(&self.uniform_buffer, off, bytemuck::bytes_of(&self.uniforms.screen_size));
746    }

784    /// Set the active weave-deformation instances (W.2c). At most
785    /// [`MAX_WEAVE_INSTANCES`] are honored; any beyond the ceiling are dropped. An
786    /// **empty slice clears all weaves** — the surface is then identical to the
787    /// no-weave surface (the deformation is additive-zero / identity at zero
788    /// instances). Instances ride in [`WaterUniforms`] and upload on the next
789    /// [`Self::update`].
790    pub fn set_weave_instances(&mut self, instances: &[WeaveInstance]) {
791        let n = instances.len().min(MAX_WEAVE_INSTANCES);
792        for (slot, inst) in self.uniforms.weave_instances.iter_mut().zip(&instances[..n]) {
793            *slot = inst.to_raw();
794        }
795        // Clear unused slots so a previously-set instance can't linger active.
796        for slot in self.uniforms.weave_instances.iter_mut().skip(n) {
797            *slot = WeaveInstanceRaw::ZERO;
798        }
799        self.uniforms.weave_count = n as u32;
800    }
801
802    /// Remove all active weave deformations (the surface returns to identity).
803    pub fn clear_weave_instances(&mut self) {
804        self.uniforms.weave_instances = [WeaveInstanceRaw::ZERO; MAX_WEAVE_INSTANCES];
805        self.uniforms.weave_count = 0;
806    }
807
808    /// Number of active weave instances this frame (diagnostics / tests).
809    pub fn weave_count(&self) -> u32 {
810        self.uniforms.weave_count
811    }

813    /// Get current water colors (deep, shallow, foam).
814    pub fn water_colors(&self) -> (Vec3, Vec3, Vec3) {
815        (
816            Vec3::from(self.uniforms.water_color_deep),
817            Vec3::from(self.uniforms.water_color_shallow),
818            Vec3::from(self.uniforms.foam_color),
819        )
820    }

822    /// Render the water surface (one instanced draw per active LOD band).
823    pub fn render<'a>(&'a self, render_pass: &mut wgpu::RenderPass<'a>) {
824        render_pass.set_pipeline(&self.pipeline);
825        render_pass.set_bind_group(0, &self.bind_group, &[]);
826        for (lod, mesh) in self.lod_meshes.iter().enumerate() {
827            let count = self.instance_counts[lod];
828            if count == 0 {
829                continue;
830            }
831            render_pass.set_vertex_buffer(0, mesh.vertex_buffer.slice(..));
832            render_pass.set_vertex_buffer(1, self.instance_buffers[lod].slice(..));
833            render_pass.set_index_buffer(mesh.index_buffer.slice(..), wgpu::IndexFormat::Uint32);
834            render_pass.draw_indexed(0..mesh.index_count, 0, 0..count);
835        }
836    }

904    fn test_water_renderer_new_and_update() {
905        pollster::block_on(async {
906            let instance = wgpu::Instance::new(&wgpu::InstanceDescriptor::default());
907            let adapter = instance
908                .request_adapter(&wgpu::RequestAdapterOptions::default())
909                .await;
910            if let Ok(adapter) = adapter {
911                let (device, queue) = adapter
912                    .request_device(&wgpu::DeviceDescriptor::default())
913                    .await
914                    .unwrap();
915                let mut renderer = WaterRenderer::new(
916                    &device,
917                    wgpu::TextureFormat::Rgba8UnormSrgb,
918                    wgpu::TextureFormat::Depth32Float,
919                );
920
921                // One pre-baked mesh per LOD band.
922                assert_eq!(renderer.lod_meshes.len(), LOD_SUBDIVS.len());
923
924                // W-FU-2 regression guard: before the first update() the renderer
925                // is dormant (no chunks) and run_water_pass must skip it. This is
926                // exactly the state the editor was stuck in until update_water was
927                // wired — a fresh WaterRenderer with has_visible_chunks() == false.
928                assert!(
929                    !renderer.has_visible_chunks(),
930                    "fresh WaterRenderer must report no visible chunks (dormant)"
931                );
932
933                let view_proj = Mat4::IDENTITY;
934                let camera_pos = Vec3::new(1.0, 2.0, 3.0);
935                let time = 10.0;
936
937                renderer.update(&queue, view_proj, camera_pos, time);
938
939                // After one update() (what update_water drives every frame) the
940                // chunk set is populated, so has_visible_chunks() flips true and
941                // the post-opaque water pass runs instead of early-returning.
942                assert!(
943                    renderer.has_visible_chunks(),
944                    "after update() the water renderer must have visible chunks"
945                );
946
947                assert_eq!(renderer.uniforms.time, 10.0);
948                assert_eq!(renderer.uniforms.camera_pos, [1.0, 2.0, 3.0]);
949
950                // The full (2*GRID_RADIUS+1)^2 block is assigned to some LOD.
951                let total: u32 = renderer.instance_counts.iter().sum();
952                assert_eq!(total as usize, MAX_CHUNKS);
953
954                // Real water-level control.
955                renderer.set_water_level(5.0);
956                assert_eq!(renderer.water_level(), 5.0);
957                assert_eq!(renderer.uniforms.water_level, 5.0);
958
959                // W.2c weave instance list: zero by default (surface = identity).
960                assert_eq!(renderer.weave_count(), 0);
961                let part = WeaveInstance {
962                    kind: WeaveKind::Part,
963                    position: Vec2::new(10.0, -4.0),
964                    radius: 20.0,
965                    orientation: 0.5,
966                    intensity: 2.0, // out-of-range — must clamp to 1.0 in to_raw
967                    phase: 0.0,
968                };
969                renderer.set_weave_instances(&[
970                    part,
971                    WeaveInstance { kind: WeaveKind::Raise, ..part },
972                    WeaveInstance { kind: WeaveKind::Freeze, ..part },
973                ]);
974                assert_eq!(renderer.weave_count(), 3);
975                // intensity clamped to [0,1] (skirt-tolerance guard).
976                assert_eq!(renderer.uniforms.weave_instances[0].intensity, 1.0);
977                assert_eq!(renderer.uniforms.weave_instances[0].kind, WeaveKind::Part as u32);
978
979                // Ceiling: more than MAX_WEAVE_INSTANCES are dropped.
980                let many = vec![part; MAX_WEAVE_INSTANCES + 5];
981                renderer.set_weave_instances(&many);
982                assert_eq!(renderer.weave_count() as usize, MAX_WEAVE_INSTANCES);
983
984                // Clear → back to identity.
985                renderer.clear_weave_instances();
986                assert_eq!(renderer.weave_count(), 0);
987                assert_eq!(renderer.uniforms.weave_instances[0].kind, WeaveKind::None as u32);
988            }
989        });
