1//! Per-frame ring buffer for transient GPU allocations.
2//!
3//! Provides fast sub-allocations from a pre-allocated GPU buffer, recycling
4//! space automatically after a configurable number of frames-in-flight.
5//! Eliminates per-frame `create_buffer_init()` calls for uniform and storage
6//! data that changes every frame.
7//!
8//! # Usage
9//! ```ignore
10//! let mut ring = StagingRing::new(&device, 4 * 1024 * 1024, wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST);
11//! // Each frame:
12//! ring.begin_frame();
13//! let alloc = ring.allocate(std::mem::size_of::<MyUniform>() as u64, 256).unwrap();
14//! queue.write_buffer(ring.buffer(), alloc.offset, bytemuck::bytes_of(&my_uniform));
15//! // Use alloc.offset as a dynamic offset or bind sub-range
16//! ```
17
18use std::collections::VecDeque;
19
20/// Minimum alignment for uniform buffer offsets (wgpu spec: 256 bytes).
21pub const MIN_UNIFORM_ALIGN: u64 = 256;
22
23/// Default staging ring capacity (4 MiB).
24pub const DEFAULT_RING_SIZE: u64 = 4 * 1024 * 1024;
25
26/// Maximum number of frames-in-flight before space is recycled.
27const MAX_FRAMES_IN_FLIGHT: usize = 3;
28
29/// Result of a sub-allocation from the ring buffer.
30#[derive(Debug, Clone, Copy)]
31pub struct SubAllocation {
32    /// Byte offset into the ring buffer.
33    pub offset: u64,
34    /// Actual size allocated (may be larger than requested due to alignment).
35    pub size: u64,
36}
37
38/// Per-frame snapshot recording how much of the ring was consumed.
39#[derive(Debug, Clone, Copy)]
40struct FrameRecord {
41    /// Cursor position at the start of this frame.
42    #[allow(dead_code)]
43    start: u64,
44    /// Cursor position at the end of this frame.
45    end: u64,
46}
47
48/// A ring buffer backed by a single GPU buffer for fast per-frame sub-allocations.
49///
50/// Space is advanced linearly each frame and recycled once the corresponding
51/// GPU work has completed (after `MAX_FRAMES_IN_FLIGHT` frames).
52pub struct StagingRing {
53    buffer: wgpu::Buffer,
54    capacity: u64,
55    /// Current write cursor into the ring.
56    cursor: u64,
57    /// Oldest byte offset that is still in use by in-flight GPU work.
58    tail: u64,
59    /// Per-frame records for tracking when regions become available.
60    frame_records: VecDeque<FrameRecord>,
61    /// Cursor at the start of the current (not-yet-submitted) frame.
62    frame_start: u64,
63    /// Total bytes allocated this frame (for diagnostics).
64    frame_bytes: u64,
65    /// High-water mark across all frames (for diagnostics).
66    peak_bytes: u64,
67}
68
69impl StagingRing {
70    /// Create a new ring buffer with the given capacity and usage flags.
71    ///
72    /// `usage` must include `COPY_DST` (for `queue.write_buffer`). Typically
73    /// also includes `UNIFORM` or `STORAGE` depending on use case.
74    pub fn new(device: &wgpu::Device, capacity: u64, usage: wgpu::BufferUsages) -> Self {
75        let buffer = device.create_buffer(&wgpu::BufferDescriptor {
76            label: Some("staging_ring"),
77            size: capacity,
78            usage: usage | wgpu::BufferUsages::COPY_DST,
79            mapped_at_creation: false,
80        });
81
82        Self {
83            buffer,
84            capacity,
85            cursor: 0,
86            tail: 0,
87            frame_records: VecDeque::with_capacity(MAX_FRAMES_IN_FLIGHT + 1),
88            frame_start: 0,
89            frame_bytes: 0,
90            peak_bytes: 0,
91        }
92    }
93
94    /// Begin a new frame. Recycles old frames that have finished GPU execution.
95    ///
96    /// Call once at the start of each frame, before any `allocate()` calls.
97    pub fn begin_frame(&mut self) {
98        // Close out the previous frame (if any work was done).
99        if self.cursor != self.frame_start {
100            self.frame_records.push_back(FrameRecord {
101                start: self.frame_start,
102                end: self.cursor,
103            });
104        }
105
106        // Retire the oldest frame if we've exceeded frames-in-flight.
107        while self.frame_records.len() > MAX_FRAMES_IN_FLIGHT {
108            if let Some(old) = self.frame_records.pop_front() {
109                self.tail = old.end;
110            }
111        }
112
113        self.frame_start = self.cursor;
114        if self.frame_bytes > self.peak_bytes {
115            self.peak_bytes = self.frame_bytes;
116        }
117        self.frame_bytes = 0;
118    }
119
120    /// Allocate a region of the given `size` with the specified byte `alignment`.
121    ///
122    /// Returns `None` if the ring is full (would overlap in-flight data).
123    /// The returned [`SubAllocation`] contains the byte offset to use with
124    /// `queue.write_buffer()` and bind group descriptors.
125    pub fn allocate(&mut self, size: u64, alignment: u64) -> Option<SubAllocation> {
126        let align = alignment.max(1);
127        let aligned_cursor = (self.cursor + align - 1) & !(align - 1);
128        let end = aligned_cursor + size;
129
130        if self.cursor >= self.tail {
131            // Cursor is ahead of tail — check if we have room before wrapping.
132            if end <= self.capacity {
133                self.cursor = end;
134                self.frame_bytes += end - aligned_cursor;
135                return Some(SubAllocation {
136                    offset: aligned_cursor,
137                    size,
138                });
139            }
140            // Try wrapping to the beginning.
141            let wrapped_start = 0u64;
142            let wrapped_end = wrapped_start + size;
143            if wrapped_end <= self.tail || self.tail == 0 && self.frame_records.is_empty() {
144                self.cursor = wrapped_end;
145                self.frame_bytes += size;
146                return Some(SubAllocation {
147                    offset: wrapped_start,
148                    size,
149                });
150            }
151        } else {
152            // Cursor is behind tail (wrapped around) — check space between.
153            if end <= self.tail {
154                self.cursor = end;
155                self.frame_bytes += end - aligned_cursor;
156                return Some(SubAllocation {
157                    offset: aligned_cursor,
158                    size,
159                });
160            }
161        }
162
163        None // Ring is full.
164    }
165
166    /// Get a reference to the backing GPU buffer.
167    pub fn buffer(&self) -> &wgpu::Buffer {
168        &self.buffer
169    }
170
171    /// Current capacity in bytes.
172    pub fn capacity(&self) -> u64 {
173        self.capacity
174    }
175
176    /// Bytes allocated in the current (unsubmitted) frame.
177    pub fn frame_bytes(&self) -> u64 {
178        self.frame_bytes
179    }
180
181    /// Peak bytes allocated in any single frame (high-water mark).
182    pub fn peak_bytes(&self) -> u64 {
183        self.peak_bytes
184    }
185
186    /// Number of in-flight frame records.
187    pub fn frames_in_flight(&self) -> usize {
188        self.frame_records.len()
189    }
190
191    /// Reset the ring completely, discarding all in-flight tracking.
192    ///
193    /// Only safe to call when no GPU work referencing the ring is pending.
194    pub fn reset(&mut self) {
195        self.cursor = 0;
196        self.tail = 0;
197        self.frame_start = 0;
198        self.frame_bytes = 0;
199        self.frame_records.clear();
200    }
201}

207    /// Stub device for unit tests — we test logic only, not GPU buffer creation.
208    /// Full integration tests require a wgpu device and are in renderer_tests.rs.
209
210    #[test]
211    fn sub_allocation_alignment() {
212        // Verify alignment math without a real GPU device.
213        let align = 256u64;
214        let cursor = 100u64;
215        let aligned = (cursor + align - 1) & !(align - 1);
216        assert_eq!(aligned, 256);
217
218        let cursor2 = 256u64;
219        let aligned2 = (cursor2 + align - 1) & !(align - 1);
220        assert_eq!(aligned2, 256);
221
222        let cursor3 = 257u64;
223        let aligned3 = (cursor3 + align - 1) & !(align - 1);
224        assert_eq!(aligned3, 512);
225    }
226
227    #[test]
228    fn frame_record_size() {
229        assert_eq!(
230            std::mem::size_of::<FrameRecord>(),
231            16,
232            "FrameRecord should be 2 × u64"
233        );
234    }
