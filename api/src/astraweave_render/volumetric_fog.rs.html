1//! Froxel-based Volumetric Fog with Light Scattering.
2//!
3//! Implements a 3-pass pipeline:
4//!
5//! 1. **Density**: Fill a 3D froxel texture with fog density (uniform + height + noise)
6//! 2. **Scatter**: Compute in-scattered light per froxel from directional lights + shadows
7//! 3. **Integrate**: Front-to-back accumulation along each pixel's view ray
8//!
9//! The result is composited over the lit scene in a final apply pass.
10//! Froxels use exponential depth distribution for better near-plane resolution.
11

99/// Configuration for volumetric fog.
100#[derive(Debug, Clone)]
101pub struct VolumetricFogConfig {
102    pub enabled: bool,
103    /// Froxel grid dimensions (x, y, z). Default: 160×90×64.
104    pub froxel_dims: [u32; 3],
105    /// Uniform base fog density.
106    pub base_density: f32,
107    /// Height-based fog density at ground level.
108    pub height_fog_density: f32,
109    /// Exponential falloff rate for height fog.
110    pub height_fog_falloff: f32,
111    /// World-space Y offset for height fog (sea level).
112    pub height_fog_offset: f32,
113    /// Noise detail scale (larger = finer detail).
114    pub noise_scale: f32,
115    /// Noise contribution intensity.
116    pub noise_intensity: f32,
117    /// Noise animation speed.
118    pub noise_speed: f32,
119    /// Wind direction (normalized).
120    pub wind_dir: Vec3,
121    /// Henyey-Greenstein anisotropy (-1 to 1). Positive = forward scattering.
122    pub anisotropy: f32,
123    /// Overall scattering strength multiplier.
124    pub scatter_strength: f32,
125    /// Temporal blend factor for scatter history.
126    pub temporal_blend: f32,
127}

223impl VolumetricFogPass {
224    pub fn new(
225        device: &wgpu::Device,
226        width: u32,
227        height: u32,
228        config: VolumetricFogConfig,
229    ) -> Self {
230        let [fx, fy, fz] = config.froxel_dims;
231        let froxel_fmt = wgpu::TextureFormat::Rgba16Float;
232
233        // 3D textures
234        let make_3d = |label: &str| {
235            let tex = device.create_texture(&wgpu::TextureDescriptor {
236                label: Some(label),
237                size: wgpu::Extent3d {
238                    width: fx,
239                    height: fy,
240                    depth_or_array_layers: fz,
241                },
242                mip_level_count: 1,
243                sample_count: 1,
244                dimension: wgpu::TextureDimension::D3,
245                format: froxel_fmt,
246                usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
247                view_formats: &[],
248            });
249            let view = tex.create_view(&wgpu::TextureViewDescriptor::default());
250            (tex, view)
251        };
252        let (density_texture, density_view) = make_3d("vol_density");
253        let (scatter_texture, scatter_view) = make_3d("vol_scatter");
254        let (scatter_history, scatter_history_view) = make_3d("vol_scatter_history");
255
256        // 2D textures
257        let make_2d = |label: &str| {
258            let tex = device.create_texture(&wgpu::TextureDescriptor {
259                label: Some(label),
260                size: wgpu::Extent3d {
261                    width,
262                    height,
263                    depth_or_array_layers: 1,
264                },
265                mip_level_count: 1,
266                sample_count: 1,
267                dimension: wgpu::TextureDimension::D2,
268                format: froxel_fmt,
269                usage: wgpu::TextureUsages::STORAGE_BINDING | wgpu::TextureUsages::TEXTURE_BINDING,
270                view_formats: &[],
271            });
272            let view = tex.create_view(&wgpu::TextureViewDescriptor::default());
273            (tex, view)
274        };
275        let (integrated_texture, integrated_view) = make_2d("vol_integrated");
276        let (applied_texture, applied_view) = make_2d("vol_applied");
277
278        // Uniform buffers
279        let density_params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
280            label: Some("vol_density_params"),
281            contents: &[0u8; std::mem::size_of::<FogDensityParams>()],
282            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
283        });
284        let scatter_params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
285            label: Some("vol_scatter_params"),
286            contents: &[0u8; std::mem::size_of::<ScatterParams>()],
287            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
288        });
289        let integrate_params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
290            label: Some("vol_integrate_params"),
291            contents: &[0u8; std::mem::size_of::<IntegrateParams>()],
292            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
293        });
294        let apply_params_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
295            label: Some("vol_apply_params"),
296            contents: &[0u8; std::mem::size_of::<ApplyParams>()],
297            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
298        });
299        let cascade_buf = device.create_buffer(&wgpu::BufferDescriptor {
300            label: Some("vol_cascades"),
301            size: 4 * std::mem::size_of::<CascadeData>() as u64,
302            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
303            mapped_at_creation: false,
304        });
305
306        // --- Bind group layouts ---
307        let density_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
308            label: Some("vol_density_bgl"),
309            entries: &[bgl_uniform(0), bgl_storage_3d_rw(1, froxel_fmt)],
310        });
311
312        let scatter_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
313            label: Some("vol_scatter_bgl"),
314            entries: &[
315                bgl_uniform(0),                   // params
316                bgl_texture_3d(1),                // density
317                bgl_depth_texture(2),             // shadow (depth)
318                bgl_storage_ro(3),                // cascades
319                bgl_sampler(4),                   // linear sampler
320                bgl_sampler_comparison(5),        // shadow sampler
321                bgl_texture_3d(6),                // history
322                bgl_storage_3d_rw(7, froxel_fmt), // output
323            ],
324        });
325
326        let integrate_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
327            label: Some("vol_integrate_bgl"),
328            entries: &[
329                bgl_uniform(0),
330                bgl_texture_3d(1), // scatter
331                bgl_texture_2d(2), // depth
332                bgl_sampler(3),
333                bgl_storage_2d_rw(4, froxel_fmt),
334            ],
335        });
336
337        let apply_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
338            label: Some("vol_apply_bgl"),
339            entries: &[
340                bgl_uniform(0),
341                bgl_texture_2d(1), // scene color
342                bgl_texture_2d(2), // volumetric result
343                bgl_sampler(3),
344                bgl_storage_2d_rw(4, froxel_fmt),
345            ],
346        });
347
348        // --- Pipelines ---
349        let density_pipeline = create_compute_pipeline(
350            device,
351            &density_bgl,
352            "vol_density",
353            include_str!("../shaders/volumetrics/fog_density.wgsl"),
354            "fog_density_main",
355        );
356        let scatter_pipeline = create_compute_pipeline(
357            device,
358            &scatter_bgl,
359            "vol_scatter",
360            concat!(include_str!("../shaders/constants.wgsl"),
361            include_str!("../shaders/volumetrics/scatter.wgsl")),
362            "scatter_main",
363        );
364        let integrate_pipeline = create_compute_pipeline(
365            device,
366            &integrate_bgl,
367            "vol_integrate",
368            include_str!("../shaders/volumetrics/integrate.wgsl"),
369            "integrate_main",
370        );
371        let apply_pipeline = create_compute_pipeline(
372            device,
373            &apply_bgl,
374            "vol_apply",
375            include_str!("../shaders/volumetrics/apply.wgsl"),
376            "apply_main",
377        );
378
379        Self {
380            config,
381            density_pipeline,
382            scatter_pipeline,
383            integrate_pipeline,
384            apply_pipeline,
385            density_bgl,
386            scatter_bgl,
387            integrate_bgl,
388            apply_bgl,
389            density_params_buf,
390            scatter_params_buf,
391            integrate_params_buf,
392            apply_params_buf,
393            cascade_buf,
394            density_texture,
395            density_view,
396            scatter_texture,
397            scatter_view,
398            scatter_history,
399            scatter_history_view,
400            integrated_texture,
401            integrated_view,
402            applied_texture,
403            applied_view,
404            frame_index: 0,
405            time: 0.0,
406            width,
407            height,
408            linear_sampler: device.create_sampler(&wgpu::SamplerDescriptor {
409                label: Some("vol_linear_sampler"),
410                mag_filter: wgpu::FilterMode::Linear,
411                min_filter: wgpu::FilterMode::Linear,
412                address_mode_u: wgpu::AddressMode::ClampToEdge,
413                address_mode_v: wgpu::AddressMode::ClampToEdge,
414                address_mode_w: wgpu::AddressMode::ClampToEdge,
415                ..Default::default()
416            }),
417            shadow_sampler: device.create_sampler(&wgpu::SamplerDescriptor {
418                label: Some("vol_shadow_sampler"),
419                compare: Some(wgpu::CompareFunction::LessEqual),
420                mag_filter: wgpu::FilterMode::Linear,
421                min_filter: wgpu::FilterMode::Linear,
422                ..Default::default()
423            }),
424            cached_density_bg: crate::bind_group_cache::CachedBindGroup::new(),
425            cached_scatter_bg: crate::bind_group_cache::CachedBindGroup::new(),
426            cached_integrate_bg: crate::bind_group_cache::CachedBindGroup::new(),
427            cached_apply_bg: crate::bind_group_cache::CachedBindGroup::new(),
428        }
429    }

465    /// Update all uniform buffers for this frame.
466    #[allow(clippy::too_many_arguments)]
467    pub fn prepare_frame(
468        &mut self,
469        queue: &wgpu::Queue,
470        inv_view_proj: Mat4,
471        view_pos: Vec3,
472        sun_dir: Vec3,
473        sun_color: Vec3,
474        sun_intensity: f32,
475        ambient_color: Vec3,
476        ambient_intensity: f32,
477        near: f32,
478        far: f32,
479        dt: f32,
480    ) {
481        self.time += dt;
482
483        let [fx, fy, fz] = self.config.froxel_dims;
484
485        // Density params
486        let density_params = FogDensityParams {
487            inv_view_proj: inv_view_proj.to_cols_array_2d(),
488            view_pos: view_pos.to_array(),
489            near_plane: near,
490            far_plane: far,
491            froxel_dims: [fx, fy, fz],
492            base_density: self.config.base_density,
493            height_fog_density: self.config.height_fog_density,
494            height_fog_falloff: self.config.height_fog_falloff,
495            height_fog_offset: self.config.height_fog_offset,
496            noise_scale: self.config.noise_scale,
497            noise_intensity: self.config.noise_intensity,
498            noise_speed: self.config.noise_speed,
499            time: self.time,
500            wind_dir: self.config.wind_dir.to_array(),
501            _pad: 0.0,
502        };
503        queue.write_buffer(
504            &self.density_params_buf,
505            0,
506            bytemuck::bytes_of(&density_params),
507        );
508
509        // Scatter params
510        let scatter_params = ScatterParams {
511            inv_view_proj: inv_view_proj.to_cols_array_2d(),
512            view_pos: view_pos.to_array(),
513            near_plane: near,
514            far_plane: far,
515            froxel_dims: [fx, fy, fz],
516            anisotropy: self.config.anisotropy,
517            sun_dir: sun_dir.to_array(),
518            sun_intensity,
519            sun_color: sun_color.to_array(),
520            ambient_intensity,
521            ambient_color: ambient_color.to_array(),
522            temporal_blend: self.config.temporal_blend,
523            frame_index: self.frame_index,
524            _pad0: 0.0,
525            _pad1: 0.0,
526            _pad2: 0.0,
527        };
528        queue.write_buffer(
529            &self.scatter_params_buf,
530            0,
531            bytemuck::bytes_of(&scatter_params),
532        );
533
534        // Integrate params
535        let integrate_params = IntegrateParams {
536            inv_view_proj: inv_view_proj.to_cols_array_2d(),
537            resolution: [self.width as f32, self.height as f32],
538            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
539            froxel_dims: [fx, fy, fz],
540            near_plane: near,
541            far_plane: far,
542            scatter_strength: self.config.scatter_strength,
543            _pad0: 0.0,
544            _pad1: 0.0,
545        };
546        queue.write_buffer(
547            &self.integrate_params_buf,
548            0,
549            bytemuck::bytes_of(&integrate_params),
550        );
551
552        // Apply params
553        let apply_params = ApplyParams {
554            resolution: [self.width as f32, self.height as f32],
555            inv_resolution: [1.0 / self.width as f32, 1.0 / self.height as f32],
556        };
557        queue.write_buffer(&self.apply_params_buf, 0, bytemuck::bytes_of(&apply_params));
558
559        self.frame_index = self.frame_index.wrapping_add(1);
560    }

562    /// Execute the full volumetric fog pipeline.
563    ///
564    /// `resource_gen` is the renderer's current generation counter; bind groups
565    /// are rebuilt only when it changes (e.g., after a resize).
566    #[allow(clippy::too_many_arguments)]
567    pub fn execute(
568        &mut self,
569        device: &wgpu::Device,
570        encoder: &mut wgpu::CommandEncoder,
571        depth_view: &wgpu::TextureView,
572        shadow_view: &wgpu::TextureView,
573        scene_color_view: &wgpu::TextureView,
574        resource_gen: crate::bind_group_cache::Generation,
575    ) {
576        if !self.config.enabled {
577            return;
578        }
579
580        let [fx, fy, fz] = self.config.froxel_dims;
581
582        // --- Rebuild cached bind groups when resource generation changes ---
583
584        // Density (only internal resources — could be gen 0, but keep uniform for simplicity)
585        if !self.cached_density_bg.is_valid(resource_gen) {
586            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
587                label: Some("vol_density_bg"),
588                layout: &self.density_bgl,
589                entries: &[
590                    wgpu::BindGroupEntry {
591                        binding: 0,
592                        resource: self.density_params_buf.as_entire_binding(),
593                    },
594                    wgpu::BindGroupEntry {
595                        binding: 1,
596                        resource: wgpu::BindingResource::TextureView(&self.density_view),
597                    },
598                ],
599            });
600            self.cached_density_bg =
601                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
602        }
603
604        // Scatter (references external shadow_view)
605        if !self.cached_scatter_bg.is_valid(resource_gen) {
606            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
607                label: Some("vol_scatter_bg"),
608                layout: &self.scatter_bgl,
609                entries: &[
610                    wgpu::BindGroupEntry {
611                        binding: 0,
612                        resource: self.scatter_params_buf.as_entire_binding(),
613                    },
614                    wgpu::BindGroupEntry {
615                        binding: 1,
616                        resource: wgpu::BindingResource::TextureView(&self.density_view),
617                    },
618                    wgpu::BindGroupEntry {
619                        binding: 2,
620                        resource: wgpu::BindingResource::TextureView(shadow_view),
621                    },
622                    wgpu::BindGroupEntry {
623                        binding: 3,
624                        resource: self.cascade_buf.as_entire_binding(),
625                    },
626                    wgpu::BindGroupEntry {
627                        binding: 4,
628                        resource: wgpu::BindingResource::Sampler(&self.linear_sampler),
629                    },
630                    wgpu::BindGroupEntry {
631                        binding: 5,
632                        resource: wgpu::BindingResource::Sampler(&self.shadow_sampler),
633                    },
634                    wgpu::BindGroupEntry {
635                        binding: 6,
636                        resource: wgpu::BindingResource::TextureView(&self.scatter_history_view),
637                    },
638                    wgpu::BindGroupEntry {
639                        binding: 7,
640                        resource: wgpu::BindingResource::TextureView(&self.scatter_view),
641                    },
642                ],
643            });
644            self.cached_scatter_bg =
645                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
646        }
647
648        // Integrate (references external depth_view)
649        if !self.cached_integrate_bg.is_valid(resource_gen) {
650            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
651                label: Some("vol_integrate_bg"),
652                layout: &self.integrate_bgl,
653                entries: &[
654                    wgpu::BindGroupEntry {
655                        binding: 0,
656                        resource: self.integrate_params_buf.as_entire_binding(),
657                    },
658                    wgpu::BindGroupEntry {
659                        binding: 1,
660                        resource: wgpu::BindingResource::TextureView(&self.scatter_view),
661                    },
662                    wgpu::BindGroupEntry {
663                        binding: 2,
664                        resource: wgpu::BindingResource::TextureView(depth_view),
665                    },
666                    wgpu::BindGroupEntry {
667                        binding: 3,
668                        resource: wgpu::BindingResource::Sampler(&self.linear_sampler),
669                    },
670                    wgpu::BindGroupEntry {
671                        binding: 4,
672                        resource: wgpu::BindingResource::TextureView(&self.integrated_view),
673                    },
674                ],
675            });
676            self.cached_integrate_bg =
677                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
678        }
679
680        // Apply (references external scene_color_view)
681        if !self.cached_apply_bg.is_valid(resource_gen) {
682            let bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
683                label: Some("vol_apply_bg"),
684                layout: &self.apply_bgl,
685                entries: &[
686                    wgpu::BindGroupEntry {
687                        binding: 0,
688                        resource: self.apply_params_buf.as_entire_binding(),
689                    },
690                    wgpu::BindGroupEntry {
691                        binding: 1,
692                        resource: wgpu::BindingResource::TextureView(scene_color_view),
693                    },
694                    wgpu::BindGroupEntry {
695                        binding: 2,
696                        resource: wgpu::BindingResource::TextureView(&self.integrated_view),
697                    },
698                    wgpu::BindGroupEntry {
699                        binding: 3,
700                        resource: wgpu::BindingResource::Sampler(&self.linear_sampler),
701                    },
702                    wgpu::BindGroupEntry {
703                        binding: 4,
704                        resource: wgpu::BindingResource::TextureView(&self.applied_view),
705                    },
706                ],
707            });
708            self.cached_apply_bg =
709                crate::bind_group_cache::CachedBindGroup::with_bind_group(bg, resource_gen);
710        }
711
712        // --- Dispatch compute passes using cached bind groups ---
713
714        // Pass 1: Density
715        {
716            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
717                label: Some("vol_density"),
718                timestamp_writes: None,
719            });
720            pass.set_pipeline(&self.density_pipeline);
721            let bg = self
722                .cached_density_bg
723                .get_or_rebuild(resource_gen, || unreachable!());
724            pass.set_bind_group(0, bg, &[]);
725            pass.dispatch_workgroups(fx.div_ceil(4), fy.div_ceil(4), fz.div_ceil(4));
726        }
727
728        // Pass 2: Scatter
729        {
730            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
731                label: Some("vol_scatter"),
732                timestamp_writes: None,
733            });
734            pass.set_pipeline(&self.scatter_pipeline);
735            let bg = self
736                .cached_scatter_bg
737                .get_or_rebuild(resource_gen, || unreachable!());
738            pass.set_bind_group(0, bg, &[]);
739            pass.dispatch_workgroups(fx.div_ceil(4), fy.div_ceil(4), fz.div_ceil(4));
740        }
741
742        // Pass 3: Integrate
743        {
744            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
745                label: Some("vol_integrate"),
746                timestamp_writes: None,
747            });
748            pass.set_pipeline(&self.integrate_pipeline);
749            let bg = self
750                .cached_integrate_bg
751                .get_or_rebuild(resource_gen, || unreachable!());
752            pass.set_bind_group(0, bg, &[]);
753            pass.dispatch_workgroups(self.width.div_ceil(8), self.height.div_ceil(8), 1);
754        }
755
756        // Pass 4: Apply
757        {
758            let mut pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
759                label: Some("vol_apply"),
760                timestamp_writes: None,
761            });
762            pass.set_pipeline(&self.apply_pipeline);
763            let bg = self
764                .cached_apply_bg
765                .get_or_rebuild(resource_gen, || unreachable!());
766            pass.set_bind_group(0, bg, &[]);
767            pass.dispatch_workgroups(self.width.div_ceil(8), self.height.div_ceil(8), 1);
768        }
769    }

771    /// Copy scatter output to history for next frame's temporal blend.
772    pub fn copy_scatter_to_history(&self, encoder: &mut wgpu::CommandEncoder) {
773        let [fx, fy, fz] = self.config.froxel_dims;
774        encoder.copy_texture_to_texture(
775            wgpu::TexelCopyTextureInfo {
776                texture: &self.scatter_texture,
777                mip_level: 0,
778                origin: wgpu::Origin3d::ZERO,
779                aspect: wgpu::TextureAspect::All,
780            },
781            wgpu::TexelCopyTextureInfo {
782                texture: &self.scatter_history,
783                mip_level: 0,
784                origin: wgpu::Origin3d::ZERO,
785                aspect: wgpu::TextureAspect::All,
786            },
787            wgpu::Extent3d {
788                width: fx,
789                height: fy,
790                depth_or_array_layers: fz,
791            },
792        );
793    }

795    /// Resize screen-space textures (does not resize froxel grid).
796    pub fn resize(&mut self, device: &wgpu::Device, width: u32, height: u32) {
797        if self.width == width && self.height == height {
798            return;
799        }
800        let config = self.config.clone();
801        *self = Self::new(device, width, height, config);
802    }

809fn create_compute_pipeline(
810    device: &wgpu::Device,
811    bgl: &wgpu::BindGroupLayout,
812    label: &str,
813    source: &str,
814    entry_point: &str,
815) -> wgpu::ComputePipeline {
816    let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
817        label: Some(label),
818        source: wgpu::ShaderSource::Wgsl(source.into()),
819    });
820    let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
821        label: Some(&format!("{label}_pl")),
822        bind_group_layouts: &[bgl],
823        push_constant_ranges: &[],
824    });
825    device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
826        label: Some(&format!("{label}_pipeline")),
827        layout: Some(&pl),
828        module: &shader,
829        entry_point: Some(entry_point),
830        compilation_options: Default::default(),
831        cache: None,
832    })
833}
