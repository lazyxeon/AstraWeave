1//! Shader permutation system — compile-time feature elimination for WGSL.
2//!
3//! Instead of runtime branching (which causes warp divergence when adjacent
4//! fragments use different material features), this module generates WGSL
5//! `const bool` preambles that allow the GPU compiler to statically eliminate
6//! dead branches at pipeline creation time.
7//!
8//! # Usage
9//!
10//! ```rust,ignore
11//! use astraweave_render::shader_permutation::ShaderPermutation;
12//!
13//! // Base PBR only (no optional lobes)
14//! let base = ShaderPermutation::NONE;
15//! let shader_src = base.generate_disney_brdf();
16//!
17//! // Car paint: clearcoat enabled
18//! let car = ShaderPermutation::CLEARCOAT;
19//! let shader_src = car.generate_disney_brdf();
20//!
21//! // Full material: all optional lobes
22//! let full = ShaderPermutation::ALL;
23//! let shader_src = full.generate_disney_brdf();
24//! ```
25
26/// Shader permutation flags — each bit enables a Disney BRDF lobe.
27///
28/// When a flag is NOT set, the corresponding WGSL `const` is `false` and the
29/// GPU compiler eliminates the branch entirely (no warp divergence).
30#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
31pub struct ShaderPermutation(u32);
32
33impl ShaderPermutation {
34    /// No optional lobes — base diffuse + specular only.
35    pub const NONE: Self = Self(0);
36
37    /// Enable clearcoat lobe (GGX with IOR 1.5).
38    pub const CLEARCOAT: Self = Self(1 << 0);
39
40    /// Enable anisotropic specular (rotated GGX).
41    pub const ANISOTROPY: Self = Self(1 << 1);
42
43    /// Enable subsurface scattering approximation (wrap lighting).
44    pub const SUBSURFACE: Self = Self(1 << 2);
45
46    /// Enable sheen lobe (Charlie distribution for fabric/velvet).
47    pub const SHEEN: Self = Self(1 << 3);
48
49    /// Enable transmission/refraction (Beer's law attenuation).
50    pub const TRANSMISSION: Self = Self(1 << 4);
51
52    /// All optional lobes enabled. Equivalent to the current runtime-branching path.
53    pub const ALL: Self = Self(0x1F);
54
55    /// Create from raw flags.
56    pub const fn from_bits(bits: u32) -> Self {
57        Self(bits & 0x1F)
58    }
59
60    /// Get raw flag bits.
61    pub const fn bits(self) -> u32 {
62        self.0
63    }
64
65    /// Check if a specific flag is set.
66    pub const fn contains(self, other: Self) -> bool {
67        (self.0 & other.0) == other.0
68    }
69
70    /// Combine two permutations.
71    pub const fn union(self, other: Self) -> Self {
72        Self(self.0 | other.0)
73    }
74
75    /// Generate the WGSL const-defines preamble for this permutation.
76    ///
77    /// Produces lines like:
78    /// ```wgsl
79    /// const ENABLE_CLEARCOAT: bool = true;
80    /// const ENABLE_ANISOTROPY: bool = false;
81    /// ```
82    ///
83    /// The GPU compiler statically eliminates branches gated on `false` consts,
84    /// avoiding warp divergence entirely.
85    pub fn generate_defines(&self) -> String {
86        format!(
87            "// Auto-generated shader permutation defines (flags=0x{:02X})\n\
88             const ENABLE_CLEARCOAT: bool = {};\n\
89             const ENABLE_ANISOTROPY: bool = {};\n\
90             const ENABLE_SUBSURFACE: bool = {};\n\
91             const ENABLE_SHEEN: bool = {};\n\
92             const ENABLE_TRANSMISSION: bool = {};\n\n",
93            self.0,
94            self.contains(Self::CLEARCOAT),
95            self.contains(Self::ANISOTROPY),
96            self.contains(Self::SUBSURFACE),
97            self.contains(Self::SHEEN),
98            self.contains(Self::TRANSMISSION),
99        )
100    }
101
102    /// Generate a complete Disney BRDF shader source with permutation defines.
103    ///
104    /// The result includes `constants.wgsl` + permutation defines + `disney_brdf.wgsl`.
105    /// Unused lobes are compiled out by the GPU compiler.
106    pub fn generate_disney_brdf(&self) -> String {
107        let constants = include_str!("../shaders/constants.wgsl");
108        let defines = self.generate_defines();
109        let brdf = include_str!("../shaders/pbr/disney_brdf.wgsl");
110        format!("{constants}\n{defines}{brdf}")
111    }
112
113    /// Return a human-readable label for this permutation (for debug/logging).
114    pub fn label(&self) -> String {
115        if self.0 == 0 {
116            return "base".to_string();
117        }
118        let mut parts = Vec::new();
119        if self.contains(Self::CLEARCOAT) {
120            parts.push("clearcoat");
121        }
122        if self.contains(Self::ANISOTROPY) {
123            parts.push("aniso");
124        }
125        if self.contains(Self::SUBSURFACE) {
126            parts.push("sss");
127        }
128        if self.contains(Self::SHEEN) {
129            parts.push("sheen");
130        }
131        if self.contains(Self::TRANSMISSION) {
132            parts.push("transmission");
133        }
134        parts.join("+")
135    }
136}

138/// Common permutation presets for typical material classes.
139impl ShaderPermutation {
140    /// Opaque dielectric — base diffuse + specular only. Most common.
141    pub const OPAQUE_DIELECTRIC: Self = Self::NONE;
142
143    /// Metal — base diffuse + specular only (metallic flag is per-material, not per-permutation).
144    pub const METAL: Self = Self::NONE;
145
146    /// Car paint — clearcoat (fixed IOR 1.5).
147    pub const CAR_PAINT: Self = Self::CLEARCOAT;
148
149    /// Fabric — sheen lobe (Charlie distribution).
150    pub const FABRIC: Self = Self::SHEEN;
151
152    /// Skin — subsurface approximation.
153    pub const SKIN: Self = Self(Self::SUBSURFACE.0);
154
155    /// Glass — transmission + optional clearcoat.
156    pub const GLASS: Self = Self(Self::TRANSMISSION.0 | Self::CLEARCOAT.0);
157
158    /// Brushed metal — anisotropic specular.
159    pub const BRUSHED_METAL: Self = Self::ANISOTROPY;
160}
