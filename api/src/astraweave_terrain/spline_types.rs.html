31/// Phase 1.X-F.1.A: Peaks-and-Valleys fold helper.
32///
33/// Implements the canonical Minecraft 1.18+ formula
34/// `pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`. Folded weirdness
35/// produces a characteristic 5-band character (Valleys / Low / Mid /
36/// High / Peaks) over a `[-1, 1]` weirdness input.
37///
38/// The PV value is the third climate-input axis the per-archetype
39/// `BootstrapSplineSet` reads (alongside `continentalness` and
40/// `erosion`); high PV drives mountain peaks, low PV drives valleys.
41///
42/// # Range
43///
44/// For `|weirdness| ∈ [0, 1]`, the formula produces `pv ∈ [-1, 1]`.
45/// Specifically:
46/// - `weirdness = 0` → `pv = -1` (valley extremum)
47/// - `weirdness = ±2/3` → `pv = +1` (peak extremum)
48/// - `weirdness = ±1/3` or `±1` → `pv = 0` (mid)
49///
50/// # Determinism
51///
52/// Pure function; same input always produces same output. No state.
53pub struct PvFold;
54
55impl PvFold {
56    /// Compute PV from weirdness via the canonical Minecraft 1.18+
57    /// formula: `pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`.
58    ///
59    /// Even-symmetric in weirdness (depends only on `|weirdness|`).
60    ///
61    /// # Examples
62    ///
63    /// ```
64    /// use astraweave_terrain::spline_types::PvFold;
65    ///
66    /// // Valley extremum at weirdness=0
67    /// let pv = PvFold::from_weirdness(0.0);
68    /// assert!((pv - (-1.0)).abs() < 1e-6);
69    ///
70    /// // Peak extremum at weirdness=2/3
71    /// let pv = PvFold::from_weirdness(2.0 / 3.0);
72    /// assert!((pv - 1.0).abs() < 1e-6);
73    /// ```
74    #[inline]
75    pub fn from_weirdness(weirdness: f32) -> f32 {
76        1.0 - ((3.0 * weirdness.abs()) - 2.0).abs()
77    }
78}

80/// Phase 1.X-F.1.A: placeholder for F.2.A's piecewise-linear 1D spline.
81///
82/// F.2.A populates with the full implementation: `evaluate(input: f32) -> f32`
83/// (binary-search + linear-interpolate between adjacent control points;
84/// clamp at endpoints), `identity()` constructor (always returns 1.0),
85/// `from_control_points(Vec<(f32, f32)>) -> Result<Self>` (validates
86/// sortedness on input). F.1.A ships only the type definition.
87///
88/// Control points are `(input, output_multiplier)` pairs sorted by input.
89/// Output multipliers are dimensionless (typically `[0.0, 4.0]`); the
90/// owning `ParamSpline` (F.2.A) multiplies these against a base value.
91#[allow(dead_code)]
92#[derive(Debug, Clone, Default)]
93pub struct Spline1D {
94    /// Sorted `(input, output_multiplier)` control points. Empty
95    /// `control_points` is valid; F.2.A's `evaluate` returns 1.0 in
96    /// that case (identity behavior).
97    pub control_points: Vec<(f32, f32)>,
98}

100/// Phase 1.X-F.1.A: placeholder enum identifying the bootstrap noise
101/// parameters that become spline outputs in F.2.A's `BootstrapSplineSet`.
102///
103/// Per campaign doc §2.3:
104/// - [`Self::MountainsAmplitude`] — `NoiseConfig.mountains.amplitude`
105///   (default 480; controls mountain layer height).
106/// - [`Self::MountainsScale`] — `NoiseConfig.mountains.scale` (default
107///   0.002; controls mountain layer feature size).
108/// - [`Self::ContinentalScale`] — `NoiseConfig.continental_scale`
109///   (default 0.0003; controls continental modulation wavelength).
110/// - [`Self::BaseElevationAmplitude`] — `NoiseConfig.base_elevation.amplitude`
111///   (default 150; controls base elevation layer height).
112///
113/// F.2.A consumes this enum; F.1.A ships type only.
114#[allow(dead_code)]
115#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
116pub enum BootstrapParam {
117    MountainsAmplitude,
118    MountainsScale,
119    ContinentalScale,
120    BaseElevationAmplitude,
121}

127    /// Phase 1.X-F.1.A: hand-verified canonical Minecraft 1.18+ PV
128    /// values per F.1 prompt §1.2. Each tuple is
129    /// `(weirdness, expected_pv, character_label)`.
130    ///
131    /// Hand-derivation (also in F.1 prompt §1.2 table):
132    /// - w=-1.0 → |w|=1.0 → ×3=3.0 → -2=1.0 → abs=1.0 → 1-1=0.0
133    /// - w=-2/3 → |w|=2/3 → ×3=2.0 → -2=0.0 → abs=0.0 → 1-0=1.0
134    /// - w=-1/3 → |w|=1/3 → ×3=1.0 → -2=-1.0 → abs=1.0 → 1-1=0.0
135    /// - w= 0.0 → |w|=0.0 → ×3=0.0 → -2=-2.0 → abs=2.0 → 1-2=-1.0
136    /// - w=+1/3 → same as -1/3 by symmetry → 0.0
137    /// - w=+2/3 → same as -2/3 → 1.0
138    /// - w=+1.0 → same as -1.0 → 0.0
139    #[test]
140    fn pv_fold_seven_canonical_points() {
141        let cases: &[(f32, f32, &str)] = &[
142            (-1.0, 0.0, "Mid (low ridge, neg extreme)"),
143            (-2.0 / 3.0, 1.0, "Peak (neg)"),
144            (-1.0 / 3.0, 0.0, "Mid (neg)"),
145            (0.0, -1.0, "Valley (zero)"),
146            (1.0 / 3.0, 0.0, "Mid (pos)"),
147            (2.0 / 3.0, 1.0, "Peak (pos)"),
148            (1.0, 0.0, "Mid (low ridge, pos extreme)"),
149        ];
150
151        for &(weirdness, expected_pv, label) in cases {
152            let actual = PvFold::from_weirdness(weirdness);
153            assert!(
154                (actual - expected_pv).abs() < 1e-6,
155                "{}: PvFold::from_weirdness({}) = {} (expected {} ± 1e-6)",
156                label,
157                weirdness,
158                actual,
159                expected_pv
160            );
161        }
162    }

164    /// PvFold is even-symmetric in weirdness (depends only on |weirdness|).
165    /// Verifies symmetry property at 5 sample magnitudes across `[0, 1]`.
166    #[test]
167    fn pv_fold_symmetry() {
168        for &mag in &[0.1f32, 0.25, 0.4, 0.55, 0.85] {
169            let pv_pos = PvFold::from_weirdness(mag);
170            let pv_neg = PvFold::from_weirdness(-mag);
171            assert!(
172                (pv_pos - pv_neg).abs() < 1e-7,
173                "symmetry violated at mag={}: pv(+)={}, pv(-)={}",
174                mag,
175                pv_pos,
176                pv_neg
177            );
178        }
179    }
180
181    /// PV output stays within `[-1, 1]` for every weirdness in `[-1, 1]`.
182    /// Sweeps 100 values to catch any out-of-range artifact.
183    #[test]
184    fn pv_fold_range_bounded() {
185        for i in 0..=100 {
186            let weirdness = -1.0 + (i as f32) * 0.02; // [-1, 1] in 0.02 steps
187            let pv = PvFold::from_weirdness(weirdness);
188            assert!(
189                (-1.0..=1.0).contains(&pv),
190                "PV out of range at weirdness={}: pv={}",
191                weirdness,
192                pv
193            );
194        }
195    }

199        let pv = PvFold::from_weirdness(0.0);
200        assert!((pv - (-1.0)).abs() < 1e-7, "pv(0) = {} (expected -1)", pv);
201    }

203    #[test]

205        let pv = PvFold::from_weirdness(2.0 / 3.0);

209    #[test]
210    fn pv_fold_at_extremes_is_zero() {
211        let pv_neg = PvFold::from_weirdness(-1.0);
212        let pv_pos = PvFold::from_weirdness(1.0);
213        assert!((pv_neg - 0.0).abs() < 1e-7, "pv(-1) = {} (expected 0)", pv_neg);
214        assert!((pv_pos - 0.0).abs() < 1e-7, "pv(+1) = {} (expected 0)", pv_pos);
215    }

241        assert_eq!(variant_id(BootstrapParam::ContinentalScale), 2);
242        assert_eq!(variant_id(BootstrapParam::BaseElevationAmplitude), 3);
