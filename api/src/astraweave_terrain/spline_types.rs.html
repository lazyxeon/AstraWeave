31/// Phase 1.X-F.1.A: Peaks-and-Valleys fold helper.
32///
33/// Implements the canonical Minecraft 1.18+ formula
34/// `pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`. Folded weirdness
35/// produces a characteristic 5-band character (Valleys / Low / Mid /
36/// High / Peaks) over a `[-1, 1]` weirdness input.
37///
38/// The PV value is the third climate-input axis the per-archetype
39/// `BootstrapSplineSet` reads (alongside `continentalness` and
40/// `erosion`); high PV drives mountain peaks, low PV drives valleys.
41///
42/// # Range
43///
44/// For `|weirdness| ∈ [0, 1]`, the formula produces `pv ∈ [-1, 1]`.
45/// Specifically:
46/// - `weirdness = 0` → `pv = -1` (valley extremum)
47/// - `weirdness = ±2/3` → `pv = +1` (peak extremum)
48/// - `weirdness = ±1/3` or `±1` → `pv = 0` (mid)
49///
50/// # Determinism
51///
52/// Pure function; same input always produces same output. No state.
53pub struct PvFold;
54
55impl PvFold {
56    /// Compute PV from weirdness via the canonical Minecraft 1.18+
57    /// formula: `pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`.
58    ///
59    /// Even-symmetric in weirdness (depends only on `|weirdness|`).
60    ///
61    /// # Examples
62    ///
63    /// ```
64    /// use astraweave_terrain::spline_types::PvFold;
65    ///
66    /// // Valley extremum at weirdness=0
67    /// let pv = PvFold::from_weirdness(0.0);
68    /// assert!((pv - (-1.0)).abs() < 1e-6);
69    ///
70    /// // Peak extremum at weirdness=2/3
71    /// let pv = PvFold::from_weirdness(2.0 / 3.0);
72    /// assert!((pv - 1.0).abs() < 1e-6);
73    /// ```
74    #[inline]
75    pub fn from_weirdness(weirdness: f32) -> f32 {
76        1.0 - ((3.0 * weirdness.abs()) - 2.0).abs()
77    }
78}

80/// Phase 1.X-F.2.A: piecewise-linear 1D spline.
81///
82/// Built from sorted `(input, output)` control points with linear
83/// interpolation between adjacent points and clamp-at-endpoint
84/// behavior for out-of-domain inputs. Used by [`ParamSpline`] to map
85/// a single climate-input axis (continentalness, erosion, or PV) to
86/// a per-archetype bootstrap noise parameter (mountains_amplitude,
87/// mountains_scale, continental_scale, base_elevation_amplitude).
88///
89/// # Construction
90///
91/// - [`Spline1D::from_control_points`] validates sortedness, NaN, and
92///   infinite values. Returns `Result<Self, SplineError>`.
93/// - [`Spline1D::identity`] returns a single-control-point spline that
94///   evaluates to `1.0` regardless of input — used as a "no-op"
95///   multiplier in separable splines.
96///
97/// # Evaluation semantics
98///
99/// - Empty `control_points`: returns `1.0` (defensive identity behavior;
100///   construction via `from_control_points` rejects empty input, but
101///   `Default::default()` produces an empty spline).
102/// - Single control point: returns the constant `y_0` regardless of
103///   input.
104/// - Multiple control points: binary-search the bracketing pair
105///   `(x_i, y_i)` and `(x_{i+1}, y_{i+1})`, linearly interpolate. For
106///   inputs below the first control point's x, return `y_0`; above the
107///   last, return `y_n`.
108/// - Duplicate `input` values produce a sharp step; the search returns
109///   the first match deterministically.
110///
111/// # Determinism
112///
113/// Pure function; same `(control_points, input)` always produces same
114/// output. f32 arithmetic determinism inherited from std.
115#[derive(Debug, Clone, Default, PartialEq)]
116pub struct Spline1D {
117    /// Sorted (ascending by input) `(input, output)` control points.
118    /// Empty is valid (returns `1.0` from `evaluate`); single-point is
119    /// valid (returns the constant `y_0`); multiple points evaluate
120    /// piecewise-linearly with clamp-at-endpoint behavior.
121    pub control_points: Vec<(f32, f32)>,
122}
123
124/// Phase 1.X-F.2.A: errors from [`Spline1D::from_control_points`]
125/// validation. Reported with the offending control-point index where
126/// applicable.
127#[derive(Debug, Clone, Copy, PartialEq, Eq)]
128pub enum SplineError {
129    /// Empty `control_points` Vec.
130    Empty,
131    /// `control_points` not sorted ascending by input. The first
132    /// out-of-order pair's index is reported.
133    NotSorted { at_index: usize },
134    /// NaN value in input or output at the reported index.
135    NaN { at_index: usize },
136    /// Infinite value in input or output at the reported index.
137    Infinite { at_index: usize },
138}
139
140impl Spline1D {
141    /// Construct a spline from sorted control points. Validates:
142    ///
143    /// - Vec is non-empty.
144    /// - Points are sorted ascending by input. Equal inputs are
145    ///   permitted (encodes a sharp step); strict descent is rejected.
146    /// - No NaN or infinite values in input or output.
147    ///
148    /// Returns `Err(SplineError)` describing the first violation.
149    pub fn from_control_points(points: Vec<(f32, f32)>) -> Result<Self, SplineError> {
150        if points.is_empty() {
151            return Err(SplineError::Empty);
152        }
153        for (i, (x, y)) in points.iter().enumerate() {
154            if x.is_nan() || y.is_nan() {
155                return Err(SplineError::NaN { at_index: i });
156            }
157            if x.is_infinite() || y.is_infinite() {
158                return Err(SplineError::Infinite { at_index: i });
159            }
160            if i > 0 {
161                let prev_x = points[i - 1].0;
162                if *x < prev_x {
163                    return Err(SplineError::NotSorted { at_index: i });
164                }
165            }
166        }
167        Ok(Self { control_points: points })
168    }
169
170    /// Identity spline: single control point at `(0.0, 1.0)`. Evaluates
171    /// to `1.0` for any input. Used as a no-op multiplier in separable
172    /// splines where one or more axes have no effect on the output.
173    ///
174    /// Cannot be a true `const fn` because `Vec::new()` / `vec!` are
175    /// not const; constructed at call time. Cheap: single allocation
176    /// of one tuple.
177    pub fn identity() -> Self {
178        Self { control_points: vec![(0.0, 1.0)] }
179    }
180
181    /// Evaluate the spline at `input` via piecewise-linear interpolation.
182    /// Out-of-domain inputs clamp to the corresponding endpoint output.
183    ///
184    /// See struct doc for full evaluation semantics.
185    pub fn evaluate(&self, input: f32) -> f32 {
186        let n = self.control_points.len();
187        if n == 0 {
188            // Defensive identity — construction normally prevents this,
189            // but Default::default() produces an empty Spline1D.
190            return 1.0;
191        }
192        if n == 1 {
193            return self.control_points[0].1;
194        }
195
196        // Clamp below the first control point's input.
197        let (first_x, first_y) = self.control_points[0];
198        if input <= first_x {
199            return first_y;
200        }
201        // Clamp above the last control point's input.
202        let (last_x, last_y) = self.control_points[n - 1];
203        if input >= last_x {
204            return last_y;
205        }
206
207        // Binary search for the bracketing pair. partition_point returns
208        // the first index where the predicate fails; we want the first
209        // index where x > input, then bracket = (index-1, index).
210        let upper = self
211            .control_points
212            .partition_point(|&(x, _)| x <= input);
213        // upper is in [1, n-1] because we handled the endpoint clamps above.
214        let (x0, y0) = self.control_points[upper - 1];
215        let (x1, y1) = self.control_points[upper];
216
217        if x1 == x0 {
218            // Duplicate input (sharp step). Return the right-side value.
219            return y1;
220        }
221        let t = (input - x0) / (x1 - x0);
222        y0 + t * (y1 - y0)
223    }
224}

226/// Phase 1.X-F.2.A: enum identifying which climate field axis a
227/// [`ParamSpline`] reads as its evaluation input.
228///
229/// Per campaign doc §2.3, each archetype's `BootstrapSplineSet` maps
230/// climate parameters `(continentalness, erosion, PV)` to bootstrap
231/// shape parameters `(mountains_amplitude, mountains_scale,
232/// continental_scale, base_elevation_amplitude)` via separable splines.
233/// This enum exposes the per-spline input-dimension choice so F.7
234/// per-archetype tuning can revise it (e.g., Boreal `mountains_amplitude`
235/// reads `pv`; Mediterranean reads `continentalness`).
236///
237/// PV is computed at sample time via [`PvFold::from_weirdness`] /
238/// [`crate::climate::ClimateSample::pv`]; reading `Pv` here invokes the
239/// accessor.
240#[derive(Debug, Clone, Copy, PartialEq, Eq)]
241pub enum ClimateInputDim {
242    Continentalness,
243    Erosion,
244    Pv,
245}
246
247impl ClimateInputDim {
248    /// Read the corresponding f32 from a [`crate::climate::ClimateSample`].
249    /// `Pv` calls the derived accessor (which routes through `PvFold`).
250    #[inline]
251    pub fn read(&self, sample: &crate::climate::ClimateSample) -> f32 {
252        match self {
253            Self::Continentalness => sample.continentalness,
254            Self::Erosion => sample.erosion,
255            Self::Pv => sample.pv(),
256        }
257    }
258}

260/// Phase 1.X-F.2.A: per-parameter spline that pairs a [`Spline1D`] with
261/// the [`ClimateInputDim`] it reads from a `ClimateSample`.
262///
263/// `ParamSpline::evaluate(sample)` is the single per-vertex per-parameter
264/// evaluation: read the configured climate dimension from the sample,
265/// then evaluate the spline at that input.
266///
267/// **Architectural simplification (logged in §10 F.2 entry)**: campaign
268/// doc §2.3 specifies "separable form (3 1D splines × multiplied)" —
269/// per `BootstrapParam`, three independent splines (one per climate
270/// axis) multiplicatively combined with a base value. F.2 ships the
271/// simpler one-spline-per-parameter shape because (a) F.2's catalog
272/// archetypes are all single-control-point constants where multi-spline
273/// product reduces to scalar product, and (b) F.7 tuning is where
274/// multi-spline product earns its keep. If F.7 needs true 3-axis
275/// separable form, it adds a `ParamSplineMulti` type or refactors
276/// `ParamSpline` to carry `[ParamSplineAxis; 3]`.
277#[derive(Debug, Clone, PartialEq)]
278pub struct ParamSpline {
279    pub climate_input: ClimateInputDim,
280    pub spline: Spline1D,
281}
282
283impl ParamSpline {
284    /// Evaluate this parameter spline against a climate sample. Reads
285    /// the configured climate dimension and evaluates the underlying
286    /// `Spline1D` at that input.
287    #[inline]
288    pub fn evaluate(&self, sample: &crate::climate::ClimateSample) -> f32 {
289        self.spline.evaluate(self.climate_input.read(sample))
290    }
291}

317/// Phase 1.X-F.2.B: per-vertex bootstrap noise parameters produced by
318/// evaluating an archetype's [`BootstrapSplineSet`] against a climate
319/// sample.
320///
321/// These four values map directly to [`crate::noise_gen::NoiseConfig`]
322/// fields:
323/// - `mountains_amplitude` → `NoiseConfig.mountains.amplitude` (f32; multiplied against `[-1, 1]` noise output, f32 precision sufficient)
324/// - `mountains_scale` → `NoiseConfig.mountains.scale` (**f64**; multiplied against world coordinates up to ~10000 WU, f64 precision required to preserve byte-identity with legacy `NoiseConfig::default()`)
325/// - `continental_scale` → `NoiseConfig.continental_scale` (**f64**; same rationale as `mountains_scale`)
326/// - `base_elevation_amplitude` → `NoiseConfig.base_elevation.amplitude` (f32)
327///
328/// **Phase 1.X-F.3.B retrofit (logged in `REGIONAL_ARCHETYPE_VARIATION_CAMPAIGN.md`
329/// §10 F.3 entry)**: `mountains_scale` changed from `f32` to `f64`
330/// because F.2.B's f32 storage produced ~60-ulp precision drift
331/// relative to the legacy `NoiseConfig.mountains.scale: f64 = 0.002`
332/// exact representation. The drift propagated through erosion (which
333/// is sensitive to small position changes per F.3-phase-3 world-coord
334/// seeding finding) and produced 103 WU divergence at chunk shared
335/// edges, breaking the phase-2 continuity grassland threshold (80 WU).
336/// `continental_scale` stays `f32` because legacy
337/// `NoiseConfig.continental_scale` is also `f32` — both sides have the
338/// same f32 representation, no drift. Amplitude fields stay `f32`
339/// because they multiply `[-1, 1]` noise output where f32 precision is
340/// sufficient.
341///
342/// F.3 wires the per-vertex `BootstrapParams` into the noise pipeline.
343/// F.2 ships the type without integration; downstream consumers (F.4
344/// mask integration, F.5 paint UI, F.7 per-archetype tuning) read this
345/// struct.
346#[derive(Debug, Clone, Copy, PartialEq)]
347pub struct BootstrapParams {
348    pub mountains_amplitude: f32,
349    pub mountains_scale: f64,
350    pub continental_scale: f32,
351    pub base_elevation_amplitude: f32,
352}

354/// Phase 1.X-F.2.B: per-archetype set of four [`ParamSpline`]s, one
355/// per [`BootstrapParam`]. Aggregates the climate-driven shape splines
356/// for a single archetype.
357///
358/// `BootstrapSplineSet::evaluate(sample)` produces a per-vertex
359/// [`BootstrapParams`] by evaluating each spline against the climate
360/// sample. Per campaign doc §2.6 / F.2 prompt §0, F.7 is where
361/// per-archetype tuning differentiates the splines; F.2 ships 6 catalog
362/// defaults all at single-control-point F.4.B.3.D.5-fix baseline values.
363#[derive(Debug, Clone, PartialEq)]
364pub struct BootstrapSplineSet {
365    pub mountains_amplitude: ParamSpline,
366    /// Phase 1.X-F.3.B retrofit: stored as plain `f64` (not a `ParamSpline`)
367    /// because the legacy `NoiseConfig.mountains.scale: f64 = 0.002` requires
368    /// f64 precision to preserve byte-identity through chunk-edge erosion.
369    /// `Spline1D` evaluates to `f32`, which drifts from the legacy f64 0.002
370    /// by ~60 ulps; that drift propagates through erosion's particle-tracking
371    /// sensitivity (per F.3-phase-3 finding) and produces ~30 WU divergence
372    /// at chunk shared edges, breaking phase-2 continuity. F.7 may replace
373    /// with a multi-control-point f64 spline (`Spline1DF64`-style) if
374    /// per-archetype scale variation surfaces a tuning need; F.2's
375    /// single-control-point catalog case is well-served by direct f64
376    /// storage.
377    pub mountains_scale: f64,
378    pub continental_scale: ParamSpline,
379    pub base_elevation_amplitude: ParamSpline,
380}
381

400    /// `Spline1D::evaluate`'s `f32` output to `f64` for the BootstrapParams
