1//! Phase 1.6-F.4.B.3.C: runevision erosion filter.
2//!
3//! Adds gradient-aligned mesoscale gully detail on top of any height function.
4//! Per F.4.B.3.A research, this is the highest-confidence visible-impact
5//! Uber Noise transform with full published formulas.
6//!
7//! **Source:** Rune Skovbo Johansen, "Fast and Gorgeous Erosion Filter"
8//! (March 2026). Blog: <https://blog.runevision.com/2026/03/fast-and-gorgeous-erosion-filter.html>.
9//! Predecessor "Phacelle" directional noise (January 2026):
10//! <https://blog.runevision.com/2026/01/phacelle-cheap-directional-noise.html>.
11//! Companion video: <https://www.youtube.com/watch?v=r4V21_uUK8Y>.
12//! Shadertoy reference: <https://www.shadertoy.com/view/wXcfWn>.
13//!
14//! **License:** Mozilla Public License v2.0 (per blog post). See
15//! `astraweave-terrain/LICENSE-runevision.md` for full license text.
16//! AstraWeave's port is a faithful interpretation of the algorithm
17//! described in the blog (gradient-aligned gully extrusion, multi-octave
18//! mask attenuation, altitude fade). Specific GLSL was not directly
19//! ported because the blog post URL was not directly fetchable from this
20//! agent environment; the implementation below is structurally
21//! consistent with the published algorithm description but parameter
22//! values and exact math may differ slightly from the canonical
23//! reference. F.4.B.3.A research summary captures the algorithmic
24//! invariants this port preserves.
25//!
26//! **Composition order in `TerrainNoise::sample_height`:**
27//! ```text
28//!   1. Base layer fBm with derivative-weighted attenuation (F.2-T-4)
29//!   2. Mountain layer fBm with derivative-weighted attenuation (F.2-T-4)
30//!   3. Detail layer Billow (F.2)
31//!   4. Domain warping applied per-octave (F.2.B)
32//!   5. Continental modulation multiplies mountain layer output (F.2.6)
33//!   6. Sum of layers
34//!   7. → runevision filter adds gully detail using output gradient (THIS) ←
35//!   8. Final height returned
36//! ```
37//! F.3 particle erosion runs AFTER `sample_height` returns. Particle
38//! droplets see the runevision-filtered terrain as their starting state.
39//!
40//! **Composition with F.2-T-4:** runevision reads gradient from
41//! morenoise's analytical-derivative output. No feedback into morenoise
42//! attenuation; no per-octave parameter interactions. The non-linear
43//! interaction that broke F.4.B.3.B (octave-emphasis weights × derivative
44//! attenuation) does not occur here — runevision is post-fBm extrusion.
45

48/// Configuration for the runevision erosion filter.
49#[derive(Debug, Clone, Serialize, Deserialize)]
50pub struct RunevisionConfig {
51    /// Master strength multiplier. 0.0 = filter off (no contribution),
52    /// 1.0 = blog-default contribution. Reasonable tuning range 0.5-1.5.
53    pub strength: f32,
54    /// Number of multi-octave iterations. Blog uses 3-4. Each octave halves
55    /// wavelength and applies finer-scale gully detail restricted to
56    /// crease/ridge regions of previous octaves via mask attenuation.
57    pub octaves: u32,
58    /// World-Y altitude below which gullies fade out (no contribution).
59    /// Calibrated to AstraWeave Target B Y range (~0-510 m). At sea level
60    /// (~0 m) gullies vanish; at peak altitude they reach full intensity.
61    pub valley_altitude: f32,
62    /// World-Y altitude above which gullies are at full intensity.
63    pub peak_altitude: f32,
64    /// Detail mask attenuation between octaves. Blog uses ~0.5. Higher
65    /// values let finer-scale gullies appear in broader regions; lower
66    /// values restrict them to steeper sub-features of previous octaves.
67    pub detail_attenuation: f32,
68    /// World-space wavelength of the largest filter octave (octave 0) in
69    /// world units. Each subsequent octave halves this. At base 100 WU
70    /// with 3 octaves: 100, 50, 25 WU wavelengths.
71    pub base_wavelength: f32,
72    /// Seed offset added to `world_seed` for filter's noise sampling.
73    /// Avoids collision with main noise generator seeds.
74    pub seed_offset: u32,
75}
76
77impl Default for RunevisionConfig {
78    fn default() -> Self {
79        // Calibrated for AstraWeave Target B Y range (~0-510 m).
80        // valley_altitude = 50 m: foothills onset.
81        // peak_altitude = 400 m: clear separation from valleys; full filter
82        //   intensity above this. Mid-mountain range typically 200-400 m
83        //   gets gradual gully fade-in.
84        // base_wavelength 100 m: ~5 cycles per 512 m chunk — mesoscale
85        //   between continental (3300 m) and detail (~10 m).
86        Self {
87            strength: 1.0,
88            octaves: 3,
89            valley_altitude: 50.0,
90            peak_altitude: 400.0,
91            detail_attenuation: 0.5,
92            base_wavelength: 100.0,
93            seed_offset: 13,
94        }
95    }
96}

98/// Apply runevision erosion filter at a single world position.
99///
100/// Layers gradient-aligned gully extrusion on top of the input `height`.
101/// Gullies orient perpendicular to the gradient (i.e., along downslope
102/// flow direction). Multi-octave with altitude fade and mask attenuation.
103///
104/// # Arguments
105/// * `height` — pre-filter height at this position (output of base + mountain
106///   + detail + continental modulation per `sample_height` stages 1-6).
107/// * `gradient` — analytical gradient `(∂h/∂x, ∂h/∂z)` at this position from
108///   `fbm_derivative_weighted_with_gradient_2d`. Approximation: AstraWeave
109///   uses the BASE layer's gradient as a proxy for the combined-output
110///   gradient (mountain layer uses opaque RidgedMulti without analytical
111///   gradient access; computing finite-difference gradient of the full
112///   output would require 4 extra noise samples per vertex). Base-layer
113///   gradient is a reasonable proxy because base is the smooth,
114///   wide-feature dominant layer.
115/// * `world_x`, `world_z` — world coordinates for filter's own noise sampling.
116/// * `world_seed` — base world seed; filter adds `config.seed_offset`.
117/// * `config` — filter parameters.
118///
119/// # Returns
120/// Modified height with gully detail added. Range: `height ± (filter_output * strength)`.
121/// Filter contribution is bounded by altitude fade (zero in valleys) and
122/// slope (zero on perfectly flat terrain).
123pub fn apply_runevision_filter(
124    height: f32,
125    gradient: (f32, f32),
126    world_x: f64,
127    world_z: f64,
128    world_seed: u64,
129    config: &RunevisionConfig,
130) -> f32 {
131    if config.strength <= 0.0 {
132        return height;
133    }
134
135    let (grad_x, grad_z) = gradient;
136    let grad_len_sq = grad_x * grad_x + grad_z * grad_z;
137    let grad_len = grad_len_sq.sqrt();
138
139    // No gradient → no flow direction → no gullies. Flat regions are
140    // intentionally left untouched (matches blog's behavior — gullies
141    // require slope to form).
142    if grad_len < 1e-6 {
143        return height;
144    }
145
146    // Slope factor: `min(grad_len, 1.0)`. Strong slopes get full gully
147    // contribution; gentle slopes get proportionally less. Real-world
148    // mesoscale gullies form on slopes ≥ ~5% (grad_len ~0.05); this
149    // unitless factor approximates that progression.
150    let slope_factor = grad_len.min(1.0);
151
152    // Altitude fade: 0 below valley_altitude, 1 above peak_altitude,
153    // smoothstep between. Gullies are concentrated on peaks.
154    let altitude_factor = inverse_lerp(config.valley_altitude, config.peak_altitude, height);
155    let altitude_smooth = smoothstep(altitude_factor);
156
157    if altitude_smooth <= 0.0 {
158        return height;
159    }
160
161    // Stripe-direction unit vector: perpendicular to gradient = downslope
162    // flow direction. Gullies extrude along this direction. (gradient
163    // points uphill; flow runs opposite, so stripes align with -gradient.)
164    let flow_x = -grad_x / grad_len;
165    let flow_z = -grad_z / grad_len;
166    // Stripe direction perpendicular to flow (so cosine wave of stripe
167    // coordinate produces gullies parallel to flow).
168    let stripe_x = -flow_z;
169    let stripe_z = flow_x;
170
171    let filter_seed = world_seed.wrapping_add(config.seed_offset as u64);
172
173    // Multi-octave gully accumulation with mask attenuation.
174    let mut filter_output = 0.0f32;
175    let mut combi_mask = 1.0f32;
176    let mut wavelength = config.base_wavelength;
177    let mut octave_amplitude = 1.0f32;
178
179    for octave in 0..config.octaves {
180        // Project world position onto stripe direction (perpendicular to
181        // flow). Coordinate along stripe normal at this wavelength.
182        let stripe_coord = ((world_x as f32 * stripe_x) + (world_z as f32 * stripe_z)) / wavelength;
183
184        // Cell hash for per-cell phase variation (Worley-style local jitter).
185        let cell = stripe_coord.floor();
186        let frac = stripe_coord - cell;
187        let smooth_frac = smoothstep(frac);
188
189        let cell_seed = filter_seed
190            .wrapping_add((octave as u64).wrapping_mul(0x9E3779B97F4A7C15))
191            .wrapping_add(cell as i64 as u64);
192        let next_cell_seed = cell_seed.wrapping_add(0x85EBCA6BE11ECC0D);
193
194        // Phase per cell — keeps gullies non-axis-aligned and varied.
195        let phase_a = hash_to_unit(cell_seed) * std::f32::consts::TAU;
196        let phase_b = hash_to_unit(next_cell_seed) * std::f32::consts::TAU;
197
198        // Cosine wave of stripe coordinate produces ridge/valley alternation
199        // perpendicular to flow direction. Sine wave gives slope variation.
200        let theta_a = stripe_coord * std::f32::consts::TAU + phase_a;
201        let theta_b = stripe_coord * std::f32::consts::TAU + phase_b;
202        let cos_a = theta_a.cos();
203        let cos_b = theta_b.cos();
204        let sin_a = theta_a.sin();
205        let sin_b = theta_b.sin();
206
207        // Smooth blend between adjacent cells.
208        let cos_blend = cos_a * (1.0 - smooth_frac) + cos_b * smooth_frac;
209        let sin_blend = sin_a * (1.0 - smooth_frac) + sin_b * smooth_frac;
210
211        // Gully height contribution: cosine wave (extrudes ridges and
212        // carves valleys perpendicular to flow). Amplitude shaped by
213        // current combi_mask (so finer octaves are restricted to crease
214        // regions of coarser octaves).
215        let gully_contribution = cos_blend * combi_mask * octave_amplitude;
216        filter_output += gully_contribution;
217
218        // Crease signal: where sin component is near zero, we're at a
219        // crease/ridge of the cosine. ease_out emphasizes the contrast.
220        let crease_signal = ease_out(1.0 - sin_blend.abs());
221
222        // Update mask for next (finer) octave: restrict next octave to
223        // crease regions of this one. pow_inv softens the restriction.
224        combi_mask = pow_inv(combi_mask, config.detail_attenuation) * crease_signal;
225
226        wavelength *= 0.5; // Lacunarity 2.0 (frequency doubles per octave).
227        octave_amplitude *= config.detail_attenuation;
228    }
229
230    // Final modifier: filter output × slope × altitude × strength.
231    // Slope and altitude are both scale-invariant 0-1 weights;
232    // strength is the user-tunable global gain.
233    let modifier = filter_output * slope_factor * altitude_smooth * config.strength;
234
235    height + modifier
236}

238#[inline]
239fn inverse_lerp(a: f32, b: f32, v: f32) -> f32 {
240    if (b - a).abs() < 1e-6 {
241        0.0
242    } else {
243        ((v - a) / (b - a)).clamp(0.0, 1.0)
244    }
245}
246
247#[inline]
248fn smoothstep(t: f32) -> f32 {
249    let t = t.clamp(0.0, 1.0);
250    t * t * (3.0 - 2.0 * t)
251}
252
253#[inline]
254fn ease_out(t: f32) -> f32 {
255    let v = 1.0 - t.clamp(0.0, 1.0);
256    1.0 - v * v
257}
258
259#[inline]
260fn pow_inv(t: f32, power: f32) -> f32 {
261    let saturated = t.clamp(0.0, 1.0);
262    1.0 - (1.0 - saturated).powf(power)
263}

265/// Hash a seed to a unit-interval float `[0, 1)`.
266#[inline]
267fn hash_to_unit(seed: u64) -> f32 {
268    // Wang-style finalizer.
269    let mut h = seed.wrapping_mul(0x9E3779B97F4A7C15);
270    h ^= h >> 32;
271    h = h.wrapping_mul(0x85EBCA6BE11ECC0D);
272    h ^= h >> 32;
273    ((h & 0xFFFF_FFFF) as f32) / (u32::MAX as f32)
274}

276#[cfg(test)]
277mod tests {
278    use super::*;
279
280    #[test]
281    fn filter_is_no_op_when_strength_zero() {
282        let mut config = RunevisionConfig::default();
283        config.strength = 0.0;
284        let h = apply_runevision_filter(100.0, (0.5, 0.3), 1234.5, 678.9, 12345, &config);
285        assert_eq!(h, 100.0);
286    }
287
288    #[test]
289    fn filter_is_no_op_in_valley() {
290        // Below valley_altitude → altitude_factor = 0 → no contribution.
291        let config = RunevisionConfig::default();
292        let h = apply_runevision_filter(10.0, (0.5, 0.3), 100.0, 100.0, 12345, &config);
293        assert_eq!(h, 10.0, "below valley altitude, filter should be no-op");
294    }
295
296    #[test]
297    fn filter_is_no_op_on_flat_terrain() {
298        // Zero gradient → no flow direction → no gullies.
299        let config = RunevisionConfig::default();
300        let h = apply_runevision_filter(300.0, (0.0, 0.0), 100.0, 100.0, 12345, &config);
301        assert_eq!(h, 300.0, "flat terrain (zero gradient), filter should be no-op");
302    }
303
304    #[test]
305    fn filter_modifies_height_above_valley_with_gradient() {
306        let config = RunevisionConfig::default();
307        // Above peak altitude with significant gradient → filter should
308        // produce non-zero contribution.
309        let original = 450.0;
310        let h = apply_runevision_filter(original, (0.5, 0.3), 100.0, 100.0, 12345, &config);
311        assert_ne!(
312            h, original,
313            "with strength=1, gradient, and altitude above peak, filter should modify height"
314        );
315    }
316
317    #[test]
318    fn filter_deterministic_at_same_position() {
319        let config = RunevisionConfig::default();
320        let h1 = apply_runevision_filter(300.0, (0.4, 0.2), 1000.0, 2000.0, 99999, &config);
321        let h2 = apply_runevision_filter(300.0, (0.4, 0.2), 1000.0, 2000.0, 99999, &config);
322        assert_eq!(h1, h2, "filter must be deterministic for same inputs");
323    }
324
325    #[test]
326    fn filter_differs_between_positions() {
327        let config = RunevisionConfig::default();
328        let h1 = apply_runevision_filter(300.0, (0.4, 0.2), 100.0, 100.0, 12345, &config);
329        let h2 = apply_runevision_filter(300.0, (0.4, 0.2), 500.0, 500.0, 12345, &config);
330        assert_ne!(
331            h1, h2,
332            "filter should produce different values at different world positions"
333        );
334    }
