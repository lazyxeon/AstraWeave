1//! Climate map generation for biome assignment.
2//!
3//! Phase 1.6-F.4.B.3.D.1 extends this module with a real-units climate-field
4//! architecture (per-vertex temperature in °C, moisture in mm/yr, continentalness
5//! in [0,1]) driven by a `WorldArchetype` envelope, latitude/elevation/water-distance
6//! modulators, and a low-frequency continentalness noise mirroring `TerrainNoise`'s
7//! continental modulation. The legacy `sample_climate`/`sample_temperature`/
8//! `sample_moisture` methods returning normalized `[0,1]` values are preserved for
9//! existing consumers (`biome_detector`, `biome_transition`, renderer overlay,
10//! benchmarks, tests) until F.4.B.3.D.3 replaces them with per-biome parameter
11//! lookup. New code should use [`ClimateMap::sample`] returning [`ClimateSample`].
12

17/// Phase 1.6-F.4.B.3.D.1: half-extent (in world units) used as the latitude
18/// normalization denominator. World Z divided by this constant produces a
19/// `[-1, 1]` latitude factor used by the temperature latitude modulator.
20///
21/// Calibrated for Target B world scale: 21 chunks of 512 WU per side =
22/// 10752 WU total extent, half-extent 5376 WU. At Target B, latitude=0 at
23/// world center, latitude=±1 at the world edges. Target C will need this
24/// recalibrated; default is parameterized via `ClimateConfig` so callers
25/// can override.
26pub const TARGET_B_LATITUDE_HALF_EXTENT_WU: f32 = 5376.0;
27
28/// Phase 1.6-F.4.B.3.D.1: standard atmospheric lapse rate (-6.5°C per 1000m
29/// elevation). Real-world value used unchanged.
30pub const ATMOSPHERIC_LAPSE_RATE_C_PER_M: f32 = -0.0065;
31
32/// Phase 1.6-F.4.B.3.D.1: documented bounds on real-units climate sample
33/// values. Worlds beyond these ranges fall outside Earth analogs; bounds
34/// chosen to span hot-desert (~+40°C, ~50mm) through polar (-30°C, ~200mm)
35/// to tropical-rainforest (~+27°C, ~4000mm).
36pub const TEMPERATURE_C_MIN: f32 = -30.0;
37pub const TEMPERATURE_C_MAX: f32 = 40.0;
38pub const MOISTURE_MM_MIN: f32 = 0.0;
39pub const MOISTURE_MM_MAX: f32 = 4000.0;

41/// Phase 1.6-F.4.B.3.D.1: a single per-vertex climate sample in real-world
42/// units. Drives Whittaker biome lookup (D.2) and per-biome parameter
43/// blending (D.3, D.4).
44///
45/// Phase 1.X-F.1.B extends with `erosion` and `weirdness` fields for the
46/// regional archetype variation campaign's per-archetype shape splines
47/// (campaign doc §2.2). PV is derived at sample time via [`Self::pv`].
48///
49/// Range invariants (enforced by `ClimateMap::sample`):
50/// - `temperature_c` ∈ `[TEMPERATURE_C_MIN, TEMPERATURE_C_MAX]`.
51/// - `moisture_mm`  ∈ `[MOISTURE_MM_MIN, MOISTURE_MM_MAX]`.
52/// - `continentalness` ∈ `[0.0, 1.0]`.
53/// - `erosion` ∈ `[-1.0, 1.0]`.
54/// - `weirdness` ∈ `[-1.0, 1.0]`.
55#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
56pub struct ClimateSample {
57    /// Mean annual temperature in degrees Celsius.
58    pub temperature_c: f32,
59    /// Mean annual precipitation in millimeters per year.
60    pub moisture_mm: f32,
61    /// Continentalness factor `[0,1]`. Higher = continental interior
62    /// (drier, more variable); lower = oceanic (wetter, more moderated).
63    /// Mirrors `TerrainNoise::sample_continental_01` semantics so D.3 can
64    /// share the same source field across both subsystems if desired.
65    pub continentalness: f32,
66    /// Phase 1.X-F.1.B: low-frequency Perlin noise field representing
67    /// flatness propensity (range `[-1, 1]`). High erosion → flat terrain;
68    /// low erosion → mountainous, per Minecraft 1.18+ canonical
69    /// interpretation. Read by per-archetype `BootstrapSplineSet` in
70    /// F.2-F.3 to produce per-region shape variation. Unused by D.2's
71    /// `lookup_biome` and D.4's biome blending; populated unconditionally.
72    pub erosion: f32,
73    /// Phase 1.X-F.1.B: low-frequency Perlin noise field that feeds the
74    /// PV (Peaks-and-Valleys) fold (range `[-1, 1]`). Use [`Self::pv`]
75    /// to compute the canonical Minecraft 1.18+ folded value
76    /// (`pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`). Read by
77    /// per-archetype `BootstrapSplineSet` in F.2-F.3. Unused by D.2 and
78    /// D.4; populated unconditionally.
79    pub weirdness: f32,
80}
81
82impl ClimateSample {
83    /// Phase 1.X-F.1.B: derived Peaks-and-Valleys field via the
84    /// canonical Minecraft 1.18+ formula
85    /// `pv = 1.0 - ((3.0 * |weirdness|) - 2.0).abs()`.
86    ///
87    /// Folded weirdness produces the characteristic 5-band character
88    /// (Valleys / Low / Mid / High / Peaks). Output range `[-1, 1]` for
89    /// `|weirdness| ∈ [0, 1]`.
90    ///
91    /// Computed at call time (not stored); see
92    /// [`crate::spline_types::PvFold::from_weirdness`] for the
93    /// underlying helper.
94    #[inline]
95    pub fn pv(&self) -> f32 {
96        crate::spline_types::PvFold::from_weirdness(self.weirdness)
97    }
98}

100/// Phase 1.6-F.4.B.3.D.1: a world archetype is a climate envelope —
101/// means and variances for the three climate fields, plus a latitude
102/// strength scalar controlling how steeply temperature drops toward
103/// the world's edge in Z.
104///
105/// Archetypes do NOT assign biomes directly. Per the F.4.B.3.D
106/// architecture, biomes emerge from per-vertex climate × elevation
107/// lookup (D.2). Archetypes shift the *distribution* of climate values
108/// across the world; specific biomes that result depend on the
109/// Whittaker lookup applied to those values.
110///
111/// D.1 ships a single `Default` impl corresponding to Continental
112/// Temperate (Veilweaver default — NC/Appalachia analog). D.5 adds the
113/// other five archetypes (Equatorial Tropical, Boreal/Subarctic,
114/// Mediterranean, Desert, Custom) and the editor UI surface.
115///
116/// Phase 1.X-F.2.C: extended with `bootstrap_splines: BootstrapSplineSet`
117/// field for the regional archetype variation campaign's per-archetype
118/// shape splines. **`Copy` derive removed** because `BootstrapSplineSet`
119/// contains `Vec<(f32, f32)>` (in `Spline1D::control_points`), which
120/// is not `Copy`. This is a documented deviation logged in
121/// `REGIONAL_ARCHETYPE_VARIATION_CAMPAIGN.md` §10 F.2 entry; F.7 will
122/// inevitably need multi-control-point splines that are intrinsically
123/// non-Copy, so removing Copy now avoids a later forced refactor.
124/// Existing `Clone` derive is preserved; call sites that previously
125/// relied on implicit copy semantics now use explicit `.clone()`.
126#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
127pub struct WorldArchetype {
128    /// Mean temperature at world center, sea level, before lapse rate.
129    pub temperature_mean_c: f32,
130    /// Per-axis temperature noise variance (±°C). Noise field is sampled
131    /// to `[-1, 1]` and multiplied by this.
132    pub temperature_variance_c: f32,
133    /// Temperature drop from equator (latitude=0) to pole-edge
134    /// (latitude=±1) in °C. Aesthetic scaling, not physically calibrated;
135    /// the world is treated as a slice of latitude rather than a globe.
136    pub latitude_temperature_drop_c: f32,
137    /// Mean moisture at world center, sea level. mm/year.
138    pub moisture_mean_mm: f32,
139    /// Per-axis moisture noise variance (±mm/yr).
140    pub moisture_variance_mm: f32,
141    /// Continentalness mean in `[0,1]`. Coastal (low) vs interior (high).
142    pub continentalness_mean: f32,
143    /// Continentalness noise variance.
144    pub continentalness_variance: f32,
145    /// Phase 1.X-F.2.C: per-archetype bootstrap noise parameter splines.
146    /// Each archetype's catalog factory function (in
147    /// `crate::world_archetypes::*`) populates this with single-control-point
148    /// splines at F.4.B.3.D.5-fix baseline values. F.7 differentiates per
149    /// archetype with multi-control-point shapes.
150    ///
151    /// `#[serde(skip, default)]` so worlds serialized before F.2.C
152    /// deserialize cleanly with the baseline `BootstrapSplineSet` as
153    /// fallback. Since `BootstrapSplineSet` contains a `Spline1D::Vec`,
154    /// it cannot be serialized via `Copy`-derived bincode/postcard;
155    /// the architectural intent is that splines live as compile-time
156    /// data populated by catalog factory calls, not as serialized world
157    /// state.
158    #[serde(skip, default)]
159    pub bootstrap_splines: crate::spline_types::BootstrapSplineSet,
160}

180impl WorldArchetype {
181    /// Validate parameter ranges. Returns `Err` with description if any
182    /// parameter is out of range. Used by tests + future archetype
183    /// validation in D.5.
184    pub fn validate(&self) -> Result<(), &'static str> {
185        if !self.temperature_mean_c.is_finite() {
186            return Err("temperature_mean_c must be finite");
187        }
188        if !(TEMPERATURE_C_MIN..=TEMPERATURE_C_MAX).contains(&self.temperature_mean_c) {
189            return Err("temperature_mean_c outside [-30, +40] range");
190        }
191        if self.temperature_variance_c < 0.0 || self.temperature_variance_c > 30.0 {
192            return Err("temperature_variance_c outside [0, 30] range");
193        }
194        if self.latitude_temperature_drop_c < 0.0 || self.latitude_temperature_drop_c > 50.0 {
195            return Err("latitude_temperature_drop_c outside [0, 50] range");
196        }
197        if !(MOISTURE_MM_MIN..=MOISTURE_MM_MAX).contains(&self.moisture_mean_mm) {
198            return Err("moisture_mean_mm outside [0, 4000] range");
199        }
200        if self.moisture_variance_mm < 0.0 || self.moisture_variance_mm > 2000.0 {
201            return Err("moisture_variance_mm outside [0, 2000] range");
202        }
203        if !(0.0..=1.0).contains(&self.continentalness_mean) {
204            return Err("continentalness_mean outside [0, 1] range");
205        }
206        if self.continentalness_variance < 0.0 || self.continentalness_variance > 0.5 {
207            return Err("continentalness_variance outside [0, 0.5] range");
208        }
209        Ok(())
210    }
211}

222    /// combined with archetype mean/variance. Wavelength chosen to mirror
223    /// `TerrainNoise::sample_continental_01` (default scale 0.0003 ≈

232    /// Latitude influence on temperature.
233    /// **Legacy**: kept for backward-compat with `sample_temperature`.
234    /// New `sample()` API uses `WorldArchetype::latitude_temperature_drop_c`.
235    pub temperature_latitude_gradient: f32,
236    /// Distance from water influence on moisture.
237    /// Used by both legacy `sample_moisture` and new `sample()` (the

240    /// distance-from-world-edge).

318/// Climate map generator that provides temperature and moisture data
319#[derive(Debug)]
320pub struct ClimateMap {
321    temperature_noise: Perlin,
322    moisture_noise: Perlin,
323    /// Phase 1.6-F.4.B.3.D.1: low-frequency continentalness noise.
324    /// Sampled by the new `sample()` API. Seed is `seed + 2000` to
325    /// decorrelate from temperature (`seed`) and moisture (`seed + 1000`).
326    continentalness_noise: Perlin,
327    /// Phase 1.X-F.1.B: low-frequency erosion noise. Seed is `seed + 3000`
328    /// to decorrelate from existing climate fields. Scale 0.0008
329    /// (wavelength ~1250 WU at Target B). Output `[-1, 1]` represents
330    /// flatness propensity per Minecraft 1.18+ canonical interpretation.
331    erosion_noise: Perlin,
332    /// Phase 1.X-F.1.B: low-frequency weirdness noise. Seed is `seed + 4000`
333    /// to decorrelate from existing climate fields. Scale 0.0006
334    /// (wavelength ~1670 WU). Output `[-1, 1]` feeds the PV fold via
335    /// [`ClimateSample::pv`].
336    weirdness_noise: Perlin,
337    config: ClimateConfig,
338}
339
340impl ClimateMap {
341    /// Create a new climate map generator
342    pub fn new(config: &ClimateConfig, seed: u64) -> Self {
343        Self {
344            temperature_noise: Perlin::new(seed as u32),
345            moisture_noise: Perlin::new((seed + 1000) as u32),
346            continentalness_noise: Perlin::new((seed + 2000) as u32),
347            // Phase 1.X-F.1.B: erosion + weirdness noise for the
348            // regional archetype variation campaign's shape splines.
349            // Offsets +3000 / +4000 verified clear of all other terrain
350            // crate seed offsets in F.1 prompt §1.1 verification step.
351            erosion_noise: Perlin::new((seed + 3000) as u32),
352            weirdness_noise: Perlin::new((seed + 4000) as u32),
353            config: config.clone(),
354        }
355    }
356
357    /// Phase 1.6-F.4.B.3.D.1: real-units climate sample at a world position.
358    ///
359    /// Returns `(temperature_c, moisture_mm, continentalness)` driven by the
360    /// configured `WorldArchetype` envelope plus three independent modulators:
361    ///
362    /// 1. **Latitude modulator** — temperature drops toward world edges in Z
363    ///    by `archetype.latitude_temperature_drop_c` × `|world_z / world_latitude_half_extent_wu|`.
364    /// 2. **Elevation lapse rate** — temperature drops by 6.5°C per 1000m
365    ///    elevation (`ATMOSPHERIC_LAPSE_RATE_C_PER_M`).
366    /// 3. **Water-distance modulator** — moisture decreases with distance
367    ///    from world edge (continental interior is drier). Falloff rate
368    ///    controlled by `config.moisture_distance_falloff`.
369    ///
370    /// Output values are clamped to `[TEMPERATURE_C_MIN, TEMPERATURE_C_MAX]`,
371    /// `[MOISTURE_MM_MIN, MOISTURE_MM_MAX]`, and `[0, 1]` respectively.
372    ///
373    /// New per-vertex API for D.2/D.3 Whittaker biome lookup. Existing
374    /// consumers (`biome_detector`, `biome_transition`, renderer overlay)
375    /// still use the legacy `[0,1]`-bounded `sample_climate`/`sample_temperature`/
376    /// `sample_moisture` methods until D.3 migrates them.
377    pub fn sample(&self, world_x: f64, world_z: f64, elevation: f32) -> ClimateSample {
378        let arch = &self.config.archetype;
379
380        // === Temperature ===
381        // Base: archetype mean + low-frequency noise variance.
382        let temp_noise_raw = self.sample_noise_signed(
383            &self.temperature_noise,
384            &self.config.temperature,
385            world_x,
386            world_z,
387        );
388        let mut temperature_c =
389            arch.temperature_mean_c + temp_noise_raw * arch.temperature_variance_c;
390
391        // Latitude modulator: temperature decreases toward poles.
392        // |z / half_extent| ∈ [0, 1+] (clamped). At equator (lat=0) no
393        // adjustment; at pole-edge (|lat|=1) full archetype drop applied.
394        let latitude_norm = self.latitude_normalized(world_z);
395        temperature_c -= latitude_norm.abs() * arch.latitude_temperature_drop_c;
396
397        // Elevation lapse rate: standard atmospheric -6.5°C per 1000m.
398        // Below sea level (negative elevation) gets a small temperature
399        // bump but the clamp at `TEMPERATURE_C_MAX` prevents runaway.
400        temperature_c += elevation * ATMOSPHERIC_LAPSE_RATE_C_PER_M;
401
402        let temperature_c = temperature_c.clamp(TEMPERATURE_C_MIN, TEMPERATURE_C_MAX);
403
404        // === Moisture ===
405        // Base: archetype mean + low-frequency noise variance.
406        let moist_noise_raw = self.sample_noise_signed(
407            &self.moisture_noise,
408            &self.config.moisture,
409            world_x,
410            world_z,
411        );
412        let mut moisture_mm =
413            arch.moisture_mean_mm + moist_noise_raw * arch.moisture_variance_mm;
414
415        // Water-distance modulator: distance from world edge proxies for
416        // continental interior. World edge → high moisture (coast); world
417        // center → low moisture (interior). `coast_distance` is the
418        // shortest distance to either x or z world edge.
419        let half_ext = self.config.world_latitude_half_extent_wu as f64;
420        let coast_distance =
421            (half_ext - world_x.abs()).min(half_ext - world_z.abs()).max(0.0) as f32;
422        let coast_factor =
423            (-coast_distance * self.config.moisture_distance_falloff).exp();
424        // Mix: 70% noise-driven, 30% coast-driven. Coastal regions get a
425        // moisture boost up to 30% of the archetype mean.
426        moisture_mm = moisture_mm * 0.7 + (arch.moisture_mean_mm * coast_factor) * 0.3;
427
428        let moisture_mm = moisture_mm.clamp(MOISTURE_MM_MIN, MOISTURE_MM_MAX);
429
430        // === Continentalness ===
431        // Single-octave low-frequency noise around archetype mean.
432        let cont_noise_raw = self.sample_noise_signed(
433            &self.continentalness_noise,
434            &self.config.continentalness,
435            world_x,
436            world_z,
437        );
438        let continentalness =
439            (arch.continentalness_mean + cont_noise_raw * arch.continentalness_variance)
440                .clamp(0.0, 1.0);
441
442        // === Erosion (Phase 1.X-F.1.B) ===
443        // Low-frequency Perlin field at scale 0.0008 (~1250 WU wavelength
444        // at Target B). Output [-1, 1] represents flatness propensity per
445        // Minecraft 1.18+ canonical interpretation. Defensive clamp
446        // against f32 boundary artifacts that occasionally produce
447        // ±1.0001 at integer lattice positions; Perlin output is
448        // mathematically bounded.
449        let erosion_raw =
450            self.erosion_noise.get([world_x * 0.0008, world_z * 0.0008]) as f32;
451        let erosion = erosion_raw.clamp(-1.0, 1.0);
452
453        // === Weirdness (Phase 1.X-F.1.B) ===
454        // Low-frequency Perlin field at scale 0.0006 (~1670 WU wavelength
455        // at Target B). Output [-1, 1] feeds the PV (Peaks-and-Valleys)
456        // fold via ClimateSample::pv. Same clamp rationale as erosion.
457        let weirdness_raw =
458            self.weirdness_noise.get([world_x * 0.0006, world_z * 0.0006]) as f32;
459        let weirdness = weirdness_raw.clamp(-1.0, 1.0);
460
461        ClimateSample {
462            temperature_c,
463            moisture_mm,
464            continentalness,
465            erosion,
466            weirdness,
467        }
468    }
469
470    /// Phase 1.6-F.4.B.3.D.1: latitude normalized to `[-1, 1]+` based on
471    /// world Z. Values beyond ±1 are possible if the world extends past
472    /// the configured half-extent (radius >10 chunks at Target B).
473    /// Exposed `pub(crate)` for D.1 unit tests.
474    #[inline]
475    pub(crate) fn latitude_normalized(&self, world_z: f64) -> f32 {
476        if self.config.world_latitude_half_extent_wu <= 0.0 {
477            0.0
478        } else {
479            (world_z as f32) / self.config.world_latitude_half_extent_wu
480        }
481    }
482
483    /// Phase 1.6-F.4.B.3.D.1: sample fBm noise as a signed value
484    /// approximately in `[-1, 1]`. Used by `sample()` so noise output can
485    /// be multiplied by archetype variance without the `+ offset` bias
486    /// the legacy `sample_noise_fbm` adds for `[0,1]`-style consumers.
487    #[inline]
488    fn sample_noise_signed(
489        &self,
490        noise: &Perlin,
491        layer: &ClimateLayer,
492        x: f64,
493        z: f64,
494    ) -> f32 {
495        let mut value = 0.0f32;
496        let mut amplitude = layer.amplitude;
497        let mut frequency = layer.scale;
498        for _ in 0..layer.octaves {
499            value += noise.get([x * frequency, 0.0, z * frequency]) as f32 * amplitude;
500            amplitude *= layer.persistence as f32;
501            frequency *= layer.lacunarity;
502        }
503        value.clamp(-1.0, 1.0)
504    }
505
506    /// Sample temperature at a world position
507    pub fn sample_temperature(&self, x: f64, z: f64, height: f32) -> f32 {
508        // Base temperature from noise
509        let mut temperature =
510            self.sample_noise_fbm(&self.temperature_noise, &self.config.temperature, x, z);
511
512        // Apply height gradient (cooler at higher elevations)
513        temperature += height * self.config.temperature_height_gradient;
514
515        // Apply latitude gradient (cooler towards poles)
516        let latitude_factor = (z * 0.00001).sin(); // Very rough latitude simulation
517        temperature += latitude_factor as f32 * self.config.temperature_latitude_gradient;
518
519        // Clamp to reasonable range
520        temperature.clamp(0.0, 1.0)
521    }
522
523    /// Sample moisture at a world position
524    pub fn sample_moisture(&self, x: f64, z: f64, height: f32) -> f32 {
525        // Base moisture from noise
526        let mut moisture = self.sample_noise_fbm(&self.moisture_noise, &self.config.moisture, x, z);
527
528        // Reduce moisture at higher elevations (rain shadow effect)
529        let height_factor = (height * 0.01).clamp(0.0, 1.0);
530        moisture *= 1.0 - height_factor * 0.3;
531
532        // Distance from water effect (simplified - in real implementation would use actual water bodies)
533        let water_distance = self.estimate_water_distance(x, z);
534        let water_factor = (-water_distance * self.config.moisture_distance_falloff).exp();
535        moisture = moisture * 0.7 + water_factor * 0.3;
536
537        // Clamp to valid range
538        moisture.clamp(0.0, 1.0)
539    }

541    /// Sample both temperature and moisture at a world position
542    pub fn sample_climate(&self, x: f64, z: f64, height: f32) -> (f32, f32) {
543        let temperature = self.sample_temperature(x, z, height);
544        let moisture = self.sample_moisture(x, z, height);
545        (temperature, moisture)
546    }

548    /// Sample climate data for an entire chunk
549    pub fn sample_chunk(
550        &self,
551        chunk_id: ChunkId,
552        chunk_size: f32,
553        resolution: u32,
554    ) -> anyhow::Result<Vec<(f32, f32)>> {
555        let world_origin = chunk_id.to_world_pos(chunk_size);
556        let step = chunk_size / (resolution - 1) as f32;
557        let mut climate_data = Vec::with_capacity((resolution * resolution) as usize);
558
559        for z in 0..resolution {
560            for x in 0..resolution {
561                let world_x = world_origin.x + x as f32 * step;
562                let world_z = world_origin.z + z as f32 * step;
563
564                // We need height data to calculate climate properly
565                // For now, use a simple height estimation based on position
566                let estimated_height = self.estimate_height(world_x as f64, world_z as f64);
567
568                let climate = self.sample_climate(world_x as f64, world_z as f64, estimated_height);
569                climate_data.push(climate);
570            }
571        }
572
573        Ok(climate_data)
574    }

576    /// Sample fractal Brownian motion noise
577    fn sample_noise_fbm(&self, noise: &Perlin, layer: &ClimateLayer, x: f64, z: f64) -> f32 {
578        let mut value = 0.0;
579        let mut amplitude = layer.amplitude;
580        let mut frequency = layer.scale;
581
582        for _ in 0..layer.octaves {
583            value += noise.get([x * frequency, 0.0, z * frequency]) as f32 * amplitude;
584            amplitude *= layer.persistence as f32;
585            frequency *= layer.lacunarity;
586        }
587
588        value + layer.offset
589    }

591    /// Estimate height at a position (temporary until we have proper integration)
592    pub fn estimate_height(&self, x: f64, z: f64) -> f32 {
593        // Simple height estimation using noise — amplitude must roughly
594        // match the maximum terrain amplitude used by noise presets (e.g.
595        // mountain base_amplitude 120 + mountains_amplitude 150 ≈ ~270).
596        let height_noise = self.sample_noise_fbm(
597            &self.temperature_noise, // Reuse temperature noise for height
598            &ClimateLayer {
599                scale: 0.002,
600                octaves: 4,
601                persistence: 0.5,
602                lacunarity: 2.0,
603                amplitude: 150.0,
604                offset: 10.0,
605            },
606            x,
607            z,
608        );
609        height_noise.max(0.0)
610    }

612    /// Estimate distance to nearest water body (simplified)
613    fn estimate_water_distance(&self, x: f64, z: f64) -> f32 {
614        // Simplified water distance using noise to create "rivers" and "lakes"
615        let water_noise = self.sample_noise_fbm(
616            &self.moisture_noise,
617            &ClimateLayer {
618                scale: 0.003,
619                octaves: 2,
620                persistence: 0.4,
621                lacunarity: 2.5,
622                amplitude: 1.0,
623                offset: 0.0,
624            },
625            x,
626            z,
627        );
628
629        // If noise is below threshold, we're "near water"
630        if water_noise.abs() < 0.1 {
631            0.0 // At water
632        } else {
633            (water_noise.abs() - 0.1) * 1000.0 // Distance in arbitrary units
634        }
635    }

648    /// Classify biome based on temperature and moisture (Whittaker biome classification)
649    pub fn classify_whittaker_biome(temperature: f32, moisture: f32) -> BiomeType {
650        match (temperature, moisture) {
651            (t, _m) if t < 0.2 => BiomeType::Tundra,
652            (t, m) if t < 0.4 && m < 0.3 => BiomeType::Tundra,
653            (t, m) if t < 0.6 && m < 0.2 => BiomeType::Desert,
654            (t, m) if t > 0.7 && m < 0.4 => BiomeType::Desert,
655            (_t, m) if m > 0.8 => BiomeType::Swamp,
656            (t, m) if t > 0.6 && m > 0.6 => BiomeType::Forest,
657            (t, m) if t > 0.4 && m > 0.4 => BiomeType::Forest,
658            _ => BiomeType::Grassland,
659        }
660    }

715    /// Calculate climate statistics for a region
716    pub fn calculate_climate_stats(
717        climate: &ClimateMap,
718        min_x: f64,
719        max_x: f64,
720        min_z: f64,
721        max_z: f64,
722        samples: u32,
723    ) -> ClimateStats {
724        let mut temperatures = Vec::new();
725        let mut moistures = Vec::new();
726
727        let step_x = (max_x - min_x) / samples as f64;
728        let step_z = (max_z - min_z) / samples as f64;
729
730        for i in 0..samples {
731            for j in 0..samples {
732                let x = min_x + i as f64 * step_x;
733                let z = min_z + j as f64 * step_z;
734                let height = climate.estimate_height(x, z);
735
736                let (temperature, moisture) = climate.sample_climate(x, z, height);
737                temperatures.push(temperature);
738                moistures.push(moisture);
739            }
740        }
741
742        ClimateStats {
743            temperature_min: temperatures.iter().copied().fold(f32::INFINITY, f32::min),
744            temperature_max: temperatures
745                .iter()
746                .copied()
747                .fold(f32::NEG_INFINITY, f32::max),
748            temperature_avg: temperatures.iter().sum::<f32>() / temperatures.len() as f32,
749            moisture_min: moistures.iter().copied().fold(f32::INFINITY, f32::min),
750            moisture_max: moistures.iter().copied().fold(f32::NEG_INFINITY, f32::max),
751            moisture_avg: moistures.iter().sum::<f32>() / moistures.len() as f32,
752        }
753    }
