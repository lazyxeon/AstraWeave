1//! Phase 1.5 (Terrain Material System Campaign) — heightmap-driven
2//! multi-biome generation.
3//!
4//! Maps vertex world-space elevation to normalized 8-slot biome weights
5//! that plug into the Phase 1 forward-lit splat pipeline. The
6//! `terrain_primary_biome` field becomes a climate bias parameter: the
7//! same heightmap produces different biome distributions under different
8//! climates.
9//!
10//! Slot order matches the editor's `TerrainVertex.biome_weights_0/1` packing
11//! and the `BiomeType` ordering in this crate:
12//!
13//! | Slot | Biome      |
14//! |------|------------|
15//! | 0    | Grassland  |
16//! | 1    | Desert     |
17//! | 2    | Forest     |
18//! | 3    | Mountain   |
19//! | 4    | Tundra     |
20//! | 5    | Swamp      |
21//! | 6    | Beach      |
22//! | 7    | River      |
23//!
24//! Slots 0-3 populate `biome_weights_0`, slots 4-7 populate `biome_weights_1`.
25
26/// Sea level Y in world space. Matches `WaterRenderer`'s hardcoded water
27/// plane at Y=2.0 per the water system audit
28/// (`docs/audits/water_system_architecture_2026-04-20.md` §3.1). Phase 1.5
29/// uses this as the pivot for elevation-to-biome mapping; Beach biome sits
30/// just above this value for climates that include a coastal band.
31pub const SEA_LEVEL: f32 = 2.0;
32
33/// Climate bias. Selects which elevation→biome mapping is used during
34/// chunk generation. The `terrain_primary_biome` string is parsed into a
35/// `ClimateBias` via [`ClimateBias::from_primary_biome_str`]; each climate
36/// has its own set of biome bands keyed to elevation relative to sea level.
37///
38/// Replaces the prior interpretation of `terrain_primary_biome` as a
39/// single-biome selector.
40#[derive(Copy, Clone, Debug, PartialEq, Eq)]
41pub enum ClimateBias {
42    /// Temperate: Beach → Grassland → Forest → Mountain.
43    Temperate,
44    /// Cold: Beach → Tundra → Mountain (no Forest, no Grassland).
45    Cold,
46    /// Arid: Beach → Desert → Mountain (no Forest, no Grassland).
47    Arid,
48    /// Tropical: Beach → Forest (heavy) → Mountain (warm, wet).
49    Tropical,
50    /// Wetland: Beach → Swamp → Grassland → Forest → Mountain (low focus).
51    Wetland,
52    /// Highland: Grassland (low) → Mountain (heavy). No Beach.
53    Highland,
54}
55
56impl ClimateBias {
57    /// Parse from the `terrain_primary_biome` string for backward compat.
58    /// Maps each biome string to the most appropriate climate bias.
59    ///
60    /// Unknown, empty, or "grassland"/"beach"/"river" strings default to
61    /// [`ClimateBias::Temperate`].
62    pub fn from_primary_biome_str(s: &str) -> Self {
63        match s.to_lowercase().as_str() {
64            "tundra" => ClimateBias::Cold,
65            "desert" => ClimateBias::Arid,
66            "forest" => ClimateBias::Tropical,
67            "swamp" => ClimateBias::Wetland,
68            "mountain" => ClimateBias::Highland,
69            // "grassland" | "beach" | "river" | "" | unknown → Temperate
70            _ => ClimateBias::Temperate,
71        }
72    }
73
74    /// Bands that define this climate's elevation → biome mapping.
75    /// Each band contributes to a single slot; weights are summed then
76    /// normalized in [`elevation_to_biome_weights`].
77    fn bands(self) -> &'static [BiomeBand] {
78        match self {
79            ClimateBias::Temperate => TEMPERATE_BANDS,
80            ClimateBias::Cold => COLD_BANDS,
81            ClimateBias::Arid => ARID_BANDS,
82            ClimateBias::Tropical => TROPICAL_BANDS,
83            ClimateBias::Wetland => WETLAND_BANDS,
84            ClimateBias::Highland => HIGHLAND_BANDS,
85        }
86    }
87
88    /// Slot to use when no band fires (e.g. extreme elevations outside
89    /// all configured bands). Keeps the output normalized.
90    fn fallback_slot(self) -> usize {
91        match self {
92            ClimateBias::Cold => 4,     // Tundra
93            ClimateBias::Highland => 3, // Mountain
94            _ => 6,                      // Beach (sea-level default)
95        }
96    }
97}

99/// Shape of a biome band along the elevation axis.
100#[derive(Copy, Clone, Debug)]
101enum BandShape {
102    /// Triangular pulse with smoothstep falloff. Peaks at `peak` with
103    /// weight 1.0; decays to zero at `peak ± width`.
104    Pulse { peak: f32, width: f32 },
105    /// High-pass. Zero at `start`, ramps smoothly to 1.0 at
106    /// `start + ramp`, then plateaus. Used for mountain-type biomes
107    /// that should dominate at any elevation above a threshold.
108    HighPass { start: f32, ramp: f32 },
109}
110
111/// A single band contributing to one biome slot.
112#[derive(Copy, Clone, Debug)]
113struct BiomeBand {
114    slot: usize,
115    shape: BandShape,
116}
117

178/// Smoothstep function `3t² - 2t³` clamped to `[0, 1]`.
179fn smoothstep(t: f32) -> f32 {
180    let t = t.clamp(0.0, 1.0);
181    t * t * (3.0 - 2.0 * t)
182}
183
184fn evaluate_band(rel_y: f32, shape: &BandShape) -> f32 {
185    match *shape {
186        BandShape::Pulse { peak, width } => {
187            if width <= 0.0 {
188                return 0.0;
189            }
190            let d = (rel_y - peak).abs();
191            if d >= width {
192                0.0
193            } else {
194                smoothstep(1.0 - d / width)
195            }
196        }
197        BandShape::HighPass { start, ramp } => {
198            if ramp <= 0.0 {
199                return if rel_y >= start { 1.0 } else { 0.0 };
200            }
201            if rel_y <= start {
202                0.0
203            } else if rel_y >= start + ramp {
204                1.0
205            } else {
206                smoothstep((rel_y - start) / ramp)
207            }
208        }
209    }
210}

212/// World-elevation-to-biome-weights mapping.
213///
214/// Returns 8 weights summing to ~1.0 in canonical slot order (see module
215/// docs). The mapping is climate-bias-dependent: each climate has its own
216/// elevation → biome curves with smooth smoothstep-falloff transitions at
217/// band boundaries so adjacent vertices produce blendable splat weights,
218/// not hard cutoffs.
219///
220/// - `world_y`: vertex Y in world space.
221/// - `sea_level`: Y coordinate of sea level (use [`SEA_LEVEL`] to match
222///   the water system's hardcoded plane).
223/// - `climate`: which elevation → biome mapping to apply.
224///
225/// If no configured band fires at the given elevation (e.g. far below sea
226/// level or far above any mountain plateau), the climate's fallback slot
227/// receives weight 1.0 so the output is always normalized.
228pub fn elevation_to_biome_weights(
229    world_y: f32,
230    sea_level: f32,
231    climate: ClimateBias,
232) -> [f32; 8] {
233    let rel = world_y - sea_level;
234    let mut weights = [0.0f32; 8];
235
236    for band in climate.bands() {
237        let w = evaluate_band(rel, &band.shape);
238        if w > 0.0 {
239            weights[band.slot] += w;
240        }
241    }
242
243    let total: f32 = weights.iter().sum();
244    if total > 1e-4 {
245        let inv = 1.0 / total;
246        for w in &mut weights {
247            *w *= inv;
248        }
249    } else {
250        weights[climate.fallback_slot()] = 1.0;
251    }
252
253    weights
254}

262    /// Test 1: sum-to-one invariant. For every climate across a wide
263    /// elevation sweep, output weights sum to within 0.001 of 1.0.
264    #[test]
265    fn weights_sum_to_one_across_elevation_sweep() {
266        let climates = [
267            ClimateBias::Temperate,
268            ClimateBias::Cold,
269            ClimateBias::Arid,
270            ClimateBias::Tropical,
271            ClimateBias::Wetland,
272            ClimateBias::Highland,
273        ];
274        let sea_level = SEA_LEVEL;
275
276        for climate in climates {
277            let mut rel = -10.0;
278            while rel <= 200.0 {
279                let world_y = sea_level + rel;
280                let w = elevation_to_biome_weights(world_y, sea_level, climate);
281                let sum: f32 = w.iter().sum();
282                assert!(
283                    (sum - 1.0).abs() < EPS,
284                    "climate {:?} at rel {}: sum = {}, weights = {:?}",
285                    climate,
286                    rel,
287                    sum,
288                    w
289                );
290                // Every individual weight is in [0, 1].
291                for (i, &v) in w.iter().enumerate() {
292                    assert!(
293                        (0.0..=1.0 + EPS).contains(&v),
294                        "climate {:?} at rel {}: slot {} = {}",
295                        climate,
296                        rel,
297                        i,
298                        v
299                    );
300                }
301                rel += 1.0;
302            }
303        }
304    }

431    fn dominant(w: &[f32; 8]) -> (usize, f32) {
432        let mut best = (0, w[0]);
433        for (i, &v) in w.iter().enumerate().skip(1) {
434            if v > best.1 {
435                best = (i, v);
436            }
437        }
438        best
