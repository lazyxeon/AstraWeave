1//! Phase 1.6-F.4.B.3.D.4: scattered-convolution biome blending.
2//!
3//! Softens the per-vertex hard biome assignments produced by D.3 so the
4//! world reads as continuous while preserving discrete biome identity at
5//! each vertex. Reference: noiseposti.ng "Fast Biome Blending Without
6//! Squareness"
7//! (<https://noiseposti.ng/posts/2021-03-13-Fast-Biome-Blending-Without-Squareness.html>).
8//!
9//! ## Algorithm
10//!
11//! For each vertex at world position `(x, z)` with bootstrap elevation `e`:
12//!
13//! 1. Generate `N` jittered sample positions within radius `r` of `(x, z)`.
14//!    Jitter is a deterministic hash of quantized world position + sample
15//!    index — same vertex always produces same sample positions.
16//! 2. At each sample, look up the biome via D.2's `lookup_biome` and
17//!    retrieve its `BiomeParameters` via D.3's `BiomeParameters::for_biome`.
18//! 3. Compute distance-weighted blend of the *numeric* parameters:
19//!    `mountains_amplitude` and `scatter_density`. Linear falloff weight.
20//! 4. The vertex's `BiomeId` is the **dominant** biome (highest summed
21//!    weight). Discrete fields (`scatter_species_set`,
22//!    `surface_color_palette`) and unwired fields (`ridge_strength`,
23//!    `runevision_config`, `erosion_preset`) take the dominant biome's
24//!    value — they don't blend.
25//!
26//! ## Why blend parameters, not biome IDs
27//!
28//! "60% TemperateDeciduous + 40% TemperateGrassland" isn't a category that
29//! exists. Blending categorical biome IDs would require defining new
30//! intermediate biomes for every pair, exploding the taxonomy. The
31//! noiseposti.ng algorithm's insight is that blending the *continuous
32//! parameter values* preserves smooth terrain transitions while keeping
33//! the per-vertex biome assignment discrete (and therefore tractable for
34//! gameplay logic, scatter species, surface coloring).
35//!
36//! ## Why this tightens continuity tolerances
37//!
38//! Per-vertex hard assignment (D.3) makes adjacent vertices' parameters
39//! flip discontinuously at biome boundaries — `mountains_amplitude` jumps
40//! from e.g. 0.8 (TemperateGrassland) to 1.2 (TemperateDeciduous). At
41//! chunk boundaries this divergence is amplified by the f32 precision
42//! discrepancy across independently-generated halos (the pre-existing
43//! 47.4 WU grassland divergence). D.4's blending makes parameter values
44//! transition smoothly across biome boundaries, eliminating the flip
45//! discontinuity. Tolerances raised in D.3 (grassland 20→150 WU, mountain
46//! 10→200 WU) should be tightenable back down toward F.4.B.2.G's 20 WU
47//! range; D.4 verification asserts targets of grassland ≤50 WU and
48//! mountain ≤100 WU.
49//!
50//! ## Determinism + position quantization
51//!
52//! Jitter offsets are computed from a Wang-style hash of the vertex's
53//! world position (quantized to 1/1024 WU = ~1mm grid) plus the sample
54//! index. Quantization is critical: f32 floating-point arithmetic across
55//! adjacent chunks' independently-derived halo coordinates can produce
56//! tiny bit-level differences at the same logical world position. Snapping
57//! to a 1mm grid before hashing ensures both chunks produce identical
58//! jitter offsets at shared-edge vertices, preserving the
59//! shared-edge-invariance property.
60

66/// Phase 1.6-F.4.B.3.D.4: configuration for scattered-convolution biome
67/// blending.
68///
69/// Defaults: 6 samples (mid-range of §1.D.4's tunable 4-9), radius 48 WU
70/// (mid-range of suggested 32-64 WU; D.6 Andrew-gate informs production
71/// value). Seed offset 31 keeps jitter decoupled from terrain noise seeds.
72#[derive(Debug, Clone, Copy, PartialEq, Serialize, Deserialize)]
73pub struct BiomeParamBlendConfig {
74    /// Number of jittered sample positions per vertex. Higher = smoother
75    /// transitions but more cost. Range 4-9; default 6.
76    pub sample_count: u32,
77    /// Maximum offset of jittered samples from the vertex (in world units).
78    /// Should span typical biome-boundary scale; too small produces no
79    /// blending, too large smears character across distant biomes.
80    /// Default 48 WU.
81    pub radius: f32,
82    /// Seed offset added to world seed for jitter determinism. Decouples
83    /// jitter pattern from terrain noise. Default 31.
84    pub seed_offset: u32,
85}
86
87impl Default for BiomeParamBlendConfig {
88    fn default() -> Self {
89        Self {
90            sample_count: 6,
91            radius: 48.0,
92            seed_offset: 31,
93        }
94    }
95}

97/// Phase 1.6-F.4.B.3.D.4: result of biome blending at a single vertex.
98///
99/// Contains:
100/// - `dominant_biome`: the discrete `BiomeId` driving surface color,
101///   scatter species, gameplay logic.
102/// - Blended numeric fields: `mountains_amplitude`, `scatter_density`.
103/// - Unblended fields: `ridge_strength`, `runevision_config`,
104///   `scatter_species_set`, `surface_color_palette`, `erosion_preset` —
105///   all take the dominant biome's value (not blended) per §1.2 plan.
106#[derive(Debug, Clone)]
107pub struct BlendedBiomeParams {
108    /// Dominant biome at this vertex (highest summed sample weight).
109    pub dominant_biome: BiomeId,
110    /// Distance-weighted blend of `BiomeParameters::mountains_amplitude`.
111    pub mountains_amplitude: f64,
112    /// Distance-weighted blend of `BiomeParameters::scatter_density`.
113    pub scatter_density: f64,
114    /// Forwarded from the dominant biome's `BiomeParameters` (no blending).
115    /// Per §1.2: unwired/discrete fields take the dominant value.
116    pub dominant_params: BiomeParameters,
117}

119/// Phase 1.6-F.4.B.3.D.4: position-quantization granularity for jitter
120/// determinism. 1/1024 WU = ~1mm. Snapping to this grid before hashing
121/// ensures adjacent chunks' halos produce identical jitter offsets at
122/// shared-edge vertices despite f32 precision discrepancies.
123const JITTER_QUANTIZATION: f32 = 1024.0;
124
125/// Phase 1.6-F.4.B.3.D.4: jitter offset for a given vertex + sample index.
126///
127/// Returns `(dx, dz)` in `[-1, 1]` (multiply by `radius` to get world-unit
128/// offset). Deterministic per `(world_x, world_z, sample_idx, seed)`.
129///
130/// Uses Wang-style integer hash on quantized world position. Position is
131/// snapped to 1mm grid before hashing so adjacent chunks computing
132/// jitter at the SAME logical world coordinate produce identical results
133/// despite f32 precision.
134#[inline]
135fn jitter_offset(world_x: f32, world_z: f32, sample_idx: u32, seed: u64) -> (f32, f32) {
136    let qx = (world_x * JITTER_QUANTIZATION).round() as i64 as u64;
137    let qz = (world_z * JITTER_QUANTIZATION).round() as i64 as u64;
138
139    let mut h = seed.wrapping_mul(0x9E37_79B9_7F4A_7C15);
140    h ^= h >> 32;
141    h = h.wrapping_add(qx).wrapping_mul(0x85EB_CA6B_E11E_CC0D);
142    h ^= h >> 32;
143    h = h.wrapping_add(qz).wrapping_mul(0xC2B2_AE35_5D9F_3B25);
144    h ^= h >> 32;
145    h = h
146        .wrapping_add(sample_idx as u64)
147        .wrapping_mul(0x9E37_79B9_7F4A_7C15);
148    h ^= h >> 32;
149
150    let u = ((h & 0xFFFF_FFFF) as f32) / (u32::MAX as f32);
151    let v = ((h >> 32) as f32) / (u32::MAX as f32);
152    (u * 2.0 - 1.0, v * 2.0 - 1.0)
153}

155/// Phase 1.6-F.4.B.3.D.4: blend biome parameters at a world position.
156///
157/// Per the noiseposti.ng algorithm:
158/// 1. Generate `config.sample_count` jittered sample positions within
159///    `config.radius` of `(world_x, world_z)`.
160/// 2. At each sample, look up biome (climate field × Whittaker) and
161///    retrieve `BiomeParameters`.
162/// 3. Linear distance-weighted blend of numeric parameters.
163/// 4. Dominant biome = highest summed weight.
164///
165/// `bootstrap_elevation` is used as the elevation input for ALL jittered
166/// climate samples. This is an approximation — strictly, the elevation at
167/// each jittered position differs slightly. Per §1.4 plan: "The bootstrap
168/// elevation used for climate sampling can remain the un-blended initial
169/// height — climate-field shape doesn't need to be smoothed at vertex
170/// level (it's already smooth at world scale). Only the *parameter values*
171/// derived from biome IDs need blending." Computing per-jittered-sample
172/// elevation would require N extra `sample_height` calls per vertex (a
173/// significant cost). The approximation is bounded because the elevation
174/// overlay layer (Alpine/SnowCap) only fires at high altitudes where
175/// nearby vertices have similar elevations.
176pub fn blend_biome_parameters(
177    world_x: f32,
178    world_z: f32,
179    bootstrap_elevation: f32,
180    climate_map: &ClimateMap,
181    config: &BiomeParamBlendConfig,
182) -> BlendedBiomeParams {
183    debug_assert!(config.sample_count > 0, "sample_count must be ≥ 1");
184
185    let seed = config.seed_offset as u64;
186
187    // Accumulators.
188    let mut weight_per_biome = [0.0f32; 19];
189    let mut sum_amp: f64 = 0.0;
190    let mut sum_density: f64 = 0.0;
191    let mut sum_w: f32 = 0.0;
192
193    for i in 0..config.sample_count {
194        let (jx, jz) = jitter_offset(world_x, world_z, i, seed);
195        let sample_x = world_x + jx * config.radius;
196        let sample_z = world_z + jz * config.radius;
197
198        let climate = climate_map.sample(sample_x as f64, sample_z as f64, bootstrap_elevation);
199        let biome = lookup_biome(
200            climate.temperature_c,
201            climate.moisture_mm,
202            bootstrap_elevation,
203        );
204        let params = BiomeParameters::for_biome(biome);
205
206        // Linear distance-weight: 1.0 at center, 0.0 at radius edge.
207        // Clamp to small floor to avoid division-by-zero when all samples
208        // happen to land at the maximum jitter offset.
209        let dist_norm = ((jx * jx + jz * jz).sqrt()).min(1.0);
210        let weight = (1.0 - dist_norm).max(0.001);
211
212        sum_amp += params.mountains_amplitude * weight as f64;
213        sum_density += params.scatter_density * weight as f64;
214        sum_w += weight;
215
216        let biome_idx = biome_id_to_index(biome);
217        weight_per_biome[biome_idx] += weight;
218    }
219
220    // Dominant biome = highest summed weight.
221    let mut dominant_idx = 0usize;
222    let mut max_w = weight_per_biome[0];
223    for (idx, &w) in weight_per_biome.iter().enumerate().skip(1) {
224        if w > max_w {
225            max_w = w;
226            dominant_idx = idx;
227        }
228    }
229    let dominant_biome = index_to_biome_id(dominant_idx);
230
231    BlendedBiomeParams {
232        dominant_biome,
233        mountains_amplitude: sum_amp / sum_w as f64,
234        scatter_density: sum_density / sum_w as f64,
235        dominant_params: BiomeParameters::for_biome(dominant_biome),
236    }
237}

239/// Phase 1.6-F.4.B.3.D.4: map `BiomeId` to a stable `[0, 19)` index for
240/// the per-biome weight accumulator. Inverse of [`index_to_biome_id`].
241#[inline]
242fn biome_id_to_index(biome: BiomeId) -> usize {
243    match biome {
244        BiomeId::TropicalRainforest => 0,
245        BiomeId::TropicalSeasonalForest => 1,
246        BiomeId::Savanna => 2,
247        BiomeId::SubtropicalDesert => 3,
248        BiomeId::TemperateRainforest => 4,
249        BiomeId::TemperateDeciduousForest => 5,
250        BiomeId::TemperateGrassland => 6,
251        BiomeId::ColdDesert => 7,
252        BiomeId::BorealForest => 8,
253        BiomeId::Tundra => 9,
254        BiomeId::Alpine => 10,
255        BiomeId::Ocean => 11,
256        BiomeId::Coast => 12,
257        BiomeId::Beach => 13,
258        BiomeId::River => 14,
259        BiomeId::Wetland => 15,
260        BiomeId::MountainRocky => 16,
261        BiomeId::SnowCap => 17,
262        BiomeId::Scree => 18,
263    }
264}

266/// Phase 1.6-F.4.B.3.D.4: inverse of [`biome_id_to_index`]. Total over
267/// `[0, 19)`; out-of-range indices fall back to TemperateGrassland (a
268/// safe-default biome that won't crash downstream rendering).
269#[inline]
270fn index_to_biome_id(idx: usize) -> BiomeId {
271    match idx {
272        0 => BiomeId::TropicalRainforest,
273        1 => BiomeId::TropicalSeasonalForest,
274        2 => BiomeId::Savanna,
275        3 => BiomeId::SubtropicalDesert,
276        4 => BiomeId::TemperateRainforest,
277        5 => BiomeId::TemperateDeciduousForest,
278        6 => BiomeId::TemperateGrassland,
279        7 => BiomeId::ColdDesert,
280        8 => BiomeId::BorealForest,
281        9 => BiomeId::Tundra,
282        10 => BiomeId::Alpine,
283        11 => BiomeId::Ocean,
284        12 => BiomeId::Coast,
285        13 => BiomeId::Beach,
286        14 => BiomeId::River,
287        15 => BiomeId::Wetland,
288        16 => BiomeId::MountainRocky,
289        17 => BiomeId::SnowCap,
290        18 => BiomeId::Scree,
291        _ => BiomeId::TemperateGrassland,
292    }

399    #[test]
400    fn phase_1_6_f4_b_3_d_4_gradient_smoothness() {
401        // Along a climate gradient (latitude effect), `mountains_amplitude`
402        // varies smoothly with bounded max derivative. This is the
403        // anti-discontinuity property — the whole point of D.4.
404        let climate = make_climate(12345);
405        let cfg = BiomeParamBlendConfig::default();
406
407        // Sample 100 points along a 4000 WU latitudinal sweep (reaches
408        // varied climate per archetype's latitude_temperature_drop).
409        const N: usize = 100;
410        let mut amps = Vec::with_capacity(N);
411        for i in 0..N {
412            let z = -2000.0 + (i as f32 / (N - 1) as f32) * 4000.0;
413            let r = blend_biome_parameters(0.0, z, 50.0, &climate, &cfg);
414            amps.push(r.mountains_amplitude);
415        }
416
417        // Compute per-step deltas; max delta should be bounded by the
418        // largest single-sample contribution / radius. Without blending,
419        // a discontinuity would produce a delta equal to the full
420        // amplitude difference between adjacent biomes (e.g., 0.8 → 1.2 = 0.4).
421        // With blending, max delta should be significantly smaller because
422        // each sample's contribution is fractional.
423        let max_delta = amps
424            .windows(2)
425            .map(|w| (w[1] - w[0]).abs())
426            .fold(0.0f64, f64::max);
427
428        // Bound: blending should keep per-step delta below the unblended
429        // hard-flip discontinuity. Without blending, adjacent vertices at
430        // a TemperateGrassland (0.8) ↔ BorealForest (1.5) boundary would
431        // produce a delta of 0.7. With blending at radius 48 WU and 6
432        // samples, the overlapping blend kernels reduce that to ~0.35-0.45
433        // depending on which biomes are involved at the sweep's boundary
434        // intersections. Threshold 0.5 leaves headroom against per-seed
435        // variance while still rejecting "blending entirely broken"
436        // failures (where deltas would equal the full hard-flip values
437        // 0.7+ for major boundary types).
438        assert!(
439            max_delta < 0.5,
440            "gradient should be smooth: max per-step amplitude delta {} > 0.5 \
441             (blending may not be working)",
442            max_delta
443        );
444    }

534        // archetype at world center should produce tropical-family
535        // dominants, not temperate.
536        let warm = WorldArchetype {
537            temperature_mean_c: 26.0,
538            temperature_variance_c: 5.0,
539            latitude_temperature_drop_c: 3.0,
540            moisture_mean_mm: 2200.0,
541            moisture_variance_mm: 800.0,
542            continentalness_mean: 0.4,
543            continentalness_variance: 0.2,
544        };
545        let mut config = ClimateConfig::default();
546        config.archetype = warm;
547        let climate = ClimateMap::new(&config, 67890);
548
549        let cfg = BiomeParamBlendConfig::default();
550        let mut tropical_count = 0;
551        let mut temperate_count = 0;
552        for i in 0..50 {
553            let x = (i as f32) * 137.0;
554            let z = (i as f32) * 211.0;
555            let r = blend_biome_parameters(x, z, 30.0, &climate, &cfg);
556            if matches!(
557                r.dominant_biome,
558                BiomeId::TropicalRainforest
559                    | BiomeId::TropicalSeasonalForest
560                    | BiomeId::Savanna
561                    | BiomeId::SubtropicalDesert
562            ) {
563                tropical_count += 1;
564            }
565            if matches!(
566                r.dominant_biome,
567                BiomeId::TemperateGrassland
568                    | BiomeId::TemperateDeciduousForest
569                    | BiomeId::TemperateRainforest
570                    | BiomeId::ColdDesert
571            ) {
572                temperate_count += 1;
573            }
574        }
575        assert!(
576            tropical_count > temperate_count,
577            "warm archetype should produce more tropical dominants ({}) than temperate ({})",
578            tropical_count,
579            temperate_count
580        );
581    }
