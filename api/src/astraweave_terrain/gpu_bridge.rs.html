1//! GPU acceleration bridge for terrain generation.
2//!
3//! Defines the [`TerrainGpuAccelerator`] trait that allows the terrain crate to
4//! dispatch expensive heightmap operations (erosion, noise generation) to GPU
5//! compute pipelines without depending on `wgpu` or the render crate directly.
6//!
7//! # Architecture
8//!
9//! ```text
10//! astraweave-terrain (defines trait)
11//!         ↑
12//!         │ implements
13//!         │
14//! astraweave-render (GpuErosionPipeline, GpuNoisePipeline)
15//! ```
16//!
17//! The terrain chunk generator checks for GPU accelerator availability at
18//! runtime, falling back to CPU noise/erosion when no GPU backend is registered.
19//!
20//! # Usage
21//!
22//! ```rust,ignore
23//! use astraweave_terrain::gpu_bridge::{TerrainGpuAccelerator, GpuHeightmapRequest};
24//!
25//! fn generate_chunk(accel: Option<&dyn TerrainGpuAccelerator>, /* ... */) {
26//!     let request = GpuHeightmapRequest {
27//!         width: 256, height: 256,
28//!         world_origin: [0.0, 0.0],
29//!         cell_size: 1.0,
30//!     };
31//!     if let Some(gpu) = accel {
32//!         if let Ok(heightmap) = gpu.generate_noise(&request, &default_noise_params()) {
33//!             // Use GPU-generated heightmap
34//!         }
35//!     }
36//! }
37//! ```

39/// Request describing a heightmap region to generate or erode.
40#[derive(Debug, Clone)]
41pub struct GpuHeightmapRequest {
42    /// Grid width in cells.
43    pub width: u32,
44    /// Grid height in cells.
45    pub height: u32,
46    /// World-space origin of the heightmap (x, z).
47    pub world_origin: [f32; 2],
48    /// World-space size of each cell (meters).
49    pub cell_size: f32,
50}
51
52/// Noise generation parameters for the GPU backend.
53#[derive(Debug, Clone)]
54pub struct GpuNoiseRequest {
55    /// Base frequency of the noise (smaller = larger features).
56    pub frequency: f32,
57    /// Output amplitude multiplier.
58    pub amplitude: f32,
59    /// Number of fBM octaves.
60    pub octaves: u32,
61    /// Persistence (amplitude falloff per octave, typically 0.5).
62    pub persistence: f32,
63    /// Lacunarity (frequency multiplier per octave, typically 2.0).
64    pub lacunarity: f32,
65    /// Random seed for deterministic generation.
66    pub seed: u32,
67}
68
69impl Default for GpuNoiseRequest {
70    fn default() -> Self {
71        Self {
72            frequency: 0.005,
73            amplitude: 50.0,
74            octaves: 6,
75            persistence: 0.5,
76            lacunarity: 2.0,
77            seed: 0,
78        }
79    }
80}

82/// Erosion parameters for the GPU backend.
83#[derive(Debug, Clone)]
84pub struct GpuErosionRequest {
85    /// Number of simulation iterations.
86    pub iterations: u32,
87    /// Simulation timestep (seconds).
88    pub dt: f32,
89    /// Rain rate (water units per cell per second).
90    pub rain_rate: f32,
91    /// Sediment carrying capacity.
92    pub sediment_capacity: f32,
93    /// Dissolution rate.
94    pub dissolution_rate: f32,
95    /// Deposition rate.
96    pub deposition_rate: f32,
97    /// Evaporation rate per second.
98    pub evaporation_rate: f32,
99}

115/// Result of a GPU heightmap operation.
116#[derive(Debug, Clone)]
117pub struct GpuHeightmapResult {
118    /// Row-major heightmap data, `width × height` entries.
119    pub heights: Vec<f32>,
120    /// Grid width.
121    pub width: u32,
122    /// Grid height.
123    pub height: u32,
124}

126/// Trait for dispatching terrain compute work to the GPU.
127///
128/// Implementors live in the render crate and wrap `GpuErosionPipeline` and
129/// `GpuNoisePipeline`. The terrain crate calls these through a trait object
130/// (`&dyn TerrainGpuAccelerator`) with no compile-time dependency on `wgpu`.
131///
132/// All methods are synchronous from the caller's perspective — the
133/// implementor is responsible for GPU dispatch, readback, and fence waiting.
134pub trait TerrainGpuAccelerator: Send + Sync {
135    /// Generate a noise heightmap on the GPU.
136    ///
137    /// Returns `Ok(result)` with the generated heights, or `Err` if the GPU
138    /// pipeline is unavailable or dispatch fails.
139    fn generate_noise(
140        &self,
141        request: &GpuHeightmapRequest,
142        params: &GpuNoiseRequest,
143    ) -> anyhow::Result<GpuHeightmapResult>;
144
145    /// Apply hydraulic erosion to an existing heightmap on the GPU.
146    ///
147    /// `input_heights` must have exactly `request.width * request.height` entries.
148    /// Returns the eroded heightmap.
149    fn erode_heightmap(
150        &self,
151        request: &GpuHeightmapRequest,
152        params: &GpuErosionRequest,
153        input_heights: &[f32],
154    ) -> anyhow::Result<GpuHeightmapResult>;
155
156    /// Check whether this accelerator is operational (GPU device + pipelines ready).
157    fn is_available(&self) -> bool;
158
159    /// Human-readable name of the GPU backend (e.g. "wgpu/Vulkan", "wgpu/DX12").
160    fn backend_name(&self) -> &str;
161}
