1//! Phase 1.6-F.4.B.3.D.2: Whittaker biome lookup table.
2//!
3//! Pure-function `(temperature, moisture, elevation) → BiomeId` mapping that
4//! drives the climate-field architecture's biome assignment. No randomness,
5//! no global state — same inputs always produce the same `BiomeId`.
6//!
7//! ## Design
8//!
9//! Real-world Whittaker biome diagrams plot vegetation against (mean annual
10//! temperature, mean annual precipitation). This module encodes the
11//! canonical placements as a deterministic lookup with three layers:
12//!
13//! 1. **Aquatic layer** (elevation-driven): below sea level → Ocean / Coast;
14//!    just above → Beach. Driven by the existing [`crate::SEA_LEVEL`]
15//!    constant for compatibility with the legacy water plane.
16//! 2. **Elevation overlay** (high-altitude exposure): above ~3500m → SnowCap;
17//!    above ~3000m → Alpine; above ~2500m with low moisture → Scree.
18//!    Overrides terrestrial Whittaker classification because at these
19//!    elevations vegetation can no longer establish regardless of climate.
20//! 3. **Terrestrial Whittaker** (temperature × moisture): the 11 standard
21//!    terrestrial biomes, plus a `Wetland` override for low-elevation
22//!    high-moisture regions.
23//!
24//! ## Polygon-vs-contract distinction
25//!
26//! The `BiomeId` enum is the **contract**: every `(temp, moisture, elevation)`
27//! tuple maps to exactly one variant. The threshold constants in this module
28//! are the **implementation**: they're tuned to match canonical Whittaker
29//! placements, not pinned by tests. If a future tuning pass moves the
30//! boundary between TemperateGrassland and TemperateDeciduousForest by
31//! 100mm of moisture, that's a polygon adjustment, not a contract break.
32//!
33//! ## Per-vertex hard assignment, blending later
34//!
35//! `lookup_biome` assigns a single `BiomeId` per vertex. Smooth transitions
36//! across biome boundaries are not handled here — that's F.4.B.3.D.4
37//! (scattered-convolution biome blending). D.4 will sample multiple jittered
38//! positions around each vertex and blend the resulting per-`BiomeId`
39//! parameters; D.2 just provides the dominant-biome lookup.
40

43/// Phase 1.6-F.4.B.3.D.2: the fixed AstraWeave biome taxonomy.
44///
45/// 11 terrestrial + 5 aquatic + 3 elevation overlay = 19 variants. Worlds
46/// get all biomes; `WorldArchetype` (D.1) determines which are common,
47/// rare, or absent at a given world's climate envelope.
48///
49/// This taxonomy is the **contract** between the climate field (D.1), the
50/// per-biome parameter system (D.3), and the biome blending system (D.4).
51/// Variant set is intentionally fixed; users do not add or remove biomes.
52/// New variants require a coordinated change across D.2/D.3/D.4 plus a §10
53/// deviation log entry.
54#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
55pub enum BiomeId {
56    // === Terrestrial (11) ===
57    /// Hot + very wet: dense closed-canopy tropical forest.
58    TropicalRainforest,
59    /// Hot + moderately wet, with a dry season: open canopy with grass understory.
60    TropicalSeasonalForest,
61    /// Hot + dry-to-moderate moisture: grassland with scattered drought-tolerant trees.
62    Savanna,
63    /// Hot + arid: cactus, sparse hardy shrub, exposed substrate.
64    SubtropicalDesert,
65    /// Cool + very wet: temperate-zone rainforest (Pacific NW analog).
66    TemperateRainforest,
67    /// Cool + moderate moisture: broadleaf deciduous forest (Appalachian analog).
68    TemperateDeciduousForest,
69    /// Cool + low-to-moderate moisture: prairie / steppe.
70    TemperateGrassland,
71    /// Cool + arid: high-altitude or rain-shadow desert (e.g., Patagonian, Gobi).
72    ColdDesert,
73    /// Cold + moderate moisture: coniferous taiga (boreal forest).
74    BorealForest,
75    /// Very cold + low moisture: dwarf shrub, lichen, permafrost soil.
76    Tundra,
77    /// Subalpine: stunted vegetation transitioning toward bare rock,
78    /// elevation-overlay biome between BorealForest/Tundra and SnowCap.
79    /// Distinct from MountainRocky (which is bare rock without vegetation).
80    Alpine,
81
82    // === Aquatic (5) ===
83    /// Open ocean: deep below sea level, away from any landmass.
84    Ocean,
85    /// Coast: shallow water just below sea level, transitioning to land.
86    Coast,
87    /// Beach: sand/gravel shoreline just above sea level.
88    Beach,
89    /// River: linear water body cutting through terrestrial biomes.
90    /// **D.2 status**: variant exists for taxonomy completeness; the
91    /// `lookup_biome` function does not currently produce River from
92    /// `(temp, moisture, elevation)` alone — rivers require hydrological
93    /// flow simulation which is out of F.4.B.3.D scope. Deferred to a
94    /// future hydrology campaign (Water System Rebuild).
95    River,
96    /// Wetland: low-elevation, very-high-moisture region. Marsh, bog, swamp.
97    Wetland,
98
99    // === Elevation overlay (3) ===
100    /// Bare-rock mountain face: vegetation absent, exposed substrate.
101    /// Distinct from Alpine (sparse vegetation) and Scree (loose rock fields).
102    MountainRocky,
103    /// Permanent snow / ice cap. Peaks above the local snowline.
104    SnowCap,
105    /// Loose-rock slope below the snowline. Mid-altitude bare slopes
106    /// where vegetation cannot establish but ice doesn't persist.
107    Scree,
108}

137    /// Whether this variant is one of the terrestrial Whittaker biomes
138    /// (excludes aquatic + elevation overlays). Useful for distribution
139    /// tests that should ignore elevation/water artifacts.
140    pub const fn is_terrestrial(self) -> bool {
141        matches!(
142            self,
143            Self::TropicalRainforest
144                | Self::TropicalSeasonalForest
145                | Self::Savanna
146                | Self::SubtropicalDesert
147                | Self::TemperateRainforest
148                | Self::TemperateDeciduousForest
149                | Self::TemperateGrassland
150                | Self::ColdDesert
151                | Self::BorealForest
152                | Self::Tundra
153                | Self::Alpine
154        )
155    }
156
157    /// Whether this variant is one of the aquatic biomes.
158    pub const fn is_aquatic(self) -> bool {
159        matches!(
160            self,
161            Self::Ocean | Self::Coast | Self::Beach | Self::River | Self::Wetland
162        )
163    }
164
165    /// Whether this variant is an elevation-overlay biome (overrides
166    /// terrestrial Whittaker at altitude).
167    pub const fn is_elevation_overlay(self) -> bool {
168        matches!(self, Self::MountainRocky | Self::SnowCap | Self::Scree)
169    }

174// Tunable to match canonical Whittaker placements + AstraWeave Target B Y range.

177/// Aquatic depth bands relative to `crate::elevation_biome::SEA_LEVEL`.
178/// Sea level is 2.0 WU (= 2.0m at the Target B 1 WU = 1 m convention).
179const OCEAN_DEPTH_THRESHOLD_M: f32 = -3.0; // 3m below sea level → Ocean (deeper than Coast band)
180const BEACH_BAND_HEIGHT_M: f32 = 3.0; // [+0, +3]m above sea level → Beach (when wet)
181const BEACH_MIN_MOISTURE_MM: f32 = 200.0; // dry beaches reclassified as Coast/Coast-fringe
182
183/// Elevation overlay thresholds. Above SnowCap → snow. Above Alpine but
184/// below SnowCap → Alpine. Above Scree but below Alpine + low moisture
185/// → Scree. These are absolute world Y coordinates.
186const SNOWCAP_THRESHOLD_M: f32 = 350.0; // Above ~350m at Target B Y range
187const ALPINE_THRESHOLD_M: f32 = 280.0; // Above ~280m
188const SCREE_THRESHOLD_M: f32 = 220.0; // Above ~220m, dry side
189const SCREE_MAX_MOISTURE_MM: f32 = 600.0;
190const SNOWCAP_MAX_TEMP_C: f32 = 18.0; // Tropical mountains can stay rocky-not-snow at extreme heat
191
192/// Wetland low-elevation override.
193const WETLAND_MAX_ELEVATION_M: f32 = 30.0; // Just above sea level
194const WETLAND_MIN_MOISTURE_MM: f32 = 1500.0;
195const WETLAND_MIN_TEMP_C: f32 = -2.0; // Frozen marshes don't qualify as Wetland (they're Tundra)
196
197/// Whittaker terrestrial polygon thresholds.
198/// These follow standard Whittaker diagram placements but the boundaries
199/// are tunable. See module docs for polygon-vs-contract distinction.
200const TUNDRA_MAX_TEMP_C: f32 = 0.0;
201const BOREAL_MAX_TEMP_C: f32 = 5.0;
202const BOREAL_MIN_MOISTURE_MM: f32 = 200.0;
203
204const COLD_DESERT_MAX_TEMP_C: f32 = 18.0;
205const COLD_DESERT_MAX_MOISTURE_MM: f32 = 300.0;
206
207const TEMPERATE_RAINFOREST_MIN_MOISTURE_MM: f32 = 1500.0;
208const TEMPERATE_RAINFOREST_MAX_TEMP_C: f32 = 18.0;
209
210const TEMPERATE_FOREST_MIN_MOISTURE_MM: f32 = 600.0;
211const TEMPERATE_FOREST_MAX_TEMP_C: f32 = 20.0;
212
213const TEMPERATE_GRASSLAND_MIN_MOISTURE_MM: f32 = 250.0;
214const TEMPERATE_GRASSLAND_MAX_TEMP_C: f32 = 22.0;
215
216const SUBTROPICAL_DESERT_MAX_MOISTURE_MM: f32 = 250.0;
217const SAVANNA_MAX_MOISTURE_MM: f32 = 1000.0;
218const TROPICAL_SEASONAL_MAX_MOISTURE_MM: f32 = 1800.0;
219
220/// Phase 1.6-F.4.B.3.D.2: deterministic biome lookup.
221///
222/// Returns the dominant `BiomeId` at a `(temperature, moisture, elevation)`
223/// tuple. Order of operations:
224///
225/// 1. **Aquatic check** (elevation-driven): Ocean / Coast / Beach for water
226///    bodies and the immediate shoreline.
227/// 2. **Wetland override** (low-elevation, high-moisture): catches marsh /
228///    swamp / bog regions before Whittaker classification overrides them.
229/// 3. **Elevation overlay** (high-altitude exposure): SnowCap / Alpine /
230///    Scree above their respective thresholds.
231/// 4. **Whittaker terrestrial**: the 11 standard biomes by
232///    `(temperature, moisture)` polygon classification.
233///
234/// **Determinism invariant**: same `(temp_c, moisture_mm, elevation_m)`
235/// always returns the same `BiomeId`. No randomness, no global state, no
236/// floating-point branching dependent on hidden state. This is a
237/// load-bearing invariant for the climate-field architecture's
238/// reproducibility.
239///
240/// **Inputs**:
241/// - `temp_c` — temperature in degrees Celsius (typically `[-30, +40]`).
242/// - `moisture_mm` — annual precipitation in mm (typically `[0, 4000]`).
243/// - `elevation_m` — world Y coordinate in meters (1 WU = 1 m at Target B).
244///
245/// Inputs outside the typical ranges are accepted and produce the
246/// nearest-matching variant (e.g., temp = -100°C → Tundra, elevation =
247/// 10000m → SnowCap).
248pub fn lookup_biome(temp_c: f32, moisture_mm: f32, elevation_m: f32) -> BiomeId {
249    let sea_level = crate::elevation_biome::SEA_LEVEL;
250
251    // === 1. Aquatic check (elevation-driven) ===
252    // Sub-sea-level: Ocean (deep) or Coast (shallow).
253    if elevation_m < sea_level + OCEAN_DEPTH_THRESHOLD_M {
254        return BiomeId::Ocean;
255    }
256    if elevation_m < sea_level {
257        return BiomeId::Coast;
258    }
259    // Just-above-sea-level Beach band, but only if moisture is high enough
260    // (dry shorelines transition into desert directly without a Beach phase).
261    if elevation_m < sea_level + BEACH_BAND_HEIGHT_M && moisture_mm >= BEACH_MIN_MOISTURE_MM {
262        return BiomeId::Beach;
263    }
264
265    // === 2. Wetland override (low-elevation, very-high-moisture) ===
266    // Caught BEFORE elevation-overlay because Wetland is a low-elevation
267    // biome; the overlay layer only fires above ~220m.
268    if elevation_m < WETLAND_MAX_ELEVATION_M
269        && moisture_mm >= WETLAND_MIN_MOISTURE_MM
270        && temp_c >= WETLAND_MIN_TEMP_C
271    {
272        return BiomeId::Wetland;
273    }
274
275    // === 3. Elevation overlay (high-altitude exposure) ===
276    // SnowCap: above the snowline AND not extreme-tropical-mountain hot.
277    if elevation_m >= SNOWCAP_THRESHOLD_M && temp_c < SNOWCAP_MAX_TEMP_C {
278        return BiomeId::SnowCap;
279    }
280    // Alpine: above the treeline but below permanent snow.
281    if elevation_m >= ALPINE_THRESHOLD_M {
282        return BiomeId::Alpine;
283    }
284    // Scree: bare-rock fields below the alpine zone, in dry conditions.
285    if elevation_m >= SCREE_THRESHOLD_M && moisture_mm < SCREE_MAX_MOISTURE_MM {
286        return BiomeId::Scree;
287    }
288
289    // === 4. Whittaker terrestrial classification ===
290    classify_whittaker_polygon(temp_c, moisture_mm)
291}

293/// Phase 1.6-F.4.B.3.D.2: deterministic Whittaker terrestrial classification.
294///
295/// Pure (`temp_c`, `moisture_mm`) lookup over the 11 standard terrestrial
296/// biomes. Caller must have already excluded aquatic and elevation-overlay
297/// cases via the threshold checks in `lookup_biome` — calling this directly
298/// for an underwater point or a high mountain produces a meaningless result.
299///
300/// Polygon order is from coldest/driest to warmest/wettest. Each branch is
301/// the first matching region; subsequent branches assume earlier checks
302/// have already filtered out colder/drier cases.
303fn classify_whittaker_polygon(temp_c: f32, moisture_mm: f32) -> BiomeId {
304    // Cold zone: Tundra / Boreal / ColdDesert.
305    if temp_c < TUNDRA_MAX_TEMP_C {
306        // Sub-zero: usually Tundra; with enough moisture and not too cold,
307        // can be BorealForest if temperature is at the warm end of cold.
308        return BiomeId::Tundra;
309    }
310    if temp_c < BOREAL_MAX_TEMP_C {
311        // Cool: BorealForest if moisture present, ColdDesert otherwise.
312        if moisture_mm >= BOREAL_MIN_MOISTURE_MM {
313            return BiomeId::BorealForest;
314        }
315        return BiomeId::ColdDesert;
316    }
317
318    // Cool-temperate zone: ColdDesert / TemperateRainforest /
319    // TemperateDeciduousForest / TemperateGrassland.
320    if temp_c < COLD_DESERT_MAX_TEMP_C {
321        if moisture_mm < COLD_DESERT_MAX_MOISTURE_MM {
322            return BiomeId::ColdDesert;
323        }
324        if moisture_mm >= TEMPERATE_RAINFOREST_MIN_MOISTURE_MM
325            && temp_c < TEMPERATE_RAINFOREST_MAX_TEMP_C
326        {
327            return BiomeId::TemperateRainforest;
328        }
329        if moisture_mm >= TEMPERATE_FOREST_MIN_MOISTURE_MM
330            && temp_c < TEMPERATE_FOREST_MAX_TEMP_C
331        {
332            return BiomeId::TemperateDeciduousForest;
333        }
334        if moisture_mm >= TEMPERATE_GRASSLAND_MIN_MOISTURE_MM {
335            return BiomeId::TemperateGrassland;
336        }
337        // Below grassland minimum moisture in cool-temperate zone: ColdDesert.
338        return BiomeId::ColdDesert;
339    }
340
341    // Warm-temperate zone: ColdDesert / TemperateDeciduousForest /
342    // TemperateGrassland (extends slightly into warm range).
343    if temp_c < TEMPERATE_GRASSLAND_MAX_TEMP_C {
344        if moisture_mm < COLD_DESERT_MAX_MOISTURE_MM {
345            return BiomeId::ColdDesert;
346        }
347        if moisture_mm >= TEMPERATE_FOREST_MIN_MOISTURE_MM
348            && temp_c < TEMPERATE_FOREST_MAX_TEMP_C
349        {
350            return BiomeId::TemperateDeciduousForest;
351        }
352        return BiomeId::TemperateGrassland;
353    }
354
355    // Tropical/subtropical zone: SubtropicalDesert / Savanna /
356    // TropicalSeasonalForest / TropicalRainforest.
357    if moisture_mm < SUBTROPICAL_DESERT_MAX_MOISTURE_MM {
358        return BiomeId::SubtropicalDesert;
359    }
360    if moisture_mm < SAVANNA_MAX_MOISTURE_MM {
361        return BiomeId::Savanna;
362    }
363    if moisture_mm < TROPICAL_SEASONAL_MAX_MOISTURE_MM {
364        return BiomeId::TropicalSeasonalForest;
365    }
366    BiomeId::TropicalRainforest
367}

590    // §1.D.2 verification: per archetype's documented climate distribution,
591    // sample 10K random points and verify the resulting biome distribution
592    // is plausible.
593    //
594    // D.1 ships only the Continental Temperate archetype as production
595    // configuration. Distribution tests for the other five archetypes are
596    // deferred to D.5 (when those archetypes land). For D.2, the
597    // Continental Temperate distribution test is the sole archetype check.
598    // ============================================================

600    /// Deterministic pseudo-random sampler for the distribution test.
601    /// Linear-congruential-like; produces the same sequence per seed for
602    /// reproducible test results.
603    fn det_random(seed: u32, idx: u32) -> f32 {
604        let mut h = seed.wrapping_mul(0x9E37_79B9).wrapping_add(idx);
605        h ^= h >> 16;
606        h = h.wrapping_mul(0x85EB_CA6B);
607        h ^= h >> 13;
608        h = h.wrapping_mul(0xC2B2_AE35);
609        h ^= h >> 16;
610        (h as f32) / (u32::MAX as f32)
611    }

613    #[test]
614    fn phase_1_6_f4_b_3_d_2_continental_temperate_distribution() {
615        // Sample 10K random world positions through the Continental Temperate
616        // archetype's climate map. Verify the resulting biome distribution
617        // is plausible: dominant biomes are TemperateDeciduousForest +
618        // TemperateGrassland; some BorealForest in cold highlands; zero
619        // TropicalRainforest; some Beach/Coast on the world edges.
620        let config = ClimateConfig::default(); // archetype = Continental Temperate
621        let climate = ClimateMap::new(&config, 12345);
622
623        let half = config.world_latitude_half_extent_wu as f64;
624        let mut counts = std::collections::HashMap::<BiomeId, u32>::new();
625        const N: u32 = 10_000;
626        for i in 0..N {
627            // Random position within the world extent.
628            let x = (det_random(7, i * 2) * 2.0 - 1.0) as f64 * half;
629            let z = (det_random(7, i * 2 + 1) * 2.0 - 1.0) as f64 * half;
630            // Random elevation in [SEA_LEVEL-10, 510m] to span the Y range
631            // (legacy estimate_height max is ~270m; Target B world Y goes to
632            // ~510m; we sample broader to cover the full overlay zone).
633            let elev_t = det_random(11, i);
634            let elevation = SEA_LEVEL - 10.0 + elev_t * 520.0;
635
636            let s = climate.sample(x, z, elevation);
637            let b = lookup_biome(s.temperature_c, s.moisture_mm, elevation);
638            *counts.entry(b).or_insert(0) += 1;
639        }
640
641        let total = N as f32;
642        let frac =
643            |b: BiomeId| (*counts.get(&b).unwrap_or(&0) as f32) / total;
644
645        // Verify zero TropicalRainforest (Continental Temperate is too cold).
646        assert!(
647            frac(BiomeId::TropicalRainforest) < 0.005,
648            "Continental Temperate should produce ~0% TropicalRainforest; got {}",
649            frac(BiomeId::TropicalRainforest)
650        );
651        // Verify zero Savanna (also too cold).
652        assert!(
653            frac(BiomeId::Savanna) < 0.005,
654            "Continental Temperate should produce ~0% Savanna; got {}",
655            frac(BiomeId::Savanna)
656        );
657
658        // Verify the temperate-zone family (TemperateDeciduousForest +
659        // TemperateGrassland + TemperateRainforest + ColdDesert + Wetland
660        // + BorealForest + Tundra) collectively dominates the terrestrial
661        // distribution (some samples will be aquatic / overlay; those are
662        // valid and not counted against the cold-zone fraction).
663        let temperate_family = frac(BiomeId::TemperateDeciduousForest)
664            + frac(BiomeId::TemperateGrassland)
665            + frac(BiomeId::TemperateRainforest)
666            + frac(BiomeId::ColdDesert)
667            + frac(BiomeId::BorealForest)
668            + frac(BiomeId::Tundra)
669            + frac(BiomeId::Wetland);
670        let elevation_overlay = frac(BiomeId::Alpine)
671            + frac(BiomeId::SnowCap)
672            + frac(BiomeId::Scree)
673            + frac(BiomeId::MountainRocky);
674        let aquatic = frac(BiomeId::Ocean)
675            + frac(BiomeId::Coast)
676            + frac(BiomeId::Beach)
677            + frac(BiomeId::River);
678        let total_classified = temperate_family + elevation_overlay + aquatic;
679
680        // 99% of samples should fall into one of the recognized non-tropical
681        // categories (a small slack accounts for boundary samples that
682        // could fall into edge-of-Whittaker variants).
683        assert!(
684            total_classified > 0.95,
685            "Continental Temperate should have ≥95% of samples in non-tropical categories; \
686             got temperate_family={temperate_family:.3}, overlay={elevation_overlay:.3}, \
687             aquatic={aquatic:.3}, total={total_classified:.3}"
688        );
689
690        // No BiomeId variant should be NaN-poisoned or panic-producing.
691        // (covered implicitly by the iteration completing.)
692    }
