200    /// Generate a terrain chunk at the given world position with vegetation and resources
201    pub fn generate_chunk_with_scatter(
202        &mut self,
203        chunk_id: ChunkId,
204    ) -> anyhow::Result<(TerrainChunk, ScatterResult)> {
205        // Generate the basic terrain chunk (lock-free)
206        let chunk = self.generate_chunk(chunk_id)?;
207
208        // Register with chunk manager
209        self.chunk_manager.add_chunk(chunk.clone());
210
211        // Generate scatter for the chunk
212        let scatter_result = self.scatter_chunk_content(&chunk)?;
213
214        Ok((chunk, scatter_result))
215    }

217    /// Generate scatter content (vegetation and resources) for an existing chunk
218    pub fn scatter_chunk_content(&mut self, chunk: &TerrainChunk) -> anyhow::Result<ScatterResult> {
219        let mut result = ScatterResult::new(chunk.id());
220
221        // Create scatter system
222        let scatter_config = ScatterConfig::default();
223        let scatter = VegetationScatter::new(scatter_config);
224
225        // Sample the biome at the chunk center to determine configuration
226        let chunk_center = chunk.id().to_center_pos(self.config.chunk_size);
227        let center_biome = chunk
228            .get_biome_at_world_pos(chunk_center, self.config.chunk_size)
229            .unwrap_or(BiomeType::Grassland);
230
231        // Find the biome configuration
232        let biome_config = self
233            .config
234            .biomes
235            .iter()
236            .find(|b| b.biome_type == center_biome)
237            .unwrap_or(&self.config.biomes[0]);
238
239        // Generate vegetation
240        result.vegetation = scatter.scatter_vegetation(
241            chunk,
242            self.config.chunk_size,
243            biome_config,
244            self.config.seed + chunk.id().x as u64 * 1000 + chunk.id().z as u64,
245        )?;
246
247        // Generate resources
248        result.resources = scatter.scatter_resources(
249            chunk,
250            self.config.chunk_size,
251            biome_config,
252            self.config.seed + chunk.id().x as u64 * 2000 + chunk.id().z as u64,
253        )?;
254
255        // Generate structures
256        let structure_result = self.structure_generator.generate_structures(
257            chunk,
258            self.config.chunk_size,
259            center_biome,
260        )?;
261        result.structures = structure_result.structures;
262
263        Ok(result)
264    }

266    /// Generate a terrain chunk at the given position (lock-free, parallel-safe)
267    /// NOTE: Does NOT add to chunk_manager - caller must handle that separately
268    pub fn generate_chunk(&self, chunk_id: ChunkId) -> anyhow::Result<TerrainChunk> {
269        // Generate heightmap for this chunk (using SIMD if enabled)
270        #[cfg(feature = "simd-noise")]
271        let heightmap = noise_simd::SimdHeightmapGenerator::generate_heightmap_simd(
272            &self.noise,
273            chunk_id,
274            self.config.chunk_size,
275            self.config.heightmap_resolution,
276        )?;
277
278        #[cfg(not(feature = "simd-noise"))]
279        let heightmap = self.noise.generate_heightmap(
280            chunk_id,
281            self.config.chunk_size,
282            self.config.heightmap_resolution,
283        )?;
284
285        // Generate climate data for biome assignment
286        let climate_data = self.climate.sample_chunk(
287            chunk_id,
288            self.config.chunk_size,
289            self.config.heightmap_resolution,
290        )?;
291
292        // Assign biomes based on height and climate
293        let biome_map = self.assign_biomes(&heightmap, &climate_data)?;
294
295        // Create the terrain chunk
296        let mut chunk = TerrainChunk::new(chunk_id, heightmap, biome_map);
297
298        // Apply erosion if enabled
299        if self.config.noise.erosion_enabled {
300            chunk.apply_erosion(self.config.noise.erosion_strength)?;
301        }
302
303        // NOTE: chunk_manager.add_chunk() removed - caller handles registration
304        Ok(chunk)
305    }

307    /// Phase 1.6-F.3-phase-1: generate a chunk with pre-erosion biome_weights
308    /// computed from the given climate bias. The returned chunk has
309    /// `biome_weights: Some(_)` populated from the PRE-erosion heightmap,
310    /// satisfying §2.5's authorial-intent invariant. The heightmap itself is
311    /// post-erosion (simple CA for phase 1; `AdvancedErosionSimulator` for
312    /// phase 2).
313    ///
314    /// Callers not requiring pre-erosion biome_weights should use the simpler
315    /// `generate_chunk` (biome_weights stays `None`, editor/consumer computes
316    /// them on-the-fly from post-erosion heights — current behavior).
317    pub fn generate_chunk_with_climate(
318        &self,
319        chunk_id: ChunkId,
320        climate_bias: crate::ClimateBias,
321    ) -> anyhow::Result<TerrainChunk> {
322        // Phase 1.6-F.3-phase-2.C: full pipeline wired.
323        //
324        // 1. Generate halo heightmap (3×3 chunks centered on target).
325        // 2. Crop the halo to the target chunk for PRE-erosion biome_weights
326        //    computation per §2.5 (authorial-intent invariant).
327        // 3. Run AdvancedErosionSimulator::apply_preset on the FULL halo —
328        //    droplets travel freely across chunk boundaries within the halo
329        //    region. §2.3 halo=1 is sufficient per phase-0's p95 = 120 world
330        //    units < 256 (one chunk width).
331        // 4. Crop AFTER erosion. This becomes the chunk's final (post-erosion)
332        //    heightmap.
333        // 5. Construct chunk with post-erosion cropped heightmap + pre-erosion
334        //    biome_weights (decoupled per §2.5).
335        //
336        // Legacy `chunk.apply_erosion(strength)` simple CA call is replaced by
337        // `AdvancedErosionSimulator` per §2.2 climate → preset mapping.
338        const HALO_CHUNKS: u32 = 1;
339
340        let mut halo = self.generate_halo_heightmap(chunk_id, HALO_CHUNKS)?;
341
342        // Phase 1.6-F.4.B.3.D.3b: per-vertex biome lookup + per-biome
343        // mountain-amplitude modulation. Iterates the freshly generated halo
344        // (each vertex currently has the default-amplitude `sample_height`
345        // value), samples the climate field at that vertex's raw elevation,
346        // looks up the dominant `BiomeId` via D.2's `lookup_biome`, looks up
347        // per-biome `BiomeParameters` from D.3a, and re-samples the height
348        // with `sample_height_with_mountain_amplitude` using
349        // `params.mountains_amplitude` as the multiplier.
350        //
351        // This is the structural replacement for the legacy
352        // `BiomeNoisePreset` whole-world configuration: every vertex gets
353        // its own biome assignment from the climate field, and the noise
354        // pipeline applies that biome's amplitude per-vertex.
355        //
356        // The pre-erosion biome IDs are computed from raw (default-amplitude)
357        // heights so that biome assignment is invariant across re-runs with
358        // different per-biome amplitudes — biome assignment shapes the world,
359        // not the other way around. Per §2.5, biome assignment uses
360        // pre-erosion heights for authorial-intent stability.
361        let halo_biome_ids = self
362            .apply_per_biome_modulation_to_halo(&mut halo, chunk_id, HALO_CHUNKS);
363
364        // Pre-erosion cropped heightmap — input to biome_weights computation.
365        // After per-biome modulation, this reflects per-vertex amplitude.
366        let pre_erosion_heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
367
368        // §2.5: biome_weights are computed from PRE-erosion heights. Once
369        // phase 2 lands real erosion, a vertex that drops from Y=50 to Y=30
370        // keeps its pre-erosion Mountain weighting — authorial intent over
371        // geological reclassification.
372        let resolution = pre_erosion_heightmap.resolution() as usize;
373        let mut biome_weights = Vec::with_capacity(resolution * resolution);
374        for z in 0..resolution {
375            for x in 0..resolution {
376                let y = pre_erosion_heightmap.get_height(x as u32, z as u32);
377                biome_weights.push(crate::elevation_to_biome_weights(
378                    y,
379                    crate::SEA_LEVEL,
380                    climate_bias,
381                ));
382            }
383        }
384
385        // D.3b: crop the per-vertex BiomeId array to the chunk-sized region
386        // matching `pre_erosion_heightmap`. Uses the same crop offsets as
387        // `crop_halo_to_chunk`.
388        let chunk_biome_ids = self.crop_halo_biome_ids_to_chunk(
389            &halo_biome_ids,
390            halo.resolution() as usize,
391            chunk_id,
392        );
393
394        // §2.2 climate → ErosionPreset mapping.
395        let preset = crate::advanced_erosion::erosion_preset_for_climate(climate_bias);
396
397        // Phase 1.6-F.3-phase-3.C: use `apply_preset_at_world_offset` for
398        // seamless chunk boundaries. Droplet spawn positions are derived
399        // from world-aligned spatial cells, so adjacent halos iterate the
400        // SAME cells in overlap → identical droplets → near-identical
401        // erosion output in overlap region.
402        //
403        // The simulator's `new(seed)` argument is no longer the primary
404        // determinism driver — per-droplet RNG comes from world-cell hash
405        // inside the new API. We pass `halo_seed` for any ancillary RNG use
406        // (currently unused inside the world-coord path).
407        let seed = Self::halo_seed(self.config.seed, chunk_id, HALO_CHUNKS);
408        let mut simulator = crate::advanced_erosion::AdvancedErosionSimulator::new(seed);
409
410        if self.config.noise.erosion_enabled {
411            // Halo's world origin (target chunk origin minus halo_chunks * chunk_size).
412            let target_origin = chunk_id.to_world_pos(self.config.chunk_size);
413            let halo_origin_x = (target_origin.x - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
414            let halo_origin_z = (target_origin.z - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
415            // Vertex spacing: chunk_size per (resolution - 1) vertices.
416            let vertex_spacing =
417                self.config.chunk_size as f64 / (self.config.heightmap_resolution - 1) as f64;
418
419            let _stats = simulator.apply_preset_at_world_offset(
420                &mut halo,
421                &preset,
422                halo_origin_x,
423                halo_origin_z,
424                vertex_spacing,
425                self.config.seed,
426            );
427            halo.recalculate_bounds();
428        }
429
430        // Post-erosion cropped heightmap — final chunk output.
431        let heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
432
433        // biome_map still computed from PRE-erosion heights + climate data
434        // (unchanged path). Phase 4 will route climate through biome_weights;
435        // biome_map currently feeds splat-rule selection elsewhere.
436        let climate_data = self.climate.sample_chunk(
437            chunk_id,
438            self.config.chunk_size,
439            self.config.heightmap_resolution,
440        )?;
441        let biome_map = self.assign_biomes(&pre_erosion_heightmap, &climate_data)?;
442
443        // Construct chunk: post-erosion heights + pre-erosion biome_weights
444        // + per-vertex BiomeId array (Phase 1.6-F.4.B.3.D.3b).
445        // No further `apply_erosion` — phase 2 replaces simple CA with
446        // AdvancedErosionSimulator.
447        let chunk = TerrainChunk::new_with_climate_field(
448            chunk_id,
449            heightmap,
450            biome_map,
451            biome_weights,
452            chunk_biome_ids,
453        );
454
455        Ok(chunk)
456    }

458    /// Phase 1.6-F.4.B.3.D.3b → D.4: apply per-vertex biome modulation
459    /// to a halo heightmap with scattered-convolution biome blending.
460    ///
461    /// For each vertex in the halo, the function:
462    /// 1. Samples the bootstrap (raw) elevation from the halo.
463    /// 2. Calls
464    ///    `biome_param_blending::blend_biome_parameters(world_x, world_z,
465    ///    bootstrap_elevation, climate_map, &blend_config)` which:
466    ///    - Generates `N` jittered sample positions within `radius` (D.4).
467    ///    - Looks up biome at each sample via climate field + Whittaker.
468    ///    - Distance-weighted blends `mountains_amplitude` and
469    ///      `scatter_density`.
470    ///    - Returns dominant biome (highest summed weight) +
471    ///      blended numeric params.
472    /// 3. Re-samples the noise pipeline with
473    ///    `sample_height_with_mountain_amplitude` using the blended
474    ///    `mountains_amplitude` (continuous across biome boundaries —
475    ///    no flip discontinuity).
476    /// 4. Records the dominant biome ID at this vertex.
477    ///
478    /// Returns a flat `Vec<BiomeId>` in row-major `(z, x)` order matching
479    /// the halo's vertex layout. Length: `halo_res × halo_res`.
480    ///
481    /// Cost analysis vs D.3b: D.3b did 1 noise sample (raw, bootstrap) +
482    /// 1 climate sample + 1 biome lookup + 1 noise sample (modulated) per
483    /// vertex. D.4 adds the blending pass: `N` climate samples + `N`
484    /// biome lookups + `N` parameter lookups per vertex. With `N=6`
485    /// default, total per-vertex cost is roughly D.3b + 5 climate samples
486    /// + 5 biome lookups. Climate sampling and biome lookup are both
487    /// pure functions of position — cache-friendly and SIMD-amenable.
488    fn apply_per_biome_modulation_to_halo(
489        &self,
490        halo: &mut heightmap::Heightmap,
491        target_chunk_id: ChunkId,
492        halo_chunks: u32,
493    ) -> Vec<crate::biome_lookup::BiomeId> {
494        let halo_res = halo.resolution() as usize;
495        let chunk_res = self.config.heightmap_resolution as usize;
496        // CRITICAL: must mirror `generate_halo_heightmap`'s f32 arithmetic
497        // so that vertex world coordinates at the shared edge between
498        // adjacent chunks match exactly. f64 arithmetic produces a
499        // slightly different `step` value than f32, breaking the
500        // shared-edge invariant under per-biome modulation. Per
501        // F.4.B.3.D.3b regression diagnosis: f64 path produced 125 WU
502        // divergence at chunk borders for the mountain test.
503        let chunk_size = self.config.chunk_size;
504        let step = chunk_size / (chunk_res as f32 - 1.0);
505
506        let target_origin = target_chunk_id.to_world_pos(chunk_size);
507        let halo_origin_x = target_origin.x - halo_chunks as f32 * chunk_size;
508        let halo_origin_z = target_origin.z - halo_chunks as f32 * chunk_size;
509
510        // F.4.B.3.D.4: blend config defaulted to 6 samples / 48 WU radius.
511        // D.6 Andrew-gate informs whether to tune.
512        let blend_config = crate::biome_param_blending::BiomeParamBlendConfig::default();
513
514        // Phase 1.X-F.3.B + F.4.E: archetype bootstrap splines.
515        //
516        // **F.3 None-mask path** (regional_archetype_mask = None): every
517        // vertex evaluates Continental Temperate's BootstrapSplineSet
518        // cached once at the start of this method; F.3 byte-identity
519        // contract preserved.
520        //
521        // **F.4 Some-mask path** (regional_archetype_mask = Some): per-
522        // vertex sample the mask + falloff via RegionalArchetypeBlend;
523        // for each contributing archetype, look up its BootstrapSplineSet
524        // from the F.4.E catalog cache (cached for all 6 archetypes
525        // once per chunk pass, indexed by WorldArchetypeId);
526        // blend_bootstrap_params produces per-vertex blended
527        // BootstrapParams per campaign doc §2.5.
528        //
529        // F.4.E catalog cache: 6 BootstrapSplineSet instances pre-fetched
530        // at the start of this method, indexed by WorldArchetypeId. The
531        // archetype_splines closure passed to blend_bootstrap_params is
532        // a simple match returning a reference to the cached entry.
533        // Avoids per-vertex factory calls.
534        let archetype_splines_continental = crate::spline_types::bootstrap_splines_continental_temperate();
535        let archetype_splines_tropical = crate::spline_types::bootstrap_splines_equatorial_tropical();
536        let archetype_splines_boreal = crate::spline_types::bootstrap_splines_boreal_subarctic();
537        let archetype_splines_mediterranean = crate::spline_types::bootstrap_splines_mediterranean();
538        let archetype_splines_desert = crate::spline_types::bootstrap_splines_desert();
539        let archetype_splines_custom = crate::spline_types::bootstrap_splines_custom();
540        let lookup_splines =
541            |id: crate::world_archetypes::WorldArchetypeId| -> crate::spline_types::BootstrapSplineSet {
542                match id {
543                    crate::world_archetypes::WorldArchetypeId::ContinentalTemperate => {
544                        archetype_splines_continental.clone()
545                    }
546                    crate::world_archetypes::WorldArchetypeId::EquatorialTropical => {
547                        archetype_splines_tropical.clone()
548                    }
549                    crate::world_archetypes::WorldArchetypeId::BorealSubarctic => {
550                        archetype_splines_boreal.clone()
551                    }
552                    crate::world_archetypes::WorldArchetypeId::Mediterranean => {
553                        archetype_splines_mediterranean.clone()
554                    }
555                    crate::world_archetypes::WorldArchetypeId::Desert => {
556                        archetype_splines_desert.clone()
557                    }
558                    crate::world_archetypes::WorldArchetypeId::Custom => {
559                        archetype_splines_custom.clone()
560                    }
561                }
562            };
563        let regional_blend = self
564            .regional_archetype_mask
565            .as_ref()
566            .map(crate::regional_archetype_mask::RegionalArchetypeBlend::new);
567
568        let mut biome_ids = Vec::with_capacity(halo_res * halo_res);
569        for z_idx in 0..halo_res {
570            for x_idx in 0..halo_res {
571                let wx_f32 = halo_origin_x + x_idx as f32 * step;
572                let wz_f32 = halo_origin_z + z_idx as f32 * step;
573                let wx = wx_f32 as f64;
574                let wz = wz_f32 as f64;
575
576                let raw_height = halo.get_height(x_idx as u32, z_idx as u32);
577
578                // F.4.B.3.D.4: scattered-convolution biome parameter
579                // blending. Produces blended `mountains_amplitude` and
580                // `scatter_density` from N jittered samples + dominant
581                // biome ID. Numeric parameters transition smoothly
582                // across biome boundaries; biome ID stays discrete.
583                let blended = crate::biome_param_blending::blend_biome_parameters(
584                    wx_f32,
585                    wz_f32,
586                    raw_height,
587                    &self.climate,
588                    &blend_config,
589                );
590
591                // Phase 1.X-F.3.B + F.4.E: evaluate archetype BootstrapSplineSet(s)
592                // at this vertex's climate sample.
593                //
594                // **None-mask path (F.3 byte-identity contract)**: use
595                // Continental Temperate's BootstrapSplineSet directly.
596                // **Some-mask path (F.4 multi-archetype)**: sample mask +
597                // falloff via RegionalArchetypeBlend, evaluate each
598                // contributing archetype's splines, blend per §2.5.
599                let climate_sample = self.climate.sample(wx, wz, raw_height);
600                let bootstrap_params = match &regional_blend {
601                    None => archetype_splines_continental.evaluate(&climate_sample),
602                    Some(blend) => {
603                        let contributors = blend.sample_at(wx_f32, wz_f32);
604                        // F.4.E empty-contributor guard: sampler invariants
605                        // guarantee non-empty (per F.4.C tests), but defensive
606                        // fallback to Continental Temperate solo prevents
607                        // all-zero BootstrapParams from reaching the noise
608                        // pipeline if a future change breaks the invariant.
609                        if contributors.is_empty() {
610                            debug_assert!(
611                                false,
612                                "RegionalArchetypeBlend::sample_at returned empty \
613                                 contributors at world ({}, {}); F.4.C invariant \
614                                 should prevent this",
615                                wx_f32, wz_f32
616                            );
617                            archetype_splines_continental.evaluate(&climate_sample)
618                        } else {
619                            crate::regional_archetype_mask::blend_bootstrap_params(
620                                &contributors,
621                                &lookup_splines,
622                                &climate_sample,
623                            )
624                        }
625                    }
626                };
627
628                // Phase 1.X-F.3.B: replace sample_height_with_mountain_amplitude
629                // with sample_height_with_params (composed with the per-biome
630                // multiplier from blended.mountains_amplitude). Per-archetype
631                // bootstrap parameters and per-biome amplitude are
632                // architecturally orthogonal layers that multiply.
633                let modulated_height = self.noise.sample_height_with_params(
634                    &bootstrap_params,
635                    wx,
636                    wz,
637                    blended.mountains_amplitude as f32,
638                );
639                halo.set_height(x_idx as u32, z_idx as u32, modulated_height);
640                biome_ids.push(blended.dominant_biome);
641            }
642        }
643        halo.recalculate_bounds();
644        biome_ids
645    }
646
647    /// Phase 1.6-F.4.B.3.D.3b: crop the per-vertex `BiomeId` halo array to
648    /// the chunk-sized center region. Mirrors `crop_halo_to_chunk` for the
649    /// heightmap.
650    fn crop_halo_biome_ids_to_chunk(
651        &self,
652        halo_biome_ids: &[crate::biome_lookup::BiomeId],
653        halo_res: usize,
654        _target_chunk_id: ChunkId,
655    ) -> Vec<crate::biome_lookup::BiomeId> {
656        let chunk_res = self.config.heightmap_resolution as usize;
657        // halo_res = chunks_per_side * (chunk_res - 1) + 1
658        // chunks_per_side = (halo_res - 1) / (chunk_res - 1)
659        // halo_chunks = (chunks_per_side - 1) / 2
660        // crop_offset = halo_chunks * (chunk_res - 1)
661        let chunks_per_side = (halo_res - 1) / (chunk_res - 1);
662        let halo_chunks = (chunks_per_side - 1) / 2;
663        let crop_offset = halo_chunks * (chunk_res - 1);
664
665        let mut chunk_ids = Vec::with_capacity(chunk_res * chunk_res);
666        for z in 0..chunk_res {
667            for x in 0..chunk_res {
668                let halo_idx =
669                    (crop_offset + z) * halo_res + (crop_offset + x);
670                chunk_ids.push(halo_biome_ids[halo_idx]);
671            }
672        }
673        chunk_ids
674    }
675
676    /// Phase 1.6-F.3-phase-1.B: generate a heightmap covering a 3×3 chunk
677    /// region centered on `target_chunk_id` (halo_chunks=1). The returned
678    /// heightmap has `(1 + 2*halo_chunks) × (heightmap_resolution - 1) + 1`
679    /// vertices per side at the same per-vertex spacing as single-chunk
680    /// generation, so the center third (sampled at the target chunk's world
681    /// coords) is byte-identical to what `generate_chunk` produces via the
682    /// SIMD heightmap generator.
683    pub(crate) fn generate_halo_heightmap(
684        &self,
685        target_chunk_id: ChunkId,
686        halo_chunks: u32,
687    ) -> anyhow::Result<heightmap::Heightmap> {
688        let chunk_res = self.config.heightmap_resolution;
689        let chunk_size = self.config.chunk_size;
690        // Per-vertex step in world units. Must match single-chunk generation.
691        let step = chunk_size / (chunk_res - 1) as f32;
692
693        // Halo region: (1 + 2*halo_chunks) sub-chunks per axis. Adjacent
694        // sub-chunks share their edge vertex so total vertex count per side is
695        // `chunks_per_side * (chunk_res - 1) + 1`.
696        let chunks_per_side = 1 + 2 * halo_chunks;
697        let halo_res = chunks_per_side * (chunk_res - 1) + 1;
698        let halo_size_world = chunks_per_side as f32 * chunk_size;
699
700        // Origin world coordinates = target chunk origin minus (halo_chunks × chunk_size).
701        let target_origin = target_chunk_id.to_world_pos(chunk_size);
702        let halo_origin_x = target_origin.x - halo_chunks as f32 * chunk_size;
703        let halo_origin_z = target_origin.z - halo_chunks as f32 * chunk_size;
704
705        // Sample the noise field directly at per-vertex world coordinates. This
706        // path uses `TerrainNoise::sample_height` rather than the SIMD
707        // heightmap generator because the SIMD path is tied to a ChunkId. For
708        // byte-identity with the SIMD path at the center crop, both routes
709        // must produce the same Y at the same world (x, z) — confirmed by the
710        // determinism invariant of `TerrainNoise`.
711        let heightmap_config = heightmap::HeightmapConfig {
712            resolution: halo_res,
713            ..Default::default()
714        };
715        let mut halo_map = heightmap::Heightmap::new(heightmap_config)?;
716        for z_idx in 0..halo_res {
717            for x_idx in 0..halo_res {
718                let wx = halo_origin_x + x_idx as f32 * step;
719                let wz = halo_origin_z + z_idx as f32 * step;
720                let y = self.noise.sample_height(wx as f64, wz as f64);
721                halo_map.set_height(x_idx, z_idx, y);
722            }
723        }
724
725        // Silence unused warning for halo_size_world — useful for phase 2
726        // perf logging and halo-overlap sanity assertions.
727        let _ = halo_size_world;
728
729        Ok(halo_map)
730    }
731
732    /// Phase 1.6-F.3-phase-1.B: crop the center chunk-sized region out of a
733    /// halo-expanded heightmap. Assumes the halo was built by
734    /// `generate_halo_heightmap` with the same `target_chunk_id` and halo
735    /// size. Returns a single-chunk heightmap (resolution matching
736    /// `WorldConfig::heightmap_resolution`).
737    pub(crate) fn crop_halo_to_chunk(
738        &self,
739        halo: &heightmap::Heightmap,
740        target_chunk_id: ChunkId,
741    ) -> anyhow::Result<heightmap::Heightmap> {
742        let _ = target_chunk_id; // centered crop — target id not needed
743        let chunk_res = self.config.heightmap_resolution;
744        let halo_res = halo.resolution();
745        // Infer halo_chunks from resolutions.
746        //   halo_res = chunks_per_side * (chunk_res - 1) + 1
747        //   chunks_per_side = (halo_res - 1) / (chunk_res - 1)
748        //   halo_chunks = (chunks_per_side - 1) / 2
749        if chunk_res == 0 || (halo_res - 1) % (chunk_res - 1) != 0 {
750            anyhow::bail!(
751                "halo resolution {} not a multiple of chunk resolution {}",
752                halo_res,
753                chunk_res
754            );
755        }
756        let chunks_per_side = (halo_res - 1) / (chunk_res - 1);
757        if chunks_per_side < 1 || chunks_per_side % 2 != 1 {
758            anyhow::bail!(
759                "halo has {} chunks per side; expected odd >= 1",
760                chunks_per_side
761            );
762        }
763        let halo_chunks = (chunks_per_side - 1) / 2;
764        let start = halo_chunks * (chunk_res - 1);
765
766        let chunk_cfg = heightmap::HeightmapConfig {
767            resolution: chunk_res,
768            ..Default::default()
769        };
770        let mut cropped = heightmap::Heightmap::new(chunk_cfg)?;
771        for z in 0..chunk_res {
772            for x in 0..chunk_res {
773                let y = halo.get_height(start + x, start + z);
774                cropped.set_height(x, z, y);
775            }
776        }
777        Ok(cropped)
778    }

780    /// Phase 1.6-F.3-phase-1.B / §2.3: derive a deterministic seed for the
781    /// halo region centered on `target_chunk_id`. Phase 2 feeds this seed to
782    /// `AdvancedErosionSimulator::new` so adjacent chunks' halos that overlap
783    /// in world space produce identical droplet trajectories in the overlap
784    /// region.
785    #[allow(dead_code)] // Wired in phase 2; validated by unit tests here.
786    pub(crate) fn halo_seed(
787        world_seed: u64,
788        target_chunk_id: ChunkId,
789        halo_chunks: u32,
790    ) -> u64 {
791        let halo_origin_x = target_chunk_id.x.wrapping_sub(halo_chunks as i32);
792        let halo_origin_z = target_chunk_id.z.wrapping_sub(halo_chunks as i32);
793        // Wang-style integer hash, deterministic across runs / platforms.
794        let mut h = world_seed;
795        h = h
796            .wrapping_add(halo_origin_x as u64)
797            .wrapping_mul(0x9E3779B97F4A7C15);
798        h ^= h >> 32;
799        h = h
800            .wrapping_add(halo_origin_z as u64)
801            .wrapping_mul(0x85EBCA6BE11ECC0D);
802        h ^= h >> 32;
803        h
804    }

821    /// Stream chunks around a center position, loading/unloading as needed
822    pub fn stream_chunks(&mut self, center: Vec3, radius: u32) -> anyhow::Result<Vec<ChunkId>> {
823        let chunks_to_load = self.chunk_manager.get_chunks_in_radius(center, radius);
824        let mut loaded = Vec::new();
825
826        for chunk_id in chunks_to_load {
827            if !self.chunk_manager.has_chunk(chunk_id) {
828                self.generate_and_register_chunk(chunk_id)?;
829                loaded.push(chunk_id);
830            }
831        }
832
833        // Unload chunks that are too far away
834        let unload_radius = radius + 2; // Keep a buffer
835        self.chunk_manager
836            .unload_distant_chunks(center, unload_radius);
837
838        Ok(loaded)
839    }

841    /// Assign biomes to heightmap points based on climate data
842    fn assign_biomes(
843        &self,
844        heightmap: &Heightmap,
845        climate_data: &[(f32, f32)], // (temperature, moisture) pairs
846    ) -> anyhow::Result<Vec<BiomeType>> {
847        let mut biome_map = Vec::with_capacity(climate_data.len());
848
849        for (i, &(temperature, moisture)) in climate_data.iter().enumerate() {
850            let height = heightmap.get_height_at_index(i);
851            let biome = self.find_best_biome(height, temperature, moisture);
852            biome_map.push(biome);
853        }
854
855        Ok(biome_map)
856    }
857
858    /// Find the best biome for given environmental conditions
859    fn find_best_biome(&self, height: f32, temperature: f32, moisture: f32) -> BiomeType {
860        let mut best_biome = BiomeType::Grassland;
861        let mut best_score = f32::NEG_INFINITY;
862
863        for biome_config in &self.config.biomes {
864            let score = biome_config.score_conditions(height, temperature, moisture);
865            if score > best_score {
866                best_score = score;
867                best_biome = biome_config.biome_type;
868            }
869        }
870
871        best_biome
872    }
