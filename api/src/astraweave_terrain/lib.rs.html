172    pub fn new(config: WorldConfig) -> Self {
173        let noise = TerrainNoise::new(&config.noise, config.seed);

175        let chunk_manager = ChunkManager::new(config.chunk_size, config.heightmap_resolution);

185            structure_generator,

189    /// Generate a terrain chunk at the given world position with vegetation and resources
190    pub fn generate_chunk_with_scatter(
191        &mut self,
192        chunk_id: ChunkId,
193    ) -> anyhow::Result<(TerrainChunk, ScatterResult)> {
194        // Generate the basic terrain chunk (lock-free)
195        let chunk = self.generate_chunk(chunk_id)?;
196
197        // Register with chunk manager
198        self.chunk_manager.add_chunk(chunk.clone());
199
200        // Generate scatter for the chunk
201        let scatter_result = self.scatter_chunk_content(&chunk)?;
202
203        Ok((chunk, scatter_result))
204    }

206    /// Generate scatter content (vegetation and resources) for an existing chunk
207    pub fn scatter_chunk_content(&mut self, chunk: &TerrainChunk) -> anyhow::Result<ScatterResult> {
208        let mut result = ScatterResult::new(chunk.id());
209
210        // Create scatter system
211        let scatter_config = ScatterConfig::default();
212        let scatter = VegetationScatter::new(scatter_config);
213
214        // Sample the biome at the chunk center to determine configuration
215        let chunk_center = chunk.id().to_center_pos(self.config.chunk_size);
216        let center_biome = chunk
217            .get_biome_at_world_pos(chunk_center, self.config.chunk_size)
218            .unwrap_or(BiomeType::Grassland);
219
220        // Find the biome configuration
221        let biome_config = self
222            .config
223            .biomes
224            .iter()
225            .find(|b| b.biome_type == center_biome)
226            .unwrap_or(&self.config.biomes[0]);
227
228        // Generate vegetation
229        result.vegetation = scatter.scatter_vegetation(
230            chunk,
231            self.config.chunk_size,
232            biome_config,
233            self.config.seed + chunk.id().x as u64 * 1000 + chunk.id().z as u64,
234        )?;
235
236        // Generate resources
237        result.resources = scatter.scatter_resources(
238            chunk,
239            self.config.chunk_size,
240            biome_config,
241            self.config.seed + chunk.id().x as u64 * 2000 + chunk.id().z as u64,
242        )?;
243
244        // Generate structures
245        let structure_result = self.structure_generator.generate_structures(
246            chunk,
247            self.config.chunk_size,
248            center_biome,
249        )?;
250        result.structures = structure_result.structures;
251
252        Ok(result)
253    }

255    /// Generate a terrain chunk at the given position (lock-free, parallel-safe)
256    /// NOTE: Does NOT add to chunk_manager - caller must handle that separately
257    pub fn generate_chunk(&self, chunk_id: ChunkId) -> anyhow::Result<TerrainChunk> {
258        // Generate heightmap for this chunk (using SIMD if enabled)
259        #[cfg(feature = "simd-noise")]
260        let heightmap = noise_simd::SimdHeightmapGenerator::generate_heightmap_simd(
261            &self.noise,
262            chunk_id,
263            self.config.chunk_size,
264            self.config.heightmap_resolution,
265        )?;
266
267        #[cfg(not(feature = "simd-noise"))]
268        let heightmap = self.noise.generate_heightmap(
269            chunk_id,
270            self.config.chunk_size,
271            self.config.heightmap_resolution,
272        )?;
273
274        // Generate climate data for biome assignment
275        let climate_data = self.climate.sample_chunk(
276            chunk_id,
277            self.config.chunk_size,
278            self.config.heightmap_resolution,
279        )?;
280
281        // Assign biomes based on height and climate
282        let biome_map = self.assign_biomes(&heightmap, &climate_data)?;
283
284        // Create the terrain chunk
285        let mut chunk = TerrainChunk::new(chunk_id, heightmap, biome_map);
286
287        // Apply erosion if enabled
288        if self.config.noise.erosion_enabled {
289            chunk.apply_erosion(self.config.noise.erosion_strength)?;
290        }
291
292        // NOTE: chunk_manager.add_chunk() removed - caller handles registration
293        Ok(chunk)
294    }

296    /// Phase 1.6-F.3-phase-1: generate a chunk with pre-erosion biome_weights
297    /// computed from the given climate bias. The returned chunk has
298    /// `biome_weights: Some(_)` populated from the PRE-erosion heightmap,
299    /// satisfying §2.5's authorial-intent invariant. The heightmap itself is
300    /// post-erosion (simple CA for phase 1; `AdvancedErosionSimulator` for
301    /// phase 2).
302    ///
303    /// Callers not requiring pre-erosion biome_weights should use the simpler
304    /// `generate_chunk` (biome_weights stays `None`, editor/consumer computes
305    /// them on-the-fly from post-erosion heights — current behavior).
306    pub fn generate_chunk_with_climate(
307        &self,
308        chunk_id: ChunkId,
309        climate_bias: crate::ClimateBias,
310    ) -> anyhow::Result<TerrainChunk> {
311        // Phase 1.6-F.3-phase-2.C: full pipeline wired.
312        //
313        // 1. Generate halo heightmap (3×3 chunks centered on target).
314        // 2. Crop the halo to the target chunk for PRE-erosion biome_weights
315        //    computation per §2.5 (authorial-intent invariant).
316        // 3. Run AdvancedErosionSimulator::apply_preset on the FULL halo —
317        //    droplets travel freely across chunk boundaries within the halo
318        //    region. §2.3 halo=1 is sufficient per phase-0's p95 = 120 world
319        //    units < 256 (one chunk width).
320        // 4. Crop AFTER erosion. This becomes the chunk's final (post-erosion)
321        //    heightmap.
322        // 5. Construct chunk with post-erosion cropped heightmap + pre-erosion
323        //    biome_weights (decoupled per §2.5).
324        //
325        // Legacy `chunk.apply_erosion(strength)` simple CA call is replaced by
326        // `AdvancedErosionSimulator` per §2.2 climate → preset mapping.
327        const HALO_CHUNKS: u32 = 1;
328
329        let mut halo = self.generate_halo_heightmap(chunk_id, HALO_CHUNKS)?;
330
331        // Phase 1.6-F.4.B.3.D.3b: per-vertex biome lookup + per-biome
332        // mountain-amplitude modulation. Iterates the freshly generated halo
333        // (each vertex currently has the default-amplitude `sample_height`
334        // value), samples the climate field at that vertex's raw elevation,
335        // looks up the dominant `BiomeId` via D.2's `lookup_biome`, looks up
336        // per-biome `BiomeParameters` from D.3a, and re-samples the height
337        // with `sample_height_with_mountain_amplitude` using
338        // `params.mountains_amplitude` as the multiplier.
339        //
340        // This is the structural replacement for the legacy
341        // `BiomeNoisePreset` whole-world configuration: every vertex gets
342        // its own biome assignment from the climate field, and the noise
343        // pipeline applies that biome's amplitude per-vertex.
344        //
345        // The pre-erosion biome IDs are computed from raw (default-amplitude)
346        // heights so that biome assignment is invariant across re-runs with
347        // different per-biome amplitudes — biome assignment shapes the world,
348        // not the other way around. Per §2.5, biome assignment uses
349        // pre-erosion heights for authorial-intent stability.
350        let halo_biome_ids = self
351            .apply_per_biome_modulation_to_halo(&mut halo, chunk_id, HALO_CHUNKS);
352
353        // Pre-erosion cropped heightmap — input to biome_weights computation.
354        // After per-biome modulation, this reflects per-vertex amplitude.
355        let pre_erosion_heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
356
357        // §2.5: biome_weights are computed from PRE-erosion heights. Once
358        // phase 2 lands real erosion, a vertex that drops from Y=50 to Y=30
359        // keeps its pre-erosion Mountain weighting — authorial intent over
360        // geological reclassification.
361        let resolution = pre_erosion_heightmap.resolution() as usize;
362        let mut biome_weights = Vec::with_capacity(resolution * resolution);
363        for z in 0..resolution {
364            for x in 0..resolution {
365                let y = pre_erosion_heightmap.get_height(x as u32, z as u32);
366                biome_weights.push(crate::elevation_to_biome_weights(
367                    y,
368                    crate::SEA_LEVEL,
369                    climate_bias,
370                ));
371            }
372        }
373
374        // D.3b: crop the per-vertex BiomeId array to the chunk-sized region
375        // matching `pre_erosion_heightmap`. Uses the same crop offsets as
376        // `crop_halo_to_chunk`.
377        let chunk_biome_ids = self.crop_halo_biome_ids_to_chunk(
378            &halo_biome_ids,
379            halo.resolution() as usize,
380            chunk_id,
381        );
382
383        // §2.2 climate → ErosionPreset mapping.
384        let preset = crate::advanced_erosion::erosion_preset_for_climate(climate_bias);
385
386        // Phase 1.6-F.3-phase-3.C: use `apply_preset_at_world_offset` for
387        // seamless chunk boundaries. Droplet spawn positions are derived
388        // from world-aligned spatial cells, so adjacent halos iterate the
389        // SAME cells in overlap → identical droplets → near-identical
390        // erosion output in overlap region.
391        //
392        // The simulator's `new(seed)` argument is no longer the primary
393        // determinism driver — per-droplet RNG comes from world-cell hash
394        // inside the new API. We pass `halo_seed` for any ancillary RNG use
395        // (currently unused inside the world-coord path).
396        let seed = Self::halo_seed(self.config.seed, chunk_id, HALO_CHUNKS);
397        let mut simulator = crate::advanced_erosion::AdvancedErosionSimulator::new(seed);
398
399        if self.config.noise.erosion_enabled {
400            // Halo's world origin (target chunk origin minus halo_chunks * chunk_size).
401            let target_origin = chunk_id.to_world_pos(self.config.chunk_size);
402            let halo_origin_x = (target_origin.x - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
403            let halo_origin_z = (target_origin.z - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
404            // Vertex spacing: chunk_size per (resolution - 1) vertices.
405            let vertex_spacing =
406                self.config.chunk_size as f64 / (self.config.heightmap_resolution - 1) as f64;
407
408            let _stats = simulator.apply_preset_at_world_offset(
409                &mut halo,
410                &preset,
411                halo_origin_x,
412                halo_origin_z,
413                vertex_spacing,
414                self.config.seed,
415            );
416            halo.recalculate_bounds();
417        }
418
419        // Post-erosion cropped heightmap — final chunk output.
420        let heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
421
422        // biome_map still computed from PRE-erosion heights + climate data
423        // (unchanged path). Phase 4 will route climate through biome_weights;
424        // biome_map currently feeds splat-rule selection elsewhere.
425        let climate_data = self.climate.sample_chunk(
426            chunk_id,
427            self.config.chunk_size,
428            self.config.heightmap_resolution,
429        )?;
430        let biome_map = self.assign_biomes(&pre_erosion_heightmap, &climate_data)?;
431
432        // Construct chunk: post-erosion heights + pre-erosion biome_weights
433        // + per-vertex BiomeId array (Phase 1.6-F.4.B.3.D.3b).
434        // No further `apply_erosion` — phase 2 replaces simple CA with
435        // AdvancedErosionSimulator.
436        let chunk = TerrainChunk::new_with_climate_field(
437            chunk_id,
438            heightmap,
439            biome_map,
440            biome_weights,
441            chunk_biome_ids,
442        );
443
444        Ok(chunk)
445    }

447    /// Phase 1.6-F.4.B.3.D.3b → D.4: apply per-vertex biome modulation
448    /// to a halo heightmap with scattered-convolution biome blending.
449    ///
450    /// For each vertex in the halo, the function:
451    /// 1. Samples the bootstrap (raw) elevation from the halo.
452    /// 2. Calls
453    ///    `biome_param_blending::blend_biome_parameters(world_x, world_z,
454    ///    bootstrap_elevation, climate_map, &blend_config)` which:
455    ///    - Generates `N` jittered sample positions within `radius` (D.4).
456    ///    - Looks up biome at each sample via climate field + Whittaker.
457    ///    - Distance-weighted blends `mountains_amplitude` and
458    ///      `scatter_density`.
459    ///    - Returns dominant biome (highest summed weight) +
460    ///      blended numeric params.
461    /// 3. Re-samples the noise pipeline with
462    ///    `sample_height_with_mountain_amplitude` using the blended
463    ///    `mountains_amplitude` (continuous across biome boundaries —
464    ///    no flip discontinuity).
465    /// 4. Records the dominant biome ID at this vertex.
466    ///
467    /// Returns a flat `Vec<BiomeId>` in row-major `(z, x)` order matching
468    /// the halo's vertex layout. Length: `halo_res × halo_res`.
469    ///
470    /// Cost analysis vs D.3b: D.3b did 1 noise sample (raw, bootstrap) +
471    /// 1 climate sample + 1 biome lookup + 1 noise sample (modulated) per
472    /// vertex. D.4 adds the blending pass: `N` climate samples + `N`
473    /// biome lookups + `N` parameter lookups per vertex. With `N=6`
474    /// default, total per-vertex cost is roughly D.3b + 5 climate samples
475    /// + 5 biome lookups. Climate sampling and biome lookup are both
476    /// pure functions of position — cache-friendly and SIMD-amenable.
477    fn apply_per_biome_modulation_to_halo(
478        &self,
479        halo: &mut heightmap::Heightmap,
480        target_chunk_id: ChunkId,
481        halo_chunks: u32,
482    ) -> Vec<crate::biome_lookup::BiomeId> {
483        let halo_res = halo.resolution() as usize;
484        let chunk_res = self.config.heightmap_resolution as usize;
485        // CRITICAL: must mirror `generate_halo_heightmap`'s f32 arithmetic
486        // so that vertex world coordinates at the shared edge between
487        // adjacent chunks match exactly. f64 arithmetic produces a
488        // slightly different `step` value than f32, breaking the
489        // shared-edge invariant under per-biome modulation. Per
490        // F.4.B.3.D.3b regression diagnosis: f64 path produced 125 WU
491        // divergence at chunk borders for the mountain test.
492        let chunk_size = self.config.chunk_size;
493        let step = chunk_size / (chunk_res as f32 - 1.0);
494
495        let target_origin = target_chunk_id.to_world_pos(chunk_size);
496        let halo_origin_x = target_origin.x - halo_chunks as f32 * chunk_size;
497        let halo_origin_z = target_origin.z - halo_chunks as f32 * chunk_size;
498
499        // F.4.B.3.D.4: blend config defaulted to 6 samples / 48 WU radius.
500        // D.6 Andrew-gate informs whether to tune.
501        let blend_config = crate::biome_param_blending::BiomeParamBlendConfig::default();
502
503        let mut biome_ids = Vec::with_capacity(halo_res * halo_res);
504        for z_idx in 0..halo_res {
505            for x_idx in 0..halo_res {
506                let wx_f32 = halo_origin_x + x_idx as f32 * step;
507                let wz_f32 = halo_origin_z + z_idx as f32 * step;
508                let wx = wx_f32 as f64;
509                let wz = wz_f32 as f64;
510
511                let raw_height = halo.get_height(x_idx as u32, z_idx as u32);
512
513                // F.4.B.3.D.4: scattered-convolution biome parameter
514                // blending. Produces blended `mountains_amplitude` and
515                // `scatter_density` from N jittered samples + dominant
516                // biome ID. Numeric parameters transition smoothly
517                // across biome boundaries; biome ID stays discrete.
518                let blended = crate::biome_param_blending::blend_biome_parameters(
519                    wx_f32,
520                    wz_f32,
521                    raw_height,
522                    &self.climate,
523                    &blend_config,
524                );
525
526                let modulated_height = self.noise.sample_height_with_mountain_amplitude(
527                    wx,
528                    wz,
529                    blended.mountains_amplitude as f32,
530                );
531                halo.set_height(x_idx as u32, z_idx as u32, modulated_height);
532                biome_ids.push(blended.dominant_biome);
533            }
534        }
535        halo.recalculate_bounds();
536        biome_ids
537    }

539    /// Phase 1.6-F.4.B.3.D.3b: crop the per-vertex `BiomeId` halo array to
540    /// the chunk-sized center region. Mirrors `crop_halo_to_chunk` for the
541    /// heightmap.
542    fn crop_halo_biome_ids_to_chunk(
543        &self,
544        halo_biome_ids: &[crate::biome_lookup::BiomeId],
545        halo_res: usize,
546        _target_chunk_id: ChunkId,
547    ) -> Vec<crate::biome_lookup::BiomeId> {
548        let chunk_res = self.config.heightmap_resolution as usize;
549        // halo_res = chunks_per_side * (chunk_res - 1) + 1
550        // chunks_per_side = (halo_res - 1) / (chunk_res - 1)
551        // halo_chunks = (chunks_per_side - 1) / 2
552        // crop_offset = halo_chunks * (chunk_res - 1)
553        let chunks_per_side = (halo_res - 1) / (chunk_res - 1);
554        let halo_chunks = (chunks_per_side - 1) / 2;
555        let crop_offset = halo_chunks * (chunk_res - 1);
556
557        let mut chunk_ids = Vec::with_capacity(chunk_res * chunk_res);
558        for z in 0..chunk_res {
559            for x in 0..chunk_res {
560                let halo_idx =
561                    (crop_offset + z) * halo_res + (crop_offset + x);
562                chunk_ids.push(halo_biome_ids[halo_idx]);
563            }
564        }
565        chunk_ids
566    }
567
568    /// Phase 1.6-F.3-phase-1.B: generate a heightmap covering a 3×3 chunk
569    /// region centered on `target_chunk_id` (halo_chunks=1). The returned
570    /// heightmap has `(1 + 2*halo_chunks) × (heightmap_resolution - 1) + 1`
571    /// vertices per side at the same per-vertex spacing as single-chunk
572    /// generation, so the center third (sampled at the target chunk's world
573    /// coords) is byte-identical to what `generate_chunk` produces via the
574    /// SIMD heightmap generator.
575    pub(crate) fn generate_halo_heightmap(
576        &self,
577        target_chunk_id: ChunkId,
578        halo_chunks: u32,
579    ) -> anyhow::Result<heightmap::Heightmap> {
580        let chunk_res = self.config.heightmap_resolution;
581        let chunk_size = self.config.chunk_size;
582        // Per-vertex step in world units. Must match single-chunk generation.
583        let step = chunk_size / (chunk_res - 1) as f32;
584
585        // Halo region: (1 + 2*halo_chunks) sub-chunks per axis. Adjacent
586        // sub-chunks share their edge vertex so total vertex count per side is
587        // `chunks_per_side * (chunk_res - 1) + 1`.
588        let chunks_per_side = 1 + 2 * halo_chunks;
589        let halo_res = chunks_per_side * (chunk_res - 1) + 1;
590        let halo_size_world = chunks_per_side as f32 * chunk_size;
591
592        // Origin world coordinates = target chunk origin minus (halo_chunks × chunk_size).
593        let target_origin = target_chunk_id.to_world_pos(chunk_size);
594        let halo_origin_x = target_origin.x - halo_chunks as f32 * chunk_size;
595        let halo_origin_z = target_origin.z - halo_chunks as f32 * chunk_size;
596
597        // Sample the noise field directly at per-vertex world coordinates. This
598        // path uses `TerrainNoise::sample_height` rather than the SIMD
599        // heightmap generator because the SIMD path is tied to a ChunkId. For
600        // byte-identity with the SIMD path at the center crop, both routes
601        // must produce the same Y at the same world (x, z) — confirmed by the
602        // determinism invariant of `TerrainNoise`.
603        let heightmap_config = heightmap::HeightmapConfig {
604            resolution: halo_res,
605            ..Default::default()
606        };
607        let mut halo_map = heightmap::Heightmap::new(heightmap_config)?;
608        for z_idx in 0..halo_res {
609            for x_idx in 0..halo_res {
610                let wx = halo_origin_x + x_idx as f32 * step;
611                let wz = halo_origin_z + z_idx as f32 * step;
612                let y = self.noise.sample_height(wx as f64, wz as f64);
613                halo_map.set_height(x_idx, z_idx, y);
614            }
615        }
616
617        // Silence unused warning for halo_size_world — useful for phase 2
618        // perf logging and halo-overlap sanity assertions.
619        let _ = halo_size_world;
620
621        Ok(halo_map)
622    }
623
624    /// Phase 1.6-F.3-phase-1.B: crop the center chunk-sized region out of a
625    /// halo-expanded heightmap. Assumes the halo was built by
626    /// `generate_halo_heightmap` with the same `target_chunk_id` and halo
627    /// size. Returns a single-chunk heightmap (resolution matching
628    /// `WorldConfig::heightmap_resolution`).
629    pub(crate) fn crop_halo_to_chunk(
630        &self,
631        halo: &heightmap::Heightmap,
632        target_chunk_id: ChunkId,
633    ) -> anyhow::Result<heightmap::Heightmap> {
634        let _ = target_chunk_id; // centered crop — target id not needed
635        let chunk_res = self.config.heightmap_resolution;
636        let halo_res = halo.resolution();
637        // Infer halo_chunks from resolutions.
638        //   halo_res = chunks_per_side * (chunk_res - 1) + 1
639        //   chunks_per_side = (halo_res - 1) / (chunk_res - 1)
640        //   halo_chunks = (chunks_per_side - 1) / 2
641        if chunk_res == 0 || (halo_res - 1) % (chunk_res - 1) != 0 {
642            anyhow::bail!(
643                "halo resolution {} not a multiple of chunk resolution {}",
644                halo_res,
645                chunk_res
646            );
647        }
648        let chunks_per_side = (halo_res - 1) / (chunk_res - 1);
649        if chunks_per_side < 1 || chunks_per_side % 2 != 1 {
650            anyhow::bail!(
651                "halo has {} chunks per side; expected odd >= 1",
652                chunks_per_side
653            );
654        }
655        let halo_chunks = (chunks_per_side - 1) / 2;
656        let start = halo_chunks * (chunk_res - 1);
657
658        let chunk_cfg = heightmap::HeightmapConfig {
659            resolution: chunk_res,
660            ..Default::default()
661        };
662        let mut cropped = heightmap::Heightmap::new(chunk_cfg)?;
663        for z in 0..chunk_res {
664            for x in 0..chunk_res {
665                let y = halo.get_height(start + x, start + z);
666                cropped.set_height(x, z, y);
667            }
668        }
669        Ok(cropped)
670    }

672    /// Phase 1.6-F.3-phase-1.B / §2.3: derive a deterministic seed for the
673    /// halo region centered on `target_chunk_id`. Phase 2 feeds this seed to
674    /// `AdvancedErosionSimulator::new` so adjacent chunks' halos that overlap
675    /// in world space produce identical droplet trajectories in the overlap
676    /// region.
677    #[allow(dead_code)] // Wired in phase 2; validated by unit tests here.
678    pub(crate) fn halo_seed(
679        world_seed: u64,
680        target_chunk_id: ChunkId,
681        halo_chunks: u32,
682    ) -> u64 {
683        let halo_origin_x = target_chunk_id.x.wrapping_sub(halo_chunks as i32);
684        let halo_origin_z = target_chunk_id.z.wrapping_sub(halo_chunks as i32);
685        // Wang-style integer hash, deterministic across runs / platforms.
686        let mut h = world_seed;
687        h = h
688            .wrapping_add(halo_origin_x as u64)
689            .wrapping_mul(0x9E3779B97F4A7C15);
690        h ^= h >> 32;
691        h = h
692            .wrapping_add(halo_origin_z as u64)
693            .wrapping_mul(0x85EBCA6BE11ECC0D);
694        h ^= h >> 32;
695        h
696    }

698    /// Generate and register a chunk (mutable version for compatibility)

701        chunk_id: ChunkId,
702    ) -> anyhow::Result<TerrainChunk> {

704        self.chunk_manager.add_chunk(chunk.clone());
705        Ok(chunk)
706    }

713    /// Stream chunks around a center position, loading/unloading as needed
714    pub fn stream_chunks(&mut self, center: Vec3, radius: u32) -> anyhow::Result<Vec<ChunkId>> {
715        let chunks_to_load = self.chunk_manager.get_chunks_in_radius(center, radius);
716        let mut loaded = Vec::new();
717
718        for chunk_id in chunks_to_load {
719            if !self.chunk_manager.has_chunk(chunk_id) {
720                self.generate_and_register_chunk(chunk_id)?;
721                loaded.push(chunk_id);
722            }
723        }
724
725        // Unload chunks that are too far away
726        let unload_radius = radius + 2; // Keep a buffer
727        self.chunk_manager
728            .unload_distant_chunks(center, unload_radius);
729
730        Ok(loaded)
731    }

733    /// Assign biomes to heightmap points based on climate data
734    fn assign_biomes(
735        &self,
736        heightmap: &Heightmap,
737        climate_data: &[(f32, f32)], // (temperature, moisture) pairs
738    ) -> anyhow::Result<Vec<BiomeType>> {
739        let mut biome_map = Vec::with_capacity(climate_data.len());
740
741        for (i, &(temperature, moisture)) in climate_data.iter().enumerate() {
742            let height = heightmap.get_height_at_index(i);
743            let biome = self.find_best_biome(height, temperature, moisture);
744            biome_map.push(biome);
745        }
746
747        Ok(biome_map)
748    }

750    /// Find the best biome for given environmental conditions
751    fn find_best_biome(&self, height: f32, temperature: f32, moisture: f32) -> BiomeType {
752        let mut best_biome = BiomeType::Grassland;
753        let mut best_score = f32::NEG_INFINITY;
754
755        for biome_config in &self.config.biomes {
756            let score = biome_config.score_conditions(height, temperature, moisture);
757            if score > best_score {
758                best_score = score;
759                best_biome = biome_config.biome_type;
760            }
761        }
762
763        best_biome
764    }
