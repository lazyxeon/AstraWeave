184    /// Generate a terrain chunk at the given world position with vegetation and resources
185    pub fn generate_chunk_with_scatter(
186        &mut self,
187        chunk_id: ChunkId,
188    ) -> anyhow::Result<(TerrainChunk, ScatterResult)> {
189        // Generate the basic terrain chunk (lock-free)
190        let chunk = self.generate_chunk(chunk_id)?;

192        // Register with chunk manager
193        self.chunk_manager.add_chunk(chunk.clone());

198        Ok((chunk, scatter_result))

201    /// Generate scatter content (vegetation and resources) for an existing chunk
202    pub fn scatter_chunk_content(&mut self, chunk: &TerrainChunk) -> anyhow::Result<ScatterResult> {
203        let mut result = ScatterResult::new(chunk.id());
204
205        // Create scatter system
206        let scatter_config = ScatterConfig::default();
207        let scatter = VegetationScatter::new(scatter_config);
208
209        // Sample the biome at the chunk center to determine configuration
210        let chunk_center = chunk.id().to_center_pos(self.config.chunk_size);
211        let center_biome = chunk
212            .get_biome_at_world_pos(chunk_center, self.config.chunk_size)
213            .unwrap_or(BiomeType::Grassland);
214
215        // Find the biome configuration
216        let biome_config = self
217            .config
218            .biomes
219            .iter()
220            .find(|b| b.biome_type == center_biome)
221            .unwrap_or(&self.config.biomes[0]);
222
223        // Generate vegetation
224        result.vegetation = scatter.scatter_vegetation(
225            chunk,
226            self.config.chunk_size,
227            biome_config,
228            self.config.seed + chunk.id().x as u64 * 1000 + chunk.id().z as u64,
229        )?;
230
231        // Generate resources
232        result.resources = scatter.scatter_resources(
233            chunk,
234            self.config.chunk_size,
235            biome_config,
236            self.config.seed + chunk.id().x as u64 * 2000 + chunk.id().z as u64,
237        )?;
238
239        // Generate structures
240        let structure_result = self.structure_generator.generate_structures(
241            chunk,
242            self.config.chunk_size,
243            center_biome,
244        )?;
245        result.structures = structure_result.structures;
246
247        Ok(result)
248    }

250    /// Generate a terrain chunk at the given position (lock-free, parallel-safe)
251    /// NOTE: Does NOT add to chunk_manager - caller must handle that separately
252    pub fn generate_chunk(&self, chunk_id: ChunkId) -> anyhow::Result<TerrainChunk> {
253        // Generate heightmap for this chunk (using SIMD if enabled)
254        #[cfg(feature = "simd-noise")]
255        let heightmap = noise_simd::SimdHeightmapGenerator::generate_heightmap_simd(
256            &self.noise,
257            chunk_id,
258            self.config.chunk_size,
259            self.config.heightmap_resolution,
260        )?;
261
262        #[cfg(not(feature = "simd-noise"))]
263        let heightmap = self.noise.generate_heightmap(
264            chunk_id,
265            self.config.chunk_size,
266            self.config.heightmap_resolution,
267        )?;
268
269        // Generate climate data for biome assignment
270        let climate_data = self.climate.sample_chunk(
271            chunk_id,
272            self.config.chunk_size,
273            self.config.heightmap_resolution,
274        )?;
275
276        // Assign biomes based on height and climate
277        let biome_map = self.assign_biomes(&heightmap, &climate_data)?;
278
279        // Create the terrain chunk
280        let mut chunk = TerrainChunk::new(chunk_id, heightmap, biome_map);
281
282        // Apply erosion if enabled
283        if self.config.noise.erosion_enabled {
284            chunk.apply_erosion(self.config.noise.erosion_strength)?;
285        }
286
287        // NOTE: chunk_manager.add_chunk() removed - caller handles registration
288        Ok(chunk)
289    }

291    /// Phase 1.6-F.3-phase-1: generate a chunk with pre-erosion biome_weights
292    /// computed from the given climate bias. The returned chunk has
293    /// `biome_weights: Some(_)` populated from the PRE-erosion heightmap,
294    /// satisfying §2.5's authorial-intent invariant. The heightmap itself is
295    /// post-erosion (simple CA for phase 1; `AdvancedErosionSimulator` for
296    /// phase 2).
297    ///
298    /// Callers not requiring pre-erosion biome_weights should use the simpler
299    /// `generate_chunk` (biome_weights stays `None`, editor/consumer computes
300    /// them on-the-fly from post-erosion heights — current behavior).
301    pub fn generate_chunk_with_climate(
302        &self,
303        chunk_id: ChunkId,
304        climate_bias: crate::ClimateBias,
305    ) -> anyhow::Result<TerrainChunk> {
306        // Phase 1.6-F.3-phase-2.C: full pipeline wired.
307        //
308        // 1. Generate halo heightmap (3×3 chunks centered on target).
309        // 2. Crop the halo to the target chunk for PRE-erosion biome_weights
310        //    computation per §2.5 (authorial-intent invariant).
311        // 3. Run AdvancedErosionSimulator::apply_preset on the FULL halo —
312        //    droplets travel freely across chunk boundaries within the halo
313        //    region. §2.3 halo=1 is sufficient per phase-0's p95 = 120 world
314        //    units < 256 (one chunk width).
315        // 4. Crop AFTER erosion. This becomes the chunk's final (post-erosion)
316        //    heightmap.
317        // 5. Construct chunk with post-erosion cropped heightmap + pre-erosion
318        //    biome_weights (decoupled per §2.5).
319        //
320        // Legacy `chunk.apply_erosion(strength)` simple CA call is replaced by
321        // `AdvancedErosionSimulator` per §2.2 climate → preset mapping.
322        const HALO_CHUNKS: u32 = 1;
323
324        let mut halo = self.generate_halo_heightmap(chunk_id, HALO_CHUNKS)?;
325
326        // Pre-erosion cropped heightmap — input to biome_weights computation.
327        let pre_erosion_heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
328
329        // §2.5: biome_weights are computed from PRE-erosion heights. Once
330        // phase 2 lands real erosion, a vertex that drops from Y=50 to Y=30
331        // keeps its pre-erosion Mountain weighting — authorial intent over
332        // geological reclassification.
333        let resolution = pre_erosion_heightmap.resolution() as usize;
334        let mut biome_weights = Vec::with_capacity(resolution * resolution);
335        for z in 0..resolution {
336            for x in 0..resolution {
337                let y = pre_erosion_heightmap.get_height(x as u32, z as u32);
338                biome_weights.push(crate::elevation_to_biome_weights(
339                    y,
340                    crate::SEA_LEVEL,
341                    climate_bias,
342                ));
343            }
344        }
345
346        // §2.2 climate → ErosionPreset mapping.
347        let preset = crate::advanced_erosion::erosion_preset_for_climate(climate_bias);
348
349        // Phase 1.6-F.3-phase-3.C: use `apply_preset_at_world_offset` for
350        // seamless chunk boundaries. Droplet spawn positions are derived
351        // from world-aligned spatial cells, so adjacent halos iterate the
352        // SAME cells in overlap → identical droplets → near-identical
353        // erosion output in overlap region.
354        //
355        // The simulator's `new(seed)` argument is no longer the primary
356        // determinism driver — per-droplet RNG comes from world-cell hash
357        // inside the new API. We pass `halo_seed` for any ancillary RNG use
358        // (currently unused inside the world-coord path).
359        let seed = Self::halo_seed(self.config.seed, chunk_id, HALO_CHUNKS);
360        let mut simulator = crate::advanced_erosion::AdvancedErosionSimulator::new(seed);
361
362        if self.config.noise.erosion_enabled {
363            // Halo's world origin (target chunk origin minus halo_chunks * chunk_size).
364            let target_origin = chunk_id.to_world_pos(self.config.chunk_size);
365            let halo_origin_x = (target_origin.x - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
366            let halo_origin_z = (target_origin.z - HALO_CHUNKS as f32 * self.config.chunk_size) as f64;
367            // Vertex spacing: chunk_size per (resolution - 1) vertices.
368            let vertex_spacing =
369                self.config.chunk_size as f64 / (self.config.heightmap_resolution - 1) as f64;
370
371            let _stats = simulator.apply_preset_at_world_offset(
372                &mut halo,
373                &preset,
374                halo_origin_x,
375                halo_origin_z,
376                vertex_spacing,
377                self.config.seed,
378            );
379            halo.recalculate_bounds();
380        }
381
382        // Post-erosion cropped heightmap — final chunk output.
383        let heightmap = self.crop_halo_to_chunk(&halo, chunk_id)?;
384
385        // biome_map still computed from PRE-erosion heights + climate data
386        // (unchanged path). Phase 4 will route climate through biome_weights;
387        // biome_map currently feeds splat-rule selection elsewhere.
388        let climate_data = self.climate.sample_chunk(
389            chunk_id,
390            self.config.chunk_size,
391            self.config.heightmap_resolution,
392        )?;
393        let biome_map = self.assign_biomes(&pre_erosion_heightmap, &climate_data)?;
394
395        // Construct chunk: post-erosion heights + pre-erosion biome_weights.
396        // No further `apply_erosion` — phase 2 replaces simple CA with
397        // AdvancedErosionSimulator.
398        let chunk = TerrainChunk::new_with_biome_weights(
399            chunk_id,
400            heightmap,
401            biome_map,
402            biome_weights,
403        );
404
405        Ok(chunk)
406    }

408    /// Phase 1.6-F.3-phase-1.B: generate a heightmap covering a 3×3 chunk
409    /// region centered on `target_chunk_id` (halo_chunks=1). The returned
410    /// heightmap has `(1 + 2*halo_chunks) × (heightmap_resolution - 1) + 1`
411    /// vertices per side at the same per-vertex spacing as single-chunk
412    /// generation, so the center third (sampled at the target chunk's world
413    /// coords) is byte-identical to what `generate_chunk` produces via the
414    /// SIMD heightmap generator.
415    pub(crate) fn generate_halo_heightmap(
416        &self,
417        target_chunk_id: ChunkId,
418        halo_chunks: u32,
419    ) -> anyhow::Result<heightmap::Heightmap> {
420        let chunk_res = self.config.heightmap_resolution;
421        let chunk_size = self.config.chunk_size;
422        // Per-vertex step in world units. Must match single-chunk generation.
423        let step = chunk_size / (chunk_res - 1) as f32;
424
425        // Halo region: (1 + 2*halo_chunks) sub-chunks per axis. Adjacent
426        // sub-chunks share their edge vertex so total vertex count per side is
427        // `chunks_per_side * (chunk_res - 1) + 1`.
428        let chunks_per_side = 1 + 2 * halo_chunks;
429        let halo_res = chunks_per_side * (chunk_res - 1) + 1;
430        let halo_size_world = chunks_per_side as f32 * chunk_size;
431
432        // Origin world coordinates = target chunk origin minus (halo_chunks × chunk_size).
433        let target_origin = target_chunk_id.to_world_pos(chunk_size);
434        let halo_origin_x = target_origin.x - halo_chunks as f32 * chunk_size;
435        let halo_origin_z = target_origin.z - halo_chunks as f32 * chunk_size;
436
437        // Sample the noise field directly at per-vertex world coordinates. This
438        // path uses `TerrainNoise::sample_height` rather than the SIMD
439        // heightmap generator because the SIMD path is tied to a ChunkId. For
440        // byte-identity with the SIMD path at the center crop, both routes
441        // must produce the same Y at the same world (x, z) — confirmed by the
442        // determinism invariant of `TerrainNoise`.
443        let heightmap_config = heightmap::HeightmapConfig {
444            resolution: halo_res,
445            ..Default::default()
446        };
447        let mut halo_map = heightmap::Heightmap::new(heightmap_config)?;
448        for z_idx in 0..halo_res {
449            for x_idx in 0..halo_res {
450                let wx = halo_origin_x + x_idx as f32 * step;
451                let wz = halo_origin_z + z_idx as f32 * step;
452                let y = self.noise.sample_height(wx as f64, wz as f64);
453                halo_map.set_height(x_idx, z_idx, y);
454            }
455        }
456
457        // Silence unused warning for halo_size_world — useful for phase 2
458        // perf logging and halo-overlap sanity assertions.
459        let _ = halo_size_world;
460
461        Ok(halo_map)
462    }

464    /// Phase 1.6-F.3-phase-1.B: crop the center chunk-sized region out of a
465    /// halo-expanded heightmap. Assumes the halo was built by
466    /// `generate_halo_heightmap` with the same `target_chunk_id` and halo
467    /// size. Returns a single-chunk heightmap (resolution matching
468    /// `WorldConfig::heightmap_resolution`).
469    pub(crate) fn crop_halo_to_chunk(
470        &self,
471        halo: &heightmap::Heightmap,
472        target_chunk_id: ChunkId,
473    ) -> anyhow::Result<heightmap::Heightmap> {
474        let _ = target_chunk_id; // centered crop — target id not needed
475        let chunk_res = self.config.heightmap_resolution;
476        let halo_res = halo.resolution();
477        // Infer halo_chunks from resolutions.
478        //   halo_res = chunks_per_side * (chunk_res - 1) + 1
479        //   chunks_per_side = (halo_res - 1) / (chunk_res - 1)
480        //   halo_chunks = (chunks_per_side - 1) / 2
481        if chunk_res == 0 || (halo_res - 1) % (chunk_res - 1) != 0 {
482            anyhow::bail!(
483                "halo resolution {} not a multiple of chunk resolution {}",
484                halo_res,
485                chunk_res
486            );
487        }
488        let chunks_per_side = (halo_res - 1) / (chunk_res - 1);
489        if chunks_per_side < 1 || chunks_per_side % 2 != 1 {
490            anyhow::bail!(
491                "halo has {} chunks per side; expected odd >= 1",
492                chunks_per_side
493            );
494        }
495        let halo_chunks = (chunks_per_side - 1) / 2;
496        let start = halo_chunks * (chunk_res - 1);
497
498        let chunk_cfg = heightmap::HeightmapConfig {
499            resolution: chunk_res,
500            ..Default::default()
501        };
502        let mut cropped = heightmap::Heightmap::new(chunk_cfg)?;
503        for z in 0..chunk_res {
504            for x in 0..chunk_res {
505                let y = halo.get_height(start + x, start + z);
506                cropped.set_height(x, z, y);
507            }
508        }
509        Ok(cropped)
510    }

512    /// Phase 1.6-F.3-phase-1.B / §2.3: derive a deterministic seed for the
513    /// halo region centered on `target_chunk_id`. Phase 2 feeds this seed to
514    /// `AdvancedErosionSimulator::new` so adjacent chunks' halos that overlap
515    /// in world space produce identical droplet trajectories in the overlap
516    /// region.
517    #[allow(dead_code)] // Wired in phase 2; validated by unit tests here.
518    pub(crate) fn halo_seed(
519        world_seed: u64,
520        target_chunk_id: ChunkId,
521        halo_chunks: u32,
522    ) -> u64 {
523        let halo_origin_x = target_chunk_id.x.wrapping_sub(halo_chunks as i32);
524        let halo_origin_z = target_chunk_id.z.wrapping_sub(halo_chunks as i32);
525        // Wang-style integer hash, deterministic across runs / platforms.
526        let mut h = world_seed;
527        h = h
528            .wrapping_add(halo_origin_x as u64)
529            .wrapping_mul(0x9E3779B97F4A7C15);
530        h ^= h >> 32;
531        h = h
532            .wrapping_add(halo_origin_z as u64)
533            .wrapping_mul(0x85EBCA6BE11ECC0D);
534        h ^= h >> 32;
535        h
536    }

553    /// Stream chunks around a center position, loading/unloading as needed
554    pub fn stream_chunks(&mut self, center: Vec3, radius: u32) -> anyhow::Result<Vec<ChunkId>> {
555        let chunks_to_load = self.chunk_manager.get_chunks_in_radius(center, radius);
556        let mut loaded = Vec::new();
557
558        for chunk_id in chunks_to_load {
559            if !self.chunk_manager.has_chunk(chunk_id) {
560                self.generate_and_register_chunk(chunk_id)?;
561                loaded.push(chunk_id);
562            }
563        }
564
565        // Unload chunks that are too far away
566        let unload_radius = radius + 2; // Keep a buffer
567        self.chunk_manager
568            .unload_distant_chunks(center, unload_radius);
569
570        Ok(loaded)
571    }
572
573    /// Assign biomes to heightmap points based on climate data
574    fn assign_biomes(
575        &self,
576        heightmap: &Heightmap,
577        climate_data: &[(f32, f32)], // (temperature, moisture) pairs
578    ) -> anyhow::Result<Vec<BiomeType>> {
579        let mut biome_map = Vec::with_capacity(climate_data.len());
580
581        for (i, &(temperature, moisture)) in climate_data.iter().enumerate() {
582            let height = heightmap.get_height_at_index(i);
583            let biome = self.find_best_biome(height, temperature, moisture);
584            biome_map.push(biome);
585        }
586
587        Ok(biome_map)
588    }
589
590    /// Find the best biome for given environmental conditions
591    fn find_best_biome(&self, height: f32, temperature: f32, moisture: f32) -> BiomeType {
592        let mut best_biome = BiomeType::Grassland;
593        let mut best_score = f32::NEG_INFINITY;
594
595        for biome_config in &self.config.biomes {
596            let score = biome_config.score_conditions(height, temperature, moisture);
597            if score > best_score {
598                best_score = score;
599                best_biome = biome_config.biome_type;
600            }
601        }
602
603        best_biome
604    }
