45/// Archetype storage: all entities with the same component signature
46///
47/// # Storage Modes
48///
49/// The archetype supports two storage modes for backward compatibility:
50///
51/// 1. **Box Mode** (legacy): Uses `Vec<Box<dyn Any>>` for each component column.
52///    - Higher overhead due to heap indirection and virtual dispatch
53///    - Used when component metadata is not available
54///
55/// 2. **BlobVec Mode** (optimized): Uses contiguous `BlobVec` for each column.
56///    - Zero heap indirection, cache-friendly iteration
57///    - Requires `ComponentMeta` to be provided at archetype creation
58///
59/// The `uses_blob` flag indicates which mode is active.
60///
61/// # Performance Note (Lazy Initialization)
62///
63/// BlobVec fields (`blob_components`, `component_metas`) use `Option` for lazy
64/// initialization. This avoids HashMap allocation overhead in Box mode, which
65/// is critical for spawn/despawn performance (previous regression: +388%).
66pub struct Archetype {
67    pub id: ArchetypeId,
68    pub signature: ArchetypeSignature,
69
70    /// Packed entity list for iteration (cache-friendly)
71    entities: Vec<Entity>,
72
73    /// O(1) entity lookup (replaces BTreeMap)
74    entity_index: SparseSet,
75
76    /// Legacy component columns: TypeId -> Vec<Box<dyn Any>>
77    /// Used when ComponentMeta is not available
78    components: HashMap<TypeId, Vec<Box<dyn std::any::Any + Send + Sync>>>,
79
80    // ========================================================================
81    // BlobVec Storage (High-Performance Path)
82    // ========================================================================
83    /// BlobVec component columns: TypeId -> BlobVec
84    /// Used when ComponentMeta is available (typed path)
85    /// NOTE: Lazy initialized (Option) to avoid allocation in Box mode
86    blob_components: Option<HashMap<TypeId, BlobVec>>,
87
88    /// Component metadata for each type (needed for clone/drop in transitions)
89    /// NOTE: Lazy initialized (Option) to avoid allocation in Box mode
90    component_metas: Option<HashMap<TypeId, ComponentMeta>>,
91
92    /// Whether this archetype uses BlobVec storage (true) or Box storage (false)
93    uses_blob: bool,
94}

98    ///
99    /// BlobVec fields are NOT allocated (lazy initialization) to avoid

101    pub fn new(id: ArchetypeId, signature: ArchetypeSignature) -> Self {
102        let mut components = HashMap::new();
103        for ty in &signature.components {

105        }

108            signature,
109            entities: Vec::new(),
110            entity_index: SparseSet::new(),
111            components,
112            // BlobVec fields NOT allocated (lazy init for performance)
113            blob_components: None,
114            component_metas: None,
115            uses_blob: false,
116        }
117    }

119    /// Create a new archetype with BlobVec storage (high-performance mode).
120    ///
121    /// # Arguments
122    /// * `id` - Unique archetype identifier
123    /// * `signature` - Component type signature
124    /// * `metas` - HashMap of TypeId -> ComponentMeta for each component
125    pub fn new_with_blob(
126        id: ArchetypeId,
127        signature: ArchetypeSignature,
128        metas: HashMap<TypeId, ComponentMeta>,
129    ) -> Self {
130        let mut blob_components = HashMap::new();
131        for ty in &signature.components {
132            if let Some(meta) = metas.get(ty) {
133                blob_components.insert(*ty, meta.create_blob_vec());
134            }
135        }
136
137        Self {
138            id,
139            signature,
140            entities: Vec::new(),
141            entity_index: SparseSet::new(),
142            components: HashMap::new(), // Not used in blob mode
143            blob_components: Some(blob_components),
144            component_metas: Some(metas),
145            uses_blob: true,
146        }
147    }

155    /// Add an entity with its components (must match signature)
156    #[allow(clippy::expect_used)] // INVARIANT: signature components always have matching columns (created together)
157    pub fn add_entity(
158        &mut self,
159        entity: Entity,
160        mut component_data: HashMap<TypeId, Box<dyn std::any::Any + Send + Sync>>,
161    ) {
162        // Use SparseSet for O(1) lookup (12-57× faster than BTreeMap)
163        self.entity_index.insert(entity);
164        self.entities.push(entity);
165
166        for ty in &self.signature.components {
167            if let Some(data) = component_data.remove(ty) {
168                // Move the Box from component_data into the column
169                let column = self
170                    .components
171                    .get_mut(ty)
172                    .expect("BUG: signature component should have column");
173                column.push(data);
174            }
175        }
176    }

182    /// Add a single typed component to an entity already in this archetype.
183    ///
184    /// This is used for the high-performance typed path when transitioning
185    /// entities between archetypes.
186    ///
187    /// # Safety
188    /// - Entity must already be added to this archetype
189    /// - Component type must be in the signature
190    /// - Archetype must use BlobVec storage
191    pub fn push_component_typed<T: Component>(&mut self, component: T) {
192        debug_assert!(self.uses_blob, "push_component_typed requires BlobVec mode");
193        let type_id = TypeId::of::<T>();
194
195        if let Some(blob_components) = &mut self.blob_components {
196            if let Some(blob) = blob_components.get_mut(&type_id) {
197                // SAFETY: Caller ensures T matches the component type registered
198                // for this column. BlobVec handles capacity internally.
199                unsafe {
200                    blob.push(component);
201                }
202            }
203        }
204    }

206    /// Add entity with typed components using BlobVec storage.
207    ///
208    /// This bypasses Box<dyn Any> for maximum performance.
209    ///
210    /// # Arguments
211    /// * `entity` - The entity to add
212    /// * `components` - Raw component data as bytes with clone functions
213    pub fn add_entity_typed_raw(&mut self, entity: Entity, components: &[(TypeId, *const u8)]) {
214        debug_assert!(self.uses_blob, "add_entity_typed_raw requires BlobVec mode");
215
216        self.entity_index.insert(entity);
217        self.entities.push(entity);
218
219        let blob_components = match &mut self.blob_components {
220            Some(bc) => bc,
221            None => return,
222        };
223        let component_metas = match &self.component_metas {
224            Some(cm) => cm,
225            None => return,
226        };
227
228        for (type_id, src_ptr) in components {
229            if let (Some(blob), Some(meta)) = (
230                blob_components.get_mut(type_id),
231                component_metas.get(type_id),
232            ) {
233                // SAFETY: `src_ptr` points to valid component data (caller guarantee).
234                // `meta.clone_fn` is the matching clone function from component registration.
235                unsafe {
236                    blob.push_raw(*src_ptr, meta.clone_fn);
237                }
238            }
239        }
240    }

242    /// Get component for entity (hybrid: checks BlobVec first, then Box)
243    pub fn get<T: Component>(&self, entity: Entity) -> Option<&T> {
244        let row = self.entity_index.get(entity)?;
245
246        // Fast path: BlobVec storage
247        if self.uses_blob {
248            let blob = self.blob_components.as_ref()?.get(&TypeId::of::<T>())?;
249            // SAFETY: T matches the type this column was created for (same TypeId lookup).
250            // `row` is a valid index from entity_index.
251            return unsafe { blob.get::<T>(row) };
252        }
253
254        // Legacy path: Box storage
255        let column = self.components.get(&TypeId::of::<T>())?;
256        let boxed = column.get(row)?;
257        boxed.downcast_ref::<T>()
258    }

260    /// Get mutable component for entity (hybrid: checks BlobVec first, then Box)
261    pub fn get_mut<T: Component>(&mut self, entity: Entity) -> Option<&mut T> {
262        let row = self.entity_index.get(entity)?;
263
264        // Fast path: BlobVec storage
265        if self.uses_blob {
266            let blob = self.blob_components.as_mut()?.get_mut(&TypeId::of::<T>())?;
267            // SAFETY: T matches the type this column was created for (same TypeId lookup).
268            // `row` is a valid index from entity_index. Mutable borrow of self prevents aliasing.
269            return unsafe { blob.get_mut::<T>(row) };
270        }
271
272        // Legacy path: Box storage
273        let column = self.components.get_mut(&TypeId::of::<T>())?;
274        let boxed = column.get_mut(row)?;
275        boxed.downcast_mut::<T>()
276    }

283    /// Remove entity from archetype and return its components
284    pub fn remove_entity_components(
285        &mut self,
286        entity: Entity,
287    ) -> HashMap<TypeId, Box<dyn std::any::Any + Send + Sync>> {
288        // NEW: O(1) removal with SparseSet
289        let row = match self.entity_index.remove(entity) {
290            Some(r) => r,
291            None => return HashMap::new(),
292        };
293
294        // Remove from packed entity list using swap_remove
295        let entities_len = self.entities.len();
296        if row < entities_len - 1 {
297            self.entities.swap(row, entities_len - 1);
298            // Update the swapped entity's index in SparseSet
299            let swapped_entity = self.entities[row];
300            self.entity_index.insert(swapped_entity); // Will update to correct row
301        }
302        self.entities.pop();
303
304        let mut components = HashMap::new();
305        for (ty, column) in self.components.iter_mut() {
306            let component = column.swap_remove(row);
307            components.insert(*ty, component);
308        }
309
310        components
311    }

326    /// Iterate over (entity, component) pairs for batch processing.
327    ///
328    /// This is much faster than repeated get() calls as it avoids per-entity lookups.
329    ///
330    /// ## Performance Notes (Week 10)
331    ///
332    /// With SparseSet integration, get() is now O(1) instead of O(log n), providing
333    /// 12-57× speedup over the old BTreeMap approach. This iterator provides additional
334    /// benefits by reducing function call overhead and improving cache locality.
335    ///
336    /// ## Mutable Iterator Limitation
337    ///
338    /// Note: A mutable version (`iter_components_mut<T>()`) is **not feasible** due to
339    /// Rust's borrow checker limitations. The issue is:
340    ///
341    /// ```rust,ignore
342    /// pub fn iter_components_mut<T>(&mut self) -> impl Iterator<Item = (Entity, &mut T)> {
343    ///     let column = self.components.get_mut(&TypeId::of::<T>())?;
344    ///     self.entities.iter().filter_map(|(idx, &entity)| {
345    ///         column.get_mut(idx)  // ❌ ERROR: captured variable escapes FnMut closure
346    ///     })
347    /// }
348    /// ```
349    ///
350    /// Rust prevents this because the closure captures `column` and tries to return
351    /// `&mut T` borrowed from it. The borrow checker rule is: **references captured in
352    /// closures cannot escape the closure scope**. This prevents dangling references.
353    ///
354    /// **Workarounds considered**:
355    /// - Unsafe raw pointers: Would work but loses safety guarantees (not worth it)
356    /// - Index-based batch API: Complex redesign with uncertain performance gains
357    /// - Type registry + BlobVec: Full solution but requires architectural changes (Week 13+)
358    ///
359    /// **Current approach**: Accept that SparseSet O(1) already provides 2.4× frame time
360    /// improvement (2.70ms → 1.144ms) and 9.4× faster movement (1,000µs → 106µs). Further
361    /// query optimization has diminishing returns vs complexity/safety trade-offs.
362    pub fn iter_components<T: Component>(&self) -> impl Iterator<Item = (Entity, &T)> + '_ {
363        // NOTE: We can't easily branch on uses_blob here because the return types differ.
364        // The BlobVec path would return a different iterator type.
365        // For now, always use the Box path in iter_components.
366        // The typed fast path is available via iter_components_blob().
367        let column = self.components.get(&TypeId::of::<T>());
368        self.entities
369            .iter()
370            .enumerate()
371            .filter_map(move |(idx, &entity)| {
372                column
373                    .and_then(|col| col.get(idx))
374                    .and_then(|boxed| boxed.downcast_ref::<T>())
375                    .map(|component| (entity, component))
376            })
377    }

379    /// High-performance iteration using BlobVec storage.
380    ///
381    /// Returns a slice of components for direct SIMD-friendly iteration.
382    /// Only works for archetypes using BlobVec mode.
383    ///
384    /// # Returns
385    /// - `Some((entities, components))` if this archetype uses BlobVec and has the component
386    /// - `None` if using Box mode or component not present
387    ///
388    /// # Safety
389    /// The returned slice is only valid while the archetype is not modified.
390    pub fn iter_components_blob<T: Component>(&self) -> Option<(&[Entity], &[T])> {
391        if !self.uses_blob {
392            return None;
393        }
394
395        let blob = self.blob_components.as_ref()?.get(&TypeId::of::<T>())?;
396        // SAFETY: T matches the column type (same TypeId). Immutable borrow of self
397        // ensures no concurrent modification. Slice is valid for the lifetime of &self.
398        let components = unsafe { blob.as_slice::<T>() };
399        Some((&self.entities, components))
400    }
401
402    /// High-performance mutable iteration using BlobVec storage.
403    ///
404    /// Returns a mutable slice of components for direct modification.
405    /// Only works for archetypes using BlobVec mode.
406    ///
407    /// # Safety
408    /// The returned slice is only valid while the archetype is not modified.
409    pub fn iter_components_blob_mut<T: Component>(&mut self) -> Option<(&[Entity], &mut [T])> {
410        if !self.uses_blob {
411            return None;
412        }
413
414        let blob = self.blob_components.as_mut()?.get_mut(&TypeId::of::<T>())?;
415        // SAFETY: T matches the column type (same TypeId). Mutable borrow of self
416        // prevents aliasing. Slice is valid for the lifetime of &mut self.
417        let components = unsafe { blob.as_slice_mut::<T>() };
418        Some((&self.entities, components))
419    }

422/// Manages all archetypes and entity->archetype mapping
423///
424/// # Determinism Guarantee
425///
426/// **CRITICAL**: This uses `BTreeMap` for archetype storage to ensure deterministic iteration.
427/// Iteration order is sorted by `ArchetypeId`, which preserves archetype creation order
428/// (IDs assigned sequentially via `next_id`).
429///
430/// **Why BTreeMap?**
431/// - HashMap iteration order is **non-deterministic** (depends on hash function, memory layout)
432/// - BTreeMap iteration order is **deterministic** (sorted by key)
433/// - For AI agents, deterministic entity iteration is **critical** for reproducible behavior
434///
435/// **Performance Note**:
436/// - BTreeMap operations are O(log n) vs HashMap O(1)
437/// - With ~100 archetypes typical, log₂(100) ≈ 7 operations (negligible)
438/// - Entity queries iterate archetypes (O(archetypes)), so iteration order matters more than lookup
439///
440/// # Zero-Allocation Hot Path
441///
442/// **CRITICAL**: Entity-to-archetype mapping uses `Vec<Option<ArchetypeId>>` indexed by entity ID
443/// instead of HashMap to ensure zero heap allocations during component access hot paths.
444/// HashMap uses RandomState hasher which can allocate thread-local state on first access.
445#[derive(Default)]
446pub struct ArchetypeStorage {
447    next_id: u64,
448    /// Map from signature to archetype ID
449    signature_to_id: HashMap<ArchetypeSignature, ArchetypeId>,
450    /// All archetypes (BTreeMap for deterministic iteration by ID)
451    archetypes: BTreeMap<ArchetypeId, Archetype>,
452    /// Entity to archetype mapping (sparse array indexed by entity ID for zero-alloc lookup)
453    /// Uses Vec<Option<ArchetypeId>> instead of HashMap for zero-alloc hot path.
454    entity_to_archetype: Vec<Option<ArchetypeId>>,
455    /// Inverted index: component TypeId → Vec of ArchetypeIds containing that component.
456    /// Turns O(n_archetypes) linear scan into O(n_matching) lookup for query resolution.
457    component_to_archetypes: HashMap<TypeId, Vec<ArchetypeId>>,
458}

460impl ArchetypeStorage {
461    pub fn new() -> Self {
462        Self {
463            next_id: 0,
464            signature_to_id: HashMap::new(),
465            archetypes: BTreeMap::new(),
466            entity_to_archetype: Vec::new(),
467            component_to_archetypes: HashMap::new(),
468        }
469    }
470
471    /// Get or create archetype for a signature
472    pub fn get_or_create_archetype(&mut self, signature: ArchetypeSignature) -> ArchetypeId {
473        if let Some(&id) = self.signature_to_id.get(&signature) {
474            return id;
475        }
476
477        let id = ArchetypeId(self.next_id);
478        self.next_id += 1;
479
480        // Update inverted index: register this archetype for each component type
481        for &ty in &signature.components {
482            self.component_to_archetypes.entry(ty).or_default().push(id);
483        }
484
485        let archetype = Archetype::new(id, signature.clone());
486        self.archetypes.insert(id, archetype);
487        self.signature_to_id.insert(signature, id);
488
489        id
490    }
491
492    /// Get or create archetype with BlobVec storage (high-performance mode).
493    ///
494    /// Unlike `get_or_create_archetype`, this creates archetypes with contiguous
495    /// BlobVec storage for faster iteration.
496    ///
497    /// # Arguments
498    /// * `signature` - Component type signature
499    /// * `metas` - Component metadata for each type (must include all signature types)
500    pub fn get_or_create_archetype_with_blob(
501        &mut self,
502        signature: ArchetypeSignature,
503        metas: HashMap<TypeId, ComponentMeta>,
504    ) -> ArchetypeId {
505        if let Some(&id) = self.signature_to_id.get(&signature) {
506            // If archetype already exists but uses Box mode, we can't upgrade it
507            // (would require migrating all existing entities)
508            return id;
509        }
510
511        let id = ArchetypeId(self.next_id);
512        self.next_id += 1;
513
514        // Update inverted index: register this archetype for each component type
515        for &ty in &signature.components {
516            self.component_to_archetypes.entry(ty).or_default().push(id);
517        }
518
519        let archetype = Archetype::new_with_blob(id, signature.clone(), metas);
520        self.archetypes.insert(id, archetype);
521        self.signature_to_id.insert(signature, id);
522
523        id
524    }

526    pub fn get_archetype(&self, id: ArchetypeId) -> Option<&Archetype> {

528    }

541    /// Set archetype for an entity (may allocate if entity ID is larger than current capacity)

544        // Grow the sparse array if needed (only allocates during warmup/setup)
545        if id >= self.entity_to_archetype.len() {
546            self.entity_to_archetype.resize(id + 1, None);
547        }
548        self.entity_to_archetype[id] = Some(archetype);

554        let id = entity.id() as usize;
555        if id < self.entity_to_archetype.len() {
556            self.entity_to_archetype[id].take()
557        } else {
558            None
559        }
560    }

577    /// Find archetypes that contain a specific component.
578    ///
579    /// Uses an inverted index (`component_to_archetypes`) for O(1) lookup
580    /// instead of scanning every archetype.
581    pub fn archetypes_with_component(&self, ty: TypeId) -> impl Iterator<Item = &Archetype> {
582        #[cfg(feature = "profiling")]
583        span!("ECS::Archetype::archetypes_with_component");
584
585        // Empty slice fallback avoids an Option branch in the iterator chain.
586        let ids = self
587            .component_to_archetypes
588            .get(&ty)
589            .map(|v| v.as_slice())
590            .unwrap_or(&[]);
591
592        ids.iter().filter_map(move |id| self.archetypes.get(id))
593    }
