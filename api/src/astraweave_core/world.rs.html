53    pub obstacles: HashSet<(i32, i32)>,

102    fn insert_entity(
103        &mut self,
104        id: Entity,
105        name: &str,
106        pos: IVec2,
107        team: Team,
108        hp: i32,
109        ammo: i32,
110    ) -> Entity {
111        debug_assert!(!self.poses.contains_key(&id), "entity {id} already exists");
112        self.poses.insert(
113            id,
114            Pose {
115                pos,
116                height: 0.0,
117                rotation: 0.0,
118                rotation_x: 0.0,
119                rotation_z: 0.0,
120                scale: 1.0,
121                scale_y: 1.0,
122                scale_z: 1.0,
123                float_x: 0.0,
124                float_z: 0.0,
125                use_float_pos: false,
126            },
127        );
128        self.health.insert(id, Health { hp });
129        self.team.insert(id, team);
130        self.ammo.insert(id, Ammo { rounds: ammo });
131        self.cds.insert(
132            id,
133            Cooldowns {
134                map: BTreeMap::new(),
135            },
136        );
137        self.names.insert(id, name.to_string());
138        id
139    }

141    pub fn tick(&mut self, dt: f32) {
142        self.t += dt;
143        for cd in self.cds.values_mut() {
144            for v in cd.map.values_mut() {
145                *v = (*v - dt).max(0.0);
146            }
147        }
148    }

150    /// Destroy an entity, removing all its components from the world.
151    /// Also cleans up hierarchy: removes from parent's children list,
152    /// and orphans any children (makes them roots).
153    /// Returns true if the entity existed and was destroyed, false otherwise.
154    pub fn destroy_entity(&mut self, e: Entity) -> bool {
155        let existed = self.poses.remove(&e).is_some();
156        if existed {
157            self.health.remove(&e);
158            self.team.remove(&e);
159            self.ammo.remove(&e);
160            self.cds.remove(&e);
161            self.names.remove(&e);
162            self.behavior_graphs.remove(&e);
163
164            // Remove from parent's children list
165            if let Some(parent) = self.parents.remove(&e) {
166                if let Some(siblings) = self.children_map.get_mut(&parent) {
167                    siblings.retain(|&child| child != e);
168                }
169            }
170
171            // Orphan any children (remove their parent reference)
172            if let Some(children) = self.children_map.remove(&e) {
173                for child in children {
174                    self.parents.remove(&child);
175                }
176            }
177        }
178        existed
179    }

232    /// Set the parent of a child entity. Removes any previous parent.
233    /// Does nothing if child == parent or if it would create a cycle.
234    pub fn set_parent(&mut self, child: Entity, parent: Entity) {
235        if child == parent {
236            return;
237        }
238        // Prevent cycles: if child is an ancestor of parent, skip
239        if self.is_ancestor_of(child, parent) {
240            return;
241        }
242        // Remove from old parent's children list
243        self.remove_parent(child);
244        // Set new parent
245        self.parents.insert(child, parent);
246        self.children_map.entry(parent).or_default().push(child);
247    }
248
249    /// Remove the parent of an entity (make it a root).
250    /// Returns the old parent if one existed.
251    pub fn remove_parent(&mut self, child: Entity) -> Option<Entity> {
252        if let Some(old_parent) = self.parents.remove(&child) {
253            if let Some(siblings) = self.children_map.get_mut(&old_parent) {
254                siblings.retain(|&e| e != child);
255                if siblings.is_empty() {
256                    self.children_map.remove(&old_parent);
257                }
258            }
259            Some(old_parent)
260        } else {
261            None
262        }
263    }

278    /// Check if `ancestor` is an ancestor of `descendant` (recursive).
279    pub fn is_ancestor_of(&self, ancestor: Entity, descendant: Entity) -> bool {
280        let mut current = descendant;
281        while let Some(parent) = self.parents.get(&current) {
282            if *parent == ancestor {
283                return true;
284            }
285            current = *parent;
286        }
287        false
288    }

290    /// Collect all descendants of an entity (recursive, depth-first).
291    pub fn descendants_of(&self, e: Entity) -> Vec<Entity> {
292        let mut result = Vec::new();
293        self.collect_descendants(e, &mut result);
294        result
295    }
296
297    fn collect_descendants(&self, e: Entity, result: &mut Vec<Entity>) {
298        for &child in self.children_of(e) {
299            result.push(child);
300            self.collect_descendants(child, result);
301        }
302    }

358    /// Get obstacles as a shared `Arc<Vec<IVec2>>`, suitable for embedding in
359    /// multiple WorldSnapshots without deep-copying. The cache is rebuilt only
360    /// when called after `invalidate_obstacles_cache()` or the first time.
361    pub fn obstacles_as_ivec2(&mut self) -> Arc<Vec<IVec2>> {
362        if let Some(ref cached) = self.obstacles_cache {
363            return Arc::clone(cached);
364        }
365        let vec: Vec<IVec2> = self
366            .obstacles
367            .iter()
368            .map(|&(x, y)| IVec2 { x, y })
369            .collect();
370        let arc = Arc::new(vec);
371        self.obstacles_cache = Some(Arc::clone(&arc));
372        arc
373    }
374
375    /// Invalidate the obstacles cache. Call after modifying `self.obstacles`.
376    pub fn invalidate_obstacles_cache(&mut self) {
377        self.obstacles_cache = None;
378    }
