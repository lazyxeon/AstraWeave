pub const DISNEY_BRDF_WGSL: &str = "// Disney Principled BRDF \u{2014} Full-featured PBR material evaluation\n//\n// Implements the following lobes from the Disney/glTF-extended model:\n//   1. Base diffuse (Burley diffuse model)\n//   2. Specular GGX (microfacet Cook-Torrance: GGX NDF, Smith-GGX geometry, Schlick Fresnel)\n//   3. Clearcoat (GGX with separate roughness, fixed IOR 1.5)\n//   4. Anisotropy (anisotropic GGX with tangent direction)\n//   5. Sheen (Charlie distribution for fabric/velvet)\n//   6. Subsurface approximation (wrap lighting)\n//   7. Transmission (refraction with Beer\'s law attenuation)\n//\n// This file is designed to be #included from the main lighting shader.\n\nconst PI: f32 = 3.14159265358979;\nconst INV_PI: f32 = 0.31830988618;\n\n// ======================================================================\n// Material data (matches MaterialGpuExtended layout)\n// ======================================================================\n\nstruct DisneyMaterial {\n    base_color:           vec3<f32>,\n    alpha:                f32,\n    metallic:             f32,\n    roughness:            f32,\n    occlusion:            f32,\n    flags:                u32,\n    emissive:             vec3<f32>,\n    // Clearcoat\n    clearcoat:            f32,\n    clearcoat_roughness:  f32,\n    // Anisotropy\n    anisotropy:           f32,\n    anisotropy_rotation:  f32,\n    // Subsurface\n    subsurface_color:     vec3<f32>,\n    subsurface_scale:     f32,\n    // Sheen\n    sheen_color:          vec3<f32>,\n    sheen_roughness:      f32,\n    // Transmission\n    transmission:         f32,\n    ior:                  f32,\n    attenuation_color:    vec3<f32>,\n    attenuation_distance: f32,\n};\n\n// Feature flags\nconst FLAG_CLEARCOAT:    u32 = 0x01u;\nconst FLAG_ANISOTROPY:   u32 = 0x02u;\nconst FLAG_SUBSURFACE:   u32 = 0x04u;\nconst FLAG_SHEEN:        u32 = 0x08u;\nconst FLAG_TRANSMISSION: u32 = 0x10u;\n\nfn has_flag(flags: u32, flag: u32) -> bool {\n    return (flags & flag) != 0u;\n}\n\n// ======================================================================\n// Common utility functions\n// ======================================================================\n\nfn pow5(x: f32) -> f32 {\n    let x2 = x * x;\n    return x2 * x2 * x;\n}\n\nfn f0_from_ior(ior: f32) -> f32 {\n    let k = (ior - 1.0) / (ior + 1.0);\n    return k * k;\n}\n\n// ======================================================================\n// 1. Fresnel\n// ======================================================================\n\nfn fresnel_schlick(cos_theta: f32, f0: vec3<f32>) -> vec3<f32> {\n    return f0 + (1.0 - f0) * pow5(saturate(1.0 - cos_theta));\n}\n\nfn fresnel_schlick_roughness(cos_theta: f32, f0: vec3<f32>, roughness: f32) -> vec3<f32> {\n    return f0 + (max(vec3<f32>(1.0 - roughness), f0) - f0) * pow5(saturate(1.0 - cos_theta));\n}\n\nfn fresnel_schlick_scalar(cos_theta: f32, f0: f32) -> f32 {\n    return f0 + (1.0 - f0) * pow5(saturate(1.0 - cos_theta));\n}\n\n// ======================================================================\n// 2. Normal Distribution Functions\n// ======================================================================\n\n// Isotropic GGX (Trowbridge-Reitz)\nfn d_ggx(n_dot_h: f32, alpha: f32) -> f32 {\n    let a2 = alpha * alpha;\n    let d = n_dot_h * n_dot_h * (a2 - 1.0) + 1.0;\n    return a2 / (PI * d * d + 1e-7);\n}\n\n// Anisotropic GGX\nfn d_ggx_aniso(n_dot_h: f32, h: vec3<f32>, t: vec3<f32>, b: vec3<f32>,\n               alpha_t: f32, alpha_b: f32) -> f32 {\n    let h_dot_t = dot(h, t);\n    let h_dot_b = dot(h, b);\n    let at2 = alpha_t * alpha_t;\n    let ab2 = alpha_b * alpha_b;\n    let d = (h_dot_t * h_dot_t / at2) + (h_dot_b * h_dot_b / ab2)\n          + n_dot_h * n_dot_h;\n    return 1.0 / (PI * alpha_t * alpha_b * d * d + 1e-7);\n}\n\n// Charlie distribution for sheen (Conty & Kulla 2017)\nfn d_charlie(n_dot_h: f32, roughness: f32) -> f32 {\n    let alpha = roughness * roughness;\n    let inv_alpha = 1.0 / alpha;\n    let cos2 = n_dot_h * n_dot_h;\n    let sin2 = 1.0 - cos2;\n    return (2.0 + inv_alpha) * pow(sin2, inv_alpha * 0.5) / (2.0 * PI);\n}\n\n// ======================================================================\n// 3. Geometry / Visibility\n// ======================================================================\n\nfn v_smith_ggx(n_dot_v: f32, n_dot_l: f32, alpha: f32) -> f32 {\n    let a2 = alpha * alpha;\n    let ggx_v = n_dot_l * sqrt(n_dot_v * n_dot_v * (1.0 - a2) + a2);\n    let ggx_l = n_dot_v * sqrt(n_dot_l * n_dot_l * (1.0 - a2) + a2);\n    return 0.5 / (ggx_v + ggx_l + 1e-7);\n}\n\nfn v_smith_ggx_aniso(n_dot_v: f32, n_dot_l: f32, v: vec3<f32>, l: vec3<f32>,\n                     t: vec3<f32>, b: vec3<f32>, alpha_t: f32, alpha_b: f32) -> f32 {\n    let tv = dot(t, v); let bv = dot(b, v);\n    let tl = dot(t, l); let bl = dot(b, l);\n    let ggx_v = n_dot_l * sqrt(alpha_t * alpha_t * tv * tv + alpha_b * alpha_b * bv * bv + n_dot_v * n_dot_v);\n    let ggx_l = n_dot_v * sqrt(alpha_t * alpha_t * tl * tl + alpha_b * alpha_b * bl * bl + n_dot_l * n_dot_l);\n    return 0.5 / (ggx_v + ggx_l + 1e-7);\n}\n\n// Kelemen visibility for clearcoat (cheap approximation)\nfn v_kelemen(l_dot_h: f32) -> f32 {\n    return 0.25 / (l_dot_h * l_dot_h + 1e-7);\n}\n\n// ======================================================================\n// 4. Diffuse models\n// ======================================================================\n\n// Burley (Disney) diffuse\nfn diffuse_burley(n_dot_v: f32, n_dot_l: f32, l_dot_h: f32, roughness: f32) -> f32 {\n    let f90 = 0.5 + 2.0 * roughness * l_dot_h * l_dot_h;\n    let light_scatter = 1.0 + (f90 - 1.0) * pow5(1.0 - n_dot_l);\n    let view_scatter = 1.0 + (f90 - 1.0) * pow5(1.0 - n_dot_v);\n    return light_scatter * view_scatter * INV_PI;\n}\n\n// Subsurface wrap lighting approximation\nfn diffuse_subsurface(n_dot_l: f32, n_dot_v: f32, l_dot_h: f32,\n                      roughness: f32, subsurface_scale: f32) -> f32 {\n    let wrap = subsurface_scale * 0.5;\n    let n_dot_l_wrap = (n_dot_l + wrap) / (1.0 + wrap);\n    let f90 = roughness * l_dot_h * l_dot_h;\n    let fl = 1.0 + (f90 - 1.0) * pow5(1.0 - n_dot_l_wrap);\n    let fv = 1.0 + (f90 - 1.0) * pow5(1.0 - n_dot_v);\n    return fl * fv * INV_PI * max(n_dot_l_wrap, 0.0);\n}\n\n// ======================================================================\n// 5. Full Disney BRDF evaluation\n// ======================================================================\n\nstruct BRDFResult {\n    diffuse:  vec3<f32>,\n    specular: vec3<f32>,\n};\n\nfn evaluate_disney_brdf(\n    mat:   DisneyMaterial,\n    n:     vec3<f32>,    // surface normal\n    v:     vec3<f32>,    // view direction\n    l:     vec3<f32>,    // light direction\n    t:     vec3<f32>,    // tangent (for anisotropy)\n    b:     vec3<f32>,    // bitangent\n) -> BRDFResult {\n    let h = normalize(v + l);\n    let n_dot_v = max(dot(n, v), 1e-5);\n    let n_dot_l = max(dot(n, l), 0.0);\n    let n_dot_h = max(dot(n, h), 0.0);\n    let l_dot_h = max(dot(l, h), 0.0);\n\n    let alpha = max(mat.roughness * mat.roughness, 0.002);\n    let f0 = mix(vec3<f32>(0.04), mat.base_color, mat.metallic);\n\n    var result: BRDFResult;\n    result.diffuse = vec3<f32>(0.0);\n    result.specular = vec3<f32>(0.0);\n\n    // --- Diffuse ---\n    if (mat.metallic < 1.0) {\n        var fd: f32;\n        if (has_flag(mat.flags, FLAG_SUBSURFACE) && mat.subsurface_scale > 0.0) {\n            fd = diffuse_subsurface(n_dot_l, n_dot_v, l_dot_h, mat.roughness, mat.subsurface_scale);\n            // Tint subsurface\n            result.diffuse = mat.base_color * mix(vec3<f32>(1.0), mat.subsurface_color, mat.subsurface_scale) * fd;\n        } else {\n            fd = diffuse_burley(n_dot_v, n_dot_l, l_dot_h, mat.roughness);\n            result.diffuse = mat.base_color * fd;\n        }\n        result.diffuse *= (1.0 - mat.metallic);\n    }\n\n    // --- Specular (base lobe) ---\n    {\n        var d: f32;\n        var vis: f32;\n\n        if (has_flag(mat.flags, FLAG_ANISOTROPY) && abs(mat.anisotropy) > 0.01) {\n            // Anisotropic specular\n            let aspect = sqrt(1.0 - mat.anisotropy * 0.9);\n            let alpha_t = alpha / aspect;\n            let alpha_b = alpha * aspect;\n\n            // Rotate tangent by anisotropy_rotation\n            let cos_r = cos(mat.anisotropy_rotation * PI * 2.0);\n            let sin_r = sin(mat.anisotropy_rotation * PI * 2.0);\n            let rt = t * cos_r + b * sin_r;\n            let rb = -t * sin_r + b * cos_r;\n\n            d = d_ggx_aniso(n_dot_h, h, rt, rb, alpha_t, alpha_b);\n            vis = v_smith_ggx_aniso(n_dot_v, n_dot_l, v, l, rt, rb, alpha_t, alpha_b);\n        } else {\n            d = d_ggx(n_dot_h, alpha);\n            vis = v_smith_ggx(n_dot_v, n_dot_l, alpha);\n        }\n\n        let f = fresnel_schlick(l_dot_h, f0);\n        result.specular = d * vis * f;\n    }\n\n    // --- Clearcoat ---\n    if (has_flag(mat.flags, FLAG_CLEARCOAT) && mat.clearcoat > 0.0) {\n        let cc_alpha = max(mat.clearcoat_roughness * mat.clearcoat_roughness, 0.002);\n        let cc_f0 = f0_from_ior(1.5); // fixed IOR for clearcoat\n        let d_cc = d_ggx(n_dot_h, cc_alpha);\n        let v_cc = v_kelemen(l_dot_h);\n        let f_cc = fresnel_schlick_scalar(l_dot_h, cc_f0);\n        result.specular += vec3<f32>(d_cc * v_cc * f_cc * mat.clearcoat);\n    }\n\n    // --- Sheen ---\n    if (has_flag(mat.flags, FLAG_SHEEN) && length(mat.sheen_color) > 0.001) {\n        let d_sheen = d_charlie(n_dot_h, mat.sheen_roughness);\n        let f_sheen = fresnel_schlick(l_dot_h, mat.sheen_color);\n        result.diffuse += d_sheen * f_sheen * (1.0 - mat.metallic);\n    }\n\n    // --- Transmission ---\n    if (has_flag(mat.flags, FLAG_TRANSMISSION) && mat.transmission > 0.0) {\n        // Simplified: treat as modulated specular transmission\n        let f_t = 1.0 - fresnel_schlick_scalar(n_dot_v, f0_from_ior(mat.ior));\n        let attenuation = exp(-mat.base_color * (1.0 / max(mat.attenuation_distance, 0.001)));\n        result.diffuse = mix(result.diffuse, mat.attenuation_color * attenuation, mat.transmission * f_t);\n    }\n\n    // Scale by n_dot_l (caller applies radiance and shadow)\n    result.diffuse *= n_dot_l;\n    result.specular *= n_dot_l;\n\n    return result;\n}\n\n// ======================================================================\n// 6. Environment (IBL) BRDF evaluation\n// ======================================================================\n\nfn evaluate_disney_ibl(\n    mat:       DisneyMaterial,\n    n:         vec3<f32>,\n    v:         vec3<f32>,\n    irradiance: vec3<f32>,    // diffuse irradiance from probe/SH\n    prefiltered: vec3<f32>,   // specular prefiltered from cubemap\n    brdf_lut:  vec2<f32>,     // (scale, bias) from BRDF integration LUT\n) -> vec3<f32> {\n    let n_dot_v = max(dot(n, v), 1e-5);\n    let f0 = mix(vec3<f32>(0.04), mat.base_color, mat.metallic);\n\n    // Diffuse IBL\n    let kd = (1.0 - mat.metallic) * (1.0 - mat.transmission);\n    let diffuse = irradiance * mat.base_color * kd;\n\n    // Specular IBL (split-sum approximation)\n    let f = fresnel_schlick_roughness(n_dot_v, f0, mat.roughness);\n    let specular = prefiltered * (f * brdf_lut.x + brdf_lut.y);\n\n    // Clearcoat IBL\n    var cc = vec3<f32>(0.0);\n    if (has_flag(mat.flags, FLAG_CLEARCOAT) && mat.clearcoat > 0.0) {\n        let cc_f0 = f0_from_ior(1.5);\n        let cc_f = fresnel_schlick_scalar(n_dot_v, cc_f0);\n        cc = prefiltered * cc_f * mat.clearcoat;\n    }\n\n    return (diffuse + specular + cc) * mat.occlusion;\n}\n";
