The LUT is parameterized by (NdotV, roughness) and stores\n// (scale, bias) for the Fresnel approximation:\n//   specular = prefiltered * (F0 * scale + bias)\n//\n// Only needs to be computed once (resolution-independent).\n\nstruct BrdfLutParams {\n    lut_size:    u32,\n    num_samples: u32,\n    _pad0:       u32,\n    _pad1:       u32,\n};\n\n@group(0) @binding(0) var<uniform>  params:   BrdfLutParams;\n@group(0) @binding(1) var           t_output: texture_storage_2d<rgba16float, write>;\n\nconst PI: f32 = 3.14159265358979;\n\n// Hammersley quasi-random sequence\nfn radical_inverse_vdc(bits_in: u32) -> f32 {\n    var bits = bits_in;\n    bits = (bits << 16u) | (bits >> 16u);\n    bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u);\n    bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u);\n    bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u);\n    bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u);\n    return f32(bits) * 2.3283064365386963e-10;\n}\n\nfn hammersley(i: u32, n: u32) -> vec2<f32> {\n    return vec2<f32>(f32(i) / f32(n), radical_inverse_vdc(i));\n}\n\n// Importance sample GGX\nfn importance_sample_ggx(xi: vec2<f32>, roughness: f32) -> vec3<f32> {\n    let a = roughness * roughness;\n    let phi = 2.0 * PI * xi.x;\n    let cos_theta = sqrt((1.0 - xi.y) / (1.0 + (a * a - 1.0) * xi.y));\n    let sin_theta = sqrt(1.0 - cos_theta * cos_theta);\n    return vec3<f32>(\n        cos(phi) * sin_theta,\n        sin(phi) * sin_theta,\n        cos_theta,\n    );\n}\n\n// Smith G1 (GGX)\nfn geometry_schlick_ggx(n_dot_x: f32, roughness: f32) -> f32 {\n    let k = (roughness * roughness) / 2.0;\n    return n_dot_x / (n_dot_x * (1.0 - k) + k);\n}\n\nfn geometry_smith(n_dot_v: f32, n_dot_l: f32, roughness: f32) -> f32 {\n    return geometry_schlick_ggx(n_dot_v, roughness) * geometry_schlick_ggx(n_dot_l, roughness);\n}\n\n@compute @workgroup_size(8, 8)\nfn brdf_lut_main(@builtin(global_invocation_id) gid: vec3<u32>) {\n    if (gid.x >= params.lut_size || gid.y >= params.lut_size) {\n        return;\n    }\n\n    let n_dot_v = max((f32(gid.x) + 0.5) / f32(params.lut_size), 0.001);\n    let roughness = max((f32(gid.y) + 0.5) / f32(params.lut_size), 0.001);\n\n    // View direction in tangent space (N = (0,0,1))\n    let v = vec3<f32>(sqrt(1.0 - n_dot_v * n_dot_v), 0.0, n_dot_v);\n    let n = vec3<f32>(0.0, 0.0, 1.0);\n\n    var scale = 0.0;\n    var bias = 0.0;\n\n    for (var i = 0u; i < params.num_samples; i++) {\n        let xi = hammersley(i, params.num_samples);\n        let h = importance_sample_ggx(xi, roughness);\n        let l = normalize(2.0 * dot(v, h) * h - v);\n\n        let n_dot_l = max(l.z, 0.0);\n        let n_dot_h = max(h.z, 0.0);\n        let v_dot_h = max(dot(v, h), 0.0);\n\n        if (n_dot_l > 0.0) {\n            let g = geometry_smith(n_dot_v, n_dot_l, roughness);\n            let g_vis = (g * v_dot_h) / (n_dot_h * n_dot_v + 1e-7);\n            let fc = pow(1.0 - v_dot_h, 5.0);\n\n            scale += (1.0 - fc) * g_vis;\n            bias += fc * g_vis;\n        }\n    }\n\n    scale /= f32(params.num_samples);\n    bias /= f32(params.num_samples);\n\n    textureStore(t_output, vec2<i32>(gid.xy), vec4<f32>(scale, bias, 0.0, 0.0));\n}\n";
