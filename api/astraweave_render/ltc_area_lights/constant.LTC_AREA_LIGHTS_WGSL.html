\u{2014} inverse M matrix coefficients [a, 0, b; 0, c, 0; d, 0, 1]\n//      packed as (a, b, c, d) in RGBA.\n//   2. ltc_amplitude: RG16Float \u{2014} (magnitude, fresnel) for energy normalization.\n//\n// Area light types: rectangular (2 half-extents), disk (radius), tube (length + radius).\n\nstruct AreaLight {\n    position: vec3<f32>,    // center of the area light\n    light_type: u32,        // 0 = rect, 1 = disk, 2 = tube\n    right: vec3<f32>,       // half-extent in X (scaled by width/2)\n    width: f32,             // full width\n    up: vec3<f32>,          // u32,\n    _pad0: u32,\n    _pad1: u32,\n    _pad2: u32,\n};\n\n// Bind group for area lights (appended to existing lighting groups)\n@group(6) @binding(0) var<storage, read> area_lights: array<AreaLight>;\n@group(6) @binding(1) var<uniform> area_params: AreaLightParams;\n@group(6) @binding(2) var ltc_matrix_tex: texture_2d<f32>;\n@group(6) @binding(3) var ltc_amplitude_tex: texture_2d<f32>;\n@group(6) @binding(4) var ltc_sampler: sampler;\n\n// Camera UBO (same as main PBR pass)\nstruct Camera {\n    view: mat4x4<f32>,\n    proj: mat4x4<f32>,\n    view_proj: mat4x4<f32>,\n    position: vec3<f32>,\n    _pad: f32,\n};\n@group(0) @binding(0) var<uniform> camera: Camera;\n\n// \u{2500}\u{2500}\u{2500} LTC Matrix Reconstruction \u{2500}\u{2500}\u{2500}\n// The LUT encodes the inverse of the linear transform M that maps\n// the clamped cosine distribution to the BRDF lobe.\n// LUT coords: (roughness, cos_theta) \u{2192} [0,1]\u{b2} \u{2192} texel (64\u{d7}64)\nfn ltc_coords(roughness: f32, cos_theta: f32) -> vec2<f32> {\n    // Roughness on X, cos_theta on Y, both clamped to [0,1]\n    let r = clamp(roughness, 0.0, 1.0);\n    let ct = clamp(cos_theta, 0.0, 1.0);\n    // Bias to texel centers for 64\u{d7}64 LUT\n    let scale = (64.0 - 1.0) / 64.0;\n    let bias = 0.5 / 64.0;\n    return vec2<f32>(r * scale + bias, ct * scale + bias);\n}\n\nfn ltc_sample_matrix(uv: vec2<f32>) -> mat3x3<f32> {\n    let t = textureSampleLevel(ltc_matrix_tex, ltc_sampler, uv, 0.0);\n    // Reconstruct 3\u{d7}3 matrix from packed (a, b, c, d)\n    // M^-1 = [a, 0, b; 0, c, 0; d, 0, 1]\n    return mat3x3<f32>(\n        vec3<f32>(t.x, 0.0, t.y),\n        vec3<f32>(0.0, t.z, 0.0),\n        vec3<f32>(t.w, 0.0, 1.0)\n    );\n}\n\nfn ltc_sample_amplitude(uv: vec2<f32>) -> vec2<f32> {\n    let t = textureSampleLevel(ltc_amplitude_tex, ltc_sampler, uv, 0.0);\n    return t.xy; // (magnitude, fresnel term)\n}\n\n// \u{2500}\u{2500}\u{2500} Polygon Clipping to Hemisphere \u{2500}\u{2500}\u{2500}\n// Clip quad polygon to upper hemisphere (z > 0) before integration.\n// Returns clipped vertex count (3-5) and writes vertices to out_poly.\nfn clip_quad_to_hemisphere(\n    v0: vec3<f32>, v1: vec3<f32>, v2: vec3<f32>, v3: vec3<f32>,\n    out_poly: ptr<function, array<vec3<f32>, 5>>,\n) -> u32 {\n    // Simple clip: for each edge, if it crosses z=0, insert intersection point.\n    // For efficiency, we handle the common case (no clipping needed) first.\n    var vertices = array<vec3<f32>, 4>(v0, v1, v2, v3);\n    var above = array<bool, 4>(v0.z > 0.0, v1.z > 0.0, v2.z > 0.0, v3.z > 0.0);\n\n    var all_above = true;\n    var all_below = true;\n    for (var i = 0u; i < 4u; i++) {\n        if (above[i]) { all_below = false; } else { all_above = false; }\n    }\n\n    if (all_below) {\n        return 0u;\n    }\n\n    if (all_above) {\n        (*out_poly)[0] = v0;\n        (*out_poly)[1] = v1;\n        (*out_poly)[2] = v2;\n        (*out_poly)[3] = v3;\n        return 4u;\n    }\n\n    // General clip: iterate edges, emit vertices\n    var count = 0u;\n    for (var i = 0u; i < 4u; i++) {\n        let j = (i + 1u) % 4u;\n        let a = vertices[i];\n        let b = vertices[j];\n\n        if (above[i]) {\n            (*out_poly)[count] = a;\n            count++;\n        }\n\n        // If edge crosses z=0, add intersection\n        if (above[i] != above[j]) {\n            let t = a.z / (a.z - b.z);\n            let intersect = a + t * (b - a);\n            (*out_poly)[count] = intersect;\n            count++;\n        }\n\n        if (count >= 5u) {\n            break;\n        }\n    }\n\n    return min(count, 5u);\n}\n\n// \u{2500}\u{2500}\u{2500} Integrate Clipped Polygon \u{2500}\u{2500}\u{2500}\n// Analytic irradiance integration of a polygon on the upper hemisphere.\n// Uses the edge-integral formula from Heitz et al. 2016 / Baum et al. 1989.\nfn integrate_edge(v0: vec3<f32>, v1: vec3<f32>) -> f32 {\n    let cos_theta = dot(v0, v1);\n    let cos_theta_clamped = clamp(cos_theta, -0.9999, 0.9999);\n    let theta = acos(cos_theta_clamped);\n    let cross_z = v0.x * v1.y - v0.y * v1.x; // z-component of cross product\n\n    // Edge integral: theta * cross(v0, v1).z / sin(theta)\n    // Use safe division to avoid NaN when theta \u{2248} 0\n    let sin_theta = sin(theta);\n    if (abs(sin_theta) < 1e-6) {\n        return 0.0;\n    }\n    return cross_z * theta / sin_theta;\n}\n\nfn ltc_integrate_polygon(poly: ptr<function, array<vec3<f32>, 5>>, count: u32) -> f32 {\n    if (count < 3u) {\n        return 0.0;\n    }\n\n    var irradiance = 0.0;\n    for (var i = 0u; i < count; i++) {\n        let j = (i + 1u) % count;\n        let a = normalize((*poly)[i]);\n        let b = normalize((*poly)[j]);\n        irradiance += integrate_edge(a, b);\n    }\n\n    // The result is the solid angle subtended, divided by 2\u{3c0}\n    return max(irradiance, 0.0) / TWO_PI;\n}\n\n// \u{2500}\u{2500}\u{2500} LTC Area Light Evaluation \u{2500}\u{2500}\u{2500}\n// Evaluates a single rectangular area light using LTC.\nfn evaluate_area_light(\n    light: AreaLight,\n    world_pos: vec3<f32>,\n    N: vec3<f32>,\n    V: vec3<f32>,\n    roughness: f32,\n    metallic: f32,\n    albedo: vec3<f32>,\n) -> vec3<f32> {\n    let cos_theta = clamp(dot(N, V), 0.0, 1.0);\n    let uv = ltc_coords(roughness, cos_theta);\n\n    let M_inv = ltc_sample_matrix(uv);\n    let amp = ltc_sample_amplitude(uv);\n\n    // Build local TBN frame\n    let T1 = normalize(V - N * dot(V, N));\n    let T2 = cross(N, T1);\n    let TBN = mat3x3<f32>(T1, T2, N);\n\n    // Transform quad vertices to local space\n    let delta = light.position - world_pos;\n    let right = light.right;\n    let up = light.up;\n\n    // Quad corners in world space (CCW winding)\n    let p0 = delta - right - up;\n    let p1 = delta + right - up;\n    let p2 = delta + right + up;\n    let p3 = delta - right + up;\n\n    // Transform to tangent space\n    let TBN_t = transpose(TBN);\n\n    // \u{2500}\u{2500} SPECULAR contribution \u{2500}\u{2500}\n    // Apply LTC inverse transform to quad vertices\n    let M_spec = M_inv;\n    let spec0 = M_spec * (TBN_t * p0);\n    let spec1 = M_spec * (TBN_t * p1);\n    let spec2 = M_spec * (TBN_t * p2);\n    let spec3 = M_spec * (TBN_t * p3);\n\n    // Clip to hemisphere and integrate\n    var spec_poly: array<vec3<f32>, 5>;\n    let spec_count = clip_quad_to_hemisphere(spec0, spec1, spec2, spec3, &spec_poly);\n    let spec_irradiance = ltc_integrate_polygon(&spec_poly, spec_count);\n\n    // \u{2500}\u{2500} DIFFUSE contribution \u{2500}\u{2500}\n    // Diffuse uses identity transform (clamped cosine = no LTC warp)\n    let diff0 = TBN_t * p0;\n    let diff1 = TBN_t * p1;\n    let diff2 = TBN_t * p2;\n    let diff3 = TBN_t * p3;\n\n    var diff_poly: array<vec3<f32>, 5>;\n    let diff_count = clip_quad_to_hemisphere(diff0, diff1, diff2, diff3, &diff_poly);\n    let diff_irradiance = ltc_integrate_polygon(&diff_poly, diff_count);\n\n    // \u{2500}\u{2500} Combine with Fresnel \u{2500}\u{2500}\n    let F0 = mix(vec3<f32>(0.04), albedo, metallic);\n    let fresnel = F0 * amp.x + (1.0 - F0) * amp.y;\n\n    let diffuse_color = albedo * (1.0 - metallic);\n    let specular = fresnel * spec_irradiance;\n    let diffuse = diffuse_color * diff_irradiance;\n\n    return (specular + diffuse) * light.color * light.intensity;\n}\n\n// \u{2500}\u{2500}\u{2500} Main Area Lighting Function \u{2500}\u{2500}\u{2500}\n// Call this from the main fragment shader after point/directional lighting.\nfn calculate_area_lighting(\n    world_pos: vec3<f32>,\n    N: vec3<f32>,\n    V: vec3<f32>,\n    albedo: vec3<f32>,\n    metallic: f32,\n    roughness: f32,\n) -> vec3<f32> {\n    var total = vec3<f32>(0.0);\n\n    for (var i = 0u; i < area_params.num_area_lights; i++) {\n        let light = area_lights[i];\n\n        // Distance-based early out\n        let delta = light.position - world_pos;\n        let dist = length(delta);\n        let max_range = max(light.width, light.height) * 5.0;\n        if (dist > max_range) {\n            continue;\n        }\n\n        total += evaluate_area_light(light, world_pos, N, V, roughness, metallic, albedo);\n    }\n\n    return total;\n}\n";
