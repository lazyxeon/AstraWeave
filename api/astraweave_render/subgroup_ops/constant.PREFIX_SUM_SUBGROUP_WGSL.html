pub const PREFIX_SUM_SUBGROUP_WGSL: &str = "// Subgroup-optimized Blelloch prefix sum (exclusive scan)\n//\n// Requires: enable subgroups;\n// Uses subgroupExclusiveAdd for intra-subgroup scan, then a small\n// workgroup-level fixup for inter-subgroup offsets.\n//\n// Performance improvement over standard Blelloch:\n// - Standard: O(log n) up-sweep + O(log n) down-sweep, each with workgroupBarrier()\n// - Subgroup: subgroupExclusiveAdd replaces log2(sg_size) steps in both phases\n// - For sg_size=32: eliminates 10 of 18 barriers for 512 elements\n\nenable subgroups;\n\nstruct PrefixSumParams {\n    element_count: u32,\n    workgroup_size: u32,\n    _pad1: u32,\n    _pad2: u32,\n}\n\n@group(0) @binding(0) var<storage, read> input: array<u32>;\n@group(0) @binding(1) var<storage, read_write> output: array<u32>;\n@group(0) @binding(2) var<uniform> params: PrefixSumParams;\n\n// Each subgroup produces a partial sum; we store those for inter-subgroup fixup\nvar<workgroup> subgroup_totals: array<u32, 16>; // max 16 subgroups per workgroup\nvar<workgroup> subgroup_offsets: array<u32, 16>;\n\n@compute @workgroup_size(256, 1, 1)\nfn prefix_sum(\n    @builtin(global_invocation_id) gid: vec3<u32>,\n    @builtin(local_invocation_id) lid: vec3<u32>,\n    @builtin(local_invocation_index) li: u32,\n    @builtin(subgroup_invocation_id) sg_id: u32,\n    @builtin(subgroup_size) sg_size: u32,\n) {\n    let tid = lid.x;\n    let gid_1d = gid.x;\n\n    // Each thread handles 2 elements (same as original)\n    let idx0 = gid_1d * 2u;\n    let idx1 = gid_1d * 2u + 1u;\n\n    var val0 = 0u;\n    var val1 = 0u;\n\n    if (idx0 < params.element_count) {\n        val0 = input[idx0];\n    }\n    if (idx1 < params.element_count) {\n        val1 = input[idx1];\n    }\n\n    // Step 1: Each thread sums its two values. We\'ll scan the per-thread sums,\n    // then reconstruct both output elements.\n    let thread_sum = val0 + val1;\n\n    // Step 2: Intra-subgroup exclusive scan using subgroupExclusiveAdd\n    // This replaces log2(sg_size) steps of the up-sweep + down-sweep\n    let sg_exclusive = subgroupExclusiveAdd(thread_sum);\n\n    // Step 3: The last thread in each subgroup stores the subgroup total\n    let subgroup_index = tid / sg_size;\n    let sg_total = sg_exclusive + thread_sum; // inclusive sum for this thread\n    if (sg_id == sg_size - 1u) {\n        subgroup_totals[subgroup_index] = sg_total;\n    }\n    workgroupBarrier();\n\n    // Step 4: First subgroup scans the subgroup totals (inter-subgroup fixup)\n    let num_subgroups = (256u + sg_size - 1u) / sg_size;\n    if (subgroup_index == 0u && sg_id < num_subgroups) {\n        let total = subgroup_totals[sg_id];\n        let offset = subgroupExclusiveAdd(total);\n        subgroup_offsets[sg_id] = offset;\n    }\n    workgroupBarrier();\n\n    // Step 5: Combine subgroup offset + intra-subgroup scan\n    let base_offset = subgroup_offsets[subgroup_index];\n    let thread_exclusive = base_offset + sg_exclusive;\n\n    // Step 6: Reconstruct both output elements\n    // thread_exclusive is the exclusive prefix sum for thread_sum\n    // val0 goes at thread_exclusive, val1 goes at thread_exclusive + val0\n    if (idx0 < params.element_count) {\n        output[idx0] = thread_exclusive;\n    }\n    if (idx1 < params.element_count) {\n        output[idx1] = thread_exclusive + val0;\n    }\n}\n";
