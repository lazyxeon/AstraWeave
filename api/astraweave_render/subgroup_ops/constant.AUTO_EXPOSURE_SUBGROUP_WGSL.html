Each thread computes its bin index\n//   2. Within the subgroup, threads voting for the same bin use subgroupAdd\n//      to aggregate their counts\n//   3. Only one representative per subgroup writes the aggregated count\n//      to the shared histogram (fewer atomic operations)\n//\n// Since subgroupAdd across different bin values is not directly useful,\n// we iterate over the 256 bins and let each thread contribute 1 if it\n// matches that bin, then subgroupAdd reduces the 1s within the subgroup.\n// This is O(256) per subgroup but each iteration is 1 instruction.\n//\n// OPTIMIZATION: Instead of iterating all 256 bins, we use the original\n// shared-memory approach but with subgroup-level pre-reduction.\n// Each thread atomicAdds to shared memory, but within a subgroup,\n// threads with the SAME bin can be summed first via ballot+popcount.\n\nvar<workgroup> local_histogram: array<atomic<u32>, 256>;\n\n@compute @workgroup_size(16, 16, 1)\nfn histogram_pass(\n    @builtin(global_invocation_id) gid: vec3<u32>,\n    @builtin(local_invocation_index) li: u32,\n    @builtin(subgroup_invocation_id) sg_id: u32,\n    @builtin(subgroup_size) sg_size: u32,\n) {\n    // Initialize local histogram\n    if (li < 256u) {\n        atomicStore(&local_histogram[li], 0u);\n    }\n    workgroupBarrier();\n\n    let dims = vec2<u32>(params.resolution);\n    if (gid.x < dims.x && gid.y < dims.y) {\n        let uv = (vec2<f32>(gid.xy) + 0.5) * params.inv_resolution;\n        let color = textureSampleLevel(hdr_tex, samp, uv, 0.0).rgb;\n        let lum = luminance(color);\n        let bin = luminance_to_bin(lum);\n\n        // Subgroup-optimized atomic add:\n        // Use subgroupAdd to sum contributions from threads in the same subgroup\n        // that map to the same bin.
