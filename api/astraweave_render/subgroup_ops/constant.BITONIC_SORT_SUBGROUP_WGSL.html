compare-and-swap operations,\n// eliminating global memory round-trips for inner sort stages where\n// stage_step <= subgroup_size.\n//\n// For a subgroup size of 32, this collapses 5 inner passes into a single\n// dispatch, reducing total host-side dispatch count by ~30-40%.\n\nenable subgroups;\n\nstruct SortParams {\n    algo_step:      u32,\n    stage_step:     u32,\n    num_particles:  u32,\n    _pad:           u32,\n    camera_pos:     vec3<f32>,\n    _pad2:          f32,\n};\n\nstruct SortEntry {\n    index:    u32,\n    distance: f32,\n};\n\n@group(0) @binding(0) var<uniform>              params:  SortParams;\n@group(0) @binding(1) var<storage, read_write>  entries: array<SortEntry>;\n\n// Subgroup-accelerated compare-and-swap\n// When stage_step fits within the subgroup, we can exchange data via\n// subgroupShuffleXor instead of reading/writing global memory.\n@compute @workgroup_size(256)\nfn bitonic_sort(\n    @builtin(global_invocation_id) gid: vec3<u32>,\n    @builtin(subgroup_invocation_id) sg_id: u32,\n    @builtin(subgroup_size) sg_size: u32,\n) {\n    let idx = gid.x;\n    if (idx >= params.num_particles) {\n        return;\n    }\n\n    let half_step = params.stage_step;\n    let step = params.algo_step;\n\n    // For inner stages where stage_step <= subgroup_size,\n    // the partner is within the same subgroup.\n    // We can use subgroupShuffleXor to exchange data directly.\n    if (half_step <= sg_size) {\n        // Load our entry\n        var my_entry = entries[idx];\n        var my_index = my_entry.index;\n        var my_dist = my_entry.distance;\n\n        // Exchange with partner via subgroup shuffle\n        // XOR with half_step gives the partner\'s subgroup lane\n        let partner_index = subgroupShuffleXor(my_index, half_step);\n        let partner_dist = subgroupShuffleXor(my_dist, half_step);\n\n        // Determine sort direction\n        let ascending = ((idx / step) % 2u) == 0u;\n\n        // Determine if we\'re in the \"lower\" half of the pair\n        let is_lower = (idx % step) < half_step;\n\n        // Compare\n        let should_swap = select(\n            my_dist < partner_dist,\n            my_dist > partner_dist,\n            ascending\n        );\n\n        // Apply swap: the lower-indexed thread takes the \"winner\"\n        if (is_lower && should_swap) {\n            my_entry.index = partner_index;\n            my_entry.distance = partner_dist;\n        } else if (!is_lower && should_swap) {\n            my_entry.index = partner_index;\n            my_entry.distance = partner_dist;\n        }\n\n        entries[idx] = my_entry;\n    } else {\n        // Fallback: standard global memory compare-and-swap for outer stages\n        let partner = }\n\n        let a = entries[idx];\n        let b = entries[partner];\n\n        let ascending = ((idx / step) % 2u) == 0u;\n        let should_swap = select(\n            a.distance < b.distance,\n  }\n    }\n}\n";
