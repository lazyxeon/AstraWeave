pub const STOCHASTIC_TILING_WGSL: &str = "// stochastic_tiling.wgsl \u{2014} Hex-tile stochastic texture sampling\n//\n// Breaks visible texture repetition by randomly rotating/offsetting samples\n// within a hexagonal grid. Based on Heitz & Neyret 2018 \"High-Performance\n// By-Example Noise using an Histogram-Preserving Blending Operator.\"\n//\n// Usage: Replace `textureSample(tex, samp, uv)` with\n//        `sample_stochastic(tex, samp, uv, seed)` wherever tiling artifacts\n//        are visible (typically large terrain surfaces).\n\n// ============================================================================\n// Hex grid helpers\n// ============================================================================\n\n// Single hash from 2D coords \u{2014} produces pseudo-random value in [0,1).\nfn hash2(p: vec2<f32>) -> f32 {\n    let h = dot(p, vec2<f32>(127.1, 311.7));\n    return fract(sin(h) * 43758.5453);\n}\n\n// Hash returning 2D pseudo-random offset.\nfn hash22(p: vec2<f32>) -> vec2<f32> {\n    let h = vec2<f32>(\n        dot(p, vec2<f32>(127.1, 311.7)),\n        dot(p, vec2<f32>(269.5, 183.3)),\n    );\n    return fract(sin(h) * 43758.5453);\n}\n\n// Hex grid coordinate decomposition. Returns the three nearest hex centers\n// and barycentric-like blend weights.\nstruct HexSample {\n    uv0: vec2<f32>,\n    uv1: vec2<f32>,\n    uv2: vec2<f32>,\n    w0: f32,\n    w1: f32,\n    w2: f32,\n}\n\nfn hex_coords(uv: vec2<f32>) -> HexSample {\n    // Transform to hexagonal grid (skewed coordinates)\n    let sqrt3 = 1.7320508;\n    let inv_sqrt3 = 0.57735026;\n\n    let skewed = vec2<f32>(uv.x + uv.y * inv_sqrt3, uv.y * 2.0 * inv_sqrt3);\n    let base = floor(skewed);\n    let f = fract(skewed);\n\n    // Determine which simplex triangle we are in\n    var v0: vec2<f32>;\n    var v1: vec2<f32>;\n    var v2: vec2<f32>;\n\n    if (f.x + f.y < 1.0) {\n        v0 = base;\n        v1 = base + vec2<f32>(1.0, 0.0);\n        v2 = base + vec2<f32>(0.0, 1.0);\n    } else {\n        v0 = base + vec2<f32>(1.0, 1.0);\n        v1 = base + vec2<f32>(1.0, 0.0);\n        v2 = base + vec2<f32>(0.0, 1.0);\n    }\n\n    // Barycentric-like weights from distance to vertices\n    let d0 = length(skewed - v0);\n    let d1 = length(skewed - v1);\n    let d2 = length(skewed - v2);\n\n    // Inverse distance weighting\n    let inv0 = 1.0 / max(d0, 0.0001);\n    let inv1 = 1.0 / max(d1, 0.0001);\n    let inv2 = 1.0 / max(d2, 0.0001);\n    let inv_total = inv0 + inv1 + inv2;\n\n    // Random UV offsets per hex cell\n    let off0 = hash22(v0);\n    let off1 = hash22(v1);\n    let off2 = hash22(v2);\n\n    // Random rotation per cell (applied as UV rotation)\n    let a0 = hash2(v0 + vec2<f32>(73.0, 19.0)) * 6.2831853;\n    let a1 = hash2(v1 + vec2<f32>(73.0, 19.0)) * 6.2831853;\n    let a2 = hash2(v2 + vec2<f32>(73.0, 19.0)) * 6.2831853;\n\n    var result: HexSample;\n    result.uv0 = rotate_uv(uv, a0) + off0;\n    result.uv1 = rotate_uv(uv, a1) + off1;\n    result.uv2 = rotate_uv(uv, a2) + off2;\n    result.w0 = inv0 / inv_total;\n    result.w1 = inv1 / inv_total;\n    result.w2 = inv2 / inv_total;\n\n    return result;\n}\n\n// Rotate UV coordinates around (0.5, 0.5) center.\nfn rotate_uv(uv: vec2<f32>, angle: f32) -> vec2<f32> {\n    let c = cos(angle);\n    let s = sin(angle);\n    let centered = uv - vec2<f32>(0.5);\n    return vec2<f32>(\n        centered.x * c - centered.y * s,\n        centered.x * s + centered.y * c,\n    ) + vec2<f32>(0.5);\n}\n\n// ============================================================================\n// Main sampling function\n// ============================================================================\n\n// Stochastic hex-tile texture sample. Drop-in replacement for textureSample\n// that eliminates visible tiling patterns.\n//\n// `scale` controls hex cell density \u{2014} higher values = smaller cells = less\n// visible pattern but more texture lookups. Typical range: 0.5 - 2.0.\nfn sample_stochastic(\n    tex: texture_2d<f32>,\n    samp: sampler,\n    uv: vec2<f32>,\n    scale: f32,\n) -> vec4<f32> {\n    let hex = hex_coords(uv * scale);\n\n    let s0 = textureSample(tex, samp, hex.uv0);\n    let s1 = textureSample(tex, samp, hex.uv1);\n    let s2 = textureSample(tex, samp, hex.uv2);\n\n    return s0 * hex.w0 + s1 * hex.w1 + s2 * hex.w2;\n}\n\n// Stochastic sample for texture arrays.\nfn sample_stochastic_array(\n    tex: texture_2d_array<f32>,\n    samp: sampler,\n    uv: vec2<f32>,\n    layer: u32,\n    scale: f32,\n) -> vec4<f32> {\n    let hex = hex_coords(uv * scale);\n\n    let s0 = textureSample(tex, samp, hex.uv0, layer);\n    let s1 = textureSample(tex, samp, hex.uv1, layer);\n    let s2 = textureSample(tex, samp, hex.uv2, layer);\n\n    return s0 * hex.w0 + s1 * hex.w1 + s2 * hex.w2;\n}\n";

WGSL source for stochastic hex-tile sampling utilities.

Contains the following functions:

  • sample_stochastic(tex, samp, uv, scale) — standard texture2d
  • sample_stochastic_array(tex, samp, uv, layer, scale) — texture2d_array
  • hex_coords(uv) — raw hex grid decomposition
  • rotate_uv(uv, angle) — UV rotation helper
