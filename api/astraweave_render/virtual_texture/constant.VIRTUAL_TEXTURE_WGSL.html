The CPU reads back the buffer\n// to decide which pages to stream from disk.\n\nstruct FeedbackParams {\n    // Virtual texture parameters\n    vt_size:           vec2<f32>,   // full virtual texture size in pixels\n    inv_vt_size:       vec2<f32>,   // 1 / vt_size\n    page_size:         f32,         // page size in texels (e.g., 128)\n    inv_page_size:     f32,         // 1 / page_size\n    max_mip_level:     f32,         // highest mip level\n    padding:           f32,\n    // Screen info\n    screen_size:       vec2<f32>,   // render target size\n    inv_screen_size:   vec2<f32>,   // 1 / screen_size\n}\n\n// Page request entry: packed (page_x, page_y, mip_level) into a u32.\n// Layout: bits [0..9] = page_x, [10..19] = page_y, [20..24] = mip_level, [25..31] = reserved\nfn pack_page_id(page_x: u32, page_y: u32, mip: u32) -> u32 {\n    return (page_x & 0x3FFu) | ((page_y & 0x3FFu) << 10u) | ((mip & 0x1Fu) << 20u);\n}\n\nfn unpack_page_x(packed: u32) -> u32 {\n    return packed & 0x3FFu;\n}\n\nfn unpack_page_y(packed: u32) -> u32 {\n    return (packed >> 10u) & 0x3FFu;\n}\n\nfn unpack_mip(packed: u32) -> u32 {\n    return (packed >> 20u) & 0x1Fu;\n}\n\n@group(0) @binding(0) var<uniform> params: FeedbackParams;\n@group(0) @binding(1) var<storage, read_write> feedback_buffer: array<atomic<u32>>;\n@group(0) @binding(2) var<storage, read_write> feedback_count: atomic<u32>;\n@group(0) @binding(3) var depth_tex: texture_2d<f32>;\n\n// Compute the mip level from UV derivatives (approximation using screen-space\n// texel density).\nfn compute_mip_level(uv: vec2<f32>, screen_pos: vec2<f32>) -> f32 {\n    // Approximate foot-print: measure UV change per pixel via central\n    // differences on the depth buffer (a rough proxy for \u{2202}uv/\u{2202}screen).\n    let texel_density = length(uv * params.vt_size) * params.inv_screen_size.x;\n    let mip = log2(max(texel_density, 1.0));\n    return clamp(mip, 0.0, params.max_mip_level);\n}\n\noverride WG_X: u32 = 8u;\noverride WG_Y: u32 = 8u;\n\n@compute @workgroup_size(WG_X, WG_Y)\nfn feedback_pass(@builtin(global_invocation_id) gid: vec3<u32>) {\n    let screen_pos = vec2<f32>(f32(gid.x), f32(gid.y));\n    if (screen_pos.x >= params.screen_size.x || screen_pos.y >= params.screen_size.y) {\n        return;\n    }\n\n    // Read depth to skip sky pixels\n    let depth = textureLoad(depth_tex, vec2<i32>(gid.xy), 0).r;\n    if (depth >= 1.0) {\n        return; // sky \u{2014} no VT page needed\n    }\n\n    // Reconstruct UV from screen position (simplified: identity mapping for terrain)\n    let uv = screen_pos * params.inv_screen_size;\n\n    // Determine required mip level\n    let mip_f = compute_mip_level(uv, screen_pos);\n    let mip = u32(floor(mip_f));\n\n    // Convert UV to page coordinates at this mip level\n    let mip_scale = 1.0 / f32(1u << mip);\n    let pages_at_mip = params.vt_size * params.inv_page_size * mip_scale;\n    let page_x = u32(clamp(uv.x * pages_at_mip.x, 0.0, pages_at_mip.x - 1.0));\n    let page_y = u32(clamp(uv.y * pages_at_mip.y, 0.0, pages_at_mip.y - 1.0));\n\n    // Write page request (deduplicated via atomic compare-and-swap would be\n    // ideal, but for simplicity we use append with a counter).\n    let page_id = pack_page_id(page_x, page_y, mip);\n    let slot = atomicAdd(&feedback_count, 1u);\n    let max_entries = arrayLength(&feedback_buffer);\n    if (slot < max_entries) {\n        atomicStore(&feedback_buffer[slot], page_id);\n    }\n}\n";
