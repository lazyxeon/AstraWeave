pub const CLIPMAP_TERRAIN_WGSL: &str = "// Geometry clipmap mesh generation for terrain LOD.\n//\n// Generates concentric \"rings\" of terrain geometry at increasing LOD levels,\n// centered on the camera. Inner rings have higher vertex density; outer rings\n// are coarser. Shared edges use transition/stitch geometry to avoid T-junctions.\n//\n// Reference: Strugar (2009) \"Continuous Distance-Dependent Level of Detail for\n// Rendering Heightmaps.\" Losasso & Hoppe (2004) \"Geometry Clipmaps.\"\n\nstruct ClipmapUniforms {\n    view_proj:        mat4x4<f32>,\n    camera_pos:       vec3<f32>,\n    clipmap_scale:    f32,          // world-space size of finest grid cell\n    morph_constants:  vec4<f32>,    // (morph_start, 1/(morph_end - morph_start), 0, 0)\n    heightmap_size:   vec2<f32>,    // width, height in texels\n    inv_heightmap_size: vec2<f32>,  // 1/width, 1/height\n}\n\n@group(0) @binding(0) var<uniform> u: ClipmapUniforms;\n@group(0) @binding(1) var heightmap_tex: texture_2d<f32>;\n@group(0) @binding(2) var heightmap_sampler: sampler;\n\nstruct VertexInput {\n    @location(0) grid_pos: vec2<f32>,   // integer grid coordinates within this ring\n    @location(1) ring_info: vec4<f32>,  // (ring_level, ring_scale, morph_start_dist, morph_range_inv)\n}\n\nstruct VertexOutput {\n    @builtin(position) clip_pos: vec4<f32>,\n    @location(0) world_pos:  vec3<f32>,\n    @location(1) uv:         vec2<f32>,\n    @location(2) normal:     vec3<f32>,\n    @location(3) morph_factor: f32,\n}\n\n// Sample height from the heightmap texture at world XZ coordinates.\nfn sample_height(world_xz: vec2<f32>) -> f32 {\n    let uv = world_xz * u.inv_heightmap_size + 0.5;\n    return textureSampleLevel(heightmap_tex, heightmap_sampler, uv, 0.0).r;\n}\n\n// Compute normal from heightmap via central differences (2 taps).\nfn compute_normal(world_xz: vec2<f32>, cell_size: f32) -> vec3<f32> {\n    let step = cell_size;\n    let hL = sample_height(world_xz - vec2<f32>(step, 0.0));\n    let hR = sample_height(world_xz + vec2<f32>(step, 0.0));\n    let hD = sample_height(world_xz - vec2<f32>(0.0, step));\n    let hU = sample_height(world_xz + vec2<f32>(0.0, step));\n    return normalize(vec3<f32>(hL - hR, 2.0 * step, hD - hU));\n}\n\n// Snap grid position to the next coarser LOD grid to enable morphing.\nfn snap_to_parent(grid_pos: vec2<f32>) -> vec2<f32> {\n    return floor(grid_pos * 0.5 + 0.5) * 2.0;\n}\n\n// Compute morph factor based on distance: 0 = full detail, 1 = fully morphed to parent LOD.\nfn compute_morph_factor(distance: f32, morph_start: f32, morph_range_inv: f32) -> f32 {\n    return clamp((distance - morph_start) * morph_range_inv, 0.0, 1.0);\n}\n\n@vertex\nfn vs_main(input: VertexInput) -> VertexOutput {\n    var out: VertexOutput;\n\n    let ring_level = input.ring_info.x;\n    let cell_size  = u.clipmap_scale * input.ring_info.y; // scale doubles per ring\n\n    // Snap to camera-aligned grid: the grid follows the camera at the\n    // resolution of the current ring level.\n    let grid_snap = cell_size * 2.0;\n    let snapped_camera = floor(u.camera_pos.xz / grid_snap + 0.5) * grid_snap;\n\n    // World position from grid coordinates\n    let world_xz = snapped_camera + input.grid_pos * cell_size;\n\n    // Distance-based morph toward parent LOD grid\n    let dist = length(world_xz - u.camera_pos.xz);\n    let morph = compute_morph_factor(dist, input.ring_info.z, input.ring_info.w);\n\n    // Interpolate between current and parent grid position\n    let parent_xz = snapped_camera + snap_to_parent(input.grid_pos) * cell_size;\n    let morphed_xz = mix(world_xz, parent_xz, morph);\n\n    // Sample height at morphed position\n    let height = sample_height(morphed_xz);\n    let world_pos = vec3<f32>(morphed_xz.x, height, morphed_xz.y);\n\n    out.world_pos = world_pos;\n    out.clip_pos  = u.view_proj * vec4<f32>(world_pos, 1.0);\n    out.uv        = morphed_xz * u.inv_heightmap_size + 0.5;\n    out.normal    = compute_normal(morphed_xz, cell_size);\n    out.morph_factor = morph;\n\n    return out;\n}\n\n@fragment\nfn fs_main(input: VertexOutput) -> @location(0) vec4<f32> {\n    // Simple shading: directional light + ambient.\n    // In production this would feed into the full PBR pipeline.\n    let light_dir = normalize(vec3<f32>(0.3, 1.0, 0.5));\n    let ndl = max(dot(input.normal, light_dir), 0.0);\n    let ambient = 0.15;\n    let diffuse = ndl * 0.85;\n    let color = vec3<f32>(0.4, 0.55, 0.3) * (ambient + diffuse);\n    return vec4<f32>(color, 1.0);\n}\n";
